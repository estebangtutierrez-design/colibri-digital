// API Key validation middleware
function apiKeyMiddleware(req, res, next) {
  const key = req.headers["x-api-key"]
  const client = apiKeys.find(k => k.key === key)
  
  if (!client) return res.status(401).json({ error: "Invalid API key" })
  
  req.client = client
  next()
}
