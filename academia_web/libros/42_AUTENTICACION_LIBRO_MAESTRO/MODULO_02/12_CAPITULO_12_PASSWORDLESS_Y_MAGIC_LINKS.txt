EJEMPLO DE CÓDIGO
----------------------------------------

// Magic Link
const token = crypto.randomBytes(32).toString("hex")
await redis.setEx(`magic:${token}`, 900, email)
// Enviar email: https://app.com/auth/magic?token=${token}

// Verificar
app.get("/auth/magic", async (req, res) => {
  const email = await redis.get(`magic:${req.query.token}`)
  if (!email) return res.status(400).send("Invalid or expired link")
  // Log in user
  req.login(user)
  res.redirect("/dashboard")
})

// WebAuthn: navegador genera key pair
// Public key stored, private key en device (TouchID/Windows Hello)
