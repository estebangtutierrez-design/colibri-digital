```typescript
// src/utilidades.ts
import { type ResultadoOperacion } from "./tipos";

// ============================================================
// GENERADORES DE IDENTIFICADORES
// ============================================================

function generarID(prefix: string): string {
  const timestamp = Date.now().toString(36);
  const random = Math.random().toString(36).substring(2, 8);
  return `${prefix}_${timestamp}_${random}`;
}

function generarISBN(): string {
  const partes = [
    Math.floor(Math.random() * 1000).toString().padStart(3, "0"),
    Math.floor(Math.random() * 10000).toString().padStart(4, "0"),
    Math.floor(Math.random() * 1000).toString().padStart(3, "0"),
  ];
  return partes.join("-");
}

// ============================================================
// VALIDACIONES
// ============================================================

function esEmailValido(email: string): boolean {
  const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return regex.test(email);
}

function esISBNValido(isbn: string): boolean {
  return /^\d{3}-\d{4}-\d{3}$/.test(isbn);
}

function crearResultadoExito<T>(mensaje: string, datos?: T): ResultadoOperacion<T> {
  return { exito: true, mensaje, datos };
}

function crearResultadoError(mensaje: string): ResultadoOperacion {
  return { exito: false, mensaje };
}

// ============================================================
// FORMATEO
// ============================================================

function formatearFecha(fecha: Date): string {
  const dia = fecha.getDate().toString().padStart(2, "0");
  const mes = (fecha.getMonth() + 1).toString().padStart(2, "0");
  const anio = fecha.getFullYear();
  return `${dia}/${mes}/${anio}`;
}

function formatearMensaje(titulo: string, lineas: string[]): void {
  const separator = "=".repeat(50);
  console.log(`\n${separator}`);
  console.log(`  ${titulo}`);
  console.log(separator);
  lineas.forEach((linea) => console.log(`  ${linea}`));
  console.log(`${separator}\n`);
}

export {
  generarID,
  generarISBN,
  esEmailValido,
  esISBNValido,
  crearResultadoExito,
  crearResultadoError,
  formatearFecha,
  formatearMensaje,
};
```

Gestión de la colección de libros.

```typescript
// src/libros.ts
import {
  type Libro,
  type ISBN,
  type GeneroLiterario,
  type ResultadoOperacion,
  type BusquedaLibro,
} from "./tipos";
import { generarISBN, crearResultadoExito, crearResultadoError } from "./utilidades";

// ============================================================
// ESTADO INTERNO
// ============================================================

let libros: Libro[] = [];

// ============================================================
// OPERACIONES CRUD
// ============================================================

function agregarLibro(
  titulo: string,
  autor: string,
  anio: number,
  genero: GeneroLiterario,
  copias: number = 1
): ResultadoOperacion<Libro> {
  // Validaciones básicas
  if (!titulo.trim()) return crearResultadoError("El título no puede estar vacío.");
  if (!autor.trim()) return crearResultadoError("El autor no puede estar vacío.");
  if (anio < 1900 || anio > new Date().getFullYear() + 1) {
    return crearResultadoError("Año de publicación inválido.");
  }
  if (copias <= 0) return crearResultadoError("Debe haber al menos una copia.");

  // Inferencia: TypeScript sabe que esto es un Libro
  const nuevoLibro: Libro = {
    isbn: generarISBN(),
    titulo: titulo.trim(),
    autor: autor.trim(),
    anioPublicacion: anio,
    genero,
    copiasDisponibles: copias,
    copiasTotales: copias,
  };

  libros.push(nuevoLibro);
  return crearResultadoExito(`Libro "${nuevoLibro.titulo}" agregado.`, nuevoLibro);
}

function buscarLibros(criterio: BusquedaLibro): Libro[] {
  const { criterio: campo, valor } = criterio;
  const valorLower = valor.toLowerCase();

  return libros.filter((libro) => {
    switch (campo) {
      case "titulo":
        return libro.titulo.toLowerCase().includes(valorLower);
      case "autor":
        return libro.autor.toLowerCase().includes(valorLower);
      case "genero":
        return libro.genero === valor;
      case "isbn":
        return libro.isbn === valor;
      default:
        return false;
    }
  });
}

function obtenerLibroPorISBN(isbn: ISBN): Libro | undefined {
  return libros.find((libro) => libro.isbn === isbn);
}

function listarLibros(): Libro[] {
  return [...libros]; // Devolver copia para evitar mutaciones externas
}

function actualizarCopias(isbn: ISBN, disponibles: number, totales: number): ResultadoOperacion<Libro> {
  const libro = obtenerLibroPorISBN(isbn);
  if (!libro) {
    return crearResultadoError(`Libro con ISBN ${isbn} no encontrado.`);
  }

  libro.copiasDisponibles = disponibles;
  libro.copiasTotales = totales;

  return crearResultadoExito(`Copias actualizadas para "${libro.titulo}".`, libro);
}

function libroTieneCopiasDisponibles(isbn: ISBN): boolean {
  const libro = obtenerLibroPorISBN(isbn);
  if (!libro) return false;
  return libro.copiasDisponibles > 0;
}

export {
  agregarLibro,
  buscarLibros,
  obtenerLibroPorISBN,
  listarLibros,
  actualizarCopias,
  libroTieneCopiasDisponibles,
};
```

```typescript
// src/usuarios.ts
import { type Usuario, type IDUsuario, type ResultadoOperacion } from "./tipos";
import {
  generarID,
  esEmailValido,
  crearResultadoExito,
  crearResultadoError,
} from "./utilidades";

// ============================================================
// ESTADO INTERNO
// ============================================================

let usuarios: Usuario[] = [];

// ============================================================
// OPERACIONES
// ============================================================

function registrarUsuario(nombre: string, email: string): ResultadoOperacion<Usuario> {
  if (!nombre.trim()) return crearResultadoError("El nombre no puede estar vacío.");
  if (!esEmailValido(email)) return crearResultadoError("Email inválido.");

  const emailExistente = usuarios.find((u) => u.email === email);
  if (emailExistente) {
    return crearResultadoError("Ya existe un usuario con ese email.");
  }

  const nuevoUsuario: Usuario = {
    id: generarID("USR"),
    nombre: nombre.trim(),
    email: email.toLowerCase().trim(),
    fechaRegistro: new Date(),
    activo: true,
  };

  usuarios.push(nuevoUsuario);
  return crearResultadoExito(`Usuario "${nuevoUsuario.nombre}" registrado.`, nuevoUsuario);
}

function buscarUsuario(id: IDUsuario): Usuario | undefined {
  return usuarios.find((u) => u.id === id);
}

function buscarUsuarioPorEmail(email: string): Usuario | undefined {
  return usuarios.find((u) => u.email === email.toLowerCase());
}

function listarUsuarios(): Usuario[] {
  return [...usuarios];
}

function desactivarUsuario(id: IDUsuario): ResultadoOperacion {
  const usuario = buscarUsuario(id);
  if (!usuario) return crearResultadoError("Usuario no encontrado.");

  usuario.activo = false;
  return crearResultadoExito(`Usuario "${usuario.nombre}" desactivado.`);
}

function activarUsuario(id: IDUsuario): ResultadoOperacion {
  const usuario = buscarUsuario(id);
  if (!usuario) return crearResultadoError("Usuario no encontrado.");

  usuario.activo = true;
  return crearResultadoExito(`Usuario "${usuario.nombre}" activado.`);
}

export {
  registrarUsuario,
  buscarUsuario,
  buscarUsuarioPorEmail,
  listarUsuarios,
  desactivarUsuario,
  activarUsuario,
};
```

Gestión de préstamos y devoluciones.

```typescript
// src/prestamos.ts
import {
  type Prestamo,
  type ISBN,
  type IDUsuario,
  type IDPrestamo,
  type ResultadoOperacion,
  type EstadoPrestamo,
} from "./tipos";
import { obtenerLibroPorISBN, libroTieneCopiasDisponibles, actualizarCopias } from "./libros";
import { buscarUsuario } from "./usuarios";
import { generarID, crearResultadoExito, crearResultadoError, formatearFecha } from "./utilidades";

// ============================================================
// ESTADO INTERNO
// ============================================================

let prestamos: Prestamo[] = [];

const DIAS_MAXIMO_PRESTAMO = 14;

// ============================================================
// OPERACIONES
// ============================================================

function realizarPrestamo(isbn: ISBN, idUsuario: IDUsuario): ResultadoOperacion<Prestamo> {
  // Validar libro
  const libro = obtenerLibroPorISBN(isbn);
  if (!libro) return crearResultadoError("Libro no encontrado.");

  // Validar disponibilidad
  if (!libroTieneCopiasDisponibles(isbn)) {
    return crearResultadoError(`No hay copias disponibles de "${libro.titulo}".`);
  }

  // Validar usuario
  const usuario = buscarUsuario(idUsuario);
  if (!usuario) return crearResultadoError("Usuario no encontrado.");
  if (!usuario.activo) return crearResultadoError("Usuario inactivo.");

  // Validar que el usuario no tenga préstamos vencidos
  const prestamosVencidos = prestamos.filter(
    (p) => p.idUsuario === idUsuario && p.estado === "vencido"
  );
  if (prestamosVencidos.length >= 3) {
    return crearResultadoError("Demasiados préstamos vencidos. No se puede prestar más.");
  }

  // Crear el préstamo
  const nuevoPrestamo: Prestamo = {
    id: generarID("PREST"),
    isbn,
    idUsuario,
    fechaPrestamo: new Date(),
    estado: "activo" as EstadoPrestamo,
  };

  // Actualizar copias
  actualizarCopias(isbn, libro.copiasDisponibles - 1, libro.copiasTotales);

  prestamos.push(nuevoPrestamo);
  return crearResultadoExito(
    `Préstamo realizado. Devolver antes del ${formatearFecha(
      calcularFechaLimite(nuevoPrestamo.fechaPrestamo)
    )}.`,
    nuevoPrestamo
  );
}

function realizarDevolucion(idPrestamo: IDPrestamo): ResultadoOperacion<Prestamo> {
  const prestamo = prestamos.find((p) => p.id === idPrestamo);
  if (!prestamo) return crearResultadoError("Préstamo no encontrado.");
  if (prestamo.estado === "devuelto") return crearResultadoError("Este préstamo ya fue devuelto.");

  // Actualizar préstamo
  prestamo.fechaDevolucion = new Date();
  prestamo.estado = "devuelto";

  // Restaurar copia del libro
  const libro = obtenerLibroPorISBN(prestamo.isbn);
  if (libro) {
    actualizarCopias(prestamo.isbn, libro.copiasDisponibles + 1, libro.copiasTotales);
  }

  const diasRetraso = calcularDiasRetraso(prestamo.fechaPrestamo, prestamo.fechaDevolucion);
  const mensaje = diasRetraso > 0
    ? `Devuelto con ${diasRetraso} día(s) de retraso.`
    : "Devuelto a tiempo.";

  return crearResultadoExito(mensaje, prestamo);
}

// ============================================================
// FUNCIONES AUXILIARES
// ============================================================

function calcularFechaLimite(fechaPrestamo: Date): Date {
  const limite = new Date(fechaPrestamo);
  limite.setDate(limite.getDate() + DIAS_MAXIMO_PRESTAMO);
  return limite;
}

function calcularDiasRetraso(fechaPrestamo: Date, fechaDevolucion: Date): number {
  const limite = calcularFechaLimite(fechaPrestamo);
  const diff = fechaDevolucion.getTime() - limite.getTime();
  return Math.max(0, Math.ceil(diff / (1000 * 60 * 60 * 24)));
}

function verificarPrestamosVencidos(): void {
  const hoy = new Date();
  prestamos.forEach((prestamo) => {
    if (prestamo.estado === "activo") {
      const limite = calcularFechaLimite(prestamo.fechaPrestamo);
      if (hoy > limite) {
        prestamo.estado = "vencido";
      }
    }
  });
}

function listarPrestamos(estado?: EstadoPrestamo): Prestamo[] {
  if (estado) return prestamos.filter((p) => p.estado === estado);
  return [...prestamos];
}

function obtenerPrestamosPorUsuario(idUsuario: IDUsuario): Prestamo[] {
  return prestamos.filter((p) => p.idUsuario === idUsuario);
}

export {
  realizarPrestamo,
  realizarDevolucion,
  listarPrestamos,
  obtenerPrestamosPorUsuario,
  verificarPrestamosVencidos,
};
```

```typescript
// src/biblioteca.ts
import { type ResultadoOperacion, type EstadisticasBiblioteca } from "./tipos";
import * as libros from "./libros";
import * as usuarios from "./usuarios";
import * as prestamos from "./prestamos";
import { formatearMensaje } from "./utilidades";

// ============================================================
// CLASE PRINCIPAL
// ============================================================

class Biblioteca {
  // ============================================================
  // GESTIÓN DE LIBROS
  // ============================================================

  agregarLibro(
    titulo: string,
    autor: string,
    anio: number,
    genero: string,
    copias: number = 1
  ): ResultadoOperacion {
    // Aserción: confiamos en que el género es válido (validado en la función)
    return libros.agregarLibro(titulo, autor, anio, genero as any, copias);
  }

  buscarLibros(criterio: string, valor: string): void {
    const resultados = libros.buscarLibros({ criterio: criterio as any, valor });

    if (resultados.length === 0) {
      console.log("No se encontraron libros.");
      return;
    }

    formatearMensaje(`Libros encontrados: ${resultados.length}`, [
      ...resultados.map(
        (libro) =>
          `📚 ${libro.titulo} - ${libro.autor} (${libro.anioPublicacion}) - ${libro.genero} - ${libro.copiasDisponibles}/${libro.copiasTotales}`
      ),
    ]);
  }

  listarLibros(): void {
    const todos = libros.listarLibros();
    if (todos.length === 0) {
      console.log("No hay libros registrados.");
      return;
    }

    formatearMensaje(`Catálogo completo (${todos.length} libros)`, [
      ...todos.map(
        (libro) => `📖 ${libro.isbn} | ${libro.titulo} | ${libro.autor} | Disp: ${libro.copiasDisponibles}`
      ),
    ]);
  }

  // ============================================================
  // GESTIÓN DE USUARIOS
  // ============================================================

  registrarUsuario(nombre: string, email: string): void {
    const resultado = usuarios.registrarUsuario(nombre, email);
    console.log(resultado.exito ? `✅ ${resultado.mensaje}` : `❌ ${resultado.mensaje}`);
  }

  listarUsuarios(): void {
    const todos = usuarios.listarUsuarios();
    if (todos.length === 0) {
      console.log("No hay usuarios registrados.");
      return;
    }

    formatearMensaje(`Usuarios registrados (${todos.length})`, [
      ...todos.map(
        (u) => `👤 ${u.nombre} (${u.email}) - ${u.activo ? "🟢 Activo" : "🔴 Inactivo"}`
      ),
    ]);
  }

  // ============================================================
  // PRÉSTAMOS
  // ============================================================

  prestarLibro(isbn: string, idUsuario: string): void {
    prestamos.verificarPrestamosVencidos();
    const resultado = prestamos.realizarPrestamo(isbn, idUsuario);
    console.log(resultado.exito ? `✅ ${resultado.mensaje}` : `❌ ${resultado.mensaje}`);
  }

  devolverLibro(idPrestamo: string): void {
    const resultado = prestamos.realizarDevolucion(idPrestamo);
    console.log(resultado.exito ? `✅ ${resultado.mensaje}` : `❌ ${resultado.mensaje}`);
  }

  listarPrestamos(estado?: string): void {
    prestamos.verificarPrestamosVencidos();
    const lista = prestamos.listarPrestamos(estado as any);

    if (lista.length === 0) {
      console.log("No hay préstamos.");
      return;
    }

    formatearMensaje(`Préstamos: ${lista.length}`, [
      ...lista.map(
        (p) => `📋 ${p.id} | ISBN: ${p.isbn} | Usuario: ${p.idUsuario} | Estado: ${p.estado}`
      ),
    ]);
  }

  // ============================================================
  // REPORTES Y ESTADÍSTICAS
  // ============================================================

  generarEstadisticas(): EstadisticasBiblioteca {
    const todosLibros = libros.listarLibros();
    const todosUsuarios = usuarios.listarUsuarios();
    const todosPrestamos = prestamos.listarPrestamos();

    // Calcular libros más prestados (simulado)
    const conteoLibros = new Map<string, number>();
    todosPrestamos.forEach((p) => {
      if (p.estado !== "devuelto") return;
      conteoLibros.set(p.isbn, (conteoLibros.get(p.isbn) ?? 0) + 1);
    });

    const librosMasPrestados = Array.from(conteoLibros.entries())
      .sort((a, b) => b[1] - a[1])
      .slice(0, 5)
      .map(([isbn, veces]) => {
        const libro = libros.obtenerLibroPorISBN(isbn);
        return { titulo: libro?.titulo ?? "Desconocido", veces };
      });

    return {
      totalLibros: todosLibros.length,
      totalUsuarios: todosUsuarios.length,
      prestamosActivos: todosPrestamos.filter((p) => p.estado === "activo").length,
      librosMasPrestados,
    };
  }

  mostrarEstadisticas(): void {
    const stats = this.generarEstadisticas();

    formatearMensaje("Estadísticas de la Biblioteca", [
      `📚 Total de libros: ${stats.totalLibros}`,
      `👥 Total de usuarios: ${stats.totalUsuarios}`,
      `🔄 Préstamos activos: ${stats.prestamosActivos}`,
      `\n  Libros más prestados:`,
      ...(stats.librosMasPrestados.length > 0
        ? stats.librosMasPrestados.map((l) => `    ${l.titulo} (${l.veces} veces)`)
        : ["    (sin datos)"]),
    ]);
  }
}

export { Biblioteca };
```

```typescript
// Ejemplo de persistencia con tipos
import * as fs from "fs";
import { type Libro, type Usuario, type Prestamo } from "./tipos";

interface DatosPersistencia {
  version: number;
  ultimaActualizacion: string;
  libros: Libro[];
  usuarios: Usuario[];
  prestamos: Prestamo[];
}

function guardarDatos(
  ruta: string,
  datos: DatosPersistencia
): void {
  fs.writeFileSync(ruta, JSON.stringify(datos, null, 2), "utf-8");
}

function cargarDatos(ruta: string): DatosPersistencia | null {
  try {
    const contenido = fs.readFileSync(ruta, "utf-8");
    const datos = JSON.parse(contenido) as DatosPersistencia;
    return datos;
  } catch {
    return null;
  }
}
```
