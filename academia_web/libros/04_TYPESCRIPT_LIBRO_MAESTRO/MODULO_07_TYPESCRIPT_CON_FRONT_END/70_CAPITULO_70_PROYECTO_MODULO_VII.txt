1. DESCRIPCIÓN DEL PROYECTO
===============================================================================

Construiremos una aplicación llamada **Dashboard Admin** que permite:

- Ver un tablero principal con estadísticas
- Gestionar usuarios (listar, crear, editar, eliminar)
- Ver publicaciones con sus comentarios
- Navegar entre vistas sin recargar la página
- Todo con tipos seguros y código optimizado

Utilizaremos la API pública JSONPlaceholder como backend simulado.

Arquitectura del proyecto:

```
src/
  types/
    index.ts          # Interfaces compartidas
    api.ts            # Tipos de API
  services/
    api.ts            # Cliente HTTP tipado

    publicaciones.ts  # Servicio de publicaciones
  components/

    UsuarioForm.ts    # Formulario de usuario
  utils/
    dom.ts            # Utilidades DOM
    debounce.ts       # Utilidades de rendimiento

// src/services/api.ts

import type { Usuario, Publicacion, Comentario, Estadisticas } from "../types";

class ApiError extends Error {
  constructor(
    public status: number,
    message: string,
    public data?: unknown
  ) {
    super(message);
    this.name = "ApiError";
  }
}

class ApiCliente {
  private baseUrl = "https://jsonplaceholder.typicode.com";

  private async peticion<T>(endpoint: string): Promise<T> {
    const respuesta = await fetch(`${this.baseUrl}${endpoint}`);

    if (!respuesta.ok) {
      throw new ApiError(
        respuesta.status,
        `Error HTTP: ${respuesta.status} ${respuesta.statusText}`
      );
    }

    return respuesta.json() as Promise<T>;
  }

  async obtenerUsuarios(): Promise<Usuario[]> {
    return this.peticion<Usuario[]>("/users");
  }

  async obtenerUsuario(id: number): Promise<Usuario> {
    return this.peticion<Usuario>(`/users/${id}`);
  }

  async obtenerPublicaciones(): Promise<Publicacion[]> {
    return this.peticion<Publicacion[]>("/posts");
  }

  async obtenerPublicacionesPorUsuario(userId: number): Promise<Publicacion[]> {
    return this.peticion<Publicacion[]>(`/posts?userId=${userId}`);
  }

  async obtenerComentarios(postId: number): Promise<Comentario[]> {
    return this.peticion<Comentario[]>(`/posts/${postId}/comments`);
  }

  async obtenerEstadisticas(): Promise<Estadisticas> {
    const [usuarios, publicaciones] = await Promise.all([
      this.obtenerUsuarios(),
      this.obtenerPublicaciones()
    ]);

    const totalPublicaciones = publicaciones.length;
    const totalUsuarios = usuarios.length;

    // Contar publicaciones por usuario
    const conteo = new Map<number, number>();
    for (const pub of publicaciones) {
      conteo.set(pub.userId, (conteo.get(pub.userId) || 0) + 1);
    }

    let usuarioMasActivo = "";
    let maxPublicaciones = 0;
    for (const [userId, count] of conteo) {
      if (count > maxPublicaciones) {
        maxPublicaciones = count;
        const user = usuarios.find(u => u.id === userId);
        usuarioMasActivo = user?.name ?? "Desconocido";
      }
    }

    return {
      totalUsuarios,
      totalPublicaciones,
      totalComentarios: 500,
      usuarioMasActivo
    };
  }

  async crearUsuario(datos: Omit<Usuario, "id">): Promise<Usuario> {
    const respuesta = await fetch(`${this.baseUrl}/users`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(datos)
    });
    return respuesta.json() as Promise<Usuario>;
  }
}

export const api = new ApiCliente();
export { ApiError };

export function $<T extends HTMLElement>(selector: string): T {
  const elemento = document.querySelector<T>(selector);
  if (!elemento) throw new Error(`Elemento no encontrado: ${selector}`);
  return elemento;
}

export function crearElemento<K extends keyof HTMLElementTagNameMap>(
  tag: K,
  clases: string[] = [],
  contenido?: string
): HTMLElementTagNameMap[K] {
  const el = document.createElement(tag);
  if (clases.length) el.classList.add(...clases);
  if (contenido) el.textContent = contenido;
  return el;
}

export function vaciarElemento(el: HTMLElement): void {
  el.innerHTML = "";
}

export function mostrarCargando(contenedor: HTMLElement): void {
  contenedor.innerHTML = `
    <div class="cargando">
      <div class="spinner"></div>
      <p>Cargando datos...</p>
    </div>
  `;
}

export function mostrarError(contenedor: HTMLElement, mensaje: string): void {
  contenedor.innerHTML = `
    <div class="error">
      <h3>Error</h3>
      <p>${mensaje}</p>
      <button class="btn-reintentar">Reintentar</button>
    </div>
  `;
}

// src/utils/debounce.ts

export function debounce<T extends (...args: any[]) => void>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let timeoutId: ReturnType<typeof setTimeout>;
  return (...args: Parameters<T>): void => {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => fn(...args), delay);
  };
}

export class UsuarioServicio {
  async listar(): Promise<Usuario[]> {
    return api.obtenerUsuarios();
  }

  async obtener(id: number): Promise<Usuario> {
    return api.obtenerUsuario(id);
  }

  async crear(datos: {
    name: string;
    email: string;
    username: string;
  }): Promise<Usuario> {
    const usuarioCompleto: Omit<Usuario, "id"> = {
      ...datos,
      address: { street: "", suite: "", city: "", zipcode: "", geo: { lat: "", lng: "" } },
      phone: "",
      website: "",
      company: { name: "", catchPhrase: "", bs: "" }
    };
    return api.crearUsuario(usuarioCompleto);
  }

  async eliminar(id: number): Promise<boolean> {
    try {
      await fetch(`https://jsonplaceholder.typicode.com/users/${id}`, {
        method: "DELETE"
      });
      return true;
    } catch {
      return false;
    }
  }
}

export const usuarioServicio = new UsuarioServicio();

import { $ } from "../utils/dom";
import type { PaginaVista } from "../types";

export class Layout {
  private navLinks: NodeListOf<HTMLAnchorElement>;

  constructor(
    private onNavigate: (pagina: PaginaVista) => void
  ) {
    this.navLinks = document.querySelectorAll<HTMLAnchorElement>("[data-nav]");

    this.navLinks.forEach(link => {
      link.addEventListener("click", (e: MouseEvent) => {
        e.preventDefault();
        const pagina = (e.currentTarget as HTMLAnchorElement).dataset
          .nav as PaginaVista;
        this.activarEnlace(pagina);
        this.onNavigate(pagina);
      });
    });
  }

  private activarEnlace(pagina: PaginaVista): void {
    this.navLinks.forEach(link => {
      link.classList.toggle(
        "activo",
        link.dataset.nav === pagina
      );
    });
  }

  obtenerContenedor(): HTMLElement {
    return $("#contenido-principal");
  }
}

export class Tablero {

// src/components/Usuarios.ts
import { usuarioServicio } from "../services/usuarios";
import { $, mostrarCargando, mostrarError } from "../utils/dom";
import type { Usuario } from "../types";

export class Usuarios {
  private contenedor: HTMLElement;

  constructor() {
    this.contenedor = $("#contenido-principal");
  }

  async montar(): Promise<void> {
    this.contenedor.innerHTML = `
      <div class="vista-usuarios">
        <h1>Usuarios</h1>
        <div class="acciones">
          <button id="btn-nuevo-usuario" class="btn btn-primario">
            + Nuevo Usuario
          </button>
        </div>
        <div id="lista-usuarios" class="lista"></div>
      </div>
    `;

    $("#btn-nuevo-usuario").addEventListener("click", () => {
      this.mostrarFormulario();
    });

    await this.cargarUsuarios();
  }

  private async cargarUsuarios(): Promise<void> {
    const lista = $("#lista-usuarios");
    mostrarCargando(lista);

    try {
      const usuarios = await usuarioServicio.listar();
      this.renderizarUsuarios(usuarios);
    } catch (error) {
      mostrarError(
        lista,
        error instanceof Error ? error.message : "Error al cargar usuarios"
      );
    }
  }

  private renderizarUsuarios(usuarios: Usuario[]): void {
    const lista = $("#lista-usuarios");
    lista.innerHTML = usuarios.map(u => `
      <div class="item-usuario" data-id="${u.id}">
        <div class="info">
          <h3>${u.name}</h3>
          <p>${u.email}</p>
          <p class="empresa">${u.company.name}</p>
        </div>
        <div class="acciones-item">
          <button class="btn-ver" data-id="${u.id}">Ver</button>
          <button class="btn-eliminar" data-id="${u.id}">Eliminar</button>
        </div>
      </div>
    `).join("");

    this.inicializarEventosUsuarios();
  }

  private inicializarEventosUsuarios(): void {
    this.contenedor.addEventListener("click", async (e: MouseEvent) => {
      const target = e.target as HTMLElement;

      if (target.classList.contains("btn-eliminar")) {
        const id = Number(target.dataset.id);
        await usuarioServicio.eliminar(id);
        target.closest(".item-usuario")?.remove();
      }
    });
  }

  private mostrarFormulario(): void {
    const lista = $("#lista-usuarios");
    lista.innerHTML = `
      <form id="form-usuario" class="formulario">
        <h2>Nuevo Usuario</h2>
        <div class="campo">
          <label>Nombre</label>
          <input type="text" id="campo-nombre" required />
        </div>
        <div class="campo">
          <label>Email</label>
          <input type="email" id="campo-email" required />
        </div>
        <div class="campo">
          <label>Username</label>
          <input type="text" id="campo-username" required />
        </div>
        <div class="acciones-form">
          <button type="submit" class="btn btn-primario">Guardar</button>
          <button type="button" id="btn-cancelar" class="btn">Cancelar</button>
        </div>
      </form>
    `;

    $("#form-usuario").addEventListener("submit", async (e: Event) => {
      e.preventDefault();
      const nombre = ($("#campo-nombre") as HTMLInputElement).value;
      const email = ($("#campo-email") as HTMLInputElement).value;
      const username = ($("#campo-username") as HTMLInputElement).value;

      await usuarioServicio.crear({ name: nombre, email, username });
      this.cargarUsuarios();
    });

    $("#btn-cancelar").addEventListener("click", () => {
      this.cargarUsuarios();
    });
  }

  desmontar(): void {
    // Limpiar
  }
}

// src/main.ts
import { Layout } from "./components/Layout";
import { Tablero } from "./components/Tablero";
import { Usuarios } from "./components/Usuarios";

import type { PaginaVista } from "./types";

class App {
  private layout: Layout;
  private componenteActual: { montar: () => Promise<void>; desmontar: () => void } | null = null;

  constructor() {
    this.layout = new Layout(this.navegar.bind(this));
  }

  private async navegar(pagina: PaginaVista): Promise<void> {
    this.componenteActual?.desmontar();

    const contenedor = this.layout.obtenerContenedor();
    contenedor.innerHTML = `
      <div class="cargando-pagina">
        <div class="spinner"></div>
        <p>Cargando ${pagina}...</p>
      </div>
    `;

    try {
      let componente: { montar: () => Promise<void>; desmontar: () => void };

      switch (pagina) {
        case "dashboard":
          componente = new Tablero();
          break;
        case "usuarios":
          componente = new Usuarios();
          break;
        case "publicaciones":
          const { Publicaciones } = await import("./components/Publicaciones");
          componente = new Publicaciones();
          break;
        default:
          componente = new Tablero();
      }

      this.componenteActual = componente;
      await componente.montar();
    } catch (error) {
      contenedor.innerHTML = `
        <div class="error">
          <h2>Error al cargar la página</h2>
          <p>${(error as Error).message}</p>
          <button onclick="location.reload()">Reintentar</button>
        </div>
      `;
    }
  }

  async iniciar(): Promise<void> {
    await this.navegar("dashboard");
  }
}

// Inicializar la aplicación cuando el DOM esté listo
document.addEventListener("DOMContentLoaded", () => {
  const app = new App();
  app.iniciar();
});
```

===============================================================================
PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Por qué usar TypeScript puro sin frameworks?**

Para entender los fundamentos. Los frameworks (React, Vue, Angular) usan
los mismos principios. Si dominas TypeScript puro, cualquier framework
se te hará más fácil.

❓ **¿El proyecto funciona con la API de JSONPlaceholder?**

Sí, la API está diseñada para pruebas. Los datos son estáticos pero las
peticiones son reales. Puedes crear, leer, actualizar y eliminar.

❓ **¿Cómo añado más vistas al proyecto?**

Crea un nuevo componente que implemente `{ montar: () => Promise<void>;
desmontar: () => void }`, agrégarlo al switch en `main.ts` y añade un
enlace en el menú de navegación.

❓ **¿Necesito un servidor para desarrollo?**

Vite incluye un servidor de desarrollo. Solo ejecuta `npm run dev` y
abre `http://localhost:5173`.
