1. VISIÓN GENERAL DEL PROYECTO
===============================================================================

Construiremos "LibroFlow", una aplicación web de gestión de biblioteca
personal con las siguientes características:

- Catálogo de libros con título, autor, género, año, portada (URL)
- Búsqueda y filtrado por título, autor y género
- Marcado de libros como leídos / no leídos
- Agregar, editar y eliminar libros
- Estadísticas: total de libros, leídos, por género

// frontend/src/application/book.service.ts
import { Libro, FiltrosLibros, GeneroLiterario } from "../types";
import { BookRepository } from "../domain/book.repository";

export class BookService {
  constructor(private readonly repository: BookRepository) {}

  async agregarLibro(datos: Omit<Libro, "id" | "fechaAgregado">): Promise<Libro> {
    this.validarLibro(datos);

    const libro: Libro = {
      ...datos,
      id: crypto.randomUUID(),
      fechaAgregado: new Date()
    };

    await this.repository.guardar(libro);
    return libro;
  }

  async editarLibro(id: string, cambios: Partial<Omit<Libro, "id" | "fechaAgregado">>): Promise<Libro> {
    const libroExistente = await this.repository.buscarPorId(id);
    if (!libroExistente) {
      throw new Error(`Libro con id ${id} no encontrado`);
    }

    const libroActualizado: Libro = {
      ...libroExistente,
      ...cambios,
      id // Aseguramos que el id no cambie
    };

    await this.repository.guardar(libroActualizado);
    return libroActualizado;
  }

  async eliminarLibro(id: string): Promise<void> {
    const existente = await this.repository.buscarPorId(id);
    if (!existente) {
      throw new Error(`Libro con id ${id} no encontrado`);
    }
    await this.repository.eliminar(id);
  }

  async buscarLibros(filtros: FiltrosLibros): Promise<Libro[]> {
    return this.repository.buscar(filtros);
  }

  async marcarComoLeido(id: string, leido: boolean): Promise<Libro> {
    return this.editarLibro(id, { leido });
  }

  private validarLibro(datos: Omit<Libro, "id" | "fechaAgregado">): void {
    if (!datos.titulo || datos.titulo.trim().length === 0) {
      throw new Error("El título es obligatorio");
    }
    if (!datos.autor || datos.autor.trim().length === 0) {
      throw new Error("El autor es obligatorio");
    }
    if (datos.año < 1000 || datos.año > new Date().getFullYear()) {
      throw new Error("El año debe ser válido");
    }
    if (datos.calificacion !== undefined && (datos.calificacion < 1 || datos.calificacion > 5)) {
      throw new Error("La calificación debe estar entre 1 y 5");
    }
  }
}

// frontend/src/application/book.stats.ts
import { Libro, EstadisticasLibros, GeneroLiterario } from "../types";
import { BookRepository } from "../domain/book.repository";

export class BookStatsService {
  constructor(private readonly repository: BookRepository) {}

  async obtenerEstadisticas(): Promise<EstadisticasLibros> {
    const libros = await this.repository.obtenerTodos();

    const leidos = libros.filter(l => l.leido);
    const porGenero = {} as Record<GeneroLiterario, number>;

    for (const libro of libros) {
      porGenero[libro.genero] = (porGenero[libro.genero] ?? 0) + 1;
    }

    const calificaciones = libros
      .filter(l => l.calificacion !== undefined)
      .map(l => l.calificacion!);

    const promedioCalificacion = calificaciones.length > 0
      ? Math.round((calificaciones.reduce((a, b) => a + b, 0) / calificaciones.length) * 10) / 10
      : 0;

    return {
      total: libros.length,
      leidos: leidos.length,
      noLeidos: libros.length - leidos.length,
      porGenero,
      promedioCalificacion
    };
  }

  async buscarPorGenero(genero: GeneroLiterario): Promise<Libro[]> {
    const libros = await this.repository.obtenerTodos();
    return libros.filter(l => l.genero === genero);
  }
}

4. INFRAESTRUCTURA: PERSISTENCIA LOCAL
===============================================================================

Implementamos el repositorio usando localStorage con tipos seguros.

```typescript
// frontend/src/infrastructure/local-storage.repository.ts
import { Libro, FiltrosLibros } from "../types";
import { BookRepository } from "../domain/book.repository";

const STORAGE_KEY = "libroflow_libros";

export class LocalStorageBookRepository implements BookRepository {
  private leerLibros(): Libro[] {
    try {
      const data = localStorage.getItem(STORAGE_KEY);
      if (!data) return [];
      const parsed = JSON.parse(data) as unknown[];
      return parsed.map(item => this.hidratarLibro(item));
    } catch {
      return [];
    }
  }

  private guardarLibros(libros: Libro[]): void {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(libros));
  }

  private hidratarLibro(item: unknown): Libro {
    const data = item as Record<string, unknown>;
    return {
      id: data.id as string,
      titulo: data.titulo as string,
      autor: data.autor as string,
      genero: data.genero as Libro["genero"],
      año: data.año as number,
      portadaUrl: data.portadaUrl as string,
      leido: data.leido as boolean,
      fechaAgregado: new Date(data.fechaAgregado as string),
      calificacion: data.calificacion as number | undefined
    };
  }

  async obtenerTodos(): Promise<Libro[]> {
    return this.leerLibros();
  }

  async buscarPorId(id: string): Promise<Libro | null> {
    const libros = this.leerLibros();
    return libros.find(l => l.id === id) ?? null;
  }

  async buscar(filtros: FiltrosLibros): Promise<Libro[]> {
    let libros = this.leerLibros();

    if (filtros.busqueda) {
      const q = filtros.busqueda.toLowerCase();
      libros = libros.filter(
        l =>
          l.titulo.toLowerCase().includes(q) ||
          l.autor.toLowerCase().includes(q)
      );
    }

    if (filtros.genero) {
      libros = libros.filter(l => l.genero === filtros.genero);
    }

    if (filtros.leido !== undefined) {
      libros = libros.filter(l => l.leido === filtros.leido);
    }

    if (filtros.ordenarPor) {
      libros.sort((a, b) => {
        const orden = filtros.ordenAscendente ?? true;
        let comparacion: number;
        switch (filtros.ordenarPor) {
          case "titulo":
            comparacion = a.titulo.localeCompare(b.titulo);
            break;
          case "autor":
            comparacion = a.autor.localeCompare(b.autor);
            break;
          case "año":
            comparacion = a.año - b.año;
            break;
          case "fecha_agregado":
            comparacion = a.fechaAgregado.getTime() - b.fechaAgregado.getTime();
            break;
        }
        return orden ? comparacion : -comparacion;
      });
    }

    return libros;
  }

  async guardar(libro: Libro): Promise<void> {
    const libros = this.leerLibros();
    const idx = libros.findIndex(l => l.id === libro.id);
    if (idx >= 0) {
      libros[idx] = libro;
    } else {
      libros.push(libro);
    }
    this.guardarLibros(libros);
  }

  async eliminar(id: string): Promise<void> {
    const libros = this.leerLibros();
    const filtrados = libros.filter(l => l.id !== id);
    this.guardarLibros(filtrados);
  }
}
```

// frontend/src/presentation/components/book-list.ts
import { Libro } from "../../types";

export function renderizarListaLibros(
  libros: Libro[],
  contenedor: HTMLElement,
  handlers: {
    onToggleLeido: (id: string) => void;
    onEliminar: (id: string) => void;
    onEditar: (id: string) => void;
  }
): void {
  if (libros.length === 0) {
    contenedor.innerHTML = `
      <div class="empty-state">
        <p>No hay libros en tu biblioteca.</p>
        <p>¡Agrega tu primer libro usando el formulario!</p>
      </div>
    `;
    return;
  }

  contenedor.innerHTML = libros
    .map(libro => `
      <article class="book-card ${libro.leido ? "leido" : ""}" data-id="${libro.id}">
        <div class="book-cover">
          <img src="${libro.portadaUrl || "https://via.placeholder.com/120x180?text=Sin+Portada"}"
               alt="${libro.titulo}"
               loading="lazy"
               onerror="this.src='https://via.placeholder.com/120x180?text=Sin+Portada'">
        </div>
        <div class="book-info">
          <h3 class="book-title">${libro.titulo}</h3>
          <p class="book-author">${libro.autor}</p>
          <p class="book-meta">
            <span class="genre-tag">${traducirGenero(libro.genero)}</span>
            <span class="year">${libro.año}</span>
          </p>
          ${libro.calificacion ? renderizarEstrellas(libro.calificacion) : ""}
        </div>
        <div class="book-actions">
          <button class="btn-toggle" data-action="toggle" title="${libro.leido ? "Marcar como no leído" : "Marcar como leído"}">
            ${libro.leido ? "✓" : "○"}
          </button>
          <button class="btn-edit" data-action="edit" title="Editar libro">✎</button>
          <button class="btn-delete" data-action="delete" title="Eliminar libro">✕</button>
        </div>
      </article>
    `)
    .join("");

  // Asignar event listeners
  contenedor.querySelectorAll("[data-action]").forEach(elemento => {
    const card = elemento.closest("[data-id]") as HTMLElement;
    if (!card) return;
    const id = card.dataset.id!;
    const accion = (elemento as HTMLElement).dataset.action!;

    elemento.addEventListener("click", (e) => {
      e.stopPropagation();
      switch (accion) {
        case "toggle": handlers.onToggleLeido(id); break;
        case "delete": handlers.onEliminar(id); break;
        case "edit": handlers.onEditar(id); break;
      }
    });
  });
}

function traducirGenero(genero: string): string {
  const mapa: Record<string, string> = {
    ficcion: "Ficción",
    no_ficcion: "No Ficción",
    ciencia_ficcion: "Ciencia Ficción",
    fantasia: "Fantasía",
    romance: "Romance",
    terror: "Terror",
    misterio: "Misterio",
    biografia: "Biografía",
    historia: "Historia",
    poesia: "Poesía"
  };
  return mapa[genero] ?? genero;
}

function renderizarEstrellas(calificacion: number): string {
  const llena = "★";
  const vacia = "☆";
  return `<p class="rating">${llena.repeat(calificacion)}${vacia.repeat(5 - calificacion)}</p>`;
}

PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Por qué usar localStorage en lugar de una base de datos?**
localStorage es ideal para aplicaciones cliente-solo sin backend. Es simple,
no requiere servidor, y los datos persisten entre sesiones. Para una app
real en producción, se reemplazaría por una API REST con base de datos.

❓ **¿Cómo manejo la sincronización entre frontend y backend?**
El backend de LibroFlow es opcional. La app funciona completamente en el
frontend. El backend se usa para exportación y potencial sincronización
futura. Puedes ampliarlo con endpoints CRUD completos.

❓ **¿Puedo desplegar esta aplicación?**
Sí. El frontend se despliega como sitio estático (Vercel, Netlify, GitHub
Pages). El backend se despliega como servicio Node.js (Railway, Render,
Fly.io).

❓ **¿Cómo puedo extender el proyecto?**
Agrega autenticación, base de datos real (SQLite con Prisma), categorías
personalizadas, importación/exportación CSV, o una API REST completa.
El límite es tu imaginación y dominio de TypeScript.
