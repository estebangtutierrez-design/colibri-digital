1.1 SISTEMA DE TIPOS

El sistema de tipos de TypeScript es la base sobre la que se construye todo
lo demás. Vamos a repasar los conceptos clave.

```typescript
// Tipos primitivos: string, number, boolean, null, undefined, void, never, any, unknown
let nombre: string = "Ana";
let edad: number = 28;
let activo: boolean = true;

// Arrays y tuplas
let numeros: number[] = [1, 2, 3];
let tupla: [string, number] = ["Ana", 28];

// Tipos unión e intersección
type Id = string | number;
type ConNombre = { nombre: string };
type ConEdad = { edad: number };
type Persona = ConNombre & ConEdad;

// Type Aliases e Interfaces
interface Usuario {
  id: Id;
  nombre: string;
  email: string;
  rol: "admin" | "usuario";
}

// Operadores clave: keyof, typeof, instanceof, satisfies
type ClavesUsuario = keyof Usuario; // "id" | "nombre" | "email" | "rol"
```

1.2 TYPE GUARDS Y NARROWING

```typescript
// Type guards definidos por el usuario
function esString(valor: unknown): valor is string {
  return typeof valor === "string";
}

function esUsuario(valor: unknown): valor is Usuario {
  return (
    typeof valor === "object" &&
    valor !== null &&
    "id" in valor &&
    "nombre" in valor &&
    "email" in valor
  );
}

// Narrowing con uniones discriminadas
type Evento =
  | { tipo: "click"; x: number; y: number }
  | { tipo: "keypress"; tecla: string }
  | { tipo: "focus"; elemento: string };

function manejarEvento(evento: Evento): string {
  switch (evento.tipo) {
    case "click":
      return `Clic en (${evento.x}, ${evento.y})`;
    case "keypress":
      return `Tecla presionada: ${evento.tecla}`;
    case "focus":
      return `Elemento enfocado: ${evento.elemento}`;
  }
}
```

===============================================================================

```typescript
// Funciones con parámetros tipados y retorno
function sumar(a: number, b: number): number {
  return a + b;
}

// Parámetros opcionales y valores por defecto
function saludar(nombre: string, saludo: string = "Hola"): string {
  return `${saludo}, ${nombre}!`;
}

// Rest parameters
function sumarTodos(...numeros: number[]): number {
  return numeros.reduce((acc, n) => acc + n, 0);
}

// Sobrecarga de funciones
function duplicar(valor: string): string;
function duplicar(valor: number): number;
function duplicar(valor: string | number): string | number {
  if (typeof valor === "string") return valor + valor;
  return valor * 2;
}

// Parámetros destructurados con tipos
interface Config {
  host: string;
  puerto: number;
  ssl?: boolean;
}

function conectar({ host, puerto, ssl = false }: Config): string {
  return `Conectando a ${host}:${puerto} (SSL: ${ssl ? "sí" : "no"})`;
}
```

```typescript
// Clases con TypeScript
abstract class Entidad {
  constructor(
    public readonly id: string,
    protected fechaCreacion: Date
  ) {}

  abstract validar(): boolean;

  obtenerInfo(): string {
    return `Entidad ${this.id} creada el ${this.fechaCreacion}`;
  }
}

class Producto extends Entidad {
  constructor(
    id: string,
    public nombre: string,
    private _precio: number,
    fechaCreacion: Date = new Date()
  ) {
    super(id, fechaCreacion);
  }

  get precio(): number {
    return this._precio;
  }

  set precio(valor: number) {
    if (valor <= 0) throw new Error("El precio debe ser positivo");
    this._precio = valor;
  }

  validar(): boolean {
    return this.nombre.length > 0 && this._precio > 0;
  }
}

// Modificadores: public, private, protected, readonly
// Clases abstractas e interfaces
// Propiedades estáticas
class ContadorGlobal {
  private static _instancias = 0;

  static get instancias(): number {
    return ContadorGlobal._instancias;
  }

  constructor() {
    ContadorGlobal._instancias++;
  }
}
```

SECCIÓN 4: REPASO DE GENÉRICOS
===============================================================================

```typescript
// Funciones genéricas
function primerElemento<T>(array: T[]): T | undefined {
  return array[0];
}

// Restricciones (constraints)
interface ConLongitud {
  length: number;
}

function mostrarLongitud<T extends ConLongitud>(item: T): void {
  console.log(`Longitud: ${item.length}`);
}

// Clases genéricas
class Pila<T> {
  private elementos: T[] = [];

  push(elemento: T): void {
    this.elementos.push(elemento);
  }

  pop(): T | undefined {
    return this.elementos.pop();
  }

  get tamaño(): number {
    return this.elementos.length;
  }
}

// Tipos condicionales
type EsArray<T> = T extends any[] ? true : false;
type ResultadoCondicional = EsArray<string[]>; // true

// Mapped Types
type Opcional<T> = {
  [P in keyof T]?: T[P];
};

type SoloLectura<T> = {
  readonly [P in keyof T]: T[P];
};

// Template Literal Types
type EventoListen = `on${Capitalize<string>}`;
```
