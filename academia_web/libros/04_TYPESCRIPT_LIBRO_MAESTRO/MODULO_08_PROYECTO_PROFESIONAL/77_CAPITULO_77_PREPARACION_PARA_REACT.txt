1. FUNDAMENTOS DE REACT CON TYPESCRIPT
===============================================================================

React y TypeScript comparten una filosofía común: ambos buscan hacer el
código más predecible y seguro. TypeScript añade tipos a los componentes,
props, estado y eventos de React.

1.1 PROPS TIPADAS

El uso más básico de TypeScript en React es tipar las props de un componente.

```typescript
// Interface para las props del componente
interface SaludoProps {
  nombre: string;
  edad?: number; // Prop opcional
}

// Componente funcional tipado
function Saludo({ nombre, edad }: SaludoProps): JSX.Element {
  return (
    <div>
      <h1>Hola, {nombre}!</h1>
      {edad !== undefined && <p>Tienes {edad} años</p>}
    </div>
  );
}

// Uso correcto
<Saludo nombre="Ana" edad={25} />

// Uso correcto (prop opcional omitida)
<Saludo nombre="Luis" />

// Error: la prop nombre es obligatoria
// <Saludo />  // Error: Property 'nombre' is missing
```

1.2 TIPADO CON REACT.FC (FUNCTIONAL COMPONENT)

React proporciona el tipo genérico `FC` (Functional Component) o `FunctionComponent`.

```typescript
import React from "react";

interface TarjetaProps {
  titulo: string;
  contenido: string;
  autor?: string;
}

// Usando React.FC - incluye automáticamente children
const Tarjeta: React.FC<TarjetaProps> = ({
  titulo,
  contenido,
  autor
}) => {
  return (
    <div className="tarjeta">
      <h2>{titulo}</h2>
      <p>{contenido}</p>
      {autor && <small>Por: {autor}</small>}
    </div>
  );
};

// Sin React.FC (alternativa más explícita)
function TarjetaV2({ titulo, contenido, autor }: TarjetaProps): JSX.Element {
  return (
    <div className="tarjeta">
      <h2>{titulo}</h2>
      <p>{contenido}</p>
      {autor && <small>Por: {autor}</small>}
    </div>
  );
}
```

1.3 CHILDREN TIPADO

```typescript
import React from "react";

// PropsWithChildren añade automáticamente la prop children
interface ContenedorProps {
  titulo: string;
  estilo?: "claro" | "oscuro";
}

function Contenedor({
  titulo,
  estilo = "claro",
  children
}: React.PropsWithChildren<ContenedorProps>): JSX.Element {
  return (
    <section className={`contenedor ${estilo}`}>
      <h2>{titulo}</h2>
      <div className="contenido">{children}</div>
    </section>
  );
}

// Uso
<Contenedor titulo="Mi Sección">
  <p>Este es el contenido hijo</p>
</Contenedor>
```

2. TIPADO DE HOOKS
===============================================================================

Los hooks de React se benefician enormemente del sistema de tipos.

2.1 useState

```typescript
import { useState } from "react";

// Inferencia automática del tipo
const [contador, setContador] = useState(0);
// contador es number, setContador es React.Dispatch<React.SetStateAction<number>>

// Tipo explícito mediante genérico
const [usuario, setUsuario] = useState<Usuario | null>(null);
// setUsuario acepta Usuario | null

// Estado con objeto complejo
interface EstadoFormulario {
  nombre: string;
  email: string;
  mensaje: string;
}

const [formulario, setFormulario] = useState<EstadoFormulario>({
  nombre: "",
  email: "",
  mensaje: ""
});

// Actualización parcial con spread
function actualizarCampo(campo: keyof EstadoFormulario, valor: string): void {
  setFormulario(prev => ({ ...prev, [campo]: valor }));
}
```

2.2 useEffect

```typescript
import { useEffect, useState } from "react";

// useEffect no requiere tipado especial, pero sus dependencias sí
function DatosUsuario({ usuarioId }: { usuarioId: string }): JSX.Element {
  const [datos, setDatos] = useState<Usuario | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelado = false;

    async function cargarDatos(): Promise<void> {
      try {
        const respuesta = await fetch(`/api/usuarios/${usuarioId}`);
        if (!respuesta.ok) throw new Error("Error al cargar");
        const datosJson: Usuario = await respuesta.json();
        if (!cancelado) {
          setDatos(datosJson);
        }
      } catch (err) {
        if (!cancelado) {
          setError(err instanceof Error ? err.message : "Error desconocido");
        }
      }
    }

    cargarDatos();

    return () => {
      cancelado = true;
    };
  }, [usuarioId]); // usuarioId debe estar en el array de dependencias

  if (error) return <div>Error: {error}</div>;
  if (!datos) return <div>Cargando...</div>;
  return <div>{datos.nombre}</div>;
}
```

2.3 useReducer

```typescript
import { useReducer } from "react";

// Estados y acciones tipadas
interface EstadoContador {
  valor: number;
  ultimaModificacion: Date | null;
}

type AccionContador =
  | { tipo: "incrementar"; cantidad: number }
  | { tipo: "decrementar"; cantidad: number }
  | { tipo: "reiniciar" };

function reducerContador(
  estado: EstadoContador,
  accion: AccionContador
): EstadoContador {
  switch (accion.tipo) {
    case "incrementar":
      return {
        valor: estado.valor + accion.cantidad,
        ultimaModificacion: new Date()
      };
    case "decrementar":
      return {
        valor: estado.valor - accion.cantidad,
        ultimaModificacion: new Date()
      };
    case "reiniciar":
      return { valor: 0, ultimaModificacion: null };
  }
}

function ContadorConReducer(): JSX.Element {
  const [estado, dispatch] = useReducer(reducerContador, {
    valor: 0,
    ultimaModificacion: null
  });

  return (
    <div>
      <p>Valor: {estado.valor}</p>
      <button onClick={() => dispatch({ tipo: "incrementar", cantidad: 1 })}>
        +1
      </button>
      <button onClick={() => dispatch({ tipo: "decrementar", cantidad: 1 })}>
        -1
      </button>
      <button onClick={() => dispatch({ tipo: "reiniciar" })}>
        Reiniciar
      </button>
    </div>
  );
}
```

2.4 useRef

```typescript
import { useRef, useEffect } from "react";

// useRef para referencia a elemento del DOM
function InputAutofoco(): JSX.Element {
  const inputRef = useRef<HTMLInputElement>(null);

  useEffect(() => {
    // TypeScript sabe que inputRef.current es HTMLInputElement | null
    inputRef.current?.focus();
  }, []);

  return <input ref={inputRef} type="text" placeholder="Escribe aquí..." />;
}

// useRef para valor mutable que no causa re-render
function Temporizador(): JSX.Element {
  const contadorRef = useRef<number>(0);
  // contadorRef.current es number y se puede modificar sin re-render
  return (
    <button onClick={() => { contadorRef.current++; console.log(contadorRef.current); }}>
      Clics (no visibles): {contadorRef.current}
    </button>
  );
}
```

```typescript
import { useState, ChangeEvent, FormEvent, MouseEvent } from "react";

// onChange en input de texto
function FormularioContacto(): JSX.Element {
  const [texto, setTexto] = useState("");

  function manejarCambio(e: ChangeEvent<HTMLInputElement>): void {
    setTexto(e.target.value);
  }

  function manejarSubmit(e: FormEvent<HTMLFormElement>): void {
    e.preventDefault();
    console.log("Enviado:", texto);
  }

  return (
    <form onSubmit={manejarSubmit}>
      <input type="text" value={texto} onChange={manejarCambio} />
      <button type="submit">Enviar</button>
    </form>
  );
}

// onClick en botón
function BotonPersonalizado(): JSX.Element {
  function manejarClick(e: MouseEvent<HTMLButtonElement>): void {
    console.log("Clic en:", e.currentTarget);
  }

  return <button onClick={manejarClick}>Haz clic</button>;
}

// onChange en select
function SelectorPais(): JSX.Element {
  const [pais, setPais] = useState("");

  function manejarCambioPais(e: ChangeEvent<HTMLSelectElement>): void {
    setPais(e.target.value);
  }

  return (
    <select value={pais} onChange={manejarCambioPais}>
      <option value="">Selecciona...</option>
      <option value="MX">México</option>
      <option value="ES">España</option>
      <option value="AR">Argentina</option>

  );
}
```

4. CUSTOM HOOKS CON TYPESCRIPT
===============================================================================

Los custom hooks se benefician enormemente de los genéricos.

```typescript
import { useState, useEffect, useCallback } from "react";

// Hook genérico para fetching de datos
interface EstadoFetch<T> {
  datos: T | null;
  cargando: boolean;
  error: string | null;
}

function useFetch<T>(url: string): EstadoFetch<T> {
  const [datos, setDatos] = useState<T | null>(null);
  const [cargando, setCargando] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelado = false;

    async function fetchDatos(): Promise<void> {
      try {
        setCargando(true);
        setError(null);
        const respuesta = await fetch(url);
        if (!respuesta.ok) throw new Error(`HTTP ${respuesta.status}`);
        const json: T = await respuesta.json();
        if (!cancelado) {
          setDatos(json);
        }
      } catch (err) {
        if (!cancelado) {
          setError(err instanceof Error ? err.message : "Error");
        }
      } finally {
        if (!cancelado) setCargando(false);
      }
    }

    fetchDatos();
    return () => { cancelado = true; };
  }, [url]);

  return { datos, cargando, error };
}

// Hook para formularios con validación
interface UseFormularioProps<T> {
  valoresIniciales: T;
  validar?: (valores: T) => Partial<Record<keyof T, string>>;
}

function useFormulario<T extends Record<string, any>>({
  valoresIniciales,
  validar
}: UseFormularioProps<T>) {
  const [valores, setValores] = useState<T>(valoresIniciales);
  const [errores, setErrores] = useState<Partial<Record<keyof T, string>>>({});

  const manejarCambio = useCallback(
    (campo: keyof T, valor: any) => {
      setValores(prev => ({ ...prev, [campo]: valor }));
      if (validar) {
        const nuevosErrores = validar({ ...valores, [campo]: valor });
        setErrores(prev => ({ ...prev, [campo]: nuevosErrores[campo] }));
      }
    },
    [valores, validar]
  );

  const reiniciar = useCallback(() => {
    setValores(valoresIniciales);
    setErrores({});
  }, [valoresIniciales]);

  return { valores, errores, manejarCambio, reiniciar };
}
```

5. CONTEXT Y PROVIDERS TIPADOS
===============================================================================

El Context API de React se tipa fácilmente con TypeScript.

```typescript
import { createContext, useContext, useState, ReactNode } from "react";

// 1. Definir el tipo del contexto
interface ContextoAutenticacion {
  usuario: Usuario | null;
  iniciarSesion: (email: string, password: string) => Promise<void>;
  cerrarSesion: () => void;
  cargando: boolean;
}

// 2. Crear el contexto con un valor por defecto
const AutenticacionContext = createContext<ContextoAutenticacion | undefined>(
  undefined
);

// 3. Crear el provider
interface ProveedorAutenticacionProps {
  children: ReactNode;
}

function ProveedorAutenticacion({
  children
}: ProveedorAutenticacionProps): JSX.Element {
  const [usuario, setUsuario] = useState<Usuario | null>(null);
  const [cargando, setCargando] = useState(false);

  async function iniciarSesion(email: string, password: string): Promise<void> {
    setCargando(true);
    try {
      const respuesta = await fetch("/api/auth/login", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ email, password })
      });
      const usuarioJson: Usuario = await respuesta.json();
      setUsuario(usuarioJson);
    } finally {
      setCargando(false);
    }
  }

  function cerrarSesion(): void {
    setUsuario(null);
  }

  return (
    <AutenticacionContext.Provider
      value={{ usuario, iniciarSesion, cerrarSesion, cargando }}
    >
      {children}
    </AutenticacionContext.Provider>
  );
}

// 4. Custom hook para usar el contexto con validación
function useAutenticacion(): ContextoAutenticacion {
  const contexto = useContext(AutenticacionContext);
  if (contexto === undefined) {
    throw new Error(
      "useAutenticacion debe usarse dentro de ProveedorAutenticacion"
    );
  }
  return contexto;
}

// Uso en componentes
function BotonLogin(): JSX.Element {
  const { usuario, cerrarSesion } = useAutenticacion();

  if (!usuario) return null;

  return (
    <div>
      <span>{usuario.nombre}</span>
      <button onClick={cerrarSesion}>Cerrar sesión</button>
    </div>
  );
}
```

PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Debo usar React.FC o la sintaxis de función directa?**
Ambas son válidas. React.FC incluye children automáticamente y proporciona
tipado implícito del retorno. La sintaxis directa es más explícita. La
comunidad está dividida; elige la que prefieras y sé consistente.

❓ **¿Cómo tipo children cuando no uso React.FC?**
Usa `React.PropsWithChildren<T>` o añade `children?: React.ReactNode` a tus
props. La primera opción es más limpia cuando solo añades children.

❓ **¿Qué tipo uso para eventos personalizados?**
Puedes crear tus propios tipos de evento extendiendo `React.SyntheticEvent`
o usando `CustomEvent` del DOM con genéricos.

❓ **¿Necesito React import en componentes con React 17+?**
Con React 17+ y el nuevo JSX transform, no necesitas importar React
explícitamente. Pero sí necesitas importar hooks y tipos desde React.
