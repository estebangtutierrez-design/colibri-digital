```typescript
// src/domain/entities/usuario.ts
export type Rol = "admin" | "miembro";

export interface Usuario {
  readonly id: string;
  readonly nombre: string;
  readonly email: string;
  readonly passwordHash: string;
  readonly rol: Rol;
  readonly creadoEn: Date;
  readonly activo: boolean;
}

// Funciones de dominio (pure functions)
export function crearUsuario(
  id: string,
  nombre: string,
  email: string,
  passwordHash: string,
  rol: Rol = "miembro"
): Usuario {
  if (!email.includes("@")) {
    throw new Error("Email inválido");
  }
  if (nombre.length < 2) {
    throw new Error("El nombre debe tener al menos 2 caracteres");
  }
  return {
    id,
    nombre,
    email,
    passwordHash,
    rol,
    creadoEn: new Date(),
    activo: true
  };
}

export function desactivarUsuario(usuario: Usuario): Usuario {
  return { ...usuario, activo: false };
}

export function cambiarRol(usuario: Usuario, nuevoRol: Rol): Usuario {
  return { ...usuario, rol: nuevoRol };
}
```

```typescript
// src/application/casos-de-uso/crear-tarea.ts
import { Tarea, crearTarea, Prioridad, EstadoTarea } from "../../domain/entities/tarea";
import { TareaRepositorio } from "../../domain/repositories/tarea.repository";
import { ProyectoRepositorio } from "../../domain/repositories/proyecto.repository";

export interface CrearTareaDTO {
  titulo: string;
  descripcion: string;
  proyectoId: string;
  prioridad: Prioridad;
  vencimiento?: Date;
}

export interface CrearTareaResultado {
  exito: boolean;
  tarea?: Tarea;
  error?: string;
}

export class CrearTareaCasoDeUso {
  constructor(
    private readonly tareaRepo: TareaRepositorio,
    private readonly proyectoRepo: ProyectoRepositorio
  ) {}

  async ejecutar(dto: CrearTareaDTO): Promise<CrearTareaResultado> {
    try {
      const proyecto = await this.proyectoRepo.buscarPorId(dto.proyectoId);
      if (!proyecto) {
        return {
          exito: false,
          error: `Proyecto con id ${dto.proyectoId} no encontrado`
        };
      }

      const tarea = crearTarea(
        crypto.randomUUID(),
        dto.titulo,
        dto.descripcion,
        dto.proyectoId,
        dto.prioridad,
        dto.vencimiento
      );

      await this.tareaRepo.guardar(tarea);
      return { exito: true, tarea };
    } catch (error) {
      return {
        exito: false,
        error: error instanceof Error ? error.message : "Error desconocido"
      };
    }
  }
}

// src/application/casos-de-uso/listar-tareas.ts
import { Tarea } from "../../domain/entities/tarea";
import { TareaRepositorio } from "../../domain/repositories/tarea.repository";

export interface FiltroTareas {
  proyectoId?: string;
  estado?: string;
  asignadoA?: string;
  prioridad?: string;
}

export class ListarTareasCasoDeUso {
  constructor(private readonly tareaRepo: TareaRepositorio) {}

  async ejecutar(filtro?: FiltroTareas): Promise<Tarea[]> {
    const todas = await this.tareaRepo.buscarTodos();

    return todas.filter(tarea => {
      if (filtro?.proyectoId && tarea.proyectoId !== filtro.proyectoId) return false;
      if (filtro?.estado && tarea.estado !== filtro.estado) return false;
      if (filtro?.asignadoA && tarea.asignadoA !== filtro.asignadoA) return false;
      if (filtro?.prioridad && tarea.prioridad !== filtro.prioridad) return false;
      return true;
    });
  }
}
```

5. INFRAESTRUCTURA: REPOSITORIO JSON
===============================================================================

Implementamos los repositorios usando archivos JSON para persistencia local.

```typescript
// src/infrastructure/persistence/json.repository.ts
import * as fs from "node:fs/promises";
import * as path from "node:path";

export class JsonRepositorio<T extends { id: string }> {
  private readonly filePath: string;

  constructor(
    private readonly nombreColeccion: string,
    private readonly basePath: string = "data"
  ) {
    this.filePath = path.join(this.basePath, `${nombreColeccion}.json`);
  }

  private async leerTodos(): Promise<T[]> {
    try {
      const data = await fs.readFile(this.filePath, "utf-8");
      return JSON.parse(data) as T[];
    } catch {
      return [];
    }
  }

  private async escribirTodos(items: T[]): Promise<void> {
    await fs.mkdir(path.dirname(this.filePath), { recursive: true });
    await fs.writeFile(this.filePath, JSON.stringify(items, null, 2), "utf-8");
  }

  async guardar(item: T): Promise<void> {
    const items = await this.leerTodos();
    const idx = items.findIndex(i => i.id === item.id);
    if (idx >= 0) {
      items[idx] = item;
    } else {
      items.push(item);
    }
    await this.escribirTodos(items);
  }

  async buscarPorId(id: string): Promise<T | null> {
    const items = await this.leerTodos();
    return items.find(i => i.id === id) ?? null;
  }

  async buscarTodos(): Promise<T[]> {
    return this.leerTodos();
  }

  async eliminar(id: string): Promise<void> {
    const items = await this.leerTodos();
    const filtrados = items.filter(i => i.id !== id);
    await this.escribirTodos(filtrados);
  }
}
```

Implementaciones concretas usando el repositorio genérico:

```typescript
// src/infrastructure/persistence/usuario.repository.impl.ts
import { Usuario } from "../../domain/entities/usuario";
import { UsuarioRepositorio } from "../../domain/repositories/usuario.repository";
import { JsonRepositorio } from "./json.repository";

export class UsuarioRepositorioJson implements UsuarioRepositorio {
  private readonly repo = new JsonRepositorio<Usuario>("usuarios");

  async guardar(usuario: Usuario): Promise<void> {
    await this.repo.guardar(usuario);
  }

  async buscarPorId(id: string): Promise<Usuario | null> {
    return this.repo.buscarPorId(id);
  }

  async buscarPorEmail(email: string): Promise<Usuario | null> {
    const usuarios = await this.repo.buscarTodos();
    return usuarios.find(u => u.email === email) ?? null;
  }

  async buscarTodos(): Promise<Usuario[]> {
    return this.repo.buscarTodos();
  }

  async eliminar(id: string): Promise<void> {
    await this.repo.eliminar(id);
  }
}
```

PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Por qué las entidades de dominio son interfaces y no clases?**
Las interfaces en TypeScript son más ligeras y no generan código en tiempo de
ejecución. Además, las funciones puras asociadas (crearUsuario, etc.) son más
fáciles de probar y componer que los métodos de clase.

❓ **¿Debo usar archivos JSON como base de datos en producción?**
No. El repositorio JSON es para desarrollo y prototipado. En producción se
reemplaza por una base de datos real (SQLite, PostgreSQL) sin cambiar el
dominio.

❓ **¿Qué es crypto.randomUUID()?**
Es una función nativa de Node.js (v19+) que genera UUIDs únicos. Si usas una
versión anterior, puedes usar librerías como `uuid`.

❓ **¿Necesito TypeScript para este proyecto o puedo usar JavaScript?**
Todo el proyecto está diseñado para aprovechar TypeScript al máximo.
Podrías usar JavaScript, pero perderías el sistema de tipos que es el
corazón de nuestra arquitectura.
