// === Sistema de esquemas con tipos condicionales ===
type SchemaType = "string" | "number" | "boolean" | "date" | "array" | "object";

interface CampoBase {
  tipo: SchemaType;
  requerido?: boolean;
  descripcion?: string;
}

interface CampoScalar extends CampoBase {
  tipo: Exclude<SchemaType, "array" | "object">;
  min?: number;
  max?: number;
  patron?: string;
}

interface CampoArray extends CampoBase {
  tipo: "array";
  elemento: SchemaDefinition;
}

interface CampoObjeto extends CampoBase {
  tipo: "object";
  propiedades: Record<string, SchemaDefinition>;
}

type SchemaDefinition = CampoScalar | CampoArray | CampoObjeto;

// === Tipo condicional que convierte SchemaDefinition a tipo TypeScript ===
type SchemaToType<T extends SchemaDefinition> =
  T extends { tipo: "string" } ? string
  : T extends { tipo: "number" } ? number
  : T extends { tipo: "boolean" } ? boolean
  : T extends { tipo: "date" } ? Date
  : T extends { tipo: "array"; elemento: infer E }
    ? E extends SchemaDefinition ? SchemaToType<E>[]
    : never
  : T extends { tipo: "object"; propiedades: infer P }
    ? { [K in keyof P]: P[K] extends SchemaDefinition ? SchemaToType<P[K]> : never }
  : never;

// === Definicion de esquemas concretos ===
const usuarioSchema = {
  tipo: "object",
  propiedades: {
    nombre: { tipo: "string", requerido: true, min: 2, max: 100 },

    edad: { tipo: "number", min: 0, max: 150 },

// === Validador de esquemas ===
type ValidationResult<T> = {
  valido: boolean;
  datos: T | null;
  errores: Record<string, string>;
};

function validarSchema<T extends Record<string, any>>(
  datos: Record<string, unknown>,
  schema: SchemaDefinition
): ValidationResult<T> {
  const errores: Record<string, string> = {};

  function validarCampo(
    valor: unknown,
    def: SchemaDefinition,
    ruta: string
  ): void {
    if (def.tipo === "string") {
      if (typeof valor !== "string") {
        errores[ruta] = `Debe ser un string`;
      } else if ("min" in def && typeof valor === "string" && valor.length < (def as any).min) {
        errores[ruta] = `Debe tener al menos ${(def as any).min} caracteres`;
      } else if ("max" in def && typeof valor === "string" && valor.length > (def as any).max) {
        errores[ruta] = `Debe tener maximo ${(def as any).max} caracteres`;
      }
    } else if (def.tipo === "number") {
      if (typeof valor !== "number") {
        errores[ruta] = `Debe ser un numero`;
      }
    } else if (def.tipo === "boolean") {
      if (typeof valor !== "boolean") {
        errores[ruta] = `Debe ser booleano`;
      }
    } else if (def.tipo === "date") {
      if (!(valor instanceof Date) && typeof valor !== "string") {
        errores[ruta] = `Debe ser una fecha`;
      }
    } else if (def.tipo === "array" && "elemento" in def) {
      if (!Array.isArray(valor)) {
        errores[ruta] = `Debe ser un arreglo`;
      } else {
        valor.forEach((item, i) => validarCampo(item, (def as any).elemento, `${ruta}[${i}]`));
      }
    } else if (def.tipo === "object" && "propiedades" in def) {
      if (typeof valor !== "object" || valor === null) {
        errores[ruta] = `Debe ser un objeto`;
      } else {
        const props = (def as any).propiedades as Record<string, SchemaDefinition>;
        for (const key of Object.keys(props)) {
          validarCampo(
            (valor as Record<string, unknown>)[key],
            props[key],
            `${ruta}.${key}`
          );
        }
      }
    }
  }

  validarCampo(datos, schema, "$");

  if (Object.keys(errores).length > 0) {
    return { valido: false, datos: null, errores };
  }

  return { valido: true, datos: datos as T, errores: {} };
}

const resultado = validarSchema<UsuarioSchemaType>(
  {
    nombre: "Ana",
    email: "ana@test.com",
    edad: 25,
    activo: true,
    fechaRegistro: new Date(),
    roles: ["admin"],
    direccion: { calle: "Av. Siempre Viva", ciudad: "Madrid" }
  },
  usuarioSchema as SchemaDefinition
);

// === Marcas de validacion ===
type Validado<T> = T & { __validated: true };
type Persistido<T> = T & { __persisted: true };

// === Validador de usuarios ===
interface CrearUsuarioDTO {
  nombre: string;
  email: string;
  password: string;
  rol?: RolUsuario;
}

interface UsuarioValidado extends CrearUsuarioDTO {
  __validated: true;
}

function validarCrearUsuario(dto: CrearUsuarioDTO): Validado<CrearUsuarioDTO> {
  const errores: string[] = [];

  if (!dto.nombre || dto.nombre.length < 2) {
    errores.push("El nombre debe tener al menos 2 caracteres");
  }
  if (!dto.email || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(dto.email)) {
    errores.push("Email invalido");
  }
  if (!dto.password || dto.password.length < 8) {
    errores.push("La contrasena debe tener al menos 8 caracteres");
  }
  if (dto.rol && !["admin", "editor", "autor", "lector"].includes(dto.rol)) {
    errores.push("Rol invalido");
  }

  if (errores.length > 0) {
    throw new Error(`Error de validacion: ${errores.join("; ")}`);
  }

  return { ...dto, __validated: true } as Validado<CrearUsuarioDTO>;
}

// === Validador de posts ===
interface CrearPostDTO {
  titulo: string;
  contenido: string;
  extracto: string;
  categoria: CategoryId;
  etiquetas: string[];
}

function validarCrearPost(dto: CrearPostDTO): Validado<CrearPostDTO> {
  const errores: string[] = [];

  if (!dto.titulo || dto.titulo.length < 3) {
    errores.push("El titulo debe tener al menos 3 caracteres");
  }
  if (!dto.contenido || dto.contenido.length < 10) {
    errores.push("El contenido debe tener al menos 10 caracteres");
  }
  if (dto.etiquetas.length > 10) {
    errores.push("Maximo 10 etiquetas");
  }

  if (errores.length > 0) {
    throw new Error(`Error de validacion: ${errores.join("; ")}`);
  }

  return { ...dto, __validated: true } as Validado<CrearPostDTO>;
}

// === Transformacion de DTO a entidad ===
function usuarioDTOtoEntity(
  dto: Validado<CrearUsuarioDTO>
): Usuario {
  const ahora = now();
  return {
    id: createUserId(),
    nombre: dto.nombre.trim(),
    email: dto.email.toLowerCase(),
    rol: dto.rol ?? "autor",
    creado: ahora,
    actualizado: ahora
  };
}

function postDTOtoEntity(
  dto: Validado<CrearPostDTO>,
  autor: UserId
): Post {
  const ahora = now();
  return {
    id: createPostId(),
    titulo: dto.titulo.trim(),
    contenido: dto.contenido,
    extracto: dto.extracto.trim(),
    slug: dto.titulo.toLowerCase().replace(/\s+/g, "-").replace(/[^a-z0-9-]/g, ""),
    autor,
    categoria: dto.categoria,
    estado: "borrador",
    etiquetas: dto.etiquetas,
    comentarios: [],
    creado: ahora,
    actualizado: ahora,
    publicado: null
  };
}

// === Dominios y acciones de eventos ===
type DominioEvento = "usuario" | "post" | "comentario" | "categoria";
type AccionEvento = "creado" | "actualizado" | "eliminado" | "publicado" | "archivado";

type NombreEvento = `${DominioEvento}:${AccionEvento}`;
// "usuario:creado" | "usuario:actualizado" | ...

// === Mapa de payloads por evento ===
interface EventPayloadMap {
  "usuario:creado": { usuario: Usuario };
  "usuario:actualizado": { usuarioId: UserId; cambios: Partial<Usuario> };
  "usuario:eliminado": { usuarioId: UserId };

  "post:creado": { post: Post };
  "post:actualizado": { postId: PostId; cambios: Partial<Post> };
  "post:eliminado": { postId: PostId };
  "post:publicado": { postId: PostId; fecha: Timestamp };
  "post:archivado": { postId: PostId };

  "comentario:creado": { comentario: Comentario };
  "comentario:eliminado": { commentId: CommentId };

  "categoria:creado": { categoria: Categoria };
  "categoria:eliminado": { categoryId: CategoryId };
}

// === Handler tipado ===
type EventHandler<E extends NombreEvento> = (
  payload: EventPayloadMap[E]
) => void;

// === Sistema de eventos ===
type ListenerStore = {
  [E in NombreEvento]?: Set<EventHandler<E>>;
};

class EventBus {
  private listeners: ListenerStore = {};

  on<E extends NombreEvento>(
    evento: E,
    handler: EventHandler<E>
  ): void {
    if (!this.listeners[evento]) {
      this.listeners[evento] = new Set();
    }
    (this.listeners[evento] as Set<EventHandler<E>>).add(handler);
  }

  off<E extends NombreEvento>(
    evento: E,
    handler: EventHandler<E>
  ): void {
    this.listeners[evento]?.delete(handler as any);
  }

  emit<E extends NombreEvento>(
    evento: E,
    payload: EventPayloadMap[E]
  ): void {
    console.log(`[EventBus] ${evento}`, payload);
    this.listeners[evento]?.forEach((handler) => {
      (handler as EventHandler<E>)(payload);
    });
  }

  // Escuchar todos los eventos de un dominio
  onDominio<D extends DominioEvento>(
    dominio: D,
    handler: <A extends AccionEvento>(
      evento: `${D}:${A}`,
      payload: EventPayloadMap[`${D}:${A}`]
    ) => void
  ): void {
    const acciones: AccionEvento[] = ["creado", "actualizado", "eliminado", "publicado", "archivado"];
    for (const accion of acciones) {
      const nombreEvento = `${dominio}:${accion}` as NombreEvento;
      this.on(nombreEvento, (payload: any) => {
        handler(nombreEvento as any, payload);
      });
    }
  }
}

const eventBus = new EventBus();

// Suscripcion tipada
eventBus.on("usuario:creado", (payload) => {
  console.log(`Nuevo usuario: ${payload.usuario.nombre}`);
});

eventBus.on("post:publicado", (payload) => {
  console.log(`Post publicado: ${payload.postId}`);
});

// === Definicion de rutas API con template literals ===
type ApiVersion = "v1";
type ApiRecurso = "usuarios" | "posts" | "comentarios" | "categorias";
type ApiAccion = "" | "/:id" | "/:id/publicar" | "/:id/archivar";

type ApiPath = `/api/${ApiVersion}/${ApiRecurso}${ApiAccion}`;
// "/api/v1/usuarios" | "/api/v1/usuarios/:id" | ...

// === Parametros de ruta extraidos ===
type ParametrosRuta<T extends string> =
  T extends `${string}:${infer Param}/${infer Rest}`
    ? Param | ParametrosRuta<`/${Rest}`>
    : T extends `${string}:${infer Param}`
      ? Param
      : never;

type RutaUsuariosId = ParametrosRuta<"/api/v1/usuarios/:id">;
// "id"

type ObjetoParams<T extends string> = {
  [K in ParametrosRuta<T>]: string;
};

// === Metodos HTTP tipados ===
type HttpMethod = "GET" | "POST" | "PUT" | "PATCH" | "DELETE";

interface ApiRequest<T extends ApiPath = ApiPath> {
  method: HttpMethod;
  path: T;
  params?: ObjetoParams<T>;
  query?: Record<string, string>;
  body?: unknown;
  headers?: Record<string, string>;
}

interface ApiResponse<T> {
  status: number;
  data: T;
  message: string;
  timestamp: Timestamp;
}

// === Controladores tipados ===
interface UsuarioController {

  crear(body: CrearUsuarioDTO): Promise<ApiResponse<Usuario>>;
  actualizar(
    params: ObjetoParams<"/api/v1/usuarios/:id">,
    body: Partial<CrearUsuarioDTO>

  eliminar(params: ObjetoParams<"/api/v1/usuarios/:id">): Promise<ApiResponse<void>>;

interface PostController {
  listar(): Promise<ApiResponse<Post[]>>;
  obtener(params: ObjetoParams<"/api/v1/posts/:id">): Promise<ApiResponse<Post>>;

  publicar(params: ObjetoParams<"/api/v1/posts/:id/publicar">): Promise<ApiResponse<Post>>;
  archivar(params: ObjetoParams<"/api/v1/posts/:id/archivar">): Promise<ApiResponse<Post>>;
  eliminar(params: ObjetoParams<"/api/v1/posts/:id">): Promise<ApiResponse<void>>;

// === Implementacion del controlador de usuarios ===
class UsuarioControllerImpl implements UsuarioController {
  private usuarios: Usuario[] = [];

  async listar(): Promise<ApiResponse<Usuario[]>> {
    return {
      status: 200,
      data: this.usuarios,
      message: "Usuarios obtenidos exitosamente",
      timestamp: now()
    };
  }

  async obtener(params: { id: string }): Promise<ApiResponse<Usuario>> {
    const usuario = this.usuarios.find(u => u.id === params.id);
    if (!usuario) {
      return {
        status: 404,
        data: null as any,
        message: "Usuario no encontrado",
        timestamp: now()
      };
    }
    return {
      status: 200,
      data: usuario,
      message: "Usuario obtenido exitosamente",
      timestamp: now()
    };
  }

  async crear(body: CrearUsuarioDTO): Promise<ApiResponse<Usuario>> {
    const validado = validarCrearUsuario(body);
    const usuario = usuarioDTOtoEntity(validado);
    this.usuarios.push(usuario);

    eventBus.emit("usuario:creado", { usuario });

    return {
      status: 201,
      data: usuario,
      message: "Usuario creado exitosamente",
      timestamp: now()
    };
  }

  async actualizar(
    params: { id: string },
    body: Partial<CrearUsuarioDTO>
  ): Promise<ApiResponse<Usuario>> {
    const idx = this.usuarios.findIndex(u => u.id === params.id);
    if (idx === -1) {
      return {
        status: 404,
        data: null as any,
        message: "Usuario no encontrado",
        timestamp: now()
      };
    }

    this.usuarios[idx] = {
      ...this.usuarios[idx],
      ...body,
      actualizado: now()
    };

    eventBus.emit("usuario:actualizado", {
      usuarioId: params.id as UserId,
      cambios: body
    });

    return {
      status: 200,
      data: this.usuarios[idx],
      message: "Usuario actualizado exitosamente",
      timestamp: now()
    };
  }

  async eliminar(params: { id: string }): Promise<ApiResponse<void>> {
    const idx = this.usuarios.findIndex(u => u.id === params.id);
    if (idx === -1) {
      return {
        status: 404,
        data: undefined as any,
        message: "Usuario no encontrado",
        timestamp: now()
      };
    }

    this.usuarios.splice(idx, 1);

    eventBus.emit("usuario:eliminado", {
      usuarioId: params.id as UserId
    });

    return {
      status: 200,
      data: undefined as any,
      message: "Usuario eliminado exitosamente",
      timestamp: now()
    };
  }
}

// === Router tipado ===
type RouteHandler<T extends ApiPath> = (
  req: ApiRequest<T>
) => Promise<ApiResponse<any>>;

class ApiRouter {
  private routes = new Map<string, RouteHandler<any>>();

  registrar<T extends ApiPath>(
    method: HttpMethod,
    path: T,
    handler: RouteHandler<T>
  ): void {
    const key = `${method}:${path}`;
    this.routes.set(key, handler);
  }

  async dispatch(req: ApiRequest): Promise<ApiResponse<any>> {
    const key = `${req.method}:${req.path}`;
    const handler = this.routes.get(key);

    if (!handler) {
      return {
        status: 404,
        data: null,
        message: `Ruta no encontrada: ${key}`,
        timestamp: now()
      };
    }

    return handler(req);
  }
}

// === Repositorio generico con cache ===
interface Cacheable<T> {
  obtener(clave: string): T | undefined;
  guardar(clave: string, valor: T): void;
  limpiar(): void;
}

class MemoriaCache<T> implements Cacheable<T> {
  private cache = new Map<string, T>();
  private tiempoLimite: number;

  constructor(tiempoLimiteMs: number = 60000) {
    this.tiempoLimite = tiempoLimiteMs;
  }

  obtener(clave: string): T | undefined {
    return this.cache.get(clave);
  }

  guardar(clave: string, valor: T): void {
    this.cache.set(clave, valor);
    setTimeout(() => this.cache.delete(clave), this.tiempoLimite);
  }

  limpiar(): void {
    this.cache.clear();
  }
}

// === Repositorio base ===
abstract class BaseRepositorio<T extends { id: string }> {
  protected items: T[] = [];
  protected cache: Cacheable<T>;

  constructor(cache?: Cacheable<T>) {
    this.cache = cache ?? new MemoriaCache<T>();
  }

  abstract validar(item: Partial<T>): boolean;

  listar(): T[] {
    return [...this.items];
  }

  obtenerPorId(id: string): T | undefined {
    const cacheado = this.cache.obtener(id);
    if (cacheado) return cacheado;

    const item = this.items.find(i => i.id === id);
    if (item) this.cache.guardar(id, item);
    return item;
  }

  crear(item: T): T {
    this.items.push(item);
    this.cache.guardar(item.id, item);
    return item;
  }

  actualizar(id: string, cambios: Partial<T>): T | undefined {
    const idx = this.items.findIndex(i => i.id === id);
    if (idx === -1) return undefined;

    this.items[idx] = { ...this.items[idx], ...cambios };
    this.cache.guardar(id, this.items[idx]);
    return this.items[idx];
  }

  eliminar(id: string): boolean {
    const idx = this.items.findIndex(i => i.id === id);
    if (idx === -1) return false;
    this.items.splice(idx, 1);
    this.cache.limpiar();
    return true;
  }
}

// === Repositorio de Usuarios ===
class UsuarioRepositorio extends BaseRepositorio<Usuario> {
  validar(item: Partial<Usuario>): boolean {
    return !!item.nombre && !!item.email && item.nombre.length >= 2;
  }

  buscarPorEmail(email: string): Usuario | undefined {
    return this.items.find(u => u.email === email);
  }

  buscarPorRol(rol: RolUsuario): Usuario[] {
    return this.items.filter(u => u.rol === rol);
  }
}

// === Sistema CMS completo ===
class CMS {
  eventos: EventBus;
  usuarios: UsuarioControllerImpl;
  usuariosRepo: UsuarioRepositorio;

  constructor() {
    this.eventos = eventBus;
    this.usuariosRepo = new UsuarioRepositorio();
    this.usuarios = new UsuarioControllerImpl();

    this.configurarEventos();
  }

  private configurarEventos(): void {
    this.eventos.on("usuario:creado", ({ usuario }) => {
      console.log(`[CMS] Nuevo usuario registrado: ${usuario.nombre}`);
      this.usuariosRepo.crear(usuario);
    });

    this.eventos.on("usuario:eliminado", ({ usuarioId }) => {
      console.log(`[CMS] Usuario eliminado: ${usuarioId}`);
      this.usuariosRepo.eliminar(usuarioId);
    });
  }

  async iniciar(): Promise<void> {
    console.log("=== CMS INICIADO ===");
    console.log("Escuchando eventos del sistema...");
  }

  async crearUsuario(dto: CrearUsuarioDTO): Promise<ApiResponse<Usuario>> {
    return this.usuarios.crear(dto);
  }

  async crearPost(dto: CrearPostDTO, autorId: UserId): Promise<Post> {
    const validado = validarCrearPost(dto);
    const post = postDTOtoEntity(validado, autorId);

    this.eventos.emit("post:creado", { post });
    return post;
  }

  async publicarPost(postId: PostId): Promise<Post | null> {
    const post = this.usuariosRepo
      .listar()
      .flatMap(u => [])
      .length > 0 ? null : null;

    // En un sistema real, buscariamos el post en un repositorio
    this.eventos.emit("post:publicado", {
      postId,
      fecha: now()
    });

    return null;
  }
}
