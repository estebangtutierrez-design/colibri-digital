===============================================================================
1. DESCRIPCIÓN DEL PROYECTO
===============================================================================

Nombre: TaskFlow — Sistema de Gestión de Tareas Colaborativo

Descripción: TaskFlow es una aplicación web que permite a equipos de trabajo
gestionar sus tareas de forma colaborativa. Los usuarios pueden registrarse,
crear proyectos, asignar tareas, comentar, adjuntar archivos y visualizar el
progreso en un tablero Kanban.

Requisitos técnicos:

  - PHP 8.0+ con PDO
  - MySQL 5.7+ con InnoDB
  - HTML5 + CSS3 (sin frameworks, pero puedes usar Bootstrap si prefieres)
  - JavaScript básico para confirmaciones y mejoras UX

Funcionalidades principales:

  - Autenticación: registro, login, logout, recordar sesión.
  - Gestión de proyectos: CRUD completo.
  - Gestión de tareas: CRUD con estados, prioridades, asignación.
  - Comentarios: agregar y listar comentarios por tarea.
  - Archivos adjuntos: subir y descargar archivos por tarea.
  - Tablero Kanban: columnas por estado (Pendiente, En Progreso, Completada).
  - Dashboard: estadísticas y reportes.
  - Perfil de usuario: editar datos personales.

3. BASE DE DATOS

```sql
CREATE DATABASE IF NOT EXISTS taskflow
    CHARACTER SET utf8mb4
    COLLATE utf8mb4_unicode_ci;

USE taskflow;

-- Tabla de usuarios
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    apellido VARCHAR(100) NOT NULL,
    email VARCHAR(150) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    avatar VARCHAR(255) DEFAULT NULL,
    rol ENUM('admin', 'miembro') DEFAULT 'miembro',
    activo BOOLEAN DEFAULT TRUE,
    ultimo_acceso TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- Tabla de proyectos
CREATE TABLE proyectos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(200) NOT NULL,
    descripcion TEXT,
    color VARCHAR(7) DEFAULT '#3498db',
    creador_id INT NOT NULL,
    activo BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (creador_id) REFERENCES usuarios(id) ON DELETE CASCADE
) ENGINE=InnoDB;

-- Tabla de miembros del proyecto
CREATE TABLE proyecto_miembros (
    proyecto_id INT NOT NULL,
    usuario_id INT NOT NULL,
    rol ENUM('lider', 'colaborador') DEFAULT 'colaborador',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (proyecto_id, usuario_id),
    FOREIGN KEY (proyecto_id) REFERENCES proyectos(id) ON DELETE CASCADE,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE
) ENGINE=InnoDB;

-- Tabla de tareas
CREATE TABLE tareas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    proyecto_id INT NOT NULL,
    titulo VARCHAR(200) NOT NULL,
    descripcion TEXT,
    estado ENUM('pendiente', 'en_progreso', 'completada') DEFAULT 'pendiente',
    prioridad ENUM('baja', 'media', 'alta', 'urgente') DEFAULT 'media',
    asignado_id INT DEFAULT NULL,
    creador_id INT NOT NULL,
    fecha_limite DATE DEFAULT NULL,
    completada_en TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (proyecto_id) REFERENCES proyectos(id) ON DELETE CASCADE,
    FOREIGN KEY (asignado_id) REFERENCES usuarios(id) ON DELETE SET NULL,
    FOREIGN KEY (creador_id) REFERENCES usuarios(id) ON DELETE CASCADE
) ENGINE=InnoDB;

-- Tabla de comentarios
CREATE TABLE comentarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tarea_id INT NOT NULL,
    usuario_id INT NOT NULL,
    contenido TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (tarea_id) REFERENCES tareas(id) ON DELETE CASCADE,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE
) ENGINE=InnoDB;

-- Tabla de archivos adjuntos
CREATE TABLE archivos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tarea_id INT NOT NULL,
    usuario_id INT NOT NULL,
    nombre_original VARCHAR(255) NOT NULL,
    nombre_archivo VARCHAR(255) NOT NULL,
    tipo_mime VARCHAR(100) NOT NULL,
    tamano INT UNSIGNED NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (tarea_id) REFERENCES tareas(id) ON DELETE CASCADE,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE
) ENGINE=InnoDB;

-- Tabla de tokens (recordar sesión)
CREATE TABLE tokens_recordar (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    token_hash VARCHAR(64) NOT NULL UNIQUE,
    expira_en TIMESTAMP NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE
) ENGINE=InnoDB;
```

```php
{
    $_SESSION['flash'][] = ['tipo' => 'error', 'texto' => $mensaje];

{
    $flash = $_SESSION['flash'] ?? [];
    unset($_SESSION['flash']);

{
    return isset($_SESSION['usuario_id']);

{

        exit;
    }

{
    return hash_equals($_SESSION['csrf_token'] ?? '', $token);
```

```php
<?php
function h(?string $valor): string
{
    return htmlspecialchars($valor ?? '', ENT_QUOTES, 'UTF-8');
}

function url(string $ruta): string
{
    return '/taskflow/' . ltrim($ruta, '/');
}

function fechaFormateada(?string $fecha, string $formato = 'd/m/Y H:i'): string
{
    if (!$fecha) return '—';
    $dt = new DateTime($fecha);
    return $dt->format($formato);
}

function tiempoRelativo(string $fecha): string
{
    $dt = new DateTime($fecha);
    $ahora = new DateTime();
    $diff = $ahora->diff($dt);

    if ($diff->y > 0) return "hace {$diff->y} año(s)";
    if ($diff->m > 0) return "hace {$diff->m} mes(es)";
    if ($diff->d > 0) return "hace {$diff->d} día(s)";
    if ($diff->h > 0) return "hace {$diff->h} hora(s)";
    if ($diff->i > 0) return "hace {$diff->i} minuto(s)";
    return "hace unos segundos";
}

function classPrioridad(string $prioridad): string
{
    return match ($prioridad) {
        'urgente' => 'prioridad-urgente',
        'alta' => 'prioridad-alta',
        'media' => 'prioridad-media',
        'baja' => 'prioridad-baja',
        default => 'prioridad-media',
    };
}
```

```php
<?php
require '../config/database.php';

$errores = [];
$datos = ['nombre' => '', 'apellido' => '', 'email' => ''];

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $datos['nombre'] = trim($_POST['nombre'] ?? '');
    $datos['apellido'] = trim($_POST['apellido'] ?? '');
    $datos['email'] = trim($_POST['email'] ?? '');
    $password = $_POST['password'] ?? '';
    $confirmar = $_POST['confirmar_password'] ?? '';

    if ($datos['nombre'] === '' || $datos['apellido'] === '') {
        $errores[] = 'Nombre y apellido son obligatorios.';
    }

    $emailValido = filter_var($datos['email'], FILTER_VALIDATE_EMAIL);
    if (!$emailValido) {
        $errores[] = 'Ingrese un correo electrónico válido.';
    }

    if (strlen($password) < 6) {
        $errores[] = 'La contraseña debe tener al menos 6 caracteres.';
    } elseif ($password !== $confirmar) {
        $errores[] = 'Las contraseñas no coinciden.';
    }

    if (empty($errores)) {
        $stmt = $pdo->prepare('SELECT id FROM usuarios WHERE email = :email');
        $stmt->execute([':email' => $datos['email']]);

        if ($stmt->fetch()) {
            $errores[] = 'Este correo ya está registrado.';
        } else {
            $hash = password_hash($password, PASSWORD_DEFAULT);
            $stmt = $pdo->prepare(
                'INSERT INTO usuarios (nombre, apellido, email, password)
                 VALUES (:nombre, :apellido, :email, :password)'
            );
            $stmt->execute([
                ':nombre' => $datos['nombre'],
                ':apellido' => $datos['apellido'],
                ':email' => $datos['email'],
                ':password' => $hash,

            $usuarioId = $pdo->lastInsertId();
            $_SESSION['usuario_id'] = (int)$usuarioId;
            $_SESSION['usuario_nombre'] = $datos['nombre'];
            $_SESSION['usuario_email'] = $datos['email'];
            session_regenerate_id(true);

            flashExito('Registro exitoso. ¡Bienvenido a TaskFlow!');
            header('Location: ../dashboard.php');
            exit;
        }
    }
}
```

```php
<?php
require '../config/database.php';
require '../config/session.php';

if (estaAutenticado()) {
    header('Location: ../dashboard.php');
    exit;
}

$error = '';
$email = '';

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    if (!verificarCsrf($_POST['csrf_token'] ?? '')) {
        die('Error CSRF.');
    }

    $email = trim($_POST['email'] ?? '');
    $password = $_POST['password'] ?? '';
    $recordar = isset($_POST['recordar']);

    if ($email === '' || $password === '') {
        $error = 'Ingrese su correo y contraseña.';
    } else {
        $stmt = $pdo->prepare('SELECT * FROM usuarios WHERE email = :email AND activo = 1');
        $stmt->execute([':email' => $email]);
        $usuario = $stmt->fetch();

        if ($usuario && password_verify($password, $usuario['password'])) {
            $_SESSION['usuario_id'] = (int)$usuario['id'];
            $_SESSION['usuario_nombre'] = $usuario['nombre'];
            $_SESSION['usuario_apellido'] = $usuario['apellido'];
            $_SESSION['usuario_email'] = $usuario['email'];
            $_SESSION['usuario_rol'] = $usuario['rol'];
            session_regenerate_id(true);

            // Actualizar último acceso
            $pdo->prepare('UPDATE usuarios SET ultimo_acceso = NOW() WHERE id = :id')
                ->execute([':id' => $usuario['id']]);

            // Recordar sesión
            if ($recordar) {
                $token = bin2hex(random_bytes(32));
                $hash = hash('sha256', $token);
                $expira = date('Y-m-d H:i:s', time() + 86400 * 30);

                $pdo->prepare('INSERT INTO tokens_recordar (usuario_id, token_hash, expira_en) VALUES (:uid, :hash, :exp)')
                    ->execute([':uid' => $usuario['id'], ':hash' => $hash, ':exp' => $expira]);

                setcookie('recordar_token', $token, [
                    'expires' => time() + 86400 * 30,
                    'path' => '/',
                    'secure' => isset($_SERVER['HTTPS']),
                    'httponly' => true,
                    'samesite' => 'Lax',
                ]);
            }

            flashExito('¡Bienvenido de nuevo, ' . $usuario['nombre'] . '!');
            header('Location: ../dashboard.php');
            exit;
        } else {
            $error = 'Credenciales incorrectas.';
        }
    }
}

require '../layouts/header.php';
?>
```

```php
<?php
require '../config/database.php';
require '../config/session.php';

header('Content-Type: application/json');

if ($_SERVER['REQUEST_METHOD'] !== 'POST') {
    echo json_encode(['success' => false, 'error' => 'Método no permitido']);
    exit;
}

if (!verificarCsrf($_POST['csrf_token'] ?? '')) {
    echo json_encode(['success' => false, 'error' => 'CSRF inválido']);
    exit;
}

$id = (int)($_POST['id'] ?? 0);
$estado = $_POST['estado'] ?? '';

$estadosPermitidos = ['pendiente', 'en_progreso', 'completada'];
if (!in_array($estado, $estadosPermitidos)) {
    echo json_encode(['success' => false, 'error' => 'Estado inválido']);
    exit;
}

try {
    $params = [':estado' => $estado, ':id' => $id];

    if ($estado === 'completada') {
        $stmt = $pdo->prepare('UPDATE tareas SET estado = :estado, completada_en = NOW() WHERE id = :id');
    } else {
        $stmt = $pdo->prepare('UPDATE tareas SET estado = :estado, completada_en = NULL WHERE id = :id');
    }

    $stmt->execute($params);

    if ($stmt->rowCount() > 0) {
        echo json_encode(['success' => true]);
    } else {
        echo json_encode(['success' => false, 'error' => 'Tarea no encontrada']);
    }
} catch (PDOException $e) {
    echo json_encode(['success' => false, 'error' => 'Error de base de datos']);
}
?>
```
