2. ARQUITECTURA DEL SISTEMA

El sistema se organiza en los siguientes componentes:

2. clases/ — Clases del dominio.
3. funciones/ — Funciones de utilidad.

Diagrama de flujo del programa:

```
inicio.php
    |
    v
autenticacion() [while + match]
    |
    v
menuPrincipal() [do...while + switch]
    |
    +---> 1. Gestión Cuentas [foreach + for + match]
    |         |
    |         +---> crearCuenta() [if/else + try/catch]
    |         +---> verCuenta() [foreach + match]
    |         +---> listarCuentas() [foreach]
    |         +---> eliminarCuenta() [if + throw]
    |
    +---> 2. Transacciones [while + for + match]
    |         |
    |         +---> depositar() [if + try/catch]
    |         +---> retirar() [if/else + validaciones]
    |         +---> transferir() [if + match]
    |
    +---> 3. Préstamos [for + while + break/continue]
    |         |
    |         +---> solicitar() [if + match]
    |         +---> pagar() [while + continue]
    |
    +---> 4. Reportes [foreach + for + match + break]
    |
    +---> 5. Salir [break del menú principal]
```

```php
<?php
declare(strict_types=1);

// ====================== CONFIGURACIÓN ======================
define("ARCHIVO_DATOS", __DIR__ . "/datos/banco.json");
define("ARCHIVO_LOG", __DIR__ . "/datos/operaciones.log");
define("TASA_INTERES_ANUAL", 0.15);
define("LIMITE_RETIRO_DIARIO", 10000.00);
define("MAX_INTENTOS_LOGIN", 3);

// ====================== CLASES ======================

class ErrorBanco extends Exception {
    public function __construct(string $mensaje, int $codigo = 400,
                                ?Throwable $previous = null) {
        parent::__construct($mensaje, $codigo, $previous);
    }
}

class ErrorSaldoInsuficiente extends ErrorBanco {
    private float $saldoActual;
    private float $montoRequerido;

    public function __construct(float $saldo, float $monto) {
        $this->saldoActual = $saldo;
        $this->montoRequerido = $monto;
        parent::__construct(
            "Saldo insuficiente. Disponible: \$$saldo, Requerido: \$$monto",
            402
        );
    }

    public function getSaldo(): float { return $this->saldoActual; }
    public function getMonto(): float { return $this->montoRequerido; }
}

class ErrorAutenticacion extends ErrorBanco {
    public function __construct(string $mensaje = "Credenciales inválidas.") {
        parent::__construct($mensaje, 401);
    }
}

class Cuenta {
    public function __construct(
        private string $numero,
        private string $titular,
        private string $email,
        private float  $saldo = 0.0,
        private bool   $activa = true,
        private string $tipo = "ahorros",
        private float  $limiteDiario = LIMITE_RETIRO_DAIRO,
        private array  $transacciones = [],
    ) {}

    // Getters
    public function getNumero(): string { return $this->numero; }
    public function getTitular(): string { return $this->titular; }
    public function getEmail(): string { return $this->email; }
    public function getSaldo(): float { return $this->saldo; }
    public function isActiva(): bool { return $this->activa; }
    public function getTipo(): string { return $this->tipo; }
    public function getTransacciones(): array { return $this->transacciones; }

    public function depositar(float $monto, string $descripcion = "Depósito"): void {
        if ($monto <= 0) {
            throw new ErrorBanco("El monto debe ser positivo.");
        }
        if (!$this->activa) {
            throw new ErrorBanco("Cuenta inactiva. No se puede depositar.");
        }

        $this->saldo += $monto;
        $this->registrarTransaccion("depósito", $monto, $descripcion);
    }

    public function retirar(float $monto, string $descripcion = "Retiro"): void {
        if ($monto <= 0) {
            throw new ErrorBanco("El monto debe ser positivo.");
        }
        if (!$this->activa) {
            throw new ErrorBanco("Cuenta inactiva.");
        }
        if ($monto > $this->saldo) {
            throw new ErrorSaldoInsuficiente($this->saldo, $monto);
        }

        $totalRetirosHoy = $this->calcularRetirosHoy();
        if ($totalRetirosHoy + $monto > $this->limiteDiario) {
            throw new ErrorBanco(
                "Límite diario excedido. Disponible: \$" .
                ($this->limiteDiario - $totalRetirosHoy)
            );
        }

        $this->saldo -= $monto;
        $this->registrarTransaccion("retiro", $monto, $descripcion);
    }

    public function transferir(Cuenta $destino, float $monto,
                               string $descripcion = "Transferencia"): void {
        $this->retirar($monto, "Transferencia a {$destino->getNumero()}");
        try {
            $destino->depositar($monto, "Transferencia de {$this->numero}");
        } catch (ErrorBanco $e) {
            $this->depositar($monto, "Reversión transferencia fallida");
            throw new ErrorBanco("Error en transferencia: " . $e->getMessage());
        }
    }

    public function aplicarInteres(float $tasa): float {
        $interes = $this->saldo * $tasa / 100;
        $this->saldo += $interes;
        $this->registrarTransaccion("interés", $interes, "Interés mensual");
        return $interes;
    }

    public function aArray(): array {
        return [
            "numero" => $this->numero,
            "titular" => $this->titular,
            "email" => $this->email,
            "saldo" => $this->saldo,
            "activa" => $this->activa,
            "tipo" => $this->tipo,
            "limiteDiario" => $this->limiteDiario,
            "transacciones" => $this->transacciones,
        ];
    }

    public static function desdeArray(array $datos): self {
        return new self(
            $datos["numero"], $datos["titular"], $datos["email"],
            $datos["saldo"], $datos["activa"], $datos["tipo"],
            $datos["limiteDiario"] ?? LIMITE_RETIRO_DIARIO,
            $datos["transacciones"] ?? []
        );
    }

    private function registrarTransaccion(string $tipo, float $monto,
                                          string $descripcion): void {
        $this->transacciones[] = [
            "fecha" => date("Y-m-d H:i:s"),
            "tipo" => $tipo,
            "monto" => $monto,
            "descripcion" => $descripcion,
            "saldoPosterior" => $this->saldo,
        ];
    }

    private function calcularRetirosHoy(): float {
        $hoy = date("Y-m-d");
        $total = 0.0;

        foreach ($this->transacciones as $tx) {
            if ($tx["tipo"] === "retiro" && str_starts_with($tx["fecha"], $hoy)) {
                $total += $tx["monto"];
            }
        }

        return $total;
    }
}

class Banco {
    private array $cuentas = [];
    private array $usuarios = [];
    private array $prestamos = [];
    private string $rutaDatos;

    public function __construct(string $rutaDatos = ARCHIVO_DATOS) {
        $this->rutaDatos = $rutaDatos;
        $this->cargarDatos();
    }

    public function autenticar(string $email, string $password): array {
        $intentos = 0;
        $usuario = null;

        foreach ($this->usuarios as $u) {
            if ($u["email"] === $email) {
                $usuario = $u;
                break;
            }
        }

        if ($usuario === null) {
            throw new ErrorAutenticacion("Usuario no encontrado: $email");
        }

        if (!password_verify($password, $usuario["password_hash"])) {
            throw new ErrorAutenticacion("Contraseña incorrecta.");
        }

        if (!$usuario["activo"]) {
            throw new ErrorAutenticacion("Cuenta de usuario desactivada.");
        }

        return $usuario;
    }

    public function crearCuenta(string $titular, string $email,
                                string $tipo = "ahorros"): Cuenta {
        $errores = [];

        if (strlen($titular) < 3) {
            $errores[] = "Nombre debe tener al menos 3 caracteres.";
        }
        if (!filter_var($email, FILTER_VALIDATE_EMAIL)) {
            $errores[] = "Email inválido.";
        }

        $tipo = strtolower($tipo);
        $tiposValidos = ["ahorros", "corriente", "empresarial"];
        if (!in_array($tipo, $tiposValidos)) {
            $errores[] = "Tipo inválido. Válidos: " . implode(", ", $tiposValidos);
        }

        if (!empty($errores)) {
            throw new ErrorBanco(implode("\n", $errores));
        }

        $numero = $this->generarNumeroCuenta();
        $cuenta = new Cuenta($numero, $titular, $email, 0.0, true, $tipo);
        $this->cuentas[$numero] = $cuenta;
        $this->guardarDatos();
        return $cuenta;
    }

    public function buscarCuenta(string $numero): ?Cuenta {
        return $this->cuentas[$numero] ?? null;
    }

    public function listarCuentas(): array {
        return $this->cuentas;
    }

    public function eliminarCuenta(string $numero): void {
        $cuenta = $this->buscarCuenta($numero);
        if ($cuenta === null) {
            throw new ErrorBanco("Cuenta no encontrada: $numero");
        }
        if ($cuenta->getSaldo() > 0) {
            throw new ErrorBanco(
                "No se puede eliminar: saldo pendiente de \${$cuenta->getSaldo()}"
            );
        }
        unset($this->cuentas[$numero]);
        $this->guardarDatos();
    }

    public function generarReporte(string $tipo): string {
        return match ($tipo) {
            "completo" => $this->reporteCompleto(),
            "resumen" => $this->reporteResumen(),
            "transacciones" => $this->reporteTransacciones(),
            "prestamos" => $this->reportePrestamos(),
            default => "Tipo de reporte no válido.",
        };
    }

    private function reporteCompleto(): string {
        $salida = "=== REPORTE COMPLETO DEL BANCO ===\n\n";
        $salida .= "Fecha: " . date("Y-m-d H:i:s") . "\n";
        $salida .= "Total cuentas: " . count($this->cuentas) . "\n";

        $totalActivas = 0;
        $totalSaldo = 0.0;
        $cuentasPorTipo = [];

        foreach ($this->cuentas as $cuenta) {
            if ($cuenta->isActiva()) $totalActivas++;
            $totalSaldo += $cuenta->getSaldo();

            $tipo = $cuenta->getTipo();
            $cuentasPorTipo[$tipo] = ($cuentasPorTipo[$tipo] ?? 0) + 1;
        }

        $salida .= "Cuentas activas: $totalActivas\n";
        $salida .= "Saldo total: \$" . number_format($totalSaldo, 2) . "\n";
        $salida .= "Cuentas por tipo:\n";

        foreach ($cuentasPorTipo as $tipo => $cantidad) {
            $salida .= "  - $tipo: $cantidad\n";
        }

        $salida .= "\n=== DETALLE DE CUENTAS ===\n";
        foreach ($this->cuentas as $cuenta) {
            $estado = $cuenta->isActiva() ? "ACTIVA" : "INACTIVA";
            $salida .= sprintf(
                "%s | %-20s | %-10s | %s | \$%s\n",
                $cuenta->getNumero(),
                $cuenta->getTitular(),
                $cuenta->getTipo(),
                $estado,
                number_format($cuenta->getSaldo(), 2)
            );
        }

        return $salida;
    }

    private function reporteResumen(): string {
        $totalSaldo = 0.0;
        $totalTransacciones = 0;
        $maxSaldo = 0.0;
        $cuentaMax = "";

        foreach ($this->cuentas as $cuenta) {
            $totalSaldo += $cuenta->getSaldo();
            $totalTransacciones += count($cuenta->getTransacciones());
            if ($cuenta->getSaldo() > $maxSaldo) {
                $maxSaldo = $cuenta->getSaldo();
                $cuentaMax = $cuenta->getNumero();
            }
        }

        $salida = "=== RESUMEN EJECUTIVO ===\n";
        $salida .= "Cuentas: " . count($this->cuentas) . "\n";
        $salida .= "Saldo total: \$" . number_format($totalSaldo, 2) . "\n";
        $salida .= "Transacciones totales: $totalTransacciones\n";
        $salida .= "Promedio por cuenta: \$" .
            number_format(count($this->cuentas) > 0 ?
                $totalSaldo / count($this->cuentas) : 0, 2) . "\n";
        $salida .= "Cuenta con mayor saldo: $cuentaMax (\$" .
            number_format($maxSaldo, 2) . ")\n";

        return $salida;
    }

    private function reporteTransacciones(): string {
        $salida = "=== HISTORIAL DE TRANSACCIONES ===\n\n";
        $totalTx = 0;
        $txPorTipo = [];

        foreach ($this->cuentas as $cuenta) {
            foreach ($cuenta->getTransacciones() as $tx) {
                $totalTx++;
                $tipo = $tx["tipo"];
                $txPorTipo[$tipo] = ($txPorTipo[$tipo] ?? 0) + 1;

                $salida .= sprintf(
                    "[%s] %s: %s \$%s - %s (Cuenta: %s)\n",
                    $tx["fecha"], strtoupper($tipo),
                    $tipo === "depósito" ? "+" : "-",
                    number_format($tx["monto"], 2),
                    $tx["descripcion"],
                    $cuenta->getNumero()
                );
            }
        }

        $salida .= "\nTotal: $totalTx transacciones\n";
        $salida .= "Por tipo:\n";
        foreach ($txPorTipo as $tipo => $cantidad) {
            $salida .= "  - $tipo: $cantidad\n";
        }

        return $salida;
    }

    private function reportePrestamos(): string {
        if (empty($this->prestamos)) {
            return "No hay préstamos registrados.\n";
        }

        $salida = "=== PRÉSTAMOS ===\n\n";
        $totalPrestado = 0.0;
        $totalPendiente = 0.0;

        foreach ($this->prestamos as $p) {
            $saldoPendiente = $p["monto"] - $p["pagado"];
            $estado = match (true) {
                $saldoPendiente <= 0 => "PAGADO",
                $p["aprobado"] === false => "PENDIENTE",
                default => "ACTIVO",
            };

            $salida .= sprintf(
                "Préstamo #%d: \$%s (%s) - %s\n",
                $p["id"], number_format($p["monto"], 2),
                $p["cuenta"], $estado
            );

            $totalPrestado += $p["monto"];
            $totalPendiente += $saldoPendiente > 0 ? $saldoPendiente : 0;
        }

        $salida .= "\nTotal prestado: \$" . number_format($totalPrestado, 2) . "\n";
        $salida .= "Total pendiente: \$" . number_format($totalPendiente, 2) . "\n";

        return $salida;
    }

    private function generarNumeroCuenta(): string {
        $prefijos = match (count($this->cuentas) % 3) {
            0 => "400",
            1 => "500",
            2 => "600",
        };

        do {
            $numero = $prefijos . str_pad((string)rand(0, 9999999), 7, "0", STR_PAD_LEFT);
        } while (isset($this->cuentas[$numero]));

        return $numero;
    }

    private function cargarDatos(): void {
        if (!file_exists($this->rutaDatos)) {
            $this->inicializarDatosPorDefecto();
            return;
        }

        try {
            $contenido = file_get_contents($this->rutaDatos);
            if ($contenido === false) {
                throw new RuntimeException("No se pudo leer el archivo.");
            }

            $datos = json_decode($contenido, true);
            if ($datos === null) {
                throw new RuntimeException("JSON inválido.");
            }

            $this->usuarios = $datos["usuarios"] ?? [];
            $this->prestamos = $datos["prestamos"] ?? [];

            $this->cuentas = [];
            foreach ($datos["cuentas"] ?? [] as $c) {
                $cuenta = Cuenta::desdeArray($c);
                $this->cuentas[$cuenta->getNumero()] = $cuenta;
            }
        } catch (RuntimeException $e) {
            echo "Error cargando datos: " . $e->getMessage() . "\n";
            echo "Usando datos por defecto.\n";
            $this->inicializarDatosPorDefecto();
        }
    }

    private function inicializarDatosPorDefecto(): void {
        $this->usuarios = [
            [
                "email" => "admin@banco.com",
                "password_hash" => password_hash("admin123", PASSWORD_DEFAULT),
                "nombre" => "Administrador",
                "rol" => "admin",
                "activo" => true,
            ],
            [
                "email" => "ana@example.com",
                "password_hash" => password_hash("ana123", PASSWORD_DEFAULT),
                "nombre" => "Ana López",
                "rol" => "usuario",
                "activo" => true,
            ],
        ];

        $this->cuentas = [];
        $this->prestamos = [];

        // Crear cuentas de ejemplo
        try {
            $cuentaAna = $this->crearCuenta("Ana López", "ana@example.com", "ahorros");
            // Forzar depósito inicial
            $ref = new ReflectionClass($cuentaAna);
            $saldo = $ref->getProperty("saldo");
            $saldo->setAccessible(true);
            $saldo->setValue($cuentaAna, 5000.0);
        } catch (ErrorBanco $e) {
            echo "Error inicializando datos: " . $e->getMessage() . "\n";
        }

        $this->guardarDatos();
    }

    private function guardarDatos(): void {
        $dir = dirname($this->rutaDatos);
        if (!is_dir($dir)) {
            mkdir($dir, 0755, true);
        }

        $cuentasArray = [];
        foreach ($this->cuentas as $cuenta) {
            $cuentasArray[] = $cuenta->aArray();
        }

        $datos = [
            "usuarios" => $this->usuarios,
            "cuentas" => $cuentasArray,
            "prestamos" => $this->prestamos,
        ];

        file_put_contents(
            $this->rutaDatos,
            json_encode($datos, JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE)
        );
    }
}

// ====================== FUNCIONES DE INTERFAZ ======================

function limpiarPantalla(): void {
    echo chr(27) . "[2J" . chr(27) . "[0;0H";
}

function pausar(): void {
    echo "\nPresiona Enter para continuar...";
    fgets(STDIN);
}

function leerOpcion(string $prompt = "Opción: "): string {
    echo $prompt;
    return trim(fgets(STDIN));
}

function leerNumero(string $prompt): float {
    while (true) {
        $entrada = leerOpcion($prompt);
        if (is_numeric($entrada)) {
            return (float)$entrada;
        }
        echo "Por favor, ingresa un número válido.\n";
    }
}

function logging(string $mensaje, string $tipo = "INFO"): void {
    $fecha = date("Y-m-d H:i:s");
    $log = "[$fecha] [$tipo] $mensaje\n";
    file_put_contents(ARCHIVO_LOG, $log, FILE_APPEND | LOCK_EX);
}

// ====================== MENÚS ======================

function menuAutenticacion(Banco &$banco): array {
    $intentos = 0;
    $maxIntentos = MAX_INTENTOS_LOGIN;

    limpiarPantalla();
    echo "========================================\n";
    echo "     SISTEMA BANCARIO INTELIGENTE\n";
    echo "========================================\n\n";

    while ($intentos < $maxIntentos) {
        echo "Inicio de sesión\n";
        echo str_repeat("-", 40) . "\n";
        $email = leerOpcion("Email: ");
        $password = leerOpcion("Contraseña: ");

        try {
            $usuario = $banco->autenticar($email, $password);
            logging("Inicio de sesión exitoso: $email");
            echo "\n¡Bienvenido, {$usuario['nombre']}!\n";
            pausar();
            return $usuario;
        } catch (ErrorAutenticacion $e) {
            $intentos++;
            $restantes = $maxIntentos - $intentos;
            echo "Error: {$e->getMessage()}\n";
            if ($restantes > 0) {
                echo "Intentos restantes: $restantes\n\n";
            }
        }
    }

    logging("Bloqueo por intentos fallidos.", "ALERTA");
    echo "\nDemasiados intentos fallidos. Sistema bloqueado.\n";
    exit(1);
}

function menuGestionCuentas(Banco &$banco): void {
    $salir = false;

    do {
        limpiarPantalla();
        echo "=== GESTIÓN DE CUENTAS ===\n";
        echo "1. Crear cuenta\n";
        echo "2. Ver cuenta\n";
        echo "3. Listar cuentas\n";
        echo "4. Eliminar cuenta\n";
        echo "5. Volver\n";
        echo str_repeat("-", 30) . "\n";

        $opcion = leerOpcion();

        switch ($opcion) {
            case "1":
                try {
                    echo "\n--- Crear cuenta ---\n";
                    $titular = leerOpcion("Titular: ");
                    $email = leerOpcion("Email: ");
                    echo "Tipos: ahorros, corriente, empresarial\n";
                    $tipo = leerOpcion("Tipo [ahorros]: ") ?: "ahorros";

                    $cuenta = $banco->crearCuenta($titular, $email, $tipo);
                    echo "\nCuenta creada exitosamente:\n";
                    echo "  Número: {$cuenta->getNumero()}\n";
                    echo "  Titular: {$cuenta->getTitular()}\n";
                    echo "  Tipo: {$cuenta->getTipo()}\n";
                    logging("Cuenta creada: {$cuenta->getNumero()} - $titular");
                } catch (ErrorBanco $e) {
                    echo "\nError: {$e->getMessage()}\n";
                }
                pausar();
                break;

            case "2":
                $numero = leerOpcion("Número de cuenta: ");
                $cuenta = $banco->buscarCuenta($numero);
                if ($cuenta === null) {
                    echo "\nCuenta no encontrada.\n";
                } else {
                    echo "\n--- Detalle de cuenta ---\n";
                    echo "Número: {$cuenta->getNumero()}\n";
                    echo "Titular: {$cuenta->getTitular()}\n";
                    echo "Email: {$cuenta->getEmail()}\n";
                    echo "Saldo: \$" . number_format($cuenta->getSaldo(), 2) . "\n";
                    echo "Tipo: {$cuenta->getTipo()}\n";
                    echo "Estado: " . ($cuenta->isActiva() ? "ACTIVA" : "INACTIVA") . "\n";
                    echo "Transacciones: " . count($cuenta->getTransacciones()) . "\n";
                }
                pausar();
                break;

            case "3":
                $cuentas = $banco->listarCuentas();
                if (empty($cuentas)) {
                    echo "\nNo hay cuentas registradas.\n";
                } else {
                    echo "\n--- Lista de cuentas ---\n";
                    echo str_pad("NÚMERO", 15) . str_pad("TITULAR", 22) .
                         str_pad("TIPO", 14) . str_pad("SALDO", 12) . "\n";
                    echo str_repeat("-", 63) . "\n";

                    foreach ($cuentas as $cuenta) {
                        printf("%-15s %-22s %-14s \$%s\n",
                            $cuenta->getNumero(),
                            substr($cuenta->getTitular(), 0, 20),
                            $cuenta->getTipo(),
                            number_format($cuenta->getSaldo(), 2)
                        );
                    }
                    echo str_repeat("-", 63) . "\n";
                    echo "Total: " . count($cuentas) . " cuentas.\n";
                }
                pausar();
                break;

            case "4":
                try {
                    $numero = leerOpcion("Número de cuenta a eliminar: ");
                    $banco->eliminarCuenta($numero);
                    echo "\nCuenta $numero eliminada.\n";
                    logging("Cuenta eliminada: $numero");
                } catch (ErrorBanco $e) {
                    echo "\nError: {$e->getMessage()}\n";
                }
                pausar();
                break;

            case "5":
                $salir = true;
                break;

            default:
                echo "Opción no válida.\n";
                pausar();
        }
    } while (!$salir);
}

function menuTransacciones(Banco &$banco): void {
    $salir = false;

    do {
        limpiarPantalla();
        echo "=== TRANSACCIONES ===\n";
        echo "1. Depósito\n";
        echo "2. Retiro\n";
        echo "3. Transferencia\n";
        echo "4. Volver\n";
        echo str_repeat("-", 30) . "\n";

        $opcion = leerOpcion();

        switch ($opcion) {
            case "1":
                try {
                    $numero = leerOpcion("Número de cuenta: ");
                    $cuenta = $banco->buscarCuenta($numero);
                    if ($cuenta === null) {
                        throw new ErrorBanco("Cuenta no encontrada.");
                    }

                    $monto = leerNumero("Monto a depositar: \$");
                    $desc = leerOpcion("Descripción [Depósito]: ") ?: "Depósito";

                    $cuenta->depositar($monto, $desc);
                    echo "\nDepósito exitoso. Nuevo saldo: \$" .
                         number_format($cuenta->getSaldo(), 2) . "\n";
                    logging("Depósito: \$$monto en $numero");
                } catch (ErrorBanco $e) {
                    echo "\nError: {$e->getMessage()}\n";
                }
                pausar();
                break;

            case "2":
                try {
                    $numero = leerOpcion("Número de cuenta: ");
                    $cuenta = $banco->buscarCuenta($numero);
                    if ($cuenta === null) {
                        throw new ErrorBanco("Cuenta no encontrada.");
                    }

                    echo "Saldo disponible: \$" .
                         number_format($cuenta->getSaldo(), 2) . "\n";
                    $monto = leerNumero("Monto a retirar: \$");
                    $desc = leerOpcion("Descripción [Retiro]: ") ?: "Retiro";

                    $cuenta->retirar($monto, $desc);
                    echo "\nRetiro exitoso. Nuevo saldo: \$" .
                         number_format($cuenta->getSaldo(), 2) . "\n";
                    logging("Retiro: \$$monto de $numero");
                } catch (ErrorSaldoInsuficiente $e) {
                    echo "\nSaldo insuficiente. Disponible: \$" .
                         number_format($e->getSaldo(), 2) . "\n";
                } catch (ErrorBanco $e) {
                    echo "\nError: {$e->getMessage()}\n";
                }
                pausar();
                break;

            case "3":
                try {
                    $origen = leerOpcion("Cuenta origen: ");
                    $cuentaOrigen = $banco->buscarCuenta($origen);
                    if ($cuentaOrigen === null) {
                        throw new ErrorBanco("Cuenta origen no encontrada.");
                    }

                    $destino = leerOpcion("Cuenta destino: ");
                    $cuentaDestino = $banco->buscarCuenta($destino);
                    if ($cuentaDestino === null) {
                        throw new ErrorBanco("Cuenta destino no encontrada.");
                    }

                    if ($origen === $destino) {
                        throw new ErrorBanco("Origen y destino son la misma cuenta.");
                    }

                    echo "Saldo disponible: \$" .
                         number_format($cuentaOrigen->getSaldo(), 2) . "\n";
                    $monto = leerNumero("Monto a transferir: \$");
                    $desc = leerOpcion("Descripción [Transferencia]: ") ?: "Transferencia";

                    $cuentaOrigen->transferir($cuentaDestino, $monto, $desc);
                    echo "\nTransferencia exitosa.\n";
                    echo "Saldo origen: \$" .
                         number_format($cuentaOrigen->getSaldo(), 2) . "\n";
                    logging("Transferencia: \$$monto de $origen a $destino");
                } catch (ErrorBanco $e) {
                    echo "\nError: {$e->getMessage()}\n";
                }
                pausar();
                break;

            case "4":
                $salir = true;
                break;

            default:
                echo "Opción no válida.\n";
                pausar();
        }
    } while (!$salir);
}

function menuPrestamos(Banco &$banco): void {
    $salir = false;

    do {
        limpiarPantalla();
        echo "=== PRÉSTAMOS ===\n";
        echo "1. Solicitar préstamo\n";
        echo "2. Pagar cuota\n";
        echo "3. Ver mis préstamos\n";
        echo "4. Volver\n";
        echo str_repeat("-", 30) . "\n";

        $opcion = leerOpcion();

        switch ($opcion) {
            case "1":
                // Simulación simplificada de solicitud
                echo "\n--- Solicitar préstamo ---\n";
                $numero = leerOpcion("Cuenta: ");
                $cuenta = $banco->buscarCuenta($numero);

                if ($cuenta === null) {
                    echo "Cuenta no encontrada.\n";
                    pausar();
                    break;
                }

                $monto = leerNumero("Monto solicitado: \$");
                $plazo = (int)leerOpcion("Plazo (meses) [12]: ") ?: 12;

                if ($monto <= 0 || $monto > 100000) {
                    echo "Monto inválido (1-100000).\n";
                    pausar();
                    break;
                }

                $plazosValidos = [6, 12, 24, 36, 48, 60];
                if (!in_array($plazo, $plazosValidos)) {
                    echo "Plazo inválido. Válidos: " . implode(", ", $plazosValidos) . "\n";
                    pausar();
                    break;
                }

                $tasaMensual = TASA_INTERES_ANUAL / 12;
                $pagoMensual = $monto * ($tasaMensual * pow(1 + $tasaMensual, $plazo))
                             / (pow(1 + $tasaMensual, $plazo) - 1);
                $totalPagar = $pagoMensual * $plazo;

                echo "\nResumen del préstamo:\n";
                echo "  Monto: \$" . number_format($monto, 2) . "\n";
                echo "  Plazo: $plazo meses\n";
                echo "  Tasa anual: " . (TASA_INTERES_ANUAL * 100) . "%\n";
                echo "  Pago mensual: \$" . number_format($pagoMensual, 2) . "\n";
                echo "  Total a pagar: \$" . number_format($totalPagar, 2) . "\n";

                $confirmar = leerOpcion("¿Aceptar? (s/n): ");
                if (strtolower($confirmar) === "s") {
                    // Guardar préstamo (simplificado)
                    echo "\nPréstamo aprobado. Los fondos se depositarán en la cuenta.\n";
                    try {
                        $cuenta->depositar($monto, "Préstamo aprobado");
                        logging("Préstamo: \$$monto a $numero a $plazo meses");
                    } catch (ErrorBanco $e) {
                        echo "Error: {$e->getMessage()}\n";
                    }
                } else {
                    echo "Solicitud cancelada.\n";
                }
                pausar();
                break;

            case "2":
                echo "\n--- Pagar cuota ---\n";
                echo "Función en desarrollo. Vuelve pronto.\n";
                pausar();
                break;

            case "3":
                echo "\n--- Mis préstamos ---\n";
                echo "No tienes préstamos activos.\n";
                pausar();
                break;

            case "4":
                $salir = true;
                break;

            default:
                echo "Opción no válida.\n";
                pausar();
        }
    } while (!$salir);
}
```

4. ESTRUCTURAS DE CONTROL UTILIZADAS

El proyecto integra todas las estructuras de control del módulo:

| Estructura     | Dónde se usa                                         |
|---------------|------------------------------------------------------|
| if/else       | Validaciones, estados, condiciones de negocio        |
| elseif        | Clasificación de respuestas, múltiples condiciones   |
| switch        | Menús principales (gestión, transacciones, reportes) |
| match         | Clasificación de tipos, menú principal, roles        |
| while         | Bucle de autenticación, entrada de datos             |
| do...while    | Menú principal, submenús (garantiza 1 iteración)     |
| for           | Rangos, tablas, generación de datos                  |
| foreach       | Recorrer cuentas, transacciones, arrays de datos     |
| break         | Salir de bucles, menús, búsquedas                    |
| break N       | Salir de bucles anidados                             |
| continue      | Saltar iteraciones en bucles                         |
| try/catch     | Manejo de excepciones en todo el sistema             |
| finally       | Cierre de recursos (archivos, BD)                    |
| throw         | Excepciones personalizadas del dominio               |

   HTTP (400, 401, 404, 500).
