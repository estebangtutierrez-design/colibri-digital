```php
{
    private string $numero;

    private string $email;

        $this->email = $email;

    {
```

```php
<?php
declare(strict_types=1);

// PARTE 1: Consultas SQL (10 puntos)

// 12. Escribe una consulta SQL que obtenga:
//     Las 5 categorías con más productos activos,
//     ordenadas de mayor a menor cantidad.
$sql = "SELECT c.nombre, COUNT(p.id) as total_productos
        FROM categorias c
        LEFT JOIN productos p ON c.id = p.categoria_id AND p.activo = 1
        GROUP BY c.id, c.nombre
        ORDER BY total_productos DESC
        LIMIT 5";

// 13. Escribe una consulta que actualice el stock de todos los productos
//     de la categoría 'Electrónica' incrementándolo en 10 unidades:
$sql = "UPDATE productos p
        INNER JOIN categorias c ON p.categoria_id = c.id
        SET p.stock = p.stock + 10
        WHERE c.nombre = 'Electrónica' AND p.activo = 1";

// PARTE 2: PDO (10 puntos)

// 14. Implementa una función que usando PDO obtenga todos los productos
//     activos cuyo precio esté entre $min y $max, con paginación:

function obtenerProductosPorRangoPrecio(
    PDO $db,
    float $min,
    float $max,
    int $page = 1,
    int $limit = 20
): array {
    $offset = ($page - 1) * $limit;

    $sql = "SELECT id, nombre, precio, stock
            FROM productos
            WHERE activo = 1
              AND precio BETWEEN :min AND :max
            ORDER BY precio ASC
            LIMIT :limit OFFSET :offset";

    $stmt = $db->prepare($sql);
    $stmt->bindValue(':min', $min, PDO::PARAM_STR);
    $stmt->bindValue(':max', $max, PDO::PARAM_STR);
    $stmt->bindValue(':limit', $limit, PDO::PARAM_INT);
    $stmt->bindValue(':offset', $offset, PDO::PARAM_INT);
    $stmt->execute();

    return $stmt->fetchAll(PDO::FETCH_ASSOC);
}

// 15. Implementa una transacción que cree una orden con sus detalles.
//     Si falla algún detalle, debe revertirse todo:

function crearOrdenCompleta(PDO $db, int $usuarioId, array $items): int
{
    try {
        $db->beginTransaction();

        // Crear orden
        $total = array_sum(array_column($items, 'subtotal'));
        $stmt = $db->prepare(
            "INSERT INTO ordenes (usuario_id, total, estado, created_at)
             VALUES (:usuario_id, :total, 'pendiente', NOW())"
        );
        $stmt->execute(['usuario_id' => $usuarioId, 'total' => $total]);
        $ordenId = (int)$db->lastInsertId();

        // Insertar detalles y actualizar stock
        $stmtDetalle = $db->prepare(
            "INSERT INTO ordenes_detalles (orden_id, producto_id, cantidad, precio_unitario)
             VALUES (:orden_id, :producto_id, :cantidad, :precio)"
        );

        $stmtStock = $db->prepare(
            "UPDATE productos SET stock = stock - :cantidad WHERE id = :id AND stock >= :cantidad2"
        );

        foreach ($items as $item) {
            $stmtDetalle->execute([
                'orden_id' => $ordenId,
                'producto_id' => $item['producto_id'],
                'cantidad' => $item['cantidad'],
                'precio' => $item['precio'],
            ]);

            $stmtStock->execute([
                'cantidad' => $item['cantidad'],
                'id' => $item['producto_id'],
                'cantidad2' => $item['cantidad'],
            ]);

            if ($stmtStock->rowCount() === 0) {
                throw new \PDOException("Stock insuficiente para el producto {$item['producto_id']}");
            }
        }

        $db->commit();
        return $ordenId;

    } catch (\Exception $e) {
        $db->rollBack();
        throw $e;
    }
}
?>
```

```php
<?php
declare(strict_types=1);

// PARTE 1: Identificar vulnerabilidades (10 puntos)

// 16. Identifica 3 vulnerabilidades de seguridad en el siguiente código:

// Código vulnerable:
// $id = $_GET['id'];
// $sql = "SELECT * FROM usuarios WHERE id = $id";
// $resultado = $db->query($sql);

// Vulnerabilidades:

if ($id === false || $id === null) {
    die('ID inválido');
}

$stmt = $db->prepare("SELECT * FROM usuarios WHERE id = :id");
$stmt->execute(['id' => $id]);
$usuario = $stmt->fetch();

$nombre = htmlspecialchars($_GET['nombre'] ?? '', ENT_QUOTES, 'UTF-8');
echo "Bienvenido, " . $nombre;

// PARTE 2: Implementar seguridad (10 puntos)

// 17. Implementa una función que almacene una contraseña de forma segura:
function almacenarPassword(string $password): string
{
    return password_hash($password, PASSWORD_BCRYPT, ['cost' => 12]);
}

// 18. Implementa la verificación:
function verificarPassword(string $password, string $hash): bool
{
    return password_verify($password, $hash);
}

// 19. ¿Qué cabeceras HTTP de seguridad añadirías a una aplicación PHP?
//     Lista al menos 5:
/*
1. Content-Security-Policy (CSP)
2. Strict-Transport-Security (HSTS)
3. X-Content-Type-Options: nosniff
4. X-Frame-Options: DENY
5. Referrer-Policy: strict-origin-when-cross-origin
6. Permissions-Policy
7. X-XSS-Protection: 1; mode=block
*/

// 20. ¿Cómo protegerías un endpoint de API contra ataques de fuerza bruta?
//     Describe la estrategia:

class BruteForceProtection
{
    private const MAX_ATTEMPTS = 5;
    private const WINDOW_MINUTES = 15;
    private const BLOCK_HOURS = 1;
    private string $storageDir;

    public function __construct()
    {
        $this->storageDir = __DIR__ . '/storage/security';
    }

    public function isBlocked(string $ip): bool
    {
        $data = $this->loadData($ip);

        // Verificar si está bloqueado permanentemente
        if (isset($data['blocked_until']) && $data['blocked_until'] > time()) {
            return true;
        }

        // Limpiar intentos fuera de la ventana
        $windowStart = time() - (self::WINDOW_MINUTES * 60);
        $data['attempts'] = array_filter(
            $data['attempts'],
            fn($t) => $t > $windowStart
        );

        return count($data['attempts']) >= self::MAX_ATTEMPTS;
    }

    public function registerAttempt(string $ip, bool $success): void
    {
        $data = $this->loadData($ip);

        if ($success) {
            $data['attempts'] = [];
            $data['blocked_until'] = null;
        } else {
            $data['attempts'][] = time();

            if (count($data['attempts']) >= self::MAX_ATTEMPTS) {
                $data['blocked_until'] = time() + (self::BLOCK_HOURS * 3600);
            }
        }

        $this->saveData($ip, $data);
    }

    private function loadData(string $ip): array
    {
        $file = $this->getFilePath($ip);
        if (!file_exists($file)) {
            return ['attempts' => [], 'blocked_until' => null];
        }
        return json_decode(file_get_contents($file), true) ?? [];
    }

    private function saveData(string $ip, array $data): void
    {
        if (!is_dir($this->storageDir)) {
            mkdir($this->storageDir, 0700, true);
        }
        file_put_contents(
            $this->getFilePath($ip),
            json_encode($data),
            LOCK_EX
        );
    }

    private function getFilePath(string $ip): string
    {
        return $this->storageDir . '/' . md5($ip) . '.json';
    }
}
?>
```

```php
<?php
// PARTE 1: Preguntas conceptuales (10 puntos)

// 21. Describe el patrón MVC y sus ventajas:
/*
MVC (Modelo-Vista-Controlador) separa la aplicación en tres capas:
- Modelo: Gestiona datos y lógica de negocio
- Vista: Presenta información al usuario
- Controlador: Procesa peticiones y coordina Modelo y Vista

Ventajas:
- Separación de responsabilidades
- Código más mantenible y testeable
- Desarrollo en paralelo
- Reutilización de componentes
*/

// 22. ¿Qué principios SOLID aplican directamente al diseño de una API REST?
/*
- S (Single Responsibility): Cada endpoint debe tener una única responsabilidad
- O (Open/Closed): Los controladores deben ser extensibles sin modificación
- L (Liskov): Los repositorios deben poder sustituirse por implementaciones mock
- I (Interface Segregation): Interfaces específicas para cada repositorio
- D (Dependency Inversion): Depender de abstracciones, no de implementaciones concretas
*/

// 23. ¿Cuándo usarías Redis en lugar de MySQL para almacenar datos?
/*
Redis es ideal para:
- Caché de consultas frecuentes
- Sesiones de usuario
- Colas de mensajes
- Rate limiting
- Datos temporales con expiración automática

MySQL es mejor para:
- Datos persistentes con relaciones
- Consultas complejas con JOINs
- Integridad referencial
- Reportes y análisis
*/

// PARTE 2: Diseño de sistema (10 puntos)

// 24. Diseña la arquitectura de un sistema de notificaciones en tiempo real
//     que soporte 100,000 usuarios concurrentes. Describe los componentes:

/*
Arquitectura propuesta:

1. Frontend (React/Vue):
   - WebSocket client (Socket.IO o Pusher)
   - Service Worker para notificaciones push
   - Fallback a long-polling

2. Backend (PHP + Laravel):
   - Eventos emitidos por el servidor
   - Broadcasting con Redis
   - Cola de notificaciones (RabbitMQ/Redis)

3. Canales de entrega:
   - WebSocket: Tiempo real, para usuarios activos
   - Email: Notificaciones no críticas (MailHog/SES)

// 25. ¿Cómo migrarías una aplicación monolítica PHP a microservicios?

Estrategia de migración (Strangler Pattern):

3. Implementar API Gateway como punto de entrada único
```
