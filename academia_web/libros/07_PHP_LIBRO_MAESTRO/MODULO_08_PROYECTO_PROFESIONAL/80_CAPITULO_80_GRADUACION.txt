        $certificado = [

            'emitido_por' => 'Profesor Búho AI',
            'academia' => 'Lucas EGA Academy',
            'url_verificacion' => "https://lucas-ega-academy.com/verificar/{$id}",
            'competencias' => [
                'Fundamentos de PHP y sintaxis moderna (PHP 8.x)',
                'Programación Orientada a Objetos y patrones de diseño',
                'Bases de datos relacionales con PDO y MySQL',
                'Desarrollo de APIs RESTful con autenticación JWT',
                'Seguridad web: CSRF, XSS, SQL Injection, CORS',
                'Arquitectura MVC y front controller',
                'Optimización de rendimiento con Redis y OPcache',
                'Integración frontend-backend con React',
                'Despliegue con Docker y CI/CD',
                'Metodologías ágiles y buenas prácticas profesionales',
            ],
        ];

        $this->guardarCertificado($certificado);
        $this->enviarCertificadoEmail($certificado);

        return $certificado;
    }

    private function generarIdUnico(): string
    {
        return strtoupper(substr(bin2hex(random_bytes(4)), 0, 8));
    }

    private function guardarCertificado(array $certificado): void
    {
        $archivo = __DIR__ . "/../../storage/certificados/{$certificado['id']}.json";
        $dir = dirname($archivo);

        if (!is_dir($dir)) {
            mkdir($dir, 0755, true);
        }

        file_put_contents(
            $archivo,
            json_encode($certificado, JSON_PRETTY_PRINT),
            LOCK_EX
        );
    }

    private function enviarCertificadoEmail(array $certificado): void
    {
        $mensaje = $this->renderizarEmail($certificado);

        // Enviar email usando PHPMailer o servicio de correo
        // mail($certificado['email'], 'Tu certificado PHP Master Book', $mensaje);

        echo "Certificado enviado a {$certificado['email']}\n";
    }

    private function renderizarEmail(array $certificado): string
    {
        return <<<HTML
        <!DOCTYPE html>
        <html>
        <head><meta charset="UTF-8"></head>
        <body style="font-family: Arial; max-width: 600px; margin: 0 auto;">
            <h1 style="color: #2563eb;">¡Felicidades, {$certificado['estudiante']}!</h1>
            <p>Has completado el <strong>PHP Master Book</strong> y ahora eres oficialmente un
            <strong>PHP Master Engineer</strong>.</p>
            <p>Tu certificado está disponible en:
            <a href="{$certificado['url_verificacion']}">{$certificado['url_verificacion']}</a></p>
            <p>ID de Certificado: <strong>{$certificado['id']}</strong></p>
            <hr>

            cosas increíbles."</em> — Profesor Búho AI</p>
