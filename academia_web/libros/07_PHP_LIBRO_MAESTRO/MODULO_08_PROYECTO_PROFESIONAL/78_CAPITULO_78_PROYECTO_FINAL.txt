1. ESPECIFICACIÓN DEL PROYECTO
===============================================================================

El **Sistema de Gestión de Aprendizaje (LMS)** debe cumplir con los
siguientes requisitos funcionales y no funcionales:

**Requisitos funcionales:**
- Registro y autenticación de usuarios con roles (admin, instructor, estudiante)
- CRUD de cursos (título, descripción, imagen, precio, categoría)
- CRUD de lecciones dentro de cada curso (video, texto, recursos)
- Inscripción de estudiantes a cursos
- Progreso de lecciones (marcar como completadas)
- Exámenes por curso con preguntas de opción múltiple
- Calificación automática de exámenes
- Dashboard con estadísticas para cada rol
- Perfil de usuario con foto y biografía

**Requisitos no funcionales:**
- API RESTful con respuestas JSON
- Autenticación JWT
- Protección CSRF
- Validación de datos en servidor
- Paginación en listados
- Caché de consultas frecuentes
- Logs de actividad
- Pruebas unitarias (mínimo 20)
- Docker para entorno de desarrollo y producción

```sql
-- ESQUEMA COMPLETO DEL LMS
CREATE DATABASE IF NOT EXISTS lms_php_master
    CHARACTER SET utf8mb4
    COLLATE utf8mb4_unicode_ci;

USE lms_php_master;

-- Usuarios
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    rol ENUM('admin', 'instructor', 'estudiante') NOT NULL DEFAULT 'estudiante',
    bio TEXT,
    avatar VARCHAR(255),
    activo TINYINT(1) NOT NULL DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Categorías de cursos
CREATE TABLE categorias (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT,
    activo TINYINT(1) NOT NULL DEFAULT 1
);

-- Cursos
CREATE TABLE cursos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(200) NOT NULL,
    descripcion TEXT,
    imagen VARCHAR(255),
    precio DECIMAL(10, 2) NOT NULL DEFAULT 0.00,
    duracion_horas INT DEFAULT 0,
    nivel ENUM('principiante', 'intermedio', 'avanzado') DEFAULT 'principiante',
    instructor_id INT NOT NULL,
    categoria_id INT,
    activo TINYINT(1) NOT NULL DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (instructor_id) REFERENCES usuarios(id),
    FOREIGN KEY (categoria_id) REFERENCES categorias(id)
);

-- Lecciones
CREATE TABLE lecciones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    curso_id INT NOT NULL,
    titulo VARCHAR(200) NOT NULL,
    contenido TEXT,
    video_url VARCHAR(500),
    duracion_minutos INT DEFAULT 0,
    orden INT NOT NULL DEFAULT 0,
    activo TINYINT(1) NOT NULL DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (curso_id) REFERENCES cursos(id) ON DELETE CASCADE
);

-- Inscripciones
CREATE TABLE inscripciones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    curso_id INT NOT NULL,
    progreso DECIMAL(5, 2) DEFAULT 0.00,
    completado TINYINT(1) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id),
    FOREIGN KEY (curso_id) REFERENCES cursos(id),
    UNIQUE KEY unique_inscripcion (usuario_id, curso_id)
);

-- Progreso de lecciones
CREATE TABLE progreso_lecciones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    leccion_id INT NOT NULL,
    completado TINYINT(1) DEFAULT 0,
    completado_at TIMESTAMP NULL,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id),
    FOREIGN KEY (leccion_id) REFERENCES lecciones(id) ON DELETE CASCADE,
    UNIQUE KEY unique_progreso (usuario_id, leccion_id)
);

-- Exámenes
CREATE TABLE examenes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    curso_id INT NOT NULL,
    titulo VARCHAR(200) NOT NULL,
    descripcion TEXT,
    tiempo_limite_minutos INT DEFAULT 30,
    puntaje_maximo INT DEFAULT 100,
    activo TINYINT(1) NOT NULL DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (curso_id) REFERENCES cursos(id) ON DELETE CASCADE
);

-- Preguntas de examen
CREATE TABLE preguntas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    examen_id INT NOT NULL,
    texto TEXT NOT NULL,
    tipo ENUM('opcion_multiple', 'verdadero_falso') DEFAULT 'opcion_multiple',
    puntaje INT NOT NULL DEFAULT 10,
    orden INT NOT NULL DEFAULT 0,
    FOREIGN KEY (examen_id) REFERENCES examenes(id) ON DELETE CASCADE
);

-- Opciones de respuesta
CREATE TABLE opciones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    pregunta_id INT NOT NULL,
    texto TEXT NOT NULL,
    es_correcta TINYINT(1) NOT NULL DEFAULT 0,
    orden INT DEFAULT 0,
    FOREIGN KEY (pregunta_id) REFERENCES preguntas(id) ON DELETE CASCADE
);

-- Intentos de examen
CREATE TABLE intentos_examen (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    examen_id INT NOT NULL,
    puntaje_obtenido INT DEFAULT 0,
    puntaje_maximo INT DEFAULT 0,
    aprobado TINYINT(1) DEFAULT 0,
    iniciado_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completado_at TIMESTAMP NULL,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id),
    FOREIGN KEY (examen_id) REFERENCES examenes(id) ON DELETE CASCADE
);

-- Respuestas del estudiante
CREATE TABLE respuestas_estudiante (
    id INT AUTO_INCREMENT PRIMARY KEY,
    intento_id INT NOT NULL,
    pregunta_id INT NOT NULL,
    opcion_id INT,
    es_correcta TINYINT(1) DEFAULT 0,
    FOREIGN KEY (intento_id) REFERENCES intentos_examen(id) ON DELETE CASCADE,
    FOREIGN KEY (pregunta_id) REFERENCES preguntas(id),
    FOREIGN KEY (opcion_id) REFERENCES opciones(id)
);

-- Índices
CREATE INDEX idx_cursos_instructor ON cursos(instructor_id);
CREATE INDEX idx_cursos_categoria ON cursos(categoria_id);
CREATE INDEX idx_lecciones_curso ON lecciones(curso_id);
CREATE INDEX idx_inscripciones_usuario ON inscripciones(usuario_id);
CREATE INDEX idx_inscripciones_curso ON inscripciones(curso_id);
CREATE INDEX idx_examenes_curso ON examenes(curso_id);
```

```php
<?php
// src/Controllers/CursoController.php
declare(strict_types=1);

namespace App\Controllers;

use App\Core\Request;
use App\Core\Response;
use App\Services\CursoService;
use App\Helpers\Validator;

class CursoController
{
    private CursoService $cursoService;
    private Validator $validator;

    public function __construct()
    {
        $this->cursoService = new CursoService();
        $this->validator = new Validator();
    }

    /**
     * GET /api/cursos
     * Listar cursos con filtros, paginación y búsqueda
     */
    public function index(Request $request): Response
    {
        $filters = [
            'page' => (int)($request->getQuery('page', 1)),
            'limit' => min((int)($request->getQuery('limit', 12)), 50),
            'categoria_id' => $request->getQuery('categoria_id'),
            'nivel' => $request->getQuery('nivel'),
            'search' => $request->getQuery('search', ''),
            'orderBy' => $request->getQuery('orderBy', 'created_at'),
            'orderDir' => $request->getQuery('orderDir', 'DESC'),
        ];

        $result = $this->cursoService->listarCursos($filters);

        return new Response(json_encode([
            'status' => 'success',
            'data' => $result['cursos'],
            'pagination' => $result['pagination'],
        ]), 200);
    }

    /**
     * GET /api/cursos/{id}
     * Obtener un curso con sus lecciones
     */
    public function show(Request $request, array $params): Response
    {
        $id = (int)($params['id'] ?? 0);
        $curso = $this->cursoService->obtenerCursoCompleto($id);

        if (!$curso) {
            return new Response(json_encode([
                'status' => 'error',
                'message' => 'Curso no encontrado',
            ]), 404);
        }

        return new Response(json_encode([
            'status' => 'success',
            'data' => $curso,
        ]), 200);
    }

    /**
     * POST /api/cursos
     * Crear un nuevo curso (solo instructores y admins)
     */
    public function store(Request $request): Response
    {
        $data = $request->getJsonBody();
        $instructorId = $request->getAttribute('usuario_id');

        $rules = [
            'titulo' => 'required|max:200',
            'descripcion' => 'required',
            'precio' => 'numeric|min:0',
            'nivel' => 'in:principiante,intermedio,avanzado',
            'categoria_id' => 'integer|min:1',
        ];

        $errors = $this->validator->validate($data, $rules);
        if (!empty($errors)) {
            return new Response(json_encode([
                'status' => 'error',
                'message' => 'Error de validación',
                'errors' => $errors,
            ]), 422);
        }

        $data['instructor_id'] = $instructorId;
        $curso = $this->cursoService->crearCurso($data);

        return new Response(json_encode([
            'status' => 'success',
            'message' => 'Curso creado exitosamente',
            'data' => $curso,
        ]), 201);
    }

    /**
     * PUT /api/cursos/{id}
     * Actualizar un curso (solo el instructor o admin)
     */
    public function update(Request $request, array $params): Response
    {
        $id = (int)($params['id'] ?? 0);
        $usuarioId = $request->getAttribute('usuario_id');
        $data = $request->getJsonBody();

        // Verificar propiedad
        $curso = $this->cursoService->obtenerCursoCompleto($id);
        if (!$curso) {
            return new Response(json_encode([
                'status' => 'error',
                'message' => 'Curso no encontrado',
            ]), 404);
        }

        $updated = $this->cursoService->actualizarCurso($id, $data, $usuarioId);

        if (!$updated) {
            return new Response(json_encode([
                'status' => 'error',
                'message' => 'Error al actualizar o no autorizado',
            ]), 403);
        }

        $cursoActualizado = $this->cursoService->obtenerCursoCompleto($id);

        return new Response(json_encode([
            'status' => 'success',
            'message' => 'Curso actualizado exitosamente',
            'data' => $cursoActualizado,
        ]), 200);
    }

    /**
     * POST /api/cursos/{id}/inscribir
     * Inscribir al usuario autenticado en un curso
     */
    public function inscribir(Request $request, array $params): Response
    {
        $cursoId = (int)($params['id'] ?? 0);
        $usuarioId = $request->getAttribute('usuario_id');

        $result = $this->cursoService->inscribirUsuario($cursoId, $usuarioId);

        $statusCode = $result['success'] ? 201 : 409;

        return new Response(json_encode($result), $statusCode);
    }
}
?>
```

```javascript
// frontend/src/pages/Cursos/CursoDetalle.jsx
import { useState, useEffect } from 'react';
import { useParams } from 'react-router-dom';
import { cursoService } from '../../services/api';
import { useAuth } from '../../context/AuthContext';
import LeccionItem from './LeccionItem';
import ExamenCard from '../Examenes/ExamenCard';

export default function CursoDetalle() {
  const { id } = useParams();
  const { user } = useAuth();
  const [curso, setCurso] = useState(null);
  const [loading, setLoading] = useState(true);
  const [inscrito, setInscrito] = useState(false);
  const [error, setError] = useState(null);

  useEffect(() => {
    loadCurso();
  }, [id]);

  const loadCurso = async () => {
    setLoading(true);
    try {
      const res = await cursoService.getById(id);
      setCurso(res.data.data);
    } catch (err) {
      setError(err.response?.data?.message || 'Error al cargar curso');
    } finally {
      setLoading(false);
    }
  };

  const handleInscribir = async () => {
    try {
      await cursoService.inscribir(id);
      setInscrito(true);
    } catch (err) {
      alert(err.response?.data?.message || 'Error al inscribirse');
    }
  };

  if (loading) return <div className="cargando">Cargando curso...</div>;
  if (error) return <div className="error">{error}</div>;
  if (!curso) return <div className="error">Curso no encontrado</div>;

  return (
    <div className="curso-detalle">
      <div className="curso-header">
        <img src={curso.imagen || '/img/curso-default.jpg'} alt={curso.titulo} />
        <div className="curso-info">
          <h1>{curso.titulo}</h1>
          <p className="curso-instructor">
            Instructor: {curso.instructor_nombre}
          </p>
          <p className="curso-descripcion">{curso.descripcion}</p>
          <div className="curso-meta">
            <span>Nivel: {curso.nivel}</span>
            <span>Duración: {curso.duracion_horas} horas</span>
            <span>Lecciones: {curso.lecciones?.length || 0}</span>
          </div>
          {user && !inscrito && (
            <button onClick={handleInscribir} className="btn btn-primary btn-lg">
              Inscribirme
            </button>
          )}
          {inscrito && <span className="inscrito-badge">✓ Inscrito</span>}
        </div>
      </div>

      <div className="curso-contenido">
        <h2>Lecciones</h2>
        {curso.lecciones?.length > 0 ? (
          <div className="lecciones-lista">
            {curso.lecciones.map((leccion, index) => (
              <LeccionItem
                key={leccion.id}
                leccion={leccion}
                numero={index + 1}
                inscrito={inscrito}
              />
            ))}
          </div>
        ) : (
          <p>No hay lecciones disponibles en este curso.</p>
        )}
      </div>

      {curso.examenes?.length > 0 && (
        <div className="curso-examenes">
          <h2>Exámenes</h2>
          <div className="examenes-lista">
            {curso.examenes.map((examen) => (
              <ExamenCard key={examen.id} examen={examen} inscrito={inscrito} />
            ))}
          </div>
        </div>
      )}
    </div>
  );
}
```
