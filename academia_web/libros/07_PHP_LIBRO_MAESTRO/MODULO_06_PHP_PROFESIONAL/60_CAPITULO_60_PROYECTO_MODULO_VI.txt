```php
namespace App\Config;

class App
{
    public static function init(): void
    {
        // Cargar variables de entorno
        $dotenv = \Dotenv\Dotenv::createImmutable(__DIR__ . '/..');
        $dotenv->load();

        // Configurar zona horaria
        date_default_timezone_set('America/Mexico_City');

        // Configurar errores según entorno
        if ($_ENV['APP_ENV'] === 'development') {
            error_reporting(E_ALL);
            ini_set('display_errors', '1');

            // Whoops para debugging
            $whoops = new \Whoops\Run();
            $whoops->pushHandler(new \Whoops\Handler\PrettyPageHandler());
            $whoops->register();
        } else {
            error_reporting(E_ALL & ~E_DEPRECATED & ~E_NOTICE);
            ini_set('display_errors', '0');
            ini_set('log_errors', '1');
            ini_set('error_log', __DIR__ . '/../storage/logs/php_errors.log');
        }

        // Configurar sesión
        ini_set('session.use_strict_mode', '1');
        ini_set('session.use_cookies', '1');
        ini_set('session.use_only_cookies', '1');
        ini_set('session.cookie_httponly', '1');
        ini_set('session.cookie_samesite', 'Lax');

        if ($_ENV['APP_ENV'] === 'production') {
            ini_set('session.cookie_secure', '1');
        }

        session_start();
    }

    public static function isProduction(): bool
    {
        return ($_ENV['APP_ENV'] ?? 'production') === 'production';
    }

    public static function isDebug(): bool
    {
        return (bool) ($_ENV['APP_DEBUG'] ?? false);
    }
```

```php
<?php
// src/Core/Application.php
namespace App\Core;

use App\Config\App;

class Application
{
    private Router $router;

    public function __construct()
    {
        App::init();
        $this->router = new Router();

        // Cargar rutas web
        require_once __DIR__ . '/../../routes/web.php';

        // Configurar manejador de errores
        $this->setupErrorHandler();
    }

    private function setupErrorHandler(): void
    {
        set_exception_handler(function (\Throwable $e) {
            // Loguear excepción
            $log = new \Monolog\Logger('app');
            $log->pushHandler(
                new \Monolog\Handler\RotatingFileHandler(
                    __DIR__ . '/../../storage/logs/error.log', 30
                )
            );
            $log->error($e->getMessage(), [
                'file'  => $e->getFile(),
                'line'  => $e->getLine(),
                'trace' => $e->getTraceAsString(),
            ]);

            http_response_code(500);
            if (App::isDebug()) {
                throw $e; // Whoops lo capturará
            }
            View::render('errors/500');
        });
    }

    public function run(): void
    {
        $request = new Request();
        $this->router->dispatch($request);
    }
}
```

```php
<?php
// src/Core/Router.php
namespace App\Core;

class Router
{
    private array $routes = [];
    private array $groupMiddleware = [];
    private string $prefix = '';

    public function group(string $prefix, callable $callback, array $middleware = []): void
    {
        $previousPrefix = $this->prefix;
        $previousMiddleware = $this->groupMiddleware;

        $this->prefix .= $prefix;
        $this->groupMiddleware = array_merge($this->groupMiddleware, $middleware);

        $callback($this);

        $this->prefix = $previousPrefix;
        $this->groupMiddleware = $previousMiddleware;
    }

    public function add(string $method, string $pattern, callable|array|string $handler, array $middleware = []): void
    {
        $fullPattern = $this->prefix . $pattern;
        $fullPattern = preg_replace('/\{(\w+)\}/', '(?P<$1>[^/]+)', $fullPattern);
        $fullPattern = '#^' . $fullPattern . '$#';

        $this->routes[] = [
            'method'     => strtoupper($method),
            'pattern'    => $fullPattern,
            'handler'    => $handler,
            'middleware' => array_merge($this->groupMiddleware, $middleware),
        ];
    }

    public function get(string $pattern, callable|array|string $handler, array $middleware = []): void
    {
        $this->add('GET', $pattern, $handler, $middleware);
    }

    public function post(string $pattern, callable|array|string $handler, array $middleware = []): void
    {
        $this->add('POST', $pattern, $handler, $middleware);
    }

    public function put(string $pattern, callable|array|string $handler, array $middleware = []): void
    {
        $this->add('PUT', $pattern, $handler, $middleware);
    }

    public function delete(string $pattern, callable|array|string $handler, array $middleware = []): void
    {
        $this->add('DELETE', $pattern, $handler, $middleware);
    }

    public function dispatch(Request $request): void
    {
        $method = $request->getMethod();
        $uri    = $request->getUri();

        foreach ($this->routes as $route) {
            if ($route['method'] !== $method) {
                continue;
            }

            if (preg_match($route['pattern'], $uri, $matches)) {
                $params = array_filter($matches, 'is_string', ARRAY_FILTER_USE_KEY);

                // Ejecutar middleware
                foreach ($route['middleware'] as $middleware) {
                    if (is_string($middleware)) {
                        $middleware = new $middleware();
                    }
                    $middleware->handle($request);
                }

                // Ejecutar handler
                $handler = $route['handler'];

                if (is_callable($handler)) {
                    $handler($request, ...array_values($params));
                    return;
                }

                if (is_array($handler)) {
                    [$controllerClass, $action] = $handler;
                }

                if (is_string($handler)) {
                    [$controllerClass, $action] = explode('@', $handler);
                }

                $controller = new $controllerClass();
                $controller->$action($request, ...array_values($params));
                return;
            }
        }

        http_response_code(404);
        View::render('errors/404');
    }
}
```

```php
<?php
// src/Core/Controller.php
namespace App\Core;

use App\Core\View;

abstract class Controller
{
    protected function render(string $view, array $data = [], string $layout = 'main'): void
    {
        View::render($view, $data, $layout);
    }

    protected function redirect(string $url): void
    {
        header('Location: ' . $url);
        exit;
    }

    protected function redirectBack(): void
    {
        $referer = $_SERVER['HTTP_REFERER'] ?? '/';
        $this->redirect($referer);
    }

    protected function withErrors(array $errors): void
    {
        $_SESSION['_errors'] = $errors;
    }

    protected function withInput(array $input): void
    {
        $_SESSION['_old_input'] = $input;
    }

    protected function withSuccess(string $message): void
    {
        $_SESSION['_success'] = $message;
    }

    protected function json(mixed $data, int $status = 200): void
    {
        http_response_code($status);
        header('Content-Type: application/json');
        echo json_encode($data, JSON_UNESCAPED_UNICODE);
        exit;
    }

    protected function validate(array $data, array $rules): array
    {
        $validator = new Validator($data, $rules);
        if ($validator->fails()) {
            $this->withErrors($validator->errors());
            $this->withInput($data);
            $this->redirectBack();
        }
        return $validator->validated();
    }
}
```

```php
<?php
// src/Core/Model.php
namespace App\Core;

abstract class Model
{
    protected \PDO $db;
    protected string $table;
    protected string $primaryKey = 'id';

    public function __construct()
    {
        $this->db = Database::connect();
    }

    public function all(string $orderBy = 'id', string $direction = 'DESC'): array
    {
        $stmt = $this->db->query("SELECT * FROM {$this->table} ORDER BY {$orderBy} {$direction}");
        return $stmt->fetchAll();
    }

    public function find(int $id): ?array
    {
        $stmt = $this->db->prepare("SELECT * FROM {$this->table} WHERE {$this->primaryKey} = :id");
        $stmt->execute([':id' => $id]);
        return $stmt->fetch() ?: null;
    }

    public function findBy(string $column, mixed $value): ?array
    {
        $stmt = $this->db->prepare("SELECT * FROM {$this->table} WHERE {$column} = :value LIMIT 1");
        $stmt->execute([':value' => $value]);
        return $stmt->fetch() ?: null;
    }

    public function create(array $data): int
    {
        $columns = implode(', ', array_keys($data));
        $params  = ':' . implode(', :', array_keys($data));

        $stmt = $this->db->prepare(
            "INSERT INTO {$this->table} ({$columns}) VALUES ({$params})"
        );
        $stmt->execute($data);

        return (int) $this->db->lastInsertId();
    }

    public function update(int $id, array $data): bool
    {
        $sets = [];
        foreach ($data as $key => $value) {
            $sets[] = "{$key} = :{$key}";
        }
        $data[$this->primaryKey] = $id;

        $sql = "UPDATE {$this->table} SET " . implode(', ', $sets) . " WHERE {$this->primaryKey} = :{$this->primaryKey}";
        $stmt = $this->db->prepare($sql);
        return $stmt->execute($data);
    }

    public function delete(int $id): bool
    {
        $stmt = $this->db->prepare("DELETE FROM {$this->table} WHERE {$this->primaryKey} = :id");
        return $stmt->execute([':id' => $id]);
    }

    public function paginate(int $page = 1, int $perPage = 15, string $where = '', array $params = []): array
    {
        $offset = ($page - 1) * $perPage;

        $countSql = "SELECT COUNT(*) FROM {$this->table}";
        if ($where) {
            $countSql .= " WHERE {$where}";
        }
        $stmt = $this->db->prepare($countSql);
        $stmt->execute($params);
        $total = (int) $stmt->fetchColumn();

        $dataSql = "SELECT * FROM {$this->table}";
        if ($where) {
            $dataSql .= " WHERE {$where}";
        }
        $dataSql .= " ORDER BY {$this->primaryKey} DESC LIMIT {$perPage} OFFSET {$offset}";

        $stmt = $this->db->prepare($dataSql);
        $stmt->execute($params);
        $data = $stmt->fetchAll();

        return [
            'data'       => $data,
            'total'      => $total,
            'page'       => $page,
            'per_page'   => $perPage,
            'last_page'  => (int) ceil($total / $perPage),
        ];
    }
}
```

```php
// src/Core/Auth.php
namespace App\Core;

class Auth
{
    public static function attempt(string $email, string $password): bool
    {
        $db = Database::connect();
        $stmt = $db->prepare('SELECT * FROM usuarios WHERE email = :email AND activo = 1');
        $stmt->execute([':email' => $email]);
        $user = $stmt->fetch();

        if (!$user || !password_verify($password, $user['password'])) {
            return false;
        }

        // Rehash si es necesario
        if (password_needs_rehash($user['password'], PASSWORD_ARGON2ID)) {
            $newHash = password_hash($password, PASSWORD_ARGON2ID);
            $update = $db->prepare('UPDATE usuarios SET password = :password WHERE id = :id');
            $update->execute([':password' => $newHash, ':id' => $user['id']]);
        }

        session_regenerate_id(true);
        $_SESSION['user_id'] = $user['id'];
        $_SESSION['user_rol'] = $user['rol'];
        $_SESSION['user_nombre'] = $user['nombre'];

        return true;
    }

    public static function check(): bool
    {
        return isset($_SESSION['user_id']);
    }

    public static function user(): ?array
    {
        if (!self::check()) {
            return null;
        }

        $cacheKey = 'user:' . $_SESSION['user_id'];
        $user = Cache::get($cacheKey);

        if (!$user) {
            $db = Database::connect();
            $stmt = $db->prepare('SELECT id, nombre, email, rol, creado_en FROM usuarios WHERE id = :id');
            $stmt->execute([':id' => $_SESSION['user_id']]);
            $user = $stmt->fetch();
            Cache::set($cacheKey, $user, 300);
        }

        return $user ?: null;
    }

    public static function id(): ?int
    {
        return $_SESSION['user_id'] ?? null;
    }

    public static function rol(): ?string
    {
        return $_SESSION['user_rol'] ?? null;
    }

    public static function isAdmin(): bool
    {
        return self::rol() === 'admin';
    }

    public static function logout(): void
    {
        Cache::delete('user:' . ($_SESSION['user_id'] ?? ''));
        $_SESSION = [];
        $params = session_get_cookie_params();
        setcookie(session_name(), '', time() - 3600, $params['path'], $params['domain'], $params['secure'], $params['httponly']);
        session_destroy();
    }
```

```php
            return false;
        }
        return hash_equals($_SESSION['_csrf_token'], $token);
    }

    {
        $filename = preg_replace('/[^\w\-\.]/', '_', $filename);
        $filename = preg_replace('/[\.]+/', '.', $filename);
        $filename = ltrim($filename, '.');
        return substr($filename, 0, 100) ?: 'file_' . bin2hex(random_bytes(8));
    }
```

```php
// src/Models/Articulo.php (método de subida)
public function subirImagen(array $file): ?string
{
    if ($file['error'] !== UPLOAD_ERR_OK) {
        throw new \RuntimeException('Error al subir la imagen');
    }

    $allowedMimes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp'];
    $maxSize = (int) ($_ENV['UPLOAD_MAX_SIZE'] ?? 5242880);

    if ($file['size'] > $maxSize) {
        throw new \RuntimeException('La imagen excede el tamaño máximo de 5MB');
    }

    $finfo = new \finfo(FILEINFO_MIME_TYPE);
    $mime = $finfo->file($file['tmp_name']);

    if (!in_array($mime, $allowedMimes)) {
        throw new \RuntimeException('Tipo de imagen no permitido');
    }

    $extension = match ($mime) {
        'image/jpeg' => 'jpg',
        'image/png'  => 'png',
        'image/gif'  => 'gif',
        'image/webp' => 'webp',
    };

    $filename = bin2hex(random_bytes(16)) . '.' . $extension;
    $uploadPath = __DIR__ . '/../../storage/uploads/' . $filename;

    if (!move_uploaded_file($file['tmp_name'], $uploadPath)) {
        throw new \RuntimeException('Error al guardar la imagen');
    }

    chmod($uploadPath, 0644);

    // Redimensionar si es muy grande
    $this->redimensionarImagen($uploadPath, 1200, 1200);

    return $filename;
}

private function redimensionarImagen(string $path, int $maxW, int $maxH): void
{
    $info = @getimagesize($path);
    if (!$info) return;

    [$w, $h, $type] = $info;
    if ($w <= $maxW && $h <= $maxH) return;

    $ratio = min($maxW / $w, $maxH / $h, 1);
    $nw = (int) ($w * $ratio);
    $nh = (int) ($h * $ratio);

    $src = match ($type) {
        IMAGETYPE_JPEG => @imagecreatefromjpeg($path),
        IMAGETYPE_PNG  => @imagecreatefrompng($path),
        IMAGETYPE_WEBP => @imagecreatefromwebp($path),
        default => null,
    };

    if (!$src) return;

    $dst = imagecreatetruecolor($nw, $nh);
    imagecopyresampled($dst, $src, 0, 0, 0, 0, $nw, $nh, $w, $h);

    match ($type) {
        IMAGETYPE_JPEG => imagejpeg($dst, $path, 85),
        IMAGETYPE_PNG  => imagepng($dst, $path, 8),
        IMAGETYPE_WEBP => imagewebp($dst, $path, 85),
    };

    imagedestroy($src);
    imagedestroy($dst);
}
```

DESAFÍO DEL ARQUITECTO
===============================================================================

🏗️ **CMS Profesional Completo + SaaS**

   Lleva el proyecto al siguiente nivel agregando características
   de Software as a Service (SaaS):

   1. Multi-tenancy: un mismo CMS sirve a múltiples sitios web.
      Cada sitio tiene su propia base de datos o prefijo de tablas.
      Los usuarios pertenecen a un tenant específico.

   2. Panel de facturación: planes de suscripción (gratuito, pro,
      enterprise) con Stripe/PayPal. Límites de almacenamiento
      y artículos según el plan.

   3. Tema personalizable: los usuarios pueden cambiar el tema
      de su sitio. Temas predefinidos + editor CSS.

   4. CDN de imágenes: las imágenes subidas se distribuyen
      automáticamente a un CDN (Cloudinary, Imgix).

   5. Editor WYSIWYG: integra TinyMCE o Quill.js para editar
      artículos con formato enriquecido.

   6. SEO automático: genera sitemap.xml, meta tags OG, schema
      org, y robots.txt automáticamente para cada sitio.

   7. Estadísticas avanzadas: gráficos de visitas por día/semana/
      mes, fuentes de tráfico, artículos populares.

   8. Modo oscuro y modo claro para el panel de administración.

   9. API GraphQL además de REST (usando graphql-php).

   10. Cola de trabajos: las operaciones pesadas (enviar emails,
       generar PDFs, procesar imágenes) se encolan y procesan
       en background.

   Arquitectura propuesta:
   ```
   ├── app/
   │   ├── Tenancy/        ← Gestión de multi-tenant
   │   ├── Billing/        ← Facturación y planes
   │   ├── Seo/            ← Meta tags, sitemaps
   │   └── Queue/          ← Cola de trabajos
   ├── themes/             ← Temas del sitio público
   ├── storage/tenants/    ← Archivos por tenant
   └── config/tenancy.php
   ```

   Este desafío te prepara para desarrollar productos SaaS
   completos, el estándar de la industria SaaS actual.
