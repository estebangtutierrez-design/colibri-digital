EJEMPLO DE CÓDIGO
----------------------------------------

// Raw SQL
var users = await _context.Users
  .FromSqlRaw("SELECT * FROM Users WHERE Name LIKE {0}", namePattern)
  .ToListAsync();

// Compiled query
private static readonly Func<AppDbContext, string, Task<List<User>>> GetUsersByName =
  EF.CompileAsyncQuery((AppDbContext ctx, string name) =>
    ctx.Users.Where(u => u.Name.Contains(name)).ToList());

// Interceptor
public class AuditInterceptor : SaveChangesInterceptor {
  public override InterceptionResult<int> SavingChanges(
    DbContextEventData data, InterceptionResult<int> result) {
    // audit logic
    return base.SavingChanges(data, result);
  }
}
