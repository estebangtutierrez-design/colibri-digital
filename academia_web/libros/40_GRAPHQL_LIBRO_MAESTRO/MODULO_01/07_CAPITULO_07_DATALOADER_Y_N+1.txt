DataLoader: batching + caching. Previene N+1 queries.

Batch load function. Cache por request.

dataloader npm package.

EJEMPLO DE CÓDIGO
----------------------------------------

import DataLoader from "dataloader"

const createUserLoader = (db) => new DataLoader(async (ids) => {
  const users = await db.user.findMany({ where: { id: { in: ids } } })
  return ids.map(id => users.find(u => u.id === id))
})

// En resolvers
User: {
  posts: async (parent, _, { postLoader }) => postLoader.load(parent.id)
}
