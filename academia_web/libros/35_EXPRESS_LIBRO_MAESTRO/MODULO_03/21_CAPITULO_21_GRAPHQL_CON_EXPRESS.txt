EJEMPLO DE CÓDIGO
----------------------------------------

const { graphqlHTTP } = require("express-graphql")
const { buildSchema } = require("graphql")

const schema = buildSchema(`
  type User { id: ID! name: String! }
  type Query { users: [User] }
`)

app.use("/graphql", graphqlHTTP({ schema, graphiql: true }))
