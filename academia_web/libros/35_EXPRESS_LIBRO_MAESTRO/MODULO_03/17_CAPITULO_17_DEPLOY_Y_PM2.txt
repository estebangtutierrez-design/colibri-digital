PM2: process manager. cluster mode. log management.

Docker: Dockerfile multi-stage. Nginx reverse proxy.

Environment: dotenv, .env.production.

EJEMPLO DE CÓDIGO
----------------------------------------

pm2 start app.js -i max
pm2 save
pm2 startup

# Dockerfile multi-stage
FROM node:18-alpine AS build
WORKDIR /app
COPY package*.json ./
RUN npm ci --production
COPY . .
EXPOSE 3000
CMD ["node", "app.js"]
