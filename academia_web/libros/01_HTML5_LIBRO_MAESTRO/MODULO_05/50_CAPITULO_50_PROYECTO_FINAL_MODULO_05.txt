// TaskFlow - Service Worker

const CACHE_NAME = 'taskflow-v2';
const URLS_CACHE = [
    '/',
    '/index.html',
    '/css/style.css',
    '/js/app.js',
    '/js/db.js',
    '/manifest.json',
    '/images/icon-72.png',
    '/images/icon-96.png',
    '/images/icon-128.png',
    '/images/icon-144.png',
    '/images/icon-152.png',
    '/images/icon-192.png',
    '/images/icon-384.png',
    '/images/icon-512.png'
];

// ============================================================ */
// INSTALACIÓN                                                  */
// ============================================================ */

self.addEventListener('install', function(event) {
    event.waitUntil(
        caches.open(CACHE_NAME).then(function(cache) {
            console.log('📦 Cacheando recursos de TaskFlow...');
            return cache.addAll(URLS_CACHE);
        })
    );
});

// ============================================================ */
// ACTIVACIÓN                                                   */
// ============================================================ */

self.addEventListener('activate', function(event) {
    event.waitUntil(
        caches.keys().then(function(cacheNames) {
            return Promise.all(
                cacheNames.map(function(cacheName) {
                    if (cacheName !== CACHE_NAME) {
                        console.log('🗑️ Eliminando cache antiguo:', cacheName);
                        return caches.delete(cacheName);
                    }
                })
            );
        })
    );
});

// ============================================================ */
// FETCH (INTERCEPTAR PETICIONES)                               */
// ============================================================ */

self.addEventListener('fetch', function(event) {
    event.respondWith(
        caches.match(event.request).then(function(response) {
            return response || fetch(event.request);
        })
    );
});

const DB_NAME = 'TaskFlowDB';
const DB_VERSION = 1;
const STORE_NAME = 'tareas';

let db = null;

// ============================================================ */
// ABRIR BASE DE DATOS                                          */
// ============================================================ */

function abrirBaseDeDatos() {
    return new Promise((resolve, reject) => {
        const request = indexedDB.open(DB_NAME, DB_VERSION);

        request.onupgradeneeded = function(event) {
            const database = event.target.result;
            if (!database.objectStoreNames.contains(STORE_NAME)) {
                const store = database.createObjectStore(STORE_NAME, {
                    keyPath: 'id',
                    autoIncrement: true
                });
                store.createIndex('por_estado', 'estado', { unique: false });
                store.createIndex('por_orden', 'orden', { unique: false });
                console.log('✅ Almacén de tareas creado');
            }
        };

        request.onsuccess = function(event) {
            db = event.target.result;
            console.log('✅ Base de datos conectada');
            resolve(db);
        };

        request.onerror = function(event) {
            console.error('❌ Error al abrir DB:', event.target.error);
            reject(event.target.error);
        };
    });
}

// ============================================================ */
// OPERACIONES CRUD                                              */
// ============================================================ */

function obtenerTodasLasTareas() {
    return new Promise((resolve, reject) => {
        if (!db) {
            reject(new Error('Base de datos no conectada'));
            return;
        }

        const transaction = db.transaction([STORE_NAME], 'readonly');
        const store = transaction.objectStore(STORE_NAME);
        const request = store.getAll();

        request.onsuccess = function() {
            resolve(request.result || []);
        };

        request.onerror = function() {
            reject(request.error);
        };
    });
}

function agregarTarea(tarea) {
    return new Promise((resolve, reject) => {
        if (!db) {
            reject(new Error('Base de datos no conectada'));
            return;
        }

        // Asignar orden automático
        tarea.orden = Date.now();

        const transaction = db.transaction([STORE_NAME], 'readwrite');
        const store = transaction.objectStore(STORE_NAME);
        const request = store.add(tarea);

        request.onsuccess = function() {
            resolve(request.result);
        };

        request.onerror = function() {
            reject(request.error);
        };
    });
}

function actualizarTarea(tarea) {
    return new Promise((resolve, reject) => {
        if (!db) {
            reject(new Error('Base de datos no conectada'));
            return;
        }

        const transaction = db.transaction([STORE_NAME], 'readwrite');
        const store = transaction.objectStore(STORE_NAME);
        const request = store.put(tarea);

        request.onsuccess = function() {
            resolve(request.result);
        };

        request.onerror = function() {
            reject(request.error);
        };
    });
}

function eliminarTarea(id) {
    return new Promise((resolve, reject) => {
        if (!db) {
            reject(new Error('Base de datos no conectada'));
            return;
        }

        const transaction = db.transaction([STORE_NAME], 'readwrite');
        const store = transaction.objectStore(STORE_NAME);
        const request = store.delete(id);

        request.onsuccess = function() {
            resolve();
        };

        request.onerror = function() {
            reject(request.error);
        };
    });
}

function actualizarOrdenTareas(tareas) {
    return new Promise((resolve, reject) => {
        if (!db) {
            reject(new Error('Base de datos no conectada'));
            return;
        }

        const transaction = db.transaction([STORE_NAME], 'readwrite');
        const store = transaction.objectStore(STORE_NAME);

        const promises = tareas.map(tarea => {
            return new Promise((res, rej) => {
                const request = store.put(tarea);
                request.onsuccess = () => res();
                request.onerror = () => rej(request.error);
            });
        });

        Promise.all(promises)
            .then(() => resolve())
            .catch(error => reject(error));
    });
}

function renderizarTareas() {
    const tareasFiltradas = filtrarTareas(tareas, filtroActual);

    let html = '';

        const estadoTexto = {
            pendiente: '📌 Pendiente',
            progreso: '⚡ En progreso',
            completada: '✅ Completada'

    configurarDragAndDrop();

let elementoArrastrado = null;

function configurarDragAndDrop() {
    const tarjetas = document.querySelectorAll('.tarjeta');

    tarjetas.forEach(tarjeta => {
        tarjeta.addEventListener('dragstart', function(event) {
            elementoArrastrado = this;
            this.classList.add('dragging');
            event.dataTransfer.effectAllowed = 'move';
            event.dataTransfer.setData('text/plain', this.dataset.id);
        });

        tarjeta.addEventListener('dragend', function(event) {
            this.classList.remove('dragging');
        });

        tarjeta.addEventListener('dragover', function(event) {
            event.preventDefault();
            event.dataTransfer.dropEffect = 'move';
            this.style.borderColor = '#58a6ff';
        });

        tarjeta.addEventListener('dragleave', function(event) {
            this.style.borderColor = '#30363d';
        });

        tarjeta.addEventListener('drop', async function(event) {
            event.preventDefault();
            this.style.borderColor = '#30363d';

            if (!elementoArrastrado) return;
            if (elementoArrastrado === this) return;

            // Obtener IDs
            const idOrigen = parseInt(elementoArrastrado.dataset.id);
            const idDestino = parseInt(this.dataset.id);

            if (isNaN(idOrigen) || isNaN(idDestino)) return;

            // Encontrar índices en la lista completa (no filtrada)
            const indexOrigen = tareas.findIndex(t => t.id === idOrigen);
            const indexDestino = tareas.findIndex(t => t.id === idDestino);

            if (indexOrigen === -1 || indexDestino === -1) return;

            // Mover en el array
            const [tareaMovida] = tareas.splice(indexOrigen, 1);
            tareas.splice(indexDestino, 0, tareaMovida);

            // Actualizar órdenes
            tareas.forEach((t, i) => t.orden = i);

            try {
                await actualizarOrdenTareas(tareas);
                renderizarTareas();
                mostrarNotificacion('🔄 Orden actualizado');
            } catch (error) {
                console.error('Error al actualizar orden:', error);
                mostrarNotificacion('❌ Error al actualizar orden');
                await cargarTareas(); // Recargar para restaurar orden
            }
        });
    });
}

function iniciarWorker() {
    try {
        // Crear Worker con código embebido
        const workerCode = `
            self.onmessage = function(event) {
                const tareas = event.data;
                const total = tareas.length;
                const pendiente = tareas.filter(t => t.estado === 'pendiente' || !t.estado).length;
                const progreso = tareas.filter(t => t.estado === 'progreso').length;
                const completada = tareas.filter(t => t.estado === 'completada').length;

                self.postMessage({
                    total: total,
                    pendiente: pendiente,
                    progreso: progreso,
                    completada: completada
                });
            };
        `;

        const blob = new Blob([workerCode], { type: 'application/javascript' });
        const url = URL.createObjectURL(blob);
        worker = new Worker(url);

        worker.onmessage = function(event) {
            const stats = event.data;
            DOM.statTotal.textContent = stats.total;
            DOM.statPendiente.textContent = stats.pendiente;
            DOM.statProgreso.textContent = stats.progreso;
            DOM.statCompletada.textContent = stats.completada;
        };
    } catch (error) {
        console.error('Error al iniciar Worker:', error);
        // Fallback sin Worker
    }
}

function actualizarEstadisticas() {
    if (worker) {
        worker.postMessage(tareas);
    } else {
        // Fallback (si no hay Worker)
        const total = tareas.length;
        const pendiente = tareas.filter(t => t.estado === 'pendiente' || !t.estado).length;
        const progreso = tareas.filter(t => t.estado === 'progreso').length;
        const completada = tareas.filter(t => t.estado === 'completada').length;
        DOM.statTotal.textContent = total;
        DOM.statPendiente.textContent = pendiente;
        DOM.statProgreso.textContent = progreso;
        DOM.statCompletada.textContent = completada;
    }
}

// NOTIFICACIONES (API de Notificaciones)                      */
// ============================================================ */

function enviarNotificacionNavegador(titulo, mensaje) {
    if (!('Notification' in window) || Notification.permission !== 'granted') return;

    try {
        const notificacion = new Notification(titulo, {
            body: mensaje,
            icon: 'images/icon-192.png',
            tag: 'taskflow-' + Date.now()
        });

        notificacion.onclick = function() {
            window.focus();
        };
    } catch (error) {
        console.error('Error al enviar notificación:', error);
    }
}

// Solicitar permiso de notificaciones (si no se ha hecho)
if ('Notification' in window && Notification.permission === 'default') {
    setTimeout(() => {
        Notification.requestPermission().then(permiso => {
            if (permiso === 'granted') {
                console.log('✅ Permiso de notificaciones concedido');
            }
        });
    }, 3000);
}

// NOTIFICACIÓN EN VIVO (aria-live)                             */
// ============================================================ */

let timeoutNotificacion = null;

function mostrarNotificacion(mensaje) {
    DOM.notificacionViva.textContent = mensaje;
    DOM.notificacionViva.classList.add('visible');

    clearTimeout(timeoutNotificacion);
    timeoutNotificacion = setTimeout(() => {
        DOM.notificacionViva.classList.remove('visible');
    }, 3000);
}

// FETCH API (FRASE MOTIVACIONAL)                               */
// ============================================================ */

async function obtenerFrase() {
    try {
        const response = await fetch('https://api.quotable.io/random');
        if (!response.ok) throw new Error('Error al obtener frase');

        const data = await response.json();
        DOM.fraseTexto.textContent = `💡 "${data.content}" — ${data.author}`;
    } catch (error) {
        console.error('Error al obtener frase:', error);
        DOM.fraseTexto.textContent = '💡 "El mejor momento para empezar fue ayer. El segundo mejor es ahora." — Anónimo';
    }
}

DOM.btnFrase.addEventListener('click', obtenerFrase);

function actualizarEstadoOffline() {
    const online = navigator.onLine;
    DOM.estadoOffline.textContent = online ? '🟢' : '🔴';
    DOM.estadoOffline.title = online ? 'Conectado' : 'Sin conexión (modo offline)';
}

window.addEventListener('online', () => {
    actualizarEstadoOffline();
    mostrarNotificacion('🟢 Conexión restablecida');
});

window.addEventListener('offline', () => {
    actualizarEstadoOffline();
    mostrarNotificacion('🔴 Sin conexión (modo offline)');
});

// PWA - INSTALACIÓN PERSONALIZADA                              */
// ============================================================ */

let deferredPrompt;

window.addEventListener('beforeinstallprompt', function(event) {
    event.preventDefault();
    deferredPrompt = event;
    DOM.btnInstalar.style.display = 'inline-block';
});

DOM.btnInstalar.addEventListener('click', function() {
    if (deferredPrompt) {
        deferredPrompt.prompt();
        deferredPrompt.userChoice.then(function(result) {
            if (result.outcome === 'accepted') {
                mostrarNotificacion('📲 TaskFlow instalada correctamente');
            } else {
                mostrarNotificacion('❌ Instalación cancelada');
            }
            deferredPrompt = null;
            DOM.btnInstalar.style.display = 'none';
        });
    }
});

window.addEventListener('appinstalled', function() {
    mostrarNotificacion('📲 ¡TaskFlow instalada!');
});
