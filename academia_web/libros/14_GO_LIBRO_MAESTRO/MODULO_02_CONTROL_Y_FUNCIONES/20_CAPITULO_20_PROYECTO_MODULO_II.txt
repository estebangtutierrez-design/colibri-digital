================================================================================
  20.1 VISIÓN GENERAL DEL PROYECTO
================================================================================

  La "Librería de Utilidades Matemáticas" (mathutil) es un proyecto
  Go que proporciona funciones matemáticas, estadísticas, geométricas
  y de conversión, todo organizado en paquetes bien definidos.

  Funcionalidades:

  MATEMÁTICA:
  - Suma, resta, multiplicación, división (variádicas).
  - Factorial, Fibonacci, MCD, MCM.
  - Primo, par/impar, dígitos.

  ESTADÍSTICA:
  - Promedio, mediana, moda, varianza, desviación estándar.
  - Mínimo, máximo, rango.

  GEOMETRÍA:
  - Área y perímetro de: cuadrado, rectángulo, círculo, triángulo.
  - Volumen de: cubo, esfera, cilindro.

  CONVERSIONES:
  - Temperatura (Celsius, Fahrenheit, Kelvin).
  - Longitud (metros, pies, pulgadas).
  - Peso (kilogramos, libras, onzas).

  UTILS:
  - Formateo de números, redondeo, clamping.
  - Generación de secuencias.

  La CLI permitirá usar estas funciones desde línea de comandos.

  ```go
  package matematica

  import "fmt"

  // Suma retorna la suma de todos los números proporcionados.
  func Suma(numeros ...float64) float64 {
      total := 0.0
      for _, n := range numeros {
          total += n
      }
      return total
  }

  // Resta retorna la resta del primero menos todos los demás.
  func Resta(primero float64, resto ...float64) float64 {
      resultado := primero
      for _, n := range resto {
          resultado -= n
      }
      return resultado
  }

  // Multiplica retorna el producto de todos los números.
  func Multiplica(numeros ...float64) float64 {
      if len(numeros) == 0 {
          return 0
      }
      total := 1.0
      for _, n := range numeros {
          total *= n
      }
      return total
  }

  // Divide retorna la división del primero entre los demás.
  // Retorna error si algún divisor es cero.
  func Divide(primero float64, resto ...float64) (float64, error) {
      resultado := primero
      for _, n := range resto {
          if n == 0 {
              return 0, fmt.Errorf("división entre cero")
          }
          resultado /= n
      }
      return resultado, nil
  }

  // Potencia calcula base^exponente (exponente entero).
  func Potencia(base float64, exponente int) float64 {
      if exponente == 0 {
          return 1
      }
      resultado := 1.0
      for i := 0; i < abs(exponente); i++ {
          resultado *= base
      }
      if exponente < 0 {
          return 1 / resultado
      }
      return resultado
  }

  func abs(x int) int {
      if x < 0 {
          return -x
      }
      return x
  }
  ```

  ```go
  package matematica

  import "errors"

  // Factorial calcula n! (n factorial).
  // Retorna error si n < 0.
  func Factorial(n int) (int, error) {
      if n < 0 {
          return 0, errors.New("factorial de número negativo")
      }
      if n <= 1 {
          return 1, nil
      }
      resultado := 1
      for i := 2; i <= n; i++ {
          resultado *= i
      }
      return resultado, nil
  }

  // Fibonacci retorna el n-ésimo número de Fibonacci.
  func Fibonacci(n int) int {
      if n <= 0 {
          return 0
      }
      if n == 1 {
          return 1
      }
      a, b := 0, 1
      for i := 2; i <= n; i++ {
          a, b = b, a+b
      }
      return b
  }

  // MCD calcula el Máximo Común Divisor (Euclides).
  func MCD(a, b int) int {
      for b != 0 {
          a, b = b, a%b
      }
      return a
  }

  // MCM calcula el Mínimo Común Múltiplo.
  func MCM(a, b int) int {
      if a == 0 || b == 0 {
          return 0
      }
      return a / MCD(a, b) * b
  }

  // EsPrimo determina si un número es primo.
  func EsPrimo(n int) bool {
      if n < 2 {
          return false
      }
      for i := 2; i*i <= n; i++ {
          if n%i == 0 {
              return false
          }
      }
      return true
  }

  // EsPar retorna true si el número es par.
  func EsPar(n int) bool {
      return n%2 == 0
  }

  // Digitos retorna la cantidad de dígitos de un número.
  func Digitos(n int) int {
      if n == 0 {
          return 1
      }
      if n < 0 {
          n = -n
      }
      count := 0
      for n > 0 {
          n /= 10
          count++
      }
      return count
  }

  // SumaDigitos suma todos los dígitos de un número.
  func SumaDigitos(n int) int {
      if n < 0 {
          n = -n
      }
      suma := 0
      for n > 0 {
          suma += n % 10
          n /= 10
      }
      return suma
  }
  ```

  ```go
  package estadistica

  import (
      "errors"
      "sort"
  )

  // Promedio calcula la media aritmética.
  func Promedio(numeros ...float64) (float64, error) {
      if len(numeros) == 0 {
          return 0, errors.New("se requiere al menos un número")
      }
      suma := 0.0
      for _, n := range numeros {
          suma += n
      }
      return suma / float64(len(numeros)), nil
  }

  // Mediana calcula el valor central de un conjunto ordenado.
  func Mediana(numeros ...float64) (float64, error) {
      if len(numeros) == 0 {
          return 0, errors.New("se requiere al menos un número")
      }

      ordenados := make([]float64, len(numeros))
      copy(ordenados, numeros)
      sort.Float64s(ordenados)

      n := len(ordenados)
      if n%2 == 0 {
          return (ordenados[n/2-1] + ordenados[n/2]) / 2, nil
      }
      return ordenados[n/2], nil
  }

  // Moda encuentra el valor que más se repite.
  func Moda(numeros ...float64) ([]float64, error) {
      if len(numeros) == 0 {
          return nil, errors.New("se requiere al menos un número")
      }

      frecuencias := make(map[float64]int)
      maxFreq := 0

      for _, n := range numeros {
          frecuencias[n]++
          if frecuencias[n] > maxFreq {
              maxFreq = frecuencias[n]
          }
      }

      if maxFreq == 1 {
          return []float64{}, nil // Sin moda
      }

      modas := []float64{}
      for n, freq := range frecuencias {
          if freq == maxFreq {
              modas = append(modas, n)
          }
      }

      sort.Float64s(modas)
      return modas, nil
  }
  ```

  ```go
  package estadistica

  import "math"

  // Minimo retorna el valor mínimo.
  func Minimo(numeros ...float64) (float64, error) {
      if len(numeros) == 0 {
          return 0, errNoDatos()
      }
      min := numeros[0]
      for _, n := range numeros[1:] {
          if n < min {
              min = n
          }
      }
      return min, nil
  }

  // Maximo retorna el valor máximo.
  func Maximo(numeros ...float64) (float64, error) {
      if len(numeros) == 0 {
          return 0, errNoDatos()
      }
      max := numeros[0]
      for _, n := range numeros[1:] {
          if n > max {
              max = n
          }
      }
      return max, nil
  }

  // Rango calcula la diferencia entre el máximo y el mínimo.
  func Rango(numeros ...float64) (float64, error) {
      if len(numeros) == 0 {
          return 0, errNoDatos()
      }
      min, _ := Minimo(numeros...)
      max, _ := Maximo(numeros...)
      return max - min, nil
  }

  // Varianza calcula la varianza muestral.
  func Varianza(numeros ...float64) (float64, error) {
      if len(numeros) < 2 {
          return 0, errMinimoDatos(2)
      }
      prom, _ := Promedio(numeros...)
      suma := 0.0
      for _, n := range numeros {
          suma += (n - prom) * (n - prom)
      }
      return suma / float64(len(numeros)-1), nil
  }

  // DesviacionEstandar calcula la desviación estándar muestral.
  func DesviacionEstandar(numeros ...float64) (float64, error) {
      varianza, err := Varianza(numeros...)
      if err != nil {
          return 0, err
      }
      return math.Sqrt(varianza), nil
  }

  func errNoDatos() error {
      return errMinimoDatos(1)
  }

  func errMinimoDatos(n int) error {
      return &ErrorDatosInsuficientes{Minimo: n}
  }

  type ErrorDatosInsuficientes struct {
      Minimo int
  }

  func (e *ErrorDatosInsuficientes) Error() string {
      return fmt.Sprintf("se requieren al menos %d datos", e.Minimo)
  }
  ```

  ```go
  package geometria

  import "math"

  type Circulo struct {
      Radio float64
  }

  func (c Circulo) Area() float64 {
      return math.Pi * c.Radio * c.Radio
  }

  func (c Circulo) Perimetro() float64 {
      return 2 * math.Pi * c.Radio
  }

  type Rectangulo struct {
      Ancho, Alto float64
  }

  func (r Rectangulo) Area() float64 {
      return r.Ancho * r.Alto
  }

  func (r Rectangulo) Perimetro() float64 {
      return 2 * (r.Ancho + r.Alto)
  }

  type Cuadrado struct {
      Lado float64
  }

  func (c Cuadrado) Area() float64 {
      return c.Lado * c.Lado
  }

  func (c Cuadrado) Perimetro() float64 {
      return 4 * c.Lado
  }

  type Triangulo struct {
      Base, Altura float64
      LadoA, LadoB, LadoC float64
  }

  func (t Triangulo) Area() float64 {
      return (t.Base * t.Altura) / 2
  }

  func (t Triangulo) Perimetro() float64 {
      return t.LadoA + t.LadoB + t.LadoC
  }

  type Figura interface {
      Area() float64
      Perimetro() float64
  }

  func Calcular(figura Figura) (area, perimetro float64) {
      area = figura.Area()
      perimetro = figura.Perimetro()
      return
  }
  ```

  ```go
  package utils

  import "math"

  // Redondear redondea un float64 a n decimales.
  func Redondear(valor float64, decimales int) float64 {
      factor := math.Pow(10, float64(decimales))
      return math.Round(valor*factor) / factor
  }

  // Clamp restringe un valor entre un mínimo y un máximo.
  func Clamp(valor, min, max float64) float64 {
      switch {
      case valor < min:
          return min
      case valor > max:
          return max
      default:
          return valor
      }
  }

  // EsCeroAbsoluto verifica si un valor es cercano a cero.
  func EsCeroAbsoluto(valor, tolerancia float64) bool {
      return math.Abs(valor) <= tolerancia
  }

  // Porcentaje calcula qué porcentaje es parte del total.
  func Porcentaje(parte, total float64) (float64, error) {
      if total == 0 {
          return 0, fmt.Errorf("el total no puede ser cero")
      }
      return (parte / total) * 100, nil
  }

  // Secuencia genera una secuencia de inicio a fin con paso dado.
  func Secuencia(inicio, fin float64, paso float64) []float64 {
      if paso == 0 {
          paso = 1
      }
      resultado := []float64{}
      if paso > 0 {
          for i := inicio; i <= fin; i += paso {
              resultado = append(resultado, i)
          }
      } else {
          for i := inicio; i >= fin; i += paso {
              resultado = append(resultado, i)
          }
      }
      return resultado
  }

  // SumaRango suma todos los números de inicio a fin.
  func SumaRango(inicio, fin int) int {
      suma := 0
      for i := inicio; i <= fin; i++ {
          suma += i
      }
      return suma
  }
  ```

  ```go
  package utils

  import "fmt"

  // FormatearNumero formatea un número con separadores de miles.
  func FormatearNumero(n float64, decimales int) string {
      redondeado := Redondear(n, decimales)
      formato := fmt.Sprintf("%%.%df", decimales)
      return fmt.Sprintf(formato, redondeado)
  }

  // FormatearMoneda formatea un número como moneda.
  func FormatearMoneda(n float64, simbolo string) string {
      return fmt.Sprintf("%s%s", simbolo, FormatearNumero(n, 2))
  }

  // FormatearPorcentaje formatea un número como porcentaje.
  func FormatearPorcentaje(n float64) string {
      return fmt.Sprintf("%.2f%%", n)
  }

  // Tabla genera una tabla formateada de valores.
  func Tabla(encabezados []string, filas [][]string) string {
      if len(filas) == 0 {
          return ""
      }

      // Calcular anchos
      anchos := make([]int, len(encabezados))
      for i, h := range encabezados {
          anchos[i] = len(h)
      }
      for _, fila := range filas {
          for i, celda := range fila {
              if len(celda) > anchos[i] {
                  anchos[i] = len(celda)
              }
          }
      }

      resultado := ""

      // Encabezados
      for i, h := range encabezados {
          resultado += fmt.Sprintf(" %-*s ", anchos[i], h)
      }
      resultado += "\n"

      // Separador
      for _, ancho := range anchos {
          resultado += fmt.Sprintf(" %s ", cadenaRepetir("-", ancho))
      }
      resultado += "\n"

      // Filas
      for _, fila := range filas {
          for i, celda := range fila {
              resultado += fmt.Sprintf(" %-*s ", anchos[i], celda)
          }
          resultado += "\n"
      }

      return resultado
  }

  func cadenaRepetir(s string, n int) string {
      resultado := ""
      for i := 0; i < n; i++ {
          resultado += s
      }
      return resultado
  }
  ```

  ```go
  func cmdMatematica() {
      if len(os.Args) < 3 {
          fmt.Println("Uso: mathutil matematica <operacion> [números...]")
          fmt.Println("Operaciones: suma, resta, multi, divide, factorial, fibonacci, primo")
          os.Exit(1)
      }

      op := os.Args[2]
      numeros := parseFloats(os.Args[3:])
      enteros := parseInts(os.Args[3:])

      switch op {
      case "suma":
          fmt.Println("Resultado:", matematica.Suma(numeros...))
      case "resta":
          if len(numeros) < 1 {
              fmt.Println("Se requiere al menos un número")
              return
          }
          fmt.Println("Resultado:", matematica.Resta(numeros[0], numeros[1:]...))
      case "multi":
          fmt.Println("Resultado:", matematica.Multiplica(numeros...))
      case "divide":
          if len(numeros) < 1 {
              fmt.Println("Se requiere al menos un número")
              return
          }
          res, err := matematica.Divide(numeros[0], numeros[1:]...)
          if err != nil {
              fmt.Println("Error:", err)
              return
          }
          fmt.Println("Resultado:", res)
      case "factorial":
          for _, n := range enteros {
              res, err := matematica.Factorial(n)
              if err != nil {
                  fmt.Printf("factorial(%d): error: %v\n", n, err)
              } else {
                  fmt.Printf("factorial(%d) = %d\n", n, res)
              }
          }
      case "fibonacci":
          for _, n := range enteros {
              fmt.Printf("fibonacci(%d) = %d\n", n, matematica.Fibonacci(n))
          }
      case "primo":
          for _, n := range enteros {
              fmt.Printf("%d es primo: %t\n", n, matematica.EsPrimo(n))
          }
      default:
          fmt.Printf("Operación desconocida: %s\n", op)
      }
  }

  func cmdEstadistica() {
      if len(os.Args) < 3 {
          fmt.Println("Uso: mathutil estadistica <operacion> [números...]")
          fmt.Println("Operaciones: promedio, mediana, moda, min, max, varianza, desviacion")
          os.Exit(1)
      }

      case "mediana":
          result, err = estadistica.Mediana(numeros...)
      case "moda":
          result, err = estadistica.Moda(numeros...)
      case "min":

      case "max":

  func cmdGeometria() {
      if len(os.Args) < 3 {
          fmt.Println("Uso: mathutil geometria <figura> [argumentos]")
          fmt.Println("Figuras: circulo <radio>, cuadrado <lado>, rectangulo <ancho> <alto>")
          fmt.Println("         triangulo <base> <altura>, esfera <radio>, cubo <lado>")
          os.Exit(1)
      }

      figura := os.Args[2]
      nums := parseFloats(os.Args[3:])

      switch figura {
      case "circulo":
          if len(nums) < 1 { fmt.Println("Se requiere radio"); return }
          c := geometria.Circulo{Radio: nums[0]}
          area, perim := geometria.Calcular(c)
          fmt.Printf("Área: %.4f, Perímetro: %.4f\n", area, perim)
      case "cuadrado":
          if len(nums) < 1 { fmt.Println("Se requiere lado"); return }
          c := geometria.Cuadrado{Lado: nums[0]}
          area, perim := geometria.Calcular(c)
          fmt.Printf("Área: %.4f, Perímetro: %.4f\n", area, perim)
      case "rectangulo":
          if len(nums) < 2 { fmt.Println("Se requiere ancho y alto"); return }
          r := geometria.Rectangulo{Ancho: nums[0], Alto: nums[1]}
          area, perim := geometria.Calcular(r)
          fmt.Printf("Área: %.4f, Perímetro: %.4f\n", area, perim)
      case "triangulo":
          if len(nums) < 2 { fmt.Println("Se requiere base y altura"); return }
          t := geometria.Triangulo{Base: nums[0], Altura: nums[1]}
          area := t.Area()
          fmt.Printf("Área: %.4f\n", area)
      case "esfera":
          if len(nums) < 1 { fmt.Println("Se requiere radio"); return }
          e := geometria.Esfera{Radio: nums[0]}
          fmt.Printf("Volumen: %.4f\n", geometria.Volumen(e))
      case "cubo":
          if len(nums) < 1 { fmt.Println("Se requiere lado"); return }
          c := geometria.Cubo{Lado: nums[0]}
          fmt.Printf("Volumen: %.4f\n", geometria.Volumen(c))
      default:
          fmt.Printf("Figura desconocida: %s\n", figura)
      }
  }

  func cmdConversiones() {
      if len(os.Args) < 4 {
          fmt.Println("Uso: mathutil conversiones <tipo> <valor>")
          fmt.Println("Tipos: c-f, f-c, c-k, k-c, f-k, k-f")
          fmt.Println("       m-ft, ft-m, m-in, in-m, kg-lb, lb-kg")
          os.Exit(1)
      }

      tipo := os.Args[2]
      valor := parseFloats(os.Args[3:4])[0]

      var resultado float64
      switch tipo {
      case "c-f":
          resultado = conversiones.CelsiusAFahrenheit(valor)
      case "f-c":
          resultado = conversiones.FahrenheitACelsius(valor)
      case "c-k":
          resultado = conversiones.CelsiusAKelvin(valor)
      case "k-c":
          resultado = conversiones.KelvinACelsius(valor)
      case "m-ft":
          resultado = conversiones.MetrosAPies(valor)
      case "ft-m":
          resultado = conversiones.PiesAMetros(valor)
      case "m-in":
          resultado = conversiones.MetrosAPulgadas(valor)
      case "in-m":
          resultado = conversiones.PulgadasAMetros(valor)
      case "kg-lb":
          resultado = conversiones.KilogramosALibras(valor)
      case "lb-kg":
          resultado = conversiones.LibrasAKilogramos(valor)
      default:
          fmt.Printf("Tipo de conversión desconocido: %s\n", tipo)
          return
      }

      fmt.Printf("Resultado: %.4f\n", resultado)
  }

  func cmdUtils() {
      if len(os.Args) < 3 {
          fmt.Println("Uso: mathutil utils <operacion> [argumentos]")
          fmt.Println("Operaciones: redondear <n> <decimales>, clamp <n> <min> <max>")
          fmt.Println("             secuencia <inicio> <fin> <paso>, sumarango <inicio> <fin>")
          os.Exit(1)
      }

      op := os.Args[2]
      nums := parseFloats(os.Args[3:])
      ints := parseInts(os.Args[3:])

      switch op {
      case "redondear":
          if len(nums) < 2 { fmt.Println("Se requiere número y decimales"); return }
          fmt.Printf("Resultado: %v\n", utils.Redondear(nums[0], int(nums[1])))
      case "clamp":
          if len(nums) < 3 { fmt.Println("Se requiere valor, min y max"); return }
          fmt.Printf("Resultado: %v\n", utils.Clamp(nums[0], nums[1], nums[2]))
      case "secuencia":
          if len(nums) < 2 { fmt.Println("Se requiere inicio y fin"); return }
          paso := 1.0
          if len(nums) >= 3 { paso = nums[2] }
          fmt.Printf("Resultado: %v\n", utils.Secuencia(nums[0], nums[1], paso))
      case "sumarango":
          if len(ints) < 2 { fmt.Println("Se requiere inicio y fin"); return }
          fmt.Printf("Resultado: %d\n", utils.SumaRango(ints[0], ints[1]))
      default:
          fmt.Printf("Operación desconocida: %s\n", op)
      }
  }

  func parseFloats(args []string) []float64 {
      nums := make([]float64, 0, len(args))
      for _, arg := range args {
          var n float64
          fmt.Sscanf(arg, "%f", &n)
          nums = append(nums, n)
      }
      return nums
  }

  func parseInts(args []string) []int {
      nums := make([]int, 0, len(args))
      for _, arg := range args {
          var n int
          fmt.Sscanf(arg, "%d", &n)
          nums = append(nums, n)
      }
      return nums
  }
  ```
