  1. Especificación del Proyecto
  ──────────────────────────────

  Nombre: Agenda de Contactos GoBook
  Descripción: Aplicación de consola que permite gestionar una agenda de
  contactos personales con las siguientes capacidades:

  • Agregar contacto (nombre, teléfono, email, dirección, grupo)
  • Listar contactos (tabla formateada)
  • Buscar contactos por nombre o grupo
  • Actualizar contacto
  • Eliminar contacto
  • Guardar/Cargar desde archivo (simulado con datos iniciales)
  • Mostrar estadísticas de la agenda

  Estructuras de datos utilizadas:
  • []Contacto (slice de structs) — almacenamiento principal
  • map[string][]int (mapa de grupos a índices) — índices rápidos
  • strings.Builder — formateo de salida
  • strconv — conversión de tipos

  package main

  import (
      "fmt"
      "sort"
      "strconv"
      "strings"
  )

  type Direccion struct {
      Calle     string
      Ciudad    string
      Pais      string
  }

  type Contacto struct {
      ID        int
      Nombre    string
      Telefono  string
      Email     string
      Direccion
      Grupo     string
      Favorito  bool
  }

  type Agenda struct {
      contactos []Contacto
      indice    map[string][]int
      siguienteID int
  }

  func NuevaAgenda() *Agenda {
      return &Agenda{
          contactos:  make([]Contacto, 0),
          indice:     make(map[string][]int),
          siguienteID: 1,
      }
  }

  func (a *Agenda) Agregar(nombre, telefono, email, calle, ciudad, pais, grupo string, fav bool) {
      c := Contacto{
          ID:       a.siguienteID,
          Nombre:   nombre,
          Telefono: telefono,
          Email:    email,
          Direccion: Direccion{
              Calle:  calle,
              Ciudad: ciudad,
              Pais:   pais,
          },
          Grupo:    grupo,
          Favorito: fav,
      }
      a.siguienteID++
      a.contactos = append(a.contactos, c)

      idx := len(a.contactos) - 1
      a.indice[grupo] = append(a.indice[grupo], idx)

      grupoNormalizado := strings.ToLower(strings.TrimSpace(nombre))
      a.indice[grupoNormalizado] = append(a.indice[grupoNormalizado], idx)
  }

  func (a *Agenda) Listar() string {
      if len(a.contactos) == 0 {
          return "No hay contactos en la agenda."
      }

      var sb strings.Builder
      sb.WriteString(strings.Repeat("=", 80))
      sb.WriteString("\n")
      sb.WriteString(fmt.Sprintf("%-4s %-20s %-15s %-25s %-15s %s\n",
          "ID", "Nombre", "Teléfono", "Email", "Grupo", "Fav"))
      sb.WriteString(strings.Repeat("-", 80))
      sb.WriteString("\n")

      for _, c := range a.contactos {
          fav := " "
          if c.Favorito {
              fav = "*"
          }
          sb.WriteString(fmt.Sprintf("%-4d %-20s %-15s %-25s %-15s %s\n",
              c.ID, c.Nombre, c.Telefono, c.Email, c.Grupo, fav))
      }
      sb.WriteString(strings.Repeat("=", 80))
      sb.WriteString(fmt.Sprintf("\nTotal: %d contactos\n", len(a.contactos)))
      return sb.String()
  }

  func (a *Agenda) BuscarPorNombre(nombre string) []Contacto {
      nombre = strings.ToLower(strings.TrimSpace(nombre))
      var resultados []Contacto
      for _, c := range a.contactos {
          if strings.Contains(strings.ToLower(c.Nombre), nombre) {
              resultados = append(resultados, c)
          }
      }
      return resultados
  }

  func (a *Agenda) BuscarPorGrupo(grupo string) []Contacto {
      grupo = strings.ToLower(strings.TrimSpace(grupo))
      var resultados []Contacto
      for _, c := range a.contactos {
          if strings.ToLower(c.Grupo) == grupo {
              resultados = append(resultados, c)
          }
      }
      return resultados
  }

  func (a *Agenda) Actualizar(id int, nombre, telefono, email, calle, ciudad, pais, grupo string, fav bool) bool {
      for i, c := range a.contactos {
          if c.ID == id {
              a.contactos[i].Nombre = nombre
              a.contactos[i].Telefono = telefono
              a.contactos[i].Email = email
              a.contactos[i].Calle = calle
              a.contactos[i].Ciudad = ciudad
              a.contactos[i].Pais = pais
              a.contactos[i].Grupo = grupo
              a.contactos[i].Favorito = fav
              return true
          }
      }
      return false
  }

  func (a *Agenda) Eliminar(id int) bool {
      for i, c := range a.contactos {
          if c.ID == id {
              a.contactos = append(a.contactos[:i], a.contactos[i+1:]...)
              a.rebuildIndex()
              return true
          }
      }
      return false
  }

  func (a *Agenda) rebuildIndex() {
      a.indice = make(map[string][]int)
      for i, c := range a.contactos {
          a.indice[c.Grupo] = append(a.indice[c.Grupo], i)
          nombreNorm := strings.ToLower(strings.TrimSpace(c.Nombre))
          a.indice[nombreNorm] = append(a.indice[nombreNorm], i)
      }
  }

  func (a *Agenda) Estadisticas() string {
      var sb strings.Builder
      sb.WriteString(strings.Repeat("=", 50))
      sb.WriteString("\nESTADÍSTICAS DE LA AGENDA\n")
      sb.WriteString(strings.Repeat("=", 50))
      sb.WriteString("\n")

      total := len(a.contactos)
      sb.WriteString(fmt.Sprintf("Total contactos: %d\n", total))

      favoritos := 0
      for _, c := range a.contactos {
          if c.Favorito {
              favoritos++
          }
      }
      sb.WriteString(fmt.Sprintf("Favoritos: %d (%.1f%%)\n",
          favoritos, float64(favoritos)/float64(total)*100))

      grupos := make(map[string]int)
      for _, c := range a.contactos {
          g := c.Grupo
          if g == "" {
              g = "Sin grupo"
          }
          grupos[g]++
      }
      sb.WriteString("Contactos por grupo:\n")
      for g, n := range grupos {
          sb.WriteString(fmt.Sprintf("  %s: %d\n", g, n))
      }

      // Contacto con nombre más largo
      masLargo := ""
      for _, c := range a.contactos {
          if len(c.Nombre) > len(masLargo) {
              masLargo = c.Nombre
          }
      }
      sb.WriteString(fmt.Sprintf("Nombre más largo: %s (%d caracteres)\n",
          masLargo, len(masLargo)))
      return sb.String()
  }

  func (a *Agenda) ListarPorGrupo() string {
      type grupoInfo struct {
          nombre string
          count  int
      }

      grupos := make(map[string][]Contacto)
      for _, c := range a.contactos {
          g := c.Grupo
          if g == "" {
              g = "Sin grupo"
          }
          grupos[g] = append(grupos[g], c)
      }

      var sb strings.Builder
      for g, contactos := range grupos {
          sb.WriteString(fmt.Sprintf("\n=== GRUPO: %s (%d contactos) ===\n", g, len(contactos)))
          for _, c := range contactos {
              fav := ""
              if c.Favorito {
                  fav = " *"
              }
              sb.WriteString(fmt.Sprintf("  [%d] %s | %s | %s%s\n",
                  c.ID, c.Nombre, c.Telefono, c.Email, fav))
          }
      }
      return sb.String()
  }

  func mostrarContactos(contactos []Contacto) string {
      if len(contactos) == 0 {
          return "  No se encontraron contactos."
      }
      var sb strings.Builder
      for _, c := range contactos {
          fav := ""
          if c.Favorito {
              fav = " *"
          }
          sb.WriteString(fmt.Sprintf("  [%d] %s | %s | %s | %s%s\n",
              c.ID, c.Nombre, c.Telefono, c.Email, c.Grupo, fav))
          sb.WriteString(fmt.Sprintf("       Dirección: %s, %s, %s\n",
              c.Calle, c.Ciudad, c.Pais))
      }
      return sb.String()
  }

  func leerInput(prompt string) string {
      fmt.Print(prompt)
      var entrada string
      fmt.Scanln(&entrada)
      return strings.TrimSpace(entrada)
  }

  func leerInt(prompt string) (int, bool) {
      entrada := leerInput(prompt)
      if entrada == "" {
          return 0, false
      }
      n, err := strconv.Atoi(entrada)
      if err != nil {
          fmt.Println("Error: debe ingresar un número válido.")
          return 0, false
      }
      return n, true
  }

  func leerBool(prompt string) bool {
      entrada := strings.ToLower(leerInput(prompt))
      return entrada == "s" || entrada == "si" || entrada == "yes" || entrada == "true" || entrada == "1"
  }

  PREGUNTAS FRECUENTES DEL MÓDULO III

  ❓ ¿Cuándo usar array vs slice?
  → Array para tamaño fijo conocido en compilación. Slice para todo lo demás.

  ❓ ¿Los slices se pasan por referencia?
  → No exactamente. Se pasa una copia del header (ptr, len, cap). El array
    subyacente se comparte.

  ❓ ¿Por qué los maps no tienen orden?
  → Por diseño. Es una tabla hash. Si necesitas orden, usa un slice aparte.

  ❓ ¿Struct embedding es lo mismo que herencia?
  → No. Es composición. Go favorece composición sobre herencia.

  ❓ ¿Cómo elijo entre map y struct?
  → Struct para campos fijos; map para datos dinámicos/claves variables.

  ❓ ¿Qué es más rápido: slice o map?
  → Depende. Slice para índice numérico (O(1) acceso). Map para claves
    arbitrarias (O(1) promedio).

  ❓ ¿Cómo iterar sobre un map en orden?
  → Extrae las claves, ordénalas, itera sobre las claves ordenadas.

  ❓ ¿Es seguro modificar un slice durante un range?
  → Modificar elementos existentes sí. Añadir/eliminar puede ser peligroso.

  ❓ ¿Cuándo usar make([]T, n) vs []T{}?
  → make con tamaño cuando conoces la cantidad. []T{} para literal.

  ❓ ¿Cómo convertir cualquier tipo a string?
  → fmt.Sprintf("%v", valor) o strconv para tipos específicos.
