```go
// Package calculadora contiene la lógica de la calculadora.
package calculadora

import (
    "fmt"
    "strings"
    "gocalc/constantes"
)

// EntradaHistorial representa una operación en el historial.
type EntradaHistorial struct {
    Operacion string
    Resultado string
}

// Historial almacena las operaciones realizadas.
type Historial struct {
    entradas []EntradaHistorial
}

// NuevaHistorial crea un nuevo historial vacío.
func NuevaHistorial() *Historial {
    return &Historial{
        entradas: make([]EntradaHistorial, 0, constantes.MaxHistorial),
    }
}

// Agregar añade una entrada al historial.
func (h *Historial) Agregar(operacion, resultado string) {
    if len(h.entradas) >= constantes.MaxHistorial {
        // Eliminar la más antigua
        h.entradas = h.entradas[1:]
    }
    h.entradas = append(h.entradas, EntradaHistorial{
        Operacion: operacion,
        Resultado: resultado,
    })
}

// Mostrar imprime el historial en pantalla.
func (h *Historial) Mostrar() {
    if len(h.entradas) == 0 {
        fmt.Println("  📭 No hay operaciones en el historial.")
        return
    }

    fmt.Println("  ╔══════════════════════════════════════════╗")
    fmt.Println("  ║           HISTORIAL DE OPERACIONES        ║")
    fmt.Println("  ╚══════════════════════════════════════════╝")
    fmt.Println()

    for i, entrada := range h.entradas {
        fmt.Printf("  %2d. %-20s = %s\n", i+1, entrada.Operacion, entrada.Resultado)
    }
    fmt.Println()
    fmt.Printf("  Total: %d operaciones\n", len(h.entradas))
}

// Limpiar vacía el historial.
func (h *Historial) Limpiar() {
    h.entradas = h.entradas[:0]
    fmt.Println("  ✅ Historial limpiado.")
}

// SeparadorHistorial imprime una línea separadora.
func SeparadorHistorial() {
    fmt.Println(strings.Repeat("─", 46))
}
```

```go
package calculadora

import (
    "fmt"
    "math"
)

// --- OPERACIONES BÁSICAS ---

// Sumar calcula a + b.
func Sumar(a, b float64) float64 {
    return a + b
}

// Restar calcula a - b.
func Restar(a, b float64) float64 {
    return a - b
}

// Multiplicar calcula a * b.
func Multiplicar(a, b float64) float64 {
    return a * b
}

// Dividir calcula a / b.
//
// Retorna error si b es 0.
func Dividir(a, b float64) (float64, error) {
    if b == 0 {
        return 0, fmt.Errorf("división por cero")
    }
    return a / b, nil
}

// --- OPERACIONES CIENTÍFICAS ---

// Potencia calcula base^exponente.
func Potencia(base, exponente float64) float64 {
    return math.Pow(base, exponente)
}

// RaizCuadrada calcula √x.
//
// Retorna error si x es negativo.
func RaizCuadrada(x float64) (float64, error) {
    if x < 0 {
        return 0, fmt.Errorf("no se puede calcular raíz cuadrada de un número negativo")
    }
    return math.Sqrt(x), nil
}

// LogaritmoNatural calcula ln(x).
//
// Retorna error si x <= 0.
func LogaritmoNatural(x float64) (float64, error) {
    if x <= 0 {
        return 0, fmt.Errorf("logaritmo indefinido para %f", x)
    }
    return math.Log(x), nil
}

// --- FUNCIONES TRIGONOMÉTRICAS ---

// Seno calcula sen(x) donde x está en radianes.
func Seno(x float64) float64 {
    return math.Sin(x)
}

// Coseno calcula cos(x) donde x está en radianes.
func Coseno(x float64) float64 {
    return math.Cos(x)
}

// Tangente calcula tan(x) donde x está en radianes.
func Tangente(x float64) float64 {
    return math.Tan(x)
}

// --- CONVERSIONES ---

// GradosARadianes convierte grados a radianes.
func GradosARadianes(grados float64) float64 {
    return grados * math.Pi / 180.0
}

// RadianesAGrados convierte radianes a grados.
func RadianesAGrados(radianes float64) float64 {
    return radianes * 180.0 / math.Pi
}
```

```go
package calculadora

import (
    "bufio"
    "fmt"
    "os"
    "strconv"
    "strings"
    "gocalc/constantes"
)

// Calculadora representa la calculadora completa.
type Calculadora struct {
    Historial *Historial
    scanner   *bufio.Scanner
}

// NuevaCalculadora crea una nueva calculadora.
func NuevaCalculadora() *Calculadora {
    return &Calculadora{
        Historial: NuevaHistorial(),
        scanner:   bufio.NewScanner(os.Stdin),
    }
}

// LeerNumero lee un número flotante del usuario.
func (c *Calculadora) LeerNumero(prompt string) (float64, error) {
    fmt.Print(prompt)
    if !c.scanner.Scan() {
        return 0, fmt.Errorf("error de lectura")
    }
    texto := strings.TrimSpace(c.scanner.Text())
    numero, err := strconv.ParseFloat(texto, 64)
    if err != nil {
        return 0, fmt.Errorf("número inválido: %s", texto)
    }
    return numero, nil
}

// LeerOpcion lee una opción del menú.
func (c *Calculadora) LeerOpcion() (int, error) {
    if !c.scanner.Scan() {
        return 0, fmt.Errorf("error de lectura")
    }
    texto := strings.TrimSpace(c.scanner.Text())
    opcion, err := strconv.Atoi(texto)
    if err != nil {
        return 0, fmt.Errorf("opción inválida: %s", texto)
    }
    return opcion, nil
}

// MostrarMenu imprime el menú principal.
func (c *Calculadora) MostrarMenu() {
    fmt.Println()
    fmt.Println("  ╔══════════════════════════════════════════╗")
    fmt.Println("  ║        🧮 CALCULADORA CIENTÍFICA         ║")
    fmt.Printf("  ║            Versión %-18s║\n", constantes.Version)
    fmt.Println("  ╚══════════════════════════════════════════╝")
    fmt.Println()
    fmt.Println("  ┌─────────────── OPERACIONES ──────────────┐")
    fmt.Println("  │                                          │")
    fmt.Println("  │  1. Sumar          7. Logaritmo natural │")
    fmt.Println("  │  2. Restar         8. Seno (radianes)   │")
    fmt.Println("  │  3. Multiplicar    9. Coseno (radianes) │")
    fmt.Println("  │  4. Dividir       10. Tangente (radianes)│")
    fmt.Println("  │  5. Potencia      11. Grados → Radianes │")
    fmt.Println("  │  6. Raíz cuadrada 12. Radianes → Grados │")
    fmt.Println("  │                                          │")
    fmt.Println("  │  13. Ver historial                       │")
    fmt.Println("  │  14. Salir                               │")
    fmt.Println("  └──────────────────────────────────────────┘")
    fmt.Println()
    fmt.Print("  Selecciona una opción [1-14]: ")
}

// Ejecutar inicia el bucle principal de la calculadora.
func (c *Calculadora) Ejecutar() {
    fmt.Println()
    fmt.Println("  ╔══════════════════════════════════════════╗")
    fmt.Println("  ║      BIENVENIDO A GOCALC v" + constantes.Version + "      ║")
    fmt.Println("  ║    Calculadora Científica en Go          ║")
    fmt.Println("  ╚══════════════════════════════════════════╝")
    fmt.Println()

    for {
        c.MostrarMenu()

        opcion, err := c.LeerOpcion()
        if err != nil {
            fmt.Println("  ❌ Error:", err)
            continue
        }

        if opcion == constantes.OpcionSalir {
            c.MostrarDespedida()
            break
        }

        c.ProcesarOpcion(opcion)
    }
}

// ProcesarOpcion procesa la opción seleccionada del menú.
func (c *Calculadora) ProcesarOpcion(opcion int) {
    var resultado float64
    var operacion string
    var err error

    switch opcion {
    case constantes.OpcionSumar:
        a, b := c.LeerDosNumeros()
        if a == nil || b == nil {
            return
        }
        resultado = Sumar(*a, *b)
        operacion = fmt.Sprintf("%.2f + %.2f", *a, *b)

    case constantes.OpcionRestar:
        a, b := c.LeerDosNumeros()
        if a == nil || b == nil {
            return
        }
        resultado = Restar(*a, *b)
        operacion = fmt.Sprintf("%.2f - %.2f", *a, *b)

    case constantes.OpcionMultiplicar:
        a, b := c.LeerDosNumeros()
        if a == nil || b == nil {
            return
        }
        resultado = Multiplicar(*a, *b)
        operacion = fmt.Sprintf("%.2f * %.2f", *a, *b)

    case constantes.OpcionDividir:
        a, b := c.LeerDosNumeros()
        if a == nil || b == nil {
            return
        }
        resultado, err = Dividir(*a, *b)
        if err != nil {
            fmt.Println("  ❌ Error:", err)
            return
        }
        operacion = fmt.Sprintf("%.2f / %.2f", *a, *b)

    case constantes.OpcionPotencia:
        a, b := c.LeerDosNumeros()
        if a == nil || b == nil {
            return
        }
        resultado = Potencia(*a, *b)
        operacion = fmt.Sprintf("%.2f ^ %.2f", *a, *b)

    case constantes.OpcionRaiz:
        x := c.LeerUnNumero()
        if x == nil {
            return
        }
        resultado, err = RaizCuadrada(*x)
        if err != nil {
            fmt.Println("  ❌ Error:", err)
            return
        }
        operacion = fmt.Sprintf("√(%.2f)", *x)

    case constantes.OpcionLogaritmo:
        x := c.LeerUnNumero()
        if x == nil {
            return
        }
        resultado, err = LogaritmoNatural(*x)
        if err != nil {
            fmt.Println("  ❌ Error:", err)
            return
        }
        operacion = fmt.Sprintf("ln(%.2f)", *x)

    case constantes.OpcionSeno:
        x := c.LeerUnNumero()
        if x == nil {
            return
        }
        resultado = Seno(*x)
        operacion = fmt.Sprintf("sen(%.4f rad)", *x)

    case constantes.OpcionCoseno:
        x := c.LeerUnNumero()
        if x == nil {
            return
        }
        resultado = Coseno(*x)
        operacion = fmt.Sprintf("cos(%.4f rad)", *x)

    case constantes.OpcionTangente:
        x := c.LeerUnNumero()
        if x == nil {
            return
        }
        resultado = Tangente(*x)
        operacion = fmt.Sprintf("tan(%.4f rad)", *x)

    case constantes.OpcionGradosARadianes:
        x := c.LeerUnNumero()
        if x == nil {
            return
        }
        resultado = GradosARadianes(*x)
        operacion = fmt.Sprintf("%.2f° a rad", *x)

    case constantes.OpcionRadianesAGrados:
        x := c.LeerUnNumero()
        if x == nil {
            return
        }
        resultado = RadianesAGrados(*x)
        operacion = fmt.Sprintf("%.4f rad a °", *x)

    case constantes.OpcionHistorial:
        c.Historial.Mostrar()
        return

    default:
        fmt.Println("  ❌ Opción no válida. Intenta de nuevo.")
        return
    }

    // Mostrar resultado
    fmt.Println()
    fmt.Printf("  🟢 %s = %.6f\n", operacion, resultado)
    fmt.Println()

    // Agregar al historial
    resultadoStr := fmt.Sprintf("%.6f", resultado)
    c.Historial.Agregar(operacion, resultadoStr)
}

// LeerDosNumeros lee dos números del usuario.
// Retorna nil si hay error en alguno.
func (c *Calculadora) LeerDosNumeros() (*float64, *float64) {
    a, err := c.LeerNumero("  Ingresa el primer número: ")
    if err != nil {
        fmt.Println("  ❌ Error:", err)
        return nil, nil
    }
    b, err := c.LeerNumero("  Ingresa el segundo número: ")
    if err != nil {
        fmt.Println("  ❌ Error:", err)
        return nil, nil
    }
    return &a, &b
}

// LeerUnNumero lee un solo número del usuario.
// Retorna nil si hay error.
func (c *Calculadora) LeerUnNumero() *float64 {
    x, err := c.LeerNumero("  Ingresa el número: ")
    if err != nil {
        fmt.Println("  ❌ Error:", err)
        return nil
    }
    return &x
}

// MostrarDespedida muestra un mensaje de despedida.
func (c *Calculadora) MostrarDespedida() {
    fmt.Println()
    fmt.Println("  ╔══════════════════════════════════════════╗")
    fmt.Println("  ║   ¡Gracias por usar GOCALC!             ║")
    fmt.Println("  ║   Nos vemos en el Módulo II.            ║")
    fmt.Println("  ╚══════════════════════════════════════════╝")
    fmt.Println()
}
```

```go
// calculadora/operaciones_test.go
package calculadora

import (
    "math"
    "testing"
)

// TestSumar prueba la función Sumar.
func TestSumar(t *testing.T) {
    casos := []struct {
        a, b, esperado float64
    }{
        {2, 3, 5},
        {-1, 1, 0},
        {0, 0, 0},
        {1.5, 2.5, 4.0},
    }

    for _, c := range casos {
        resultado := Sumar(c.a, c.b)
        if math.Abs(resultado-c.esperado) > 1e-10 {
            t.Errorf("Sumar(%.2f, %.2f) = %.2f; esperado %.2f",
                c.a, c.b, resultado, c.esperado)
        }
    }
}

// TestDividir prueba la función Dividir.
func TestDividir(t *testing.T) {
    // Caso exitoso
    resultado, err := Dividir(10, 2)
    if err != nil {
        t.Errorf("Error inesperado: %v", err)
    }
    if resultado != 5 {
        t.Errorf("Dividir(10, 2) = %.2f; esperado 5", resultado)
    }

    // Caso de error
    _, err = Dividir(10, 0)
    if err == nil {
        t.Error("Dividir(10, 0) debería retornar error")
    }
}

// TestRaizCuadrada prueba la función RaizCuadrada.
func TestRaizCuadrada(t *testing.T) {
    resultado, err := RaizCuadrada(9)
    if err != nil {
        t.Errorf("Error inesperado: %v", err)
    }
    if math.Abs(resultado-3) > 1e-10 {
        t.Errorf("RaizCuadrada(9) = %.2f; esperado 3", resultado)
    }

    _, err = RaizCuadrada(-1)
    if err == nil {
        t.Error("RaizCuadrada(-1) debería retornar error")
    }
}

// TestHistorial prueba el historial.
func TestHistorial(t *testing.T) {
    h := NuevaHistorial()
    if len(h.entradas) != 0 {
        t.Error("El historial nuevo debería estar vacío")
    }

    h.Agregar("2 + 3", "5")
    if len(h.entradas) != 1 {
        t.Error("El historial debería tener 1 entrada")
    }

    if h.entradas[0].Operacion != "2 + 3" {
        t.Error("Operación incorrecta en historial")
    }
    if h.entradas[0].Resultado != "5" {
        t.Error("Resultado incorrecto en historial")
    }
}
```

❓ **¿Puedo modificar la calculadora gocalc para mi proyecto final?**
No. El proyecto final debe ser `goagenda`, un gestor de contactos.
Puedes usar gocalc como referencia, pero el proyecto final es
diferente.

❓ **¿Cuánto tiempo debería tomar el proyecto final?**
Entre 4 y 8 horas para un principiante. Si te toma más, no te
preocupes; cada persona aprende a su ritmo.

❓ **¿Qué hago si me atoro en el proyecto final?**
1. Relee los capítulos relevantes del módulo.
2. Divide el problema en partes pequeñas.
3. Resuelve cada parte por separado.
4. Busca en la documentación oficial (go.dev/doc).
5. Pregunta en la comunidad (foros, Discord, Reddit r/golang).

❓ **¿Debo usar bibliotecas externas para el proyecto?**
No. Usa solo la biblioteca estándar de Go. Esa es la condición
del proyecto.

❓ **¿Puedo empezar el Módulo II sin completar el proyecto?**
Puedes, pero no es recomendable. El proyecto está diseñado para
reforzar los conceptos del módulo. Sin esa práctica, el Módulo II
será más difícil.

sientes cómodo con los conceptos del módulo, estás listo.

pueda reutilizarse en todos tus proyectos Go. El sistema debe:

- ¿Qué interfaces definirías para un menú reutilizable?
