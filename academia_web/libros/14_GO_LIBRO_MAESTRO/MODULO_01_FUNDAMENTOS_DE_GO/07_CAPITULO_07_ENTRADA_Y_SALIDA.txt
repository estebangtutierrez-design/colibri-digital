1. SALIDA CON FMT
===============================================================================

```go
package main

import "fmt"

func main() {
    // Println: imprime y añade salto de línea
    fmt.Println("Hola, mundo!")
    fmt.Println("Línea 1")
    fmt.Println("Línea 2")

    // Print: imprime sin salto de línea
    fmt.Print("Hola ")
    fmt.Print("mundo")
    fmt.Print("!\n")

    // Printf: imprime con formato
    nombre := "Lucas"
    edad := 30
    fmt.Printf("Nombre: %s, Edad: %d\n", nombre, edad)

    // Múltiples valores
    fmt.Println("Uno", "Dos", "Tres")
    fmt.Println(1, 2, 3, "cuatro")
}
```

1.1 Format verbs principales

```go
package main

import "fmt"

func main() {
    nombre := "Lucas"
    edad := 30
    altura := 1.75
    activo := true

    // %v - valor por defecto
    fmt.Printf("%%v: %v %v %v %v\n", nombre, edad, altura, activo)

    // %#v - valor con sintaxis Go
    fmt.Printf("%%#v: %#v %#v %#v %#v\n", nombre, edad, altura, activo)

    // %T - tipo del valor
    fmt.Printf("%%T: %T %T %T %T\n", nombre, edad, altura, activo)

    // %d - entero decimal
    fmt.Printf("%%d: %d\n", edad)

    // %s - string
    fmt.Printf("%%s: %s\n", nombre)

    // %q - string con comillas
    fmt.Printf("%%q: %q\n", nombre)

    // %f - float
    fmt.Printf("%%f: %f\n", altura)
    fmt.Printf("%%.2f: %.2f\n", altura)
    fmt.Printf("%%6.2f: %6.2f\n", altura)

    // %t - booleano
    fmt.Printf("%%t: %t\n", activo)

    // %% - signo de porcentaje literal
    fmt.Println("100%%")

    // %b - binario
    fmt.Printf("%%b: %b\n", edad)

    // %x - hexadecimal
    fmt.Printf("%%x: %x\n", edad)
    fmt.Printf("%%X: %X\n", edad)

    // %o - octal
    fmt.Printf("%%o: %o\n", edad)

    // %c - carácter (rune)
    letra := 'A'
    fmt.Printf("%%c: %c\n", letra)
    fmt.Printf("%%U: %U\n", letra)
}
```

1.2 Ancho y alineación

```go
package main

import "fmt"

func main() {
    // Alineación a la derecha (ancho mínimo)
    fmt.Printf("|%10s|%10s|\n", "Nombre", "Edad")
    fmt.Println("|----------|----------|")
    fmt.Printf("|%10s|%10d|\n", "Lucas", 30)
    fmt.Printf("|%10s|%10d|\n", "Ana", 25)
    fmt.Printf("|%10s|%10d|\n", "Carlos", 35)

    fmt.Println()

    // Alineación a la izquierda (signo negativo)
    fmt.Printf("|%-10s|%-10s|\n", "Nombre", "Edad")
    fmt.Println("|----------|----------|")
    fmt.Printf("|%-10s|%-10d|\n", "Lucas", 30)
    fmt.Printf("|%-10s|%-10d|\n", "Ana", 25)
    fmt.Printf("|%-10s|%-10d|\n", "Carlos", 35)

    fmt.Println()

    // Ancho con precisión para floats
    fmt.Printf("|%10s|%10s|\n", "Producto", "Precio")
    fmt.Println("|----------|----------|")
    fmt.Printf("|%-10s|%10.2f|\n", "Manzana", 1.5)
    fmt.Printf("|%-10s|%10.2f|\n", "Pan", 0.75)
    fmt.Printf("|%-10s|%10.2f|\n", "Leche", 2.99)

    // Ceros a la izquierda
    for i := 0; i < 5; i++ {
        fmt.Printf("Item-%03d\n", i)
    }
}
```

1.3 Sprintf: formatear sin imprimir


2. ENTRADA CON FMT
===============================================================================

```go
package main

import "fmt"

func main() {
    var nombre string
    var edad int

    // Scan: lee valores separados por espacios
    fmt.Print("Ingresa tu nombre y edad: ")
    n, err := fmt.Scan(&nombre, &edad)
    if err != nil {
        fmt.Println("Error:", err)
        return
    }
    fmt.Printf("Leídos %d valores: %s, %d años\n", n, nombre, edad)
}
```

3. BUFIO.SCANNER PARA ENTRADA FLEXIBLE
===============================================================================

`bufio.Scanner` es más flexible y recomendado para entrada del usuario:

```go
package main

import (
    "bufio"
    "fmt"
    "os"
    "strconv"
    "strings"
)

func main() {
    lector := bufio.NewReader(os.Stdin)

    fmt.Print("¿Cómo te llamas? ")
    nombre, _ := lector.ReadString('\n')
    nombre = strings.TrimSpace(nombre)
    fmt.Printf("¡Hola, %s!\n", nombre)

    fmt.Print("¿Cuántos años tienes? ")
    edadStr, _ := lector.ReadString('\n')
    edadStr = strings.TrimSpace(edadStr)
    edad, err := strconv.Atoi(edadStr)
    if err != nil {
        fmt.Println("Edad inválida:", err)
        return
    }
    fmt.Printf("Tienes %d años\n", edad)
}
```

3.1 Scanner con bufio.Scanner

```go
package main

import (
    "bufio"
    "fmt"
    "os"
    "strconv"
    "strings"
)

func main() {
    scanner := bufio.NewScanner(os.Stdin)

    fmt.Print("Ingresa tu nombre: ")
    scanner.Scan()
    nombre := scanner.Text()
    nombre = strings.TrimSpace(nombre)

    fmt.Print("Ingresa tu edad: ")
    scanner.Scan()
    edadStr := scanner.Text()
    edad, err := strconv.Atoi(strings.TrimSpace(edadStr))
    if err != nil {
        fmt.Println("Edad inválida")
        return
    }

    fmt.Printf("Hola %s, tienes %d años.\n", nombre, edad)
}
```

3.3 Leer múltiples líneas

```go
package main

import (
    "bufio"
    "fmt"
    "os"
)

func main() {
    scanner := bufio.NewScanner(os.Stdin)

    fmt.Println("Ingresa líneas de texto (escribe 'fin' para terminar):")

    for scanner.Scan() {
        linea := scanner.Text()
        if linea == "fin" {
            break
        }
        fmt.Println("Leíste:", linea)
    }

    if err := scanner.Err(); err != nil {
        fmt.Println("Error:", err)
    }
}
```

3.4 Leer números de una línea

```go
package main

import (
    "bufio"
    "fmt"
    "os"
    "strconv"
    "strings"
)

func main() {
    scanner := bufio.NewScanner(os.Stdin)

    fmt.Print("Ingresa números separados por espacio: ")
    scanner.Scan()
    line := scanner.Text()
    tokens := strings.Fields(line)

    suma := 0
    for _, token := range tokens {
        num, err := strconv.Atoi(token)
        if err != nil {
            fmt.Println("Error al convertir:", token)
            continue
        }
        suma += num
    }

    fmt.Printf("Los números suman: %d\n", suma)
    fmt.Printf("Promedio: %.2f\n", float64(suma)/float64(len(tokens)))
}
```

```go
package main

import (
    "fmt"
    "os"
)

func main() {
    // Escribir a stdout
    fmt.Fprint(os.Stdout, "Esto va a stdout\n")
    fmt.Fprintln(os.Stdout, "Esto también va a stdout")

    // Escribir a stderr
    fmt.Fprint(os.Stderr, "Esto es un error\n")

    // Leer de stdin directamente
    var buffer [100]byte
    fmt.Print("Ingresa algo: ")
    n, err := os.Stdin.Read(buffer[:])
    if err != nil {
        fmt.Println("Error:", err)
        return
    }
    fmt.Printf("Leídos %d bytes: %q\n", n, string(buffer[:n]))
}
```

```go
package main

import "fmt"

// Implementar la interfaz Stringer para formato personalizado
type Persona struct {
    Nombre string
    Edad   int
}

func (p Persona) String() string {
    return fmt.Sprintf("%s (%d años)", p.Nombre, p.Edad)
}

// Formato personalizado con Format (requiere Go 1.24+)
type Punto struct {
    X, Y int
}

func (p Punto) String() string {
    return fmt.Sprintf("(%d, %d)", p.X, p.Y)
}

func main() {
    p := Persona{Nombre: "Lucas", Edad: 30}
    fmt.Println(p)  // Usa String()

    pt := Punto{X: 10, Y: 20}
    fmt.Println(pt)

    // También puedes formatear manualmente
    fmt.Printf("Persona: %s\n", p)
    fmt.Printf("Punto: %s\n", pt)
}
```

9. FORMATEO AVANZADO
===============================================================================

```go
package main

import "fmt"

func main() {
    // Formato de números grandes
    n := 1234567
    fmt.Printf("Número: %d\n", n)

    // Sin formato de miles (Go no tiene %,'d como Python)
    // Puedes usar un helper o formatear manualmente

    // Formato hexadecimal
    fmt.Printf("Hex: %x\n", 255)
    fmt.Printf("Hex con 0x: %#x\n", 255)
    fmt.Printf("Hex mayúscula: %X\n", 255)

    // Formato octal
    fmt.Printf("Octal: %o\n", 64)
    fmt.Printf("Octal con 0o: %#o\n", 64)

    // Formato binario
    fmt.Printf("Binario: %b\n", 12)

    // Formato científico
    fmt.Printf("Científico: %e\n", 1234.5678)
    fmt.Printf("Científico mayúscula: %E\n", 1234.5678)

    // Selección automática de formato
    fmt.Printf("%%g: %g\n", 1234.5678)
    fmt.Printf("%%g: %g\n", 0.00001234)

    // Relleno con espacios
    fmt.Printf("'%6d'\n", 42)

    // Relleno con ceros
    fmt.Printf("'%06d'\n", 42)

    // Asterisco para ancho dinámico
    ancho := 10
    valor := 42
    fmt.Printf("'%*d'\n", ancho, valor)
    fmt.Printf("'%-*d'\n", ancho, valor)
}
```

10. EJEMPLO COMPLETO: CALCULADORA INTERACTIVA
===============================================================================

```go
package main

import (
    "bufio"
    "fmt"
    "os"
    "strconv"
    "strings"
)

func main() {
    scanner := bufio.NewScanner(os.Stdin)

    fmt.Println("========================================")
    fmt.Println("  CALCULADORA GO")
    fmt.Println("========================================")
    fmt.Println("Operaciones: +, -, *, /")
    fmt.Println("Escribe 'salir' para terminar.")
    fmt.Println("========================================")

    for {
        fmt.Print("\n>>> ")
        scanner.Scan()
        entrada := strings.TrimSpace(scanner.Text())

        if entrada == "salir" {
            fmt.Println("¡Hasta luego!")
            break
        }

        if entrada == "" {
            continue
        }

        partes := strings.Fields(entrada)
        if len(partes) != 3 {
            fmt.Println("Formato: número operador número (ej: 5 + 3)")
            continue
        }

        a, err := strconv.ParseFloat(partes[0], 64)
        if err != nil {
            fmt.Println("Error: primer número inválido")
            continue
        }

        operador := partes[1]

        b, err := strconv.ParseFloat(partes[2], 64)
        if err != nil {
            fmt.Println("Error: segundo número inválido")
            continue
        }

        var resultado float64
        var operacion string

        switch operador {
        case "+":
            resultado = a + b
            operacion = fmt.Sprintf("%.2f + %.2f", a, b)
        case "-":
            resultado = a - b
            operacion = fmt.Sprintf("%.2f - %.2f", a, b)
        case "*":
            resultado = a * b
            operacion = fmt.Sprintf("%.2f * %.2f", a, b)
        case "/":
            if b == 0 {
                fmt.Println("Error: división por cero")
                continue
            }
            resultado = a / b
            operacion = fmt.Sprintf("%.2f / %.2f", a, b)
        default:
            fmt.Println("Operador no válido. Usa: +, -, *, /")
            continue
        }

        fmt.Printf("%s = %.2f\n", operacion, resultado)
    }

    fmt.Println("========================================")
}
```

PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Cuál es la diferencia entre Print y Println?**
Println añade un espacio entre argumentos y un salto de línea
al final. Print no añade espacio ni salto de línea.

❓ **¿Por qué fmt.Scan deja basura en el buffer?**
Scan lee hasta el siguiente espacio o salto de línea. Si hay
datos sobrantes, pueden interferir con lecturas posteriores.
Usa bufio.Scanner para un control más preciso.

❓ **¿Qué es %v y %+v?**
%v muestra el valor por defecto. %+v muestra los nombres de
los campos en estructuras. %#v muestra la sintaxis Go completa
del valor.

❓ **¿Cómo leo una línea completa con espacios?**
Usa `bufio.Scanner` con `scanner.Scan()` y `scanner.Text()`.
fmt.Scan solo lee hasta el primer espacio.

❓ **¿Cómo formateo números con separadores de miles?**
Go no tiene un format verb para separadores de miles. Puedes
usar el paquete `golang.org/x/text/message` o implementar
tu propia función.

❓ **¿Qué significan los %*d?**
El asterisco permite pasar el ancho como argumento.
`fmt.Printf("%*d", ancho, valor)` es equivalente a
`fmt.Printf("%Xd", valor)` donde X es el valor de ancho.
