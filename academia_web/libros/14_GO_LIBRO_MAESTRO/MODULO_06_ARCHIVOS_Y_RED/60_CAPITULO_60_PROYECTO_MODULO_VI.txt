  2. MONITOR.GO — LÓGICA DE MONITOREO
  ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

  ```go
  package main

  import (
      "encoding/csv"
      "encoding/json"
      "fmt"
      "log"
      "os"
      "sync"
      "time"
  )

  type Monitor struct {
      config     Config
      resultados []Resultado
      mu         sync.RWMutex
      archivoCSV string
  }

  func NewMonitor(rutaConfig string) (*Monitor, error) {
      datos, err := os.ReadFile(rutaConfig)
      if err != nil {
          return nil, fmt.Errorf("error al leer config: %w", err)
      }

      var config Config
      if err := json.Unmarshal(datos, &config); err != nil {
          return nil, fmt.Errorf("error al parsear config: %w", err)
      }

      if config.Puerto == 0 {
          config.Puerto = 8080
      }
      if config.Intervalo == 0 {
          config.Intervalo = 60
      }
      if config.Token == "" {
          config.Token = "api-monitor-token"
      }
      for i := range config.APIs {
          if config.APIs[i].Metodo == "" {
              config.APIs[i].Metodo = "GET"
          }
          if config.APIs[i].Timeout == 0 {
              config.APIs[i].Timeout = 10
          }
          if config.APIs[i].Esperado == 0 {
              config.APIs[i].Esperado = 200
          }
      }

      m := &Monitor{
          config:     config,
          archivoCSV: "historial.csv",
      }

      // Inicializar CSV con cabeceras
      m.inicializarCSV()

      return m, nil
  }

  func (m *Monitor) GetConfig() Config {
      return m.config
  }

  func (m *Monitor) GetResultados() []Resultado {
      m.mu.RLock()
      defer m.mu.RUnlock()
      copia := make([]Resultado, len(m.resultados))
      copy(copia, m.resultados)
      return copia
  }

  func (m *Monitor) inicializarCSV() {
      if _, err := os.Stat(m.archivoCSV); os.IsNotExist(err) {
          f, err := os.Create(m.archivoCSV)
          if err != nil {
              log.Printf("Error al crear CSV: %v", err)
              return
          }
          defer f.Close()

          writer := csv.NewWriter(f)
          writer.Write([]string{
              "timestamp", "nombre", "url", "estado",
              "status_code", "latencia_ms", "error",
          })
          writer.Flush()
      }
  }

  func (m *Monitor) AgregarResultado(r Resultado) {
      m.mu.Lock()
      m.resultados = append(m.resultados, r)
      if len(m.resultados) > 1000 {
          m.resultados = m.resultados[len(m.resultados)-1000:]
      }
      m.mu.Unlock()

      // Guardar en CSV
      m.guardarEnCSV(r)
  }

  func (m *Monitor) guardarEnCSV(r Resultado) {
      f, err := os.OpenFile(m.archivoCSV,
          os.O_APPEND|os.O_WRONLY|os.O_CREATE, 0644)
      if err != nil {
          log.Printf("Error al abrir CSV: %v", err)
          return
      }
      defer f.Close()

      writer := csv.NewWriter(f)
      defer writer.Flush()

      errMsg := r.Error
      if errMsg == "" {
          errMsg = "-"
      }

      writer.Write([]string{
          r.Verificado.Format(time.RFC3339),
          r.Nombre,
          r.URL,
          r.Estado,
          fmt.Sprintf("%d", r.StatusCode),
          fmt.Sprintf("%d", r.LatenciaMs),
          errMsg,
      })
  }

  func (m *Monitor) Iniciar() {
      log.Printf("Iniciando monitoreo de %d APIs cada %d segundos",
          len(m.config.APIs), m.config.Intervalo)

      for _, api := range m.config.APIs {
          resultado := m.verificarAPI(api)
          m.AgregarResultado(resultado)
          log.Printf("[%s] %s (%dms)",
              resultado.Estado, api.Nombre, resultado.LatenciaMs)
      }

      ticker := time.NewTicker(time.Duration(m.config.Intervalo) * time.Second)
      go func() {
          for range ticker.C {
              for _, api := range m.config.APIs {
                  resultado := m.verificarAPI(api)
                  m.AgregarResultado(resultado)
                  log.Printf("[%s] %s (%dms)",
                      resultado.Estado, api.Nombre, resultado.LatenciaMs)
              }
          }
      }()
  }

  func (m *Monitor) verificarAPI(api APIConfig) Resultado {
      return VerificarAPI(api)
  }

  func (m *Monitor) GenerarReporte() Reporte {
      resultados := m.GetResultados()
      reporte := Reporte{Resultados: make([]Resultado, 0)}

      // Solo el último resultado por API
      vistos := make(map[string]bool)
      for i := len(resultados) - 1; i >= 0; i-- {
          if vistos[resultados[i].Nombre] {
              continue
          }
          vistos[resultados[i].Nombre] = true
          reporte.Resultados = append(reporte.Resultados, resultados[i])
          if resultados[i].Estado == "ok" {
              reporte.Exitosos++
          } else {
              reporte.Fallidos++
          }
          reporte.Total++
      }

      return reporte
  }

  func (m *Monitor) GenerarHistorial() []Resultado {
      return m.GetResultados()
  }

  func (m *Monitor) ExportarCSV(ruta string) error {
      resultados := m.GetResultados()
      f, err := os.Create(ruta)
      if err != nil {
          return err
      }
      defer f.Close()

      writer := csv.NewWriter(f)
      defer writer.Flush()

      writer.Write([]string{
          "timestamp", "nombre", "url", "estado",
          "status_code", "latencia_ms", "error",
      })

      for _, r := range resultados {
          errMsg := r.Error
          if errMsg == "" {
              errMsg = "-"
          }
          writer.Write([]string{
              r.Verificado.Format(time.RFC3339),
              r.Nombre,
              r.URL,
              r.Estado,
              fmt.Sprintf("%d", r.StatusCode),
              fmt.Sprintf("%d", r.LatenciaMs),
              errMsg,
          })
      }

      return writer.Error()
  }

  func (m *Monitor) ExportarJSON(ruta string) error {
      reporte := m.GenerarReporte()
      datos, err := json.MarshalIndent(reporte, "", "  ")
      if err != nil {
          return err
      }
      return os.WriteFile(ruta, datos, 0644)
  }
  ```

  3. CLIENT.GO — CLIENTE HTTP PARA VERIFICAR APIs
  ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

  ```go
  package main

  import (
      "fmt"
      "io"
      "net/http"
      "time"
  )

  func VerificarAPI(api APIConfig) Resultado {
      inicio := time.Now()

      resultado := Resultado{
          Nombre:     api.Nombre,
          URL:        api.URL,
          Verificado: time.Now(),
      }

      timeout := time.Duration(api.Timeout) * time.Second
      client := &http.Client{
          Timeout: timeout,
          Transport: &http.Transport{
              MaxIdleConnsPerHost: 1,
              DisableKeepAlives:   true,
          },
      }

      req, err := http.NewRequest(api.Metodo, api.URL, nil)
      if err != nil {
          resultado.Estado = "error"
          resultado.Error = fmt.Sprintf("error al crear request: %v", err)
          resultado.LatenciaMs = time.Since(inicio).Milliseconds()
          return resultado
      }

      req.Header.Set("User-Agent", "APIMonitor/1.0")

      resp, err := client.Do(req)
      if err != nil {
          resultado.Estado = "error"
          resultado.Error = fmt.Sprintf("error en petición: %v", err)
          resultado.LatenciaMs = time.Since(inicio).Milliseconds()
          return resultado
      }
      defer resp.Body.Close()

      // Leer y descartar body (necesario para reciclar conexión)
      io.Copy(io.Discard, resp.Body)

      latencia := time.Since(inicio).Milliseconds()
      resultado.LatenciaMs = latencia
      resultado.StatusCode = resp.StatusCode

      if resp.StatusCode == api.Esperado {
          resultado.Estado = "ok"
      } else {
          resultado.Estado = "error"
          resultado.Error = fmt.Sprintf(
              "código esperado %d, recibido %d",
              api.Esperado, resp.StatusCode)
      }

      return resultado
  }
  ```

  4. MIDDLEWARE.GO — MIDDLEWARE REUTILIZABLE
  ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

  ```go
  package main

  import (
      "context"
      "encoding/json"
      "log"
      "net/http"
      "strings"
      "time"
  )

  type contextKey string

  const (
      KeyRequestID contextKey = "request_id"
      KeyStartTime contextKey = "start_time"
  )

  type Middleware func(http.Handler) http.Handler

  func Chain(handler http.Handler, middlewares ...Middleware) http.Handler {
      for i := len(middlewares) - 1; i >= 0; i-- {
          handler = middlewares[i](handler)
      }
      return handler
  }

  func RequestID(next http.Handler) http.Handler {
      return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
          id := r.Header.Get("X-Request-ID")
          if id == "" {
              id = fmt.Sprintf("%d", time.Now().UnixNano())
          }
          ctx := context.WithValue(r.Context(), KeyRequestID, id)
          w.Header().Set("X-Request-ID", id)
          next.ServeHTTP(w, r.WithContext(ctx))
      })
  }

  func Logger(next http.Handler) http.Handler {
      return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
          inicio := time.Now()
          lw := &logWriter{ResponseWriter: w, statusCode: http.StatusOK}

          next.ServeHTTP(lw, r)

          reqID := r.Context().Value(KeyRequestID)
          log.Printf("[%v] %s %s %d %d %v",
              reqID, r.Method, r.URL.Path,
              lw.statusCode, lw.bytesEscritos, time.Since(inicio))
      })
  }

  type logWriter struct {
      http.ResponseWriter
      statusCode    int
      bytesEscritos int
  }

  func (w *logWriter) WriteHeader(code int) {
      w.statusCode = code
      w.ResponseWriter.WriteHeader(code)
  }

  func (w *logWriter) Write(b []byte) (int, error) {
      n, err := w.ResponseWriter.Write(b)
      w.bytesEscritos += n
      return n, err
  }

  func Recoverer(next http.Handler) http.Handler {
      return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
          defer func() {
              if err := recover(); err != nil {
                  log.Printf("PANIC: %v", err)
                  w.Header().Set("Content-Type", "application/json")
                  w.WriteHeader(http.StatusInternalServerError)
                  json.NewEncoder(w).Encode(map[string]string{
                      "error": "error interno del servidor",
                  })
              }
          }()
          next.ServeHTTP(w, r)
      })
  }

  func Auth(token string) Middleware {
      return func(next http.Handler) http.Handler {
          return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
              auth := r.Header.Get("Authorization")
              if !strings.HasPrefix(auth, "Bearer ") || strings.TrimPrefix(auth, "Bearer ") != token {
                  w.Header().Set("Content-Type", "application/json")
                  w.WriteHeader(http.StatusUnauthorized)
                  json.NewEncoder(w).Encode(map[string]string{
                      "error": "no autorizado: token inválido o ausente",
                  })
                  return
              }
              next.ServeHTTP(w, r)
          })
      }
  }

  func CORS(next http.Handler) http.Handler {
      return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
          w.Header().Set("Access-Control-Allow-Origin", "*")
          w.Header().Set("Access-Control-Allow-Methods", "GET, POST, OPTIONS")
          w.Header().Set("Access-Control-Allow-Headers", "Content-Type, Authorization")

          if r.Method == "OPTIONS" {
              w.WriteHeader(http.StatusNoContent)
              return
          }

          next.ServeHTTP(w, r)
      })
  }
  ```

  5. SERVER.GO — SERVIDOR HTTP CON RUTAS

  ```go
  package main

  import (
      "encoding/json"
      "fmt"
      "log"
      "net/http"
      "strconv"
      "time"
  )

  type Server struct {
      monitor *Monitor
      mux     *http.ServeMux
  }

  func NewServer(monitor *Monitor) *Server {
      s := &Server{
          monitor: monitor,
          mux:     http.NewServeMux(),
      }
      s.registrarRutas()
      return s
  }

  func (s *Server) registrarRutas() {
      config := s.monitor.GetConfig()

      // Rutas públicas
      s.mux.HandleFunc("GET /", s.handleHome)
      s.mux.HandleFunc("GET /health", s.handleHealth)
      s.mux.HandleFunc("GET /api/v1/status", s.handleStatus)

      // Rutas protegidas (requieren token)
      s.mux.HandleFunc("GET /api/v1/apis", s.handleListarAPIs)
      s.mux.HandleFunc("GET /api/v1/report", s.handleReport)
      s.mux.HandleFunc("GET /api/v1/history", s.handleHistory)
      s.mux.HandleFunc("GET /api/v1/export/json", s.handleExportJSON)
      s.mux.HandleFunc("GET /api/v1/export/csv", s.handleExportCSV)
      s.mux.HandleFunc("GET /api/v1/config", s.handleConfig)

      // Crear handler con middleware
      handler := Chain(
          s.mux,
          RequestID,
          Logger,
          Recoverer,
          CORS,
          Auth(config.Token),
      )

      // Reemplazar mux con handler protegido
      // Las rutas públicas se manejan antes
      publicHandler := Chain(
          http.HandlerFunc(s.handlePublic),
          RequestID,
          Logger,
          Recoverer,
          CORS,
      )

      muxFinal := http.NewServeMux()
      muxFinal.HandleFunc("GET /", func(w http.ResponseWriter, r *http.Request) {
          if r.URL.Path == "/" || r.URL.Path == "/health" || r.URL.Path == "/api/v1/status" {
              publicHandler.ServeHTTP(w, r)
          } else {
              handler.ServeHTTP(w, r)
          }
      })

      s.mux = muxFinal
  }

  func (s *Server) handlePublic(w http.ResponseWriter, r *http.Request) {
      switch r.URL.Path {
      case "/":
          s.handleHome(w, r)
      case "/health":
          s.handleHealth(w, r)
      case "/api/v1/status":
          s.handleStatus(w, r)
      default:
          http.NotFound(w, r)
      }
  }

  func (s *Server) Handler() http.Handler {
      return s.mux
  }

  func (s *Server) handleHome(w http.ResponseWriter, r *http.Request) {
      w.Header().Set("Content-Type", "application/json")
      json.NewEncoder(w).Encode(map[string]interface{}{
          "nombre":      "API Monitor",
          "version":     "1.0.0",
          "modulo":      "Módulo VI - Archivos y Red",
          "libro":       "GO MASTER BOOK",
          "endpoints": map[string]string{
              "/health":          "Health check",
              "/api/v1/status":   "Estado del monitor",
              "/api/v1/apis":     "Lista de APIs monitoreadas",
              "/api/v1/report":   "Reporte actual",
              "/api/v1/history":  "Historial de verificaciones",
              "/api/v1/export/json": "Exportar reporte JSON",
              "/api/v1/export/csv":  "Exportar historial CSV",
              "/api/v1/config":   "Configuración actual",
          },
      })
  }

  func (s *Server) handleHealth(w http.ResponseWriter, r *http.Request) {
      w.Header().Set("Content-Type", "application/json")
      json.NewEncoder(w).Encode(map[string]interface{}{
          "status":    "healthy",
          "timestamp": time.Now().Format(time.RFC3339),
          "apis_monitoreadas": len(s.monitor.GetConfig().APIs),
      })
  }

  func (s *Server) handleStatus(w http.ResponseWriter, r *http.Request) {
      w.Header().Set("Content-Type", "application/json")
      resultados := s.monitor.GetResultados()
      if len(resultados) == 0 {
          json.NewEncoder(w).Encode(map[string]string{
              "mensaje": "No hay verificaciones aún. Espera al primer ciclo.",
          })
          return
      }
      ultimo := resultados[len(resultados)-1]
      reporte := s.monitor.GenerarReporte()
      json.NewEncoder(w).Encode(map[string]interface{}{
          "ultima_verificacion": ultimo.Verificado,
          "total_apis":          reporte.Total,
          "exitosos":            reporte.Exitosos,
          "fallidos":            reporte.Fallidos,
          "resultados":          reporte.Resultados,
      })
  }

  func (s *Server) handleListarAPIs(w http.ResponseWriter, r *http.Request) {
      w.Header().Set("Content-Type", "application/json")
      config := s.monitor.GetConfig()
      json.NewEncoder(w).Encode(config.APIs)
  }

  func (s *Server) handleReport(w http.ResponseWriter, r *http.Request) {
      w.Header().Set("Content-Type", "application/json")
      reporte := s.monitor.GenerarReporte()
      json.NewEncoder(w).Encode(reporte)
  }

  func (s *Server) handleHistory(w http.ResponseWriter, r *http.Request) {
      limiteStr := r.URL.Query().Get("limite")
      limite := 100
      if limiteStr != "" {
          if l, err := strconv.Atoi(limiteStr); err == nil && l > 0 && l <= 1000 {
              limite = l
          }
      }

      historia := s.monitor.GenerarHistorial()
      if len(historia) > limite {
          historia = historia[len(historia)-limite:]
      }

      w.Header().Set("Content-Type", "application/json")
      json.NewEncoder(w).Encode(historia)
  }

  func (s *Server) handleExportJSON(w http.ResponseWriter, r *http.Request) {
      w.Header().Set("Content-Type", "application/json")
      w.Header().Set("Content-Disposition", "attachment; filename=reporte.json")
      reporte := s.monitor.GenerarReporte()
      json.NewEncoder(w).Encode(reporte)
  }

  func (s *Server) handleExportCSV(w http.ResponseWriter, r *http.Request) {
      w.Header().Set("Content-Type", "text/csv")
      w.Header().Set("Content-Disposition", "attachment; filename=historial.csv")

      resultados := s.monitor.GenerarHistorial()
      // Escribir CSV directamente a la respuesta
      fmt.Fprintf(w, "timestamp,nombre,url,estado,status_code,latencia_ms,error\n")
      for _, r := range resultados {
          errMsg := r.Error
          if errMsg == "" {
              errMsg = "-"
          }
          fmt.Fprintf(w, "%s,%s,%s,%s,%d,%d,%s\n",
              r.Verificado.Format(time.RFC3339),
              r.Nombre, r.URL, r.Estado,
              r.StatusCode, r.LatenciaMs, errMsg)
      }
  }

  func (s *Server) handleConfig(w http.ResponseWriter, r *http.Request) {
      w.Header().Set("Content-Type", "application/json")
      config := s.monitor.GetConfig()
      // Ocultar token por seguridad
      config.Token = "****"
      json.NewEncoder(w).Encode(config)
  }

  func (s *Server) Iniciar() {
      config := s.monitor.GetConfig()
      addr := fmt.Sprintf(":%d", config.Puerto)

      server := &http.Server{
          Addr:         addr,
          Handler:      s.Handler(),
          ReadTimeout:  15 * time.Second,
          WriteTimeout: 15 * time.Second,
          IdleTimeout:  60 * time.Second,
      }

      log.Printf("Servidor HTTP iniciado en %s", addr)
      log.Printf("Endpoints disponibles:")
      log.Printf("  GET /")
      log.Printf("  GET /health")
      log.Printf("  GET /api/v1/status")
      log.Printf("  GET /api/v1/apis")
      log.Printf("  GET /api/v1/report")
      log.Printf("  GET /api/v1/history?limite=100")
      log.Printf("  GET /api/v1/export/json")
      log.Printf("  GET /api/v1/export/csv")
      log.Printf("  GET /api/v1/config")
      log.Printf("Usa header 'Authorization: Bearer <token>' para rutas protegidas")

      if err := server.ListenAndServe(); err != nil {
          log.Fatalf("Error en servidor: %v", err)
      }
  }
  ```

  6. EXPORTER.GO — EXPORTACIÓN

  ```go
  package main

  import (
      "encoding/csv"
      "encoding/json"
      "fmt"
      "os"
      "time"
  )

  type Exporter struct {
      monitor *Monitor
  }

  func NewExporter(monitor *Monitor) *Exporter {
      return &Exporter{monitor: monitor}
  }

  func (e *Exporter) ExportarJSON(ruta string) error {
      reporte := e.monitor.GenerarReporte()
      datos, err := json.MarshalIndent(reporte, "", "  ")
      if err != nil {
          return fmt.Errorf("error al serializar JSON: %w", err)
      }

      return os.WriteFile(ruta, datos, 0644)
  }

  func (e *Exporter) ExportarCSV(ruta string) error {
      f, err := os.Create(ruta)
      if err != nil {
          return fmt.Errorf("error al crear archivo: %w", err)
      }
      defer f.Close()

      writer := csv.NewWriter(f)
      defer writer.Flush()

      writer.Write([]string{
          "timestamp", "nombre", "url", "estado",
          "status_code", "latencia_ms", "error",
      })

      resultados := e.monitor.GenerarHistorial()
      for _, r := range resultados {
          errMsg := r.Error
          if errMsg == "" {
              errMsg = "-"
          }
          writer.Write([]string{
              r.Verificado.Format(time.RFC3339),
              r.Nombre,
              r.URL,
              r.Estado,
              fmt.Sprintf("%d", r.StatusCode),
              fmt.Sprintf("%d", r.LatenciaMs),
              errMsg,
          })
      }

      return writer.Error()
  }

  func (e *Exporter) ExportarReporteTexto(ruta string) error {
      reporte := e.monitor.GenerarReporte()

      f, err := os.Create(ruta)
      if err != nil {
          return err
      }
      defer f.Close()

      fmt.Fprintf(f, "===================================\n")
      fmt.Fprintf(f, "  REPORTE DE MONITOREO\n")
      fmt.Fprintf(f, "  Generado: %s\n", time.Now().Format(time.RFC1123))
      fmt.Fprintf(f, "===================================\n\n")
      fmt.Fprintf(f, "Total APIs: %d\n", reporte.Total)
      fmt.Fprintf(f, "Exitosas:   %d\n", reporte.Exitosos)
      fmt.Fprintf(f, "Fallidas:   %d\n", reporte.Fallidos)
      fmt.Fprintf(f, "Disponibilidad: %.1f%%\n\n",
          float64(reporte.Exitosos)/float64(reporte.Total)*100)

      fmt.Fprintf(f, "Detalle:\n")
      fmt.Fprintf(f, "--------\n")
      for _, r := range reporte.Resultados {
          estado := "✅"
          if r.Estado != "ok" {
              estado = "❌"
          }
          fmt.Fprintf(f, "%s %s (%dms)\n", estado, r.Nombre, r.LatenciaMs)
          if r.Error != "" {
              fmt.Fprintf(f, "   Error: %s\n", r.Error)
          }
      }

      return nil
  }
  ```

  ```go
  package main

  import (
      "flag"
      "log"
      "os"
      "os/signal"

  func main() {
      configPath := flag.String("config", "config.json", "Ruta al archivo de configuración")
      flag.Parse()

      log.SetFlags(log.LstdFlags | log.Lshortfile)
      log.Println("=== API MONITOR - GO MASTER BOOK ===")

      // Cargar configuración
      monitor, err := NewMonitor(*configPath)
      if err != nil {
          log.Fatalf("Error al iniciar monitor: %v", err)
      }
      log.Printf("Configuración cargada: %d APIs, puerto %d",
          len(monitor.GetConfig().APIs),
          monitor.GetConfig().Puerto)

      // Iniciar monitoreo en segundo plano
      monitor.Iniciar()

      // Iniciar servidor HTTP
      server := NewServer(monitor)

      // Manejar señales para graceful shutdown
      sigChan := make(chan os.Signal, 1)
      signal.Notify(sigChan, syscall.SIGINT, syscall.SIGTERM)

      go func() {
          <-sigChan
          log.Println("Apagando servidor...")
          os.Exit(0)
      }()

      server.Iniciar()
  }
  ```

  ```go
  func TestMiddlewareAuth(t *testing.T) {
      handler := http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
          w.Write([]byte(`{"mensaje":"ok"}`))
      })

      middleware := Auth("token123")(handler)

      // Sin token
      req := httptest.NewRequest("GET", "/test", nil)
      rec := httptest.NewRecorder()
      middleware.ServeHTTP(rec, req)
      if rec.Code != http.StatusUnauthorized {
          t.Errorf("esperado 401, obtenido %d", rec.Code)
      }

      // Con token correcto
      req = httptest.NewRequest("GET", "/test", nil)
      req.Header.Set("Authorization", "Bearer token123")
      rec = httptest.NewRecorder()
      middleware.ServeHTTP(rec, req)
      if rec.Code != http.StatusOK {
          t.Errorf("esperado 200, obtenido %d", rec.Code)
      }

      tests := []struct {
          ruta   string

          {"/api/v1/status", 200, false},

          {"/api/v1/report", 200, true},
          {"/api/v1/history", 200, true},
          {"/api/v1/config", 200, true},

          req := httptest.NewRequest("GET", tt.ruta, nil)
          if tt.auth {
              req.Header.Set("Authorization", "Bearer test-token")
          }
          rec := httptest.NewRecorder()
          server.Handler().ServeHTTP(rec, req)

          if rec.Code != tt.codigo {
              t.Errorf("%s: esperado %d, obtenido %d", tt.ruta, tt.codigo, rec.Code)
          }
      }
  ```

  PREGUNTAS FRECUENTES (FAQ) DEL MÓDULO
  ----------------------------------------------------------------------------

  ❓ ¿Cómo elijo entre bufio.Scanner y bufio.Reader?
  R: Scanner para texto línea por línea; Reader para control
  binario o de bytes.

  ❓ ¿Cuándo usar Marshal vs Encoder?
  R: Marshal para []byte en memoria; Encoder para streams.

  ❓ ¿ServeMux o Chi?
  R: ServeMux para proyectos simples (Go 1.22+); Chi para
  proyectos con middleware y organización compleja.

  ❓ ¿Cómo asegurar mi API?
  R: Usa middleware de autenticación, valida entradas, limita
  tamaños, configura timeouts, usa HTTPS.
