================================================================================

¿Qué es React?
--------------
React es una biblioteca de JavaScript para construir interfaces de usuario,
creada por Facebook (ahora Meta) en 2013. Se basa en componentes reutilizables
y un modelo de renderizado eficiente mediante el Virtual DOM. Es el framework
frontend más popular del mundo.

Características principales:
- Componentes: piezas de UI autónomas y reutilizables.
- JSX: extensión de JavaScript que permite escribir HTML dentro de JS.
- Estado (state) y propiedades (props) para manejar datos.
- Hooks (a partir de 16.8) para manejar estado y efectos sin clases.
- Flujo de datos unidireccional (de padre a hijo).
- Virtual DOM para actualizaciones eficientes.
- Gran ecosistema: React Router, Redux, Next.js, etc.

Filosofía: "Learn once, write anywhere" - Aprende React y úsalo en web, móvil
(React Native) y escritorio.

Versión actual: React 18+ (2022+) con nuevas características como Concurrent
Rendering y Transitions.

CAPITULO 5: ESTADO Y HOOKS (USESTATE, USEEFFECT)
================================================================================

useState (manejo de estado local):
----------------------------------
import { useState } from 'react';

function Contador() {
    const [contador, setContador] = useState(0); // valor inicial 0

    const incrementar = () => {
        setContador(contador + 1);
    };

    const decrementar = () => {
        setContador(contador - 1);
    };

    return (
        <div>
            <p>Contador: {contador}</p>
            <button onClick={incrementar}>+1</button>
            <button onClick={decrementar}>-1</button>
        </div>
    );
}

Estado con objeto:
------------------
const [usuario, setUsuario] = useState({ nombre: "", edad: 0 });

// Actualizar propiedad (preservando el resto):
setUsuario({ ...usuario, nombre: "Ana" });

// Usar el estado anterior (para actualizaciones seguras):
setContador(prev => prev + 1);

Estado con array:
-----------------
const [tareas, setTareas] = useState([]);

// Agregar:
setTareas([...tareas, nuevaTarea]);

// Eliminar:
setTareas(tareas.filter(t => t.id !== id));

// Actualizar:
setTareas(tareas.map(t => t.id === id ? { ...t, completada: true } : t));

useEffect (efectos secundarios):
--------------------------------
import { useEffect } from 'react';

// Ejecutar al montar (una vez):
useEffect(() => {
    console.log("Componente montado");
    // Función de limpieza (opcional):
    return () => {
        console.log("Componente desmontado");
    };
}, []); // Array vacío = solo una vez

// Ejecutar cuando cambia una dependencia:
useEffect(() => {
    document.title = `Contador: ${contador}`;
}, [contador]); // Se ejecuta cada vez que contador cambia

// Ejecutar en cada renderizado (sin dependencias):
useEffect(() => {
    console.log("Renderizado");
});

// Efecto con limpieza (ej: suscripciones, timers):
useEffect(() => {
    const timer = setInterval(() => {
        console.log("Tick");
    }, 1000);
    return () => clearInterval(timer);
}, []);

function Input() {

function Formulario() {

            <input value={texto} onChange={(e) => setTexto(e.target.value)} />

        </form>

Renderizado condicional:

// Con if:
function Mensaje({ usuario }) {
    if (!usuario) return <p>No hay usuario</p>;
    return <p>Bienvenido {usuario.nombre}</p>;
}

// Con operador ternario:
return <div>{usuario ? <Saludo /> : <Login />}</div>;

// Con && (cuando la primera parte es true):
return <div>{isAdmin && <PanelAdmin />}</div>;

// Con switch o múltiples condiciones:
let contenido;
switch (estado) {
    case 'cargando': contenido = <Spinner />; break;
    case 'error': contenido = <Error />; break;
    default: contenido = <Datos />;
}
return <div>{contenido}</div>;

Renderizar listas (map):
------------------------
function Lista({ items }) {
    return (
        <ul>
            {items.map((item) => (
                <li key={item.id}>{item.nombre}</li>
            ))}
        </ul>
    );
}
// Siempre usar una key única y estable (no usar index a menos que sea necesario)

// Lista con filtrado:
function ListaFiltrada({ items, filtro }) {
    const filtrados = items.filter(item => item.nombre.includes(filtro));
    return (
        <ul>
            {filtrados.map(item => (
                <li key={item.id}>{item.nombre}</li>
            ))}
        </ul>
    );
}

CAPITULO 9: CONTEXT API (ESTADO GLOBAL)
================================================================================

Context permite compartir estado sin prop drilling (pasar props por múltiples niveles).

Crear contexto:
---------------
import { createContext, useContext } from 'react';

const ThemeContext = createContext();

Proveedor (en el componente padre):
-----------------------------------
function App() {
    const [theme, setTheme] = useState('light');

    return (
        <ThemeContext.Provider value={{ theme, setTheme }}>
            <Componente />
        </ThemeContext.Provider>
    );
}

Consumir con useContext:
------------------------
function Componente() {
    const { theme, setTheme } = useContext(ThemeContext);

    const toggleTheme = () => {
        setTheme(theme === 'light' ? 'dark' : 'light');
    };

    return (
        <div className={theme}>
            <p>Tema actual: {theme}</p>
            <button onClick={toggleTheme}>Cambiar tema</button>
        </div>
    );
}

Context con valor por defecto y proveedor opcional:
---------------------------------------------------
const UserContext = createContext(null);

function App() {
    const [user, setUser] = useState(null);
    return (
        <UserContext.Provider value={user}>
            <Profile />
        </UserContext.Provider>
    );
}

function Profile() {
    const user = useContext(UserContext);
    if (!user) return <p>No autenticado</p>;
    return <p>Bienvenido {user.nombre}</p>;
}

Context para múltiples valores (recomendado crear contextos separados):

useReducer (para estados complejos con múltiples acciones):
-----------------------------------------------------------
import { useReducer } from 'react';

// Estado inicial
const initialState = { count: 0 };

// Reducer (función que maneja acciones)
function reducer(state, action) {
    switch (action.type) {
        case 'incrementar':
            return { count: state.count + 1 };
        case 'decrementar':
            return { count: state.count - 1 };
        case 'reset':
            return { count: 0 };
        default:
            return state;
    }
}

function Contador() {
    const [state, dispatch] = useReducer(reducer, initialState);

    return (
        <div>
            <p>Contador: {state.count}</p>
            <button onClick={() => dispatch({ type: 'incrementar' })}>+</button>
            <button onClick={() => dispatch({ type: 'decrementar' })}>-</button>
            <button onClick={() => dispatch({ type: 'reset' })}>Reset</button>
        </div>
    );
}

useCallback (memoización de funciones - evita recrear funciones innecesariamente):
----------------------------------------------------------------------------------
import { useCallback } from 'react';

function Componente({ onSave }) {
    // Esta función solo se recrea si cambian las dependencias
    const handleClick = useCallback(() => {
        onSave('datos');
    }, [onSave]);

    return <button onClick={handleClick}>Guardar</button>;
}

useMemo (memoización de valores - evita cálculos costosos):
-----------------------------------------------------------
import { useMemo } from 'react';

function Componente({ items, filtro }) {
    // Solo recalcula si items o filtro cambian
    const itemsFiltrados = useMemo(() => {
        console.log("Filtrando...");
        return items.filter(item => item.nombre.includes(filtro));
    }, [items, filtro]);

    return <Lista items={itemsFiltrados} />;
}

useRef (referencias a elementos DOM y valores mutables):
--------------------------------------------------------
import { useRef } from 'react';

function Input() {
    const inputRef = useRef(null);

    const focusInput = () => {
        inputRef.current.focus();  // Enfoca el input
        inputRef.current.value = "Enfocado!";
    };

    return (
        <>
            <input ref={inputRef} type="text" />
            <button onClick={focusInput}>Enfocar</button>
        </>
    );
}

// useRef para valores mutables (sin causar rerender):
const timerRef = useRef(null);
useEffect(() => {
    timerRef.current = setInterval(() => {
        console.log("Tick");
    }, 1000);
    return () => clearInterval(timerRef.current);
}, []);

Custom Hook para peticiones (reutilizable):
-------------------------------------------
function useFetch(url) {
    const [data, setData] = useState(null);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);

    useEffect(() => {
        setLoading(true);
        axios.get(url)
            .then(res => {
                setData(res.data);
                setError(null);
            })
            .catch(err => setError(err.message))
            .finally(() => setLoading(false));
    }, [url]);

    return { data, loading, error };
}

// Uso:

CAPITULO 14: BUENAS PRÁCTICAS Y PATRONES
================================================================================

1. Mantén componentes pequeños y enfocados (Single Responsibility).
2. Usa nombres descriptivos para componentes y funciones.
3. Desestructura props en la firma de la función.
4. Utiliza hooks para lógica reutilizable (custom hooks).
5. Evita efectos secundarios en el renderizado (useEffect para eso).
6. Normaliza el estado, evita duplicación de datos.
7. Optimiza con useMemo/useCallback solo cuando sea necesario (no prematuramente).

function useLocalStorage(key, initialValue) {
    const [value, setValue] = useState(() => {
        const stored = localStorage.getItem(key);
        return stored ? JSON.parse(stored) : initialValue;
    });

    useEffect(() => {
        localStorage.setItem(key, JSON.stringify(value));
    }, [key, value]);

    return [value, setValue];
}

Context de autenticación (AuthContext.jsx):
-------------------------------------------
import { createContext, useContext, useState, useEffect } from 'react';
import api from '../services/api';

const AuthContext = createContext();

export function AuthProvider({ children }) {
    const [user, setUser] = useState(null);
    const [loading, setLoading] = useState(true);

    useEffect(() => {
        const token = localStorage.getItem('token');
        if (token) {
            // Verificar token con el backend
            api.get('/auth/me')
                .then(res => setUser(res.data))
                .catch(() => localStorage.removeItem('token'))
                .finally(() => setLoading(false));
        } else {
            setLoading(false);
        }
    }, []);

    const login = async (email, password) => {
        const res = await api.post('/auth/login', { email, password });
        localStorage.setItem('token', res.data.token);
        setUser(res.data.user);
        return res.data;
    };

    const register = async (nombre, email, password) => {
        const res = await api.post('/auth/register', { nombre, email, password });
        localStorage.setItem('token', res.data.token);
        setUser(res.data.user);
        return res.data;
    };

    const logout = () => {
        localStorage.removeItem('token');
        setUser(null);
    };

    return (
        <AuthContext.Provider value={{ user, loading, login, register, logout }}>
            {children}
        </AuthContext.Provider>
    );
}

export const useAuth = () => useContext(AuthContext);

import { Navigate } from 'react-router-dom';
import { useAuth } from '../contexts/AuthContext';

function PrivateRoute({ children }) {
    const { user, loading } = useAuth();
    if (loading) return <p>Cargando...</p>;
    return user ? children : <Navigate to="/login" />;
}
