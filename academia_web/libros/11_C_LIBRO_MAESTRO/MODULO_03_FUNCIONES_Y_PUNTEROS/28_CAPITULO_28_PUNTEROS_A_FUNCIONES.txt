================================================================================
1. ¿QUÉ ES UN PUNTERO A FUNCIÓN?
================================================================================

Un puntero a función almacena la dirección de memoria donde
comienza el código de una función. Permite llamar a la función
sin usar su nombre directamente.

DECLARACIÓN:

  tipo_retorno (*nombre_puntero)(tipos_parametros);

Ejemplo:
  int (*op)(int, int);
  op es un puntero a una función que recibe dos ints y retorna int.

--------------------------------------------------------------------------------
  #include <stdio.h>

  int suma(int a, int b) { return a + b; }

  int main(void) {
      // Declarar puntero a funcion
      int (*pf)(int, int);

      // Asignar direccion de la funcion
      pf = suma;  // o pf = &suma;

      // Llamar a la funcion a traves del puntero
      int resultado = pf(10, 5); // o (*pf)(10, 5)
      printf("Resultado: %d\n", resultado); // 15

      return 0;
  }
--------------------------------------------------------------------------------

SINTAXIS DE LLAMADA:
  pf(10, 5)    → sintaxis directa (preferida, C99+)
  (*pf)(10, 5) → sintaxis clásica

Ambas son equivalentes.

================================================================================
2. DECLARACIÓN EN DETALLE
--------------------------------------------------------------------------------

Desglose de la declaración:

  int (*pf)(int, int);
  ^    ^   ^  ^      ^
  |    |   |  |      |
  |    |   |  |      +-- parámetros de la función
  |    |   |  +--------- lista de parámetros
  |    |   +------------ nombre del puntero
  |    +--------------- * indica que es puntero
  +------------------- tipo de retorno

Los PARÉNTESIS (*pf) son OBLIGATORIOS:
  int *pf(int, int);  // pf es FUNCIÓN que retorna int*
  int (*pf)(int, int); // pf es PUNTERO A FUNCIÓN que retorna int

EJEMPLOS DE DECLARACIONES:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <math.h>

  void (*p1)(void);           // puntero a void f(void)
  int (*p2)(void);            // puntero a int f(void)
  double (*p3)(double);       // puntero a double f(double)
  const char* (*p4)(int);     // puntero a const char* f(int)
  int (*p5)(int, int);        // puntero a int f(int, int)

  int main(void) {
      p3 = sqrt;
      printf("sqrt(9) = %.2f\n", p3(9.0));

      p2 = rand;
      printf("rand = %d\n", p2());

      return 0;
  }
--------------------------------------------------------------------------------

3. CALLBACKS: FUNCIONES COMO PARÁMETROS
--------------------------------------------------------------------------------

Un callback es una función que se pasa como argumento a otra
función para que sea llamada en el momento adecuado.

EJEMPLO CLÁSICO:

--------------------------------------------------------------------------------
  #include <stdio.h>

  // Operación que recibe un callback
  int operar(int a, int b, int (*callback)(int, int)) {
      printf("Operando %d y %d...\n", a, b);
      return callback(a, b);
  }

  // Callbacks posibles
  int sumar(int x, int y) { return x + y; }
  int restar(int x, int y) { return x - y; }
  int multiplicar(int x, int y) { return x * y; }
  int dividir(int x, int y) { return y != 0 ? x / y : 0; }

  int main(void) {
      printf("Suma: %d\n", operar(10, 5, sumar));
      printf("Resta: %d\n", operar(10, 5, restar));
      printf("Multi: %d\n", operar(10, 5, multiplicar));
      printf("Div: %d\n", operar(10, 5, dividir));
      return 0;
  }
--------------------------------------------------------------------------------

CALLBACK CON DATOS ADICIONALES (void* context):

--------------------------------------------------------------------------------
  #include <stdio.h>

  void recorrer(int arr[], int n,
                void (*accion)(int, void*),
                void* contexto) {
      for (int i = 0; i < n; i++) {
          accion(arr[i], contexto);
      }
  }

  void imprimir_elemento(int valor, void* ctx) {
      printf("%d ", valor);
  }

  void acumular(int valor, void* ctx) {
      int* suma = (int*)ctx;
      *suma += valor;
  }

  int main(void) {
      int datos[] = {10, 20, 30, 40, 50};
      int n = sizeof(datos) / sizeof(datos[0]);

      recorrer(datos, n, imprimir_elemento, NULL);
      printf("\n");

      int total = 0;
      recorrer(datos, n, acumular, &total);
      printf("Suma: %d\n", total); // 150
      return 0;
  }
--------------------------------------------------------------------------------

4. USO DE QSORT CON PUNTEROS A FUNCIÓN
--------------------------------------------------------------------------------

qsort() es la función de ordenamiento de la biblioteca estándar.
Su prototipo:

  void qsort(void* base, size_t nmemb, size_t size,
             int (*compar)(const void*, const void*));

El cuarto parámetro es un puntero a función comparadora.

ORDENAR ENTEROS:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <stdlib.h>

  int compar_int(const void* a, const void* b) {
      int ia = *(const int*)a;
      int ib = *(const int*)b;
      if (ia < ib) return -1;
      if (ia > ib) return 1;
      return 0;
  }

  int main(void) {
      int arr[] = {42, 15, 8, 23, 4, 16, 30};
      int n = sizeof(arr) / sizeof(arr[0]);

      qsort(arr, n, sizeof(int), compar_int);

      for (int i = 0; i < n; i++)
          printf("%d ", arr[i]);
      printf("\n");
      return 0;
  }
--------------------------------------------------------------------------------

ORDENAR ENTEROS (versión compacta):

--------------------------------------------------------------------------------
  int compar_int(const void* a, const void* b) {
      return *(const int*)a - *(const int*)b;
  }
--------------------------------------------------------------------------------

ORDENAR DESCENDENTE:

--------------------------------------------------------------------------------
  int compar_desc(const void* a, const void* b) {
      return *(const int*)b - *(const int*)a;
  }
  // Solo cambia el orden de a y b
--------------------------------------------------------------------------------

ORDENAR STRINGS:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <stdlib.h>
  #include <string.h>

  int compar_string(const void* a, const void* b) {
      const char* sa = *(const char**)a;
      const char* sb = *(const char**)b;
      return strcmp(sa, sb);
  }

  int main(void) {
      const char* nombres[] = {
          "Carlos", "Ana", "Beatriz", "David"
      };
      int n = sizeof(nombres) / sizeof(nombres[0]);

      qsort(nombres, n, sizeof(const char*), compar_string);

      for (int i = 0; i < n; i++)
          printf("%s\n", nombres[i]);
      return 0;
  }
--------------------------------------------------------------------------------

ORDENAR ESTRUCTURAS:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <stdlib.h>
  #include <string.h>

  typedef struct {
      char nombre[50];
      int edad;
  } Persona;

  int compar_por_edad(const void* a, const void* b) {
      const Persona* pa = (const Persona*)a;
      const Persona* pb = (const Persona*)b;
      return pa->edad - pb->edad;
  }

  int compar_por_nombre(const void* a, const void* b) {
      const Persona* pa = (const Persona*)a;
      const Persona* pb = (const Persona*)b;
      return strcmp(pa->nombre, pb->nombre);
  }

  int main(void) {
      Persona gente[] = {
          {"Carlos", 30},
          {"Ana", 25},
          {"Beatriz", 35},
          {"David", 28}
      };
      int n = sizeof(gente) / sizeof(gente[0]);

      qsort(gente, n, sizeof(Persona), compar_por_edad);
      for (int i = 0; i < n; i++)
          printf("%s: %d\n", gente[i].nombre, gente[i].edad);

      printf("---\n");
      qsort(gente, n, sizeof(Persona), compar_por_nombre);
      for (int i = 0; i < n; i++)
          printf("%s: %d\n", gente[i].nombre, gente[i].edad);
      return 0;
  }
--------------------------------------------------------------------------------

5. TYPEDEF PARA PUNTEROS A FUNCIÓN
--------------------------------------------------------------------------------

typedef simplifica la sintaxis de punteros a función.

SIN TYPEDEF (cada declaración es verbosa):

  void ordenar(int arr[], int n,
               int (*compar)(const void*, const void*));

CON TYPEDEF:

  typedef int (*Comparador)(const void*, const void*);

  void ordenar(int arr[], int n, Comparador compar);

EJEMPLO COMPLETO:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <stdlib.h>

  typedef int (*Comparador)(const void*, const void*);

  int ascendente(const void* a, const void* b) {
      return *(const int*)a - *(const int*)b;
  }

  int descendente(const void* a, const void* b) {
      return *(const int*)b - *(const int*)a;
  }

  void imprimir(int arr[], int n) {
      for (int i = 0; i < n; i++)
          printf("%d ", arr[i]);
      printf("\n");
  }

  int main(void) {
      int datos[] = {5, 3, 8, 1, 9, 2, 7};
      int n = sizeof(datos) / sizeof(datos[0]);

      Comparador comps[] = {ascendente, descendente};
      const char* nombres[] = {"ASC", "DESC"};

      for (int i = 0; i < 2; i++) {
          qsort(datos, n, sizeof(int), comps[i]);
          printf("%s: ", nombres[i]);
          imprimir(datos, n);
      }
      return 0;
  }
--------------------------------------------------------------------------------

MÁS TYPEDEFS ÚTILES:

--------------------------------------------------------------------------------
  #include <math.h>

  typedef double (*FuncionUnaria)(double);
  typedef int    (*FuncionBinaria)(int, int);
  typedef void   (*Accion)(int);
  typedef int    (*Predicado)(int);

  double aplicar(FuncionUnaria f, double x) {
      return f(x);
  }

  int main(void) {
      FuncionUnaria ops[] = {sqrt, sin, cos, fabs};

      for (int i = 0; i < 4; i++) {
          printf("%.2f ", aplicar(ops[i], -9.0));
      }
      // sqrt(-9)=NaN  sin(-9)=-0.41  cos(-9)=-0.91  fabs(-9)=9
      return 0;
  }
--------------------------------------------------------------------------------

6. TABLAS DE DESPACHO (DISPATCH TABLES)
--------------------------------------------------------------------------------

Un arreglo de punteros a función permite seleccionar qué función
ejecutar basado en un índice, simulando polimorfismo.

CALCULADORA CON TABLA:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <stdbool.h>

  // Operaciones
  int suma(int a, int b)     { return a + b; }
  int resta(int a, int b)    { return a - b; }
  int mult(int a, int b)     { return a * b; }
  int division(int a, int b) { return b != 0 ? a / b : 0; }
  int modulo(int a, int b)   { return b != 0 ? a % b : 0; }

  int main(void) {
      // Tabla de despacho
      int (*operaciones[])(int, int) = {
          suma, resta, mult, division, modulo
      };
      const char* nombres[] = {
          "Suma", "Resta", "Multiplicacion",
          "Division", "Modulo"
      };
      int num_ops = sizeof(operaciones) / sizeof(operaciones[0]);

      printf("Calculadora de tabla:\n");
      for (int i = 0; i < num_ops; i++) {
          printf("%s(10, 3) = %d\n",
                 nombres[i], operaciones[i](10, 3));
      }
      return 0;
  }
--------------------------------------------------------------------------------

MENÚ INTERACTIVO:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <stdlib.h>

  typedef int (*OpBinaria)(int, int);

  int suma(int a, int b) { return a + b; }
  int resta(int a, int b) { return a - b; }
  int mult(int a, int b) { return a * b; }
  int dividir(int a, int b) { return b ? a / b : 0; }

  int main(void) {
      OpBinaria ops[] = {suma, resta, mult, dividir};
      const char* menu[] = {
          "Salir", "Sumar", "Restar",
          "Multiplicar", "Dividir"
      };
      int opcion, a, b;

      while (1) {
          printf("\n--- CALCULADORA ---\n");
          for (int i = 1; i <= 4; i++)
              printf("%d. %s\n", i, menu[i]);
          printf("0. %s\n", menu[0]);
          printf("Opcion: ");
          scanf("%d", &opcion);

          if (opcion == 0) break;
          if (opcion < 1 || opcion > 4) continue;

          printf("a = "); scanf("%d", &a);
          printf("b = "); scanf("%d", &b);
          printf("Resultado: %d\n", ops[opcion - 1](a, b));
      }
      return 0;
  }
--------------------------------------------------------------------------------

7. PUNTEROS A FUNCIÓN COMO MIEMBROS DE STRUCT
--------------------------------------------------------------------------------

Podemos almacenar punteros a función en estructuras, simulando
métodos de objetos.

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <string.h>
  #include <ctype.h>

  typedef struct {
      const char* nombre;
      int (*transformar)(int);
  } Transformacion;

  int main(void) {
      Transformacion trans[] = {
          {"toupper", toupper},
          {"tolower", tolower},
          {"toascii", toascii}
      };
      int n = sizeof(trans) / sizeof(trans[0]);

      char c = 'm';
      for (int i = 0; i < n; i++) {
          printf("%s('%c') = '%c'\n",
                 trans[i].nombre, c,
                 trans[i].transformar(c));
      }
      return 0;
  }
--------------------------------------------------------------------------------

OBJETO SIMULADO (Struct con métodos):

--------------------------------------------------------------------------------
  #include <stdio.h>

  typedef struct Calculadora {
      int memoria;
      int (*sumar)(struct Calculadora*, int);
      int (*restar)(struct Calculadora*, int);
      void (*reset)(struct Calculadora*);
  } Calculadora;

  int calc_sumar(Calculadora* self, int valor) {
      self->memoria += valor;
      return self->memoria;
  }

  int calc_restar(Calculadora* self, int valor) {
      self->memoria -= valor;
      return self->memoria;
  }

  void calc_reset(Calculadora* self) {
      self->memoria = 0;
  }

  Calculadora crear_calc(void) {
      Calculadora c;
      c.memoria = 0;
      c.sumar = calc_sumar;
      c.restar = calc_restar;
      c.reset = calc_reset;
      return c;
  }

  int main(void) {
      Calculadora calc = crear_calc();
      calc.sumar(&calc, 10);
      calc.sumar(&calc, 20);
      printf("Valor: %d\n", calc.restar(&calc, 5)); // 25
      calc.reset(&calc);
      printf("Reset: %d\n", calc.memoria); // 0
      return 0;
  }
--------------------------------------------------------------------------------

8. ARREGLOS DE PUNTEROS A FUNCIÓN CON DIFERENTES FIRMAS
--------------------------------------------------------------------------------

Si las funciones tienen diferentes firmas, usamos void* y unión.

EJEMPLO: funciones con 0, 1 y 2 parámetros:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <stdarg.h>

  int cero_params(void) { return 0; }
  int un_param(int x) { return x; }
  int dos_params(int x, int y) { return x + y; }

  // Solucion con union (mas segura)
  typedef union {
      int (*f0)(void);
      int (*f1)(int);
      int (*f2)(int, int);
  } FuncionFlexible;

  typedef struct {
      const char* nombre;
      FuncionFlexible func;
      int num_params;
  } Entrada;

  int main(void) {
      Entrada tabla[] = {
          {"cero",  {.f0 = cero_params}, 0},
          {"uno",   {.f1 = un_param},    1},
          {"dos",   {.f2 = dos_params},  2}
      };
      int n = sizeof(tabla) / sizeof(tabla[0]);

      for (int i = 0; i < n; i++) {
          printf("%s: ", tabla[i].nombre);
          switch (tabla[i].num_params) {
              case 0: printf("%d\n", tabla[i].func.f0()); break;
              case 1: printf("%d\n", tabla[i].func.f1(42)); break;
              case 2: printf("%d\n", tabla[i].func.f2(10, 20)); break;
          }
      }
      return 0;
  }
--------------------------------------------------------------------------------

9. PUNTEROS A FUNCIÓN Y RECURSIÓN INDIRECTA
--------------------------------------------------------------------------------

Los punteros a función habilitan recursión indirecta dinámica:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <stdbool.h>

  void paso1(int*, bool*, void (*siguiente)(int*, bool*, void (*)(int*, bool*, void (*)(int*, bool*))));
  // Sintaxis complicada... mejor con typedef

  typedef void (*Estado)(int*, bool*);

  void estado_par(int* cont, bool* activo) {
      if (!*activo || *cont <= 0) { *activo = false; return; }
      printf("Par: %d\n", *cont);
      (*cont)--;
      // Automata: alterna entre par e impar
  }

  void estado_impar(int* cont, bool* activo) {
      if (!*activo || *cont <= 0) { *activo = false; return; }
      printf("Impar: %d\n", *cont);
      (*cont)--;
  }

  int main(void) {
      // Maquina de estados simple con punteros a funcion
      Estado estados[] = {NULL, NULL};
      estados[0] = estado_par;
      estados[1] = estado_impar;

      int contador = 5;
      bool activo = true;
      int turno = 0;

      while (activo) {
          estados[turno](&contador, &activo);
          turno = (contador % 2 == 0) ? 0 : 1;
      }
      return 0;
  }
--------------------------------------------------------------------------------

10. BÚSQUEDA BINARIA CON CALLBACK (bsearch)
--------------------------------------------------------------------------------

bsearch() también usa puntero a función comparadora:

--------------------------------------------------------------------------------
  #include <stdio.h>
  #include <stdlib.h>

  int compar_int(const void* a, const void* b) {
      return *(const int*)a - *(const int*)b;
  }

  int main(void) {
      int arr[] = {4, 8, 15, 16, 23, 42};
      int n = sizeof(arr) / sizeof(arr[0]);

      int clave = 16;
      int* resultado = bsearch(&clave, arr, n, sizeof(int),
                                compar_int);

      if (resultado != NULL) {
          printf("Encontrado: %d en posicion %ld\n",
                 *resultado, resultado - arr);
      } else {
          printf("No encontrado\n");
      }
      return 0;
  }
--------------------------------------------------------------------------------

11. PASANDO CONTEXTO A CALLBACKS (void*)
--------------------------------------------------------------------------------

El patrón void* para datos de contexto es muy común en bibliotecas:

--------------------------------------------------------------------------------
  #include <stdio.h>

  typedef void (*Callback)(int, void*);

  void generar(int inicio, int fin, int paso,
               Callback cb, void* ctx) {
      for (int i = inicio; i < fin; i += paso) {
          cb(i, ctx);
      }
  }

  void imprimir(int valor, void* ctx) {
      const char* prefijo = (const char*)ctx;
      printf("%s %d\n", prefijo, valor);
  }

  void contar_pares(int valor, void* ctx) {
      int* contador = (int*)ctx;
      if (valor % 2 == 0) (*contador)++;
  }

  int main(void) {
      generar(1, 11, 1, imprimir, "[NUM]");

      int pares = 0;
      generar(1, 101, 1, contar_pares, &pares);
      printf("Pares entre 1 y 100: %d\n", pares); // 50
      return 0;
  }
--------------------------------------------------------------------------------

12. ERRORES COMUNES CON PUNTEROS A FUNCIÓN
--------------------------------------------------------------------------------

ERROR 1: Olvidar los paréntesis en la declaración.

--------------------------------------------------------------------------------
  int *pf(int, int);  // pf es FUNCION que retorna int*
  int (*pf)(int, int); // pf es PUNTERO A FUNCION
--------------------------------------------------------------------------------

ERROR 2: Firmas incompatibles.

--------------------------------------------------------------------------------
  int suma(int, int);
  void (*pf)(int, int) = suma; // WARNING: tipos incompatibles
  // pf espera void, suma retorna int
--------------------------------------------------------------------------------

ERROR 3: Llamar sin paréntesis.

--------------------------------------------------------------------------------
  int (*pf)(int, int) = suma;
  // int r = pf;   // ERROR: pf no es int
  int r = pf(1, 2); // correcto
--------------------------------------------------------------------------------

ERROR 4: Asignar función sin el & (no es error pero confunde).

--------------------------------------------------------------------------------
  pf = suma;   // correcto (el nombre decae a puntero)
  pf = &suma;  // correcto (explícito)
  // Nunca pf = suma(); // ESTO LLAMA la funcion y asigna resultado
--------------------------------------------------------------------------------

ERROR 5: Usar typedef incorrectamente.

--------------------------------------------------------------------------------
  typedef int (*PF)(int); // PF es un tipo: puntero a funcion
  PF p = mi_funcion;       // correcto
  // PF pfunc();           // pfunc seria funcion que retorna PF
  PF* pp = &p;             // puntero a puntero a funcion (avanzado)
--------------------------------------------------------------------------------

RESUMEN DEL CAPÍTULO
================================================================================

  • Un puntero a función almacena la dirección de una función.
  • Declaración: tipo_retorno (*nombre)(parametros);
  • Llamada: pf(args) o (*pf)(args).
  • Callbacks: funciones pasadas como argumentos.
  • qsort() y bsearch() usan punteros a función para comparar.
  • typedef simplifica la sintaxis.
  • Tablas de despacho: arreglos de punteros a función.
  • void* permite pasar contexto a callbacks.
  • Los punteros a función permiten polimorfismo en C.

================================================================================
PREGUNTAS FRECUENTES (FAQ)
================================================================================

  P: ¿Cómo se lee una declaración compleja de puntero a función?
  R: De adentro hacia afuera: int (*pf)(int, int) →
     "pf es un puntero a una función que toma (int, int) y
     retorna int".

  P: ¿Por qué usar punteros a función en lugar de switch?
  R: Las tablas de despacho son más flexibles: se pueden
     modificar en tiempo de ejecución, agregar nuevas funciones
     sin cambiar el código, y pasar como argumentos.

  P: ¿Puedo tener un arreglo de punteros a función con
     diferentes firmas?
  R: Técnicamente no directamente. Necesitas usar uniones,
     void* para parámetros, o funciones con firmas compatibles.

  P: ¿Cómo funcionan los callbacks asíncronos en C?
  R: C no tiene soporte nativo de asincronía, pero puedes
     almacenar punteros a función para llamarlos más tarde
     (ej: signal(), pthread_create()).

  P: ¿Es seguro usar punteros a función?
  R: Sí, si se usan correctamente. El riesgo principal es
     llamar a una función con una firma incorrecta (undefined
     behavior).

  P: ¿Se puede modificar el código al que apunta un puntero
     a función?
  R: En la mayoría de sistemas, el código está en memoria de
     solo lectura. Modificarlo causa violación de segmento.
     Existen técnicas avanzadas (JIT) pero no son estándar.

DESAFÍO DEL ARQUITECTO
================================================================================

  Implementa un framework de plugins en C usando punteros a
  función. Requisitos:

    • Define una estructura Plugin con:
        - char nombre[50]
        - char version[10]
        - int (*inicializar)(void* contexto)
        - int (*ejecutar)(void* contexto)
        - void (*finalizar)(void)

    • Crea un gestor de plugins que:
        - Mantenga un arreglo de plugins (máximo 10).
        - Función registrar_plugin(Plugin p) para agregar.
        - Función ejecutar_todos() que ejecute cada plugin.
        - Función buscar_plugin(const char* nombre) que
          retorne un Plugin* o NULL.

    • Implementa 3 plugins concretos:
        - Plugin "Saludo": imprime "Hola desde plugin".
        - Plugin "Matematicas": recibe un int* y lo duplica.
        - Plugin "Logger": imprime fecha/hora al ejecutar.

    • main() debe registrar los plugins, ejecutarlos, y
      demostrar la búsqueda por nombre.

  Puntos extra:
    - Carga dinámica con dlopen() (Linux) o LoadLibrary()
      (Windows).
    - Configuración de plugins desde archivo.

CIERRE DEL CAPÍTULO
================================================================================

  Los punteros a función son una de las características más
  sofisticadas de C. Permiten escribir código genérico,
  extensible y elegante. Son la base de los callbacks, las
  tablas de despacho, y el polimorfismo en C. Dominarlos te
  coloca en la élite de los programadores de C.
