¡Bienvenido al Módulo V de C MASTER BOOK!

Hasta ahora has dominado variables, funciones, punteros y cadenas. Pero en el
mundo real los datos rara vez viajan solos. Un "cliente" tiene nombre, edad y
saldo. Un "punto" tiene coordenadas X e Y. Una "fecha" tiene día, mes y año.

Para modelar estas entidades del mundo real necesitamos algo más poderoso que
variables sueltas: necesitamos ESTRUCTURAS.

En este capítulo aprenderás a crear tus propios tipos de datos compuestos
usando la palabra clave `struct`. Las estructuras son el cimiento sobre el
que se construyen sistemas complejos en C. Dominarlas te abrirá las puertas
a listas enlazadas, árboles, archivos binarios y mucho más.

Prepárate para dar el salto de datos sueltos a datos con estructura.

--- INTRODUCCIÓN ---

Imagina que trabajas en un sistema de gestión bancaria. Necesitas almacenar
los datos de un cliente: su nombre (cadena), su edad (entero) y su saldo
(flotante). Sin estructuras tendrías tres variables separadas:

    char nombre[50];
    int edad;
    float saldo;

Esto funciona... hasta que tienes 1000 clientes. Necesitarías 3000 variables
o engorrosos arrays paralelos. Un desastre.

La solución es `struct`: un tipo de dato que agrupa múltiples variables
relacionadas bajo un mismo nombre.

--- DECLARACIÓN DE UNA ESTRUCTURA ---

La sintaxis básica es:

    struct etiqueta {
        tipo1 miembro1;
        tipo2 miembro2;
        // ...
    };

La `etiqueta` (tag) es opcional pero recomendada. Sirve para referirse al
tipo en otras partes del código. Los `miembros` son las variables que forman
parte de la estructura.

--- DECLARACIÓN Y DEFINICIÓN DE VARIABLES ---

Hay varias formas de crear variables de tipo estructura:

1. Declarar el tipo y luego la variable:

    struct Persona {
        char nombre[50];
        int edad;
        float altura;
    };

    struct Persona p1;

2. Declarar variable junto con el tipo:

    struct Persona {
        char nombre[50];
        int edad;
    } p1, p2;

3. Usando typedef (lo veremos a fondo en el capítulo 42):

    typedef struct {
        char nombre[50];
        int edad;
    } Persona;

    Persona p1;

--- INICIALIZACIÓN DE ESTRUCTURAS ---

Puedes inicializar una estructura al declararla usando llaves `{}`:

    struct Persona p1 = {"Ana García", 28, 1.65};

El orden de los valores debe coincidir con el orden de los miembros.

También puedes usar inicialización designada (C99):

    struct Persona p2 = {.edad = 30, .nombre = "Luis", .altura = 1.80};

--- OPERADOR PUNTO (.) ---

Para acceder a los miembros de una estructura usamos el operador punto:

    struct Persona p1;
    p1.edad = 28;
    strcpy(p1.nombre, "Ana García");
    p1.altura = 1.65;

    printf("Nombre: %s\n", p1.nombre);
    printf("Edad: %d\n", p1.edad);

--- TAMAÑO DE UNA ESTRUCTURA ---

El operador `sizeof` aplicado a una estructura devuelve su tamaño en bytes.
Importante: puede haber padding (relleno) entre miembros por alineación.

    printf("Tamaño de Persona: %zu bytes\n", sizeof(struct Persona));

--- COPY DE ESTRUCTURAS ---

Las estructuras se copian miembro a miembro con el operador de asignación:

    struct Persona p1 = {"Ana", 28, 1.65};
    struct Persona p2;

    p2 = p1;  // Copia miembro a miembro

Esto es válido y seguro para estructuras simples.

--- EJEMPLO COMPLETO ---

A continuación un programa completo que demuestra todo lo aprendido:

```c
#include <stdio.h>
#include <string.h>

// Declaración de la estructura
struct Libro {
    char titulo[100];
    char autor[50];
    int anio;
    float precio;
    int paginas;
};

int main() {
    // Declaración e inicialización
    struct Libro lib1 = {
        "C Programming: A Modern Approach",
        "K. N. King",
        2008,
        45.99,
        832
    };

    // Declaración y asignación miembro por miembro
    struct Libro lib2;
    strcpy(lib2.titulo, "El Lenguaje de Programacion C");
    strcpy(lib2.autor, "Brian W. Kernighan");
    lib2.anio = 1978;
    lib2.precio = 39.95;
    lib2.paginas = 272;

    // Mostrar datos
    printf("============= BIBLIOTECA =============\n\n");

    printf("Libro 1:\n");
    printf("  Titulo : %s\n", lib1.titulo);
    printf("  Autor  : %s\n", lib1.autor);
    printf("  Anio   : %d\n", lib1.anio);
    printf("  Precio : $%.2f\n", lib1.precio);
    printf("  Paginas: %d\n", lib1.paginas);

    printf("\nLibro 2:\n");
    printf("  Titulo : %s\n", lib2.titulo);
    printf("  Autor  : %s\n", lib2.autor);
    printf("  Anio   : %d\n", lib2.anio);
    printf("  Precio : $%.2f\n", lib2.precio);
    printf("  Paginas: %d\n", lib2.paginas);

    // Modificar un miembro
    lib2.precio = 34.50;
    printf("\nPrecio actualizado de lib2: $%.2f\n", lib2.precio);

    // Copiar estructura
    struct Libro lib3;
    lib3 = lib1;
    printf("\nCopia (lib3 = lib1): %s - $%.2f\n",
           lib3.titulo, lib3.precio);

    // Tamaño de la estructura
    printf("\nTamanio de struct Libro: %zu bytes\n",
           sizeof(struct Libro));
    printf("Tamanio de lib1: %zu bytes\n", sizeof(lib1));

    // Array de estructuras
    struct Libro biblioteca[3];
    biblioteca[0] = lib1;
    biblioteca[1] = lib2;
    biblioteca[2] = lib3;

    printf("\n--- Recorriendo array de estructuras ---\n");
    for (int i = 0; i < 3; i++) {
        printf("[%d] %s por %s\n",
               i + 1, biblioteca[i].titulo, biblioteca[i].autor);
    }

    return 0;
}
```

Salida del programa:

```
============= BIBLIOTECA =============

Libro 1:
  Titulo : C Programming: A Modern Approach
  Autor  : K. N. King
  Anio   : 2008
  Precio : $45.99
  Paginas: 832

Libro 2:
  Titulo : El Lenguaje de Programacion C
  Autor  : Brian W. Kernighan
  Anio   : 1978
  Precio : $39.95
  Paginas: 272

Precio actualizado de lib2: $34.50

Copia (lib3 = lib1): C Programming: A Modern Approach - $45.99

Tamanio de struct Libro: 168 bytes
Tamanio de lib1: 168 bytes

--- Recorriendo array de estructuras ---
[1] C Programming: A Modern Approach por K. N. King
[2] El Lenguaje de Programacion C por Brian W. Kernighan
[3] C Programming: A Modern Approach por K. N. King
```

--- PADDING Y ALINEACIÓN ---

El compilador puede añadir bytes de relleno entre miembros para alinearlos
en direcciones de memoria que optimicen el acceso. Esto se llama "padding".

```c
#include <stdio.h>

struct Ejemplo1 {
    char c;    // 1 byte
    int i;     // 4 bytes
    char d;    // 1 byte
};

struct Ejemplo2 {
    int i;     // 4 bytes
    char c;    // 1 byte
    char d;    // 1 byte
};

int main() {
    printf("Ejemplo1 (char,int,char): %zu bytes\n",
           sizeof(struct Ejemplo1));
    printf("Ejemplo2 (int,char,char): %zu bytes\n",
           sizeof(struct Ejemplo2));
    return 0;
}
```

Salida típica:
```
Ejemplo1 (char,int,char): 12 bytes
Ejemplo2 (int,char,char): 8 bytes
```

El orden de los miembros afecta el tamaño total por el padding.

--- BUENAS PRÁCTICAS ---

1. Agrupa los miembros del mismo tipo para minimizar el padding.
2. Usa nombres descriptivos para la estructura y sus miembros.
3. Inicializa siempre las estructuras (aunque sea con `= {0}`).
4. Prefiere `typedef` (capítulo 42) para no escribir `struct` cada vez.
5. No pases estructuras grandes por valor a funciones; usa punteros.

--- PASAR ESTRUCTURAS A FUNCIONES ---

Puedes pasar estructuras por valor (copia) o por referencia (puntero):

Por valor (lento para estructuras grandes):
```c
void imprimirLibro(struct Libro lib) {
    printf("%s - $%.2f\n", lib.titulo, lib.precio);
}
```

Por referencia (eficiente, permite modificar):
```c
void actualizarPrecio(struct Libro *lib, float nuevo) {
    lib->precio = nuevo;  // operador flecha ->
}
```

El operador `->` (flecha) equivale a `(*puntero).miembro`. Se usa cuando
tenemos un puntero a estructura.

```c
#include <stdio.h>
#include <string.h>

struct Producto {
    char nombre[50];
    float precio;
    int stock;
};

// Paso por valor
void mostrarProducto(struct Producto p) {
    printf("Producto: %s - $%.2f (stock: %d)\n",
           p.nombre, p.precio, p.stock);
}

// Paso por referencia (puntero)
void aplicarDescuento(struct Producto *p, float porcentaje) {
    p->precio *= (1.0f - porcentaje / 100.0f);
}

int main() {
    struct Producto prod = {"Laptop", 1200.00, 10};

    mostrarProducto(prod);
    aplicarDescuento(&prod, 15.0f);
    printf("Despues del descuento:\n");
    mostrarProducto(prod);

    return 0;
}
```

Salida:
```
Producto: Laptop - $1200.00 (stock: 10)
Despues del descuento:
Producto: Laptop - $1020.00 (stock: 10)
```

═══════════════════════════════════════════════════════════════════════════════
                    PREGUNTAS FRECUENTES (FAQ)
═══════════════════════════════════════════════════════════════════════════════

P: ¿Puedo tener una función dentro de una estructura?
R: No. En C las estructuras solo contienen datos, no funciones. Para
   eso necesitas C++ o lenguajes orientados a objetos. Puedes tener
   punteros a función como miembros.

P: ¿Puedo inicializar una estructura con `= {0}`?
R: Sí. Esto pone a cero todos los miembros (NULL para punteros, 0 para
   enteros, 0.0 para flotantes). Es una práctica recomendada.

P: ¿Las estructuras se pasan por valor o por referencia?
R: Por defecto se pasan por valor (se copia toda la estructura). Si
   quieres evitar la copia (recomendado para estructuras grandes) o
   modificar la original, pasa un puntero.

P: ¿Qué diferencia hay entre `struct` y `typedef`?
R: `struct` declara el tipo. `typedef` crea un alias para no tener que
   escribir `struct` cada vez. Lo veremos en el capítulo 42.

P: ¿Puedo comparar dos estructuras con `==`?
R: No directamente. El operador `==` no está definido para estructuras.
   Debes comparar miembro por miembro o usar `memcmp()` (con cuidado
   por el padding).

P: ¿Qué es el padding y por qué existe?
R: El padding son bytes de relleno que el compilador inserta para
   alinear los miembros en direcciones de memoria que optimicen el
   acceso. La CPU accede más rápido a datos alineados.

P: ¿Puedo tener un array dentro de una estructura?
R: Sí, completamente. De hecho es muy común. Un miembro puede ser un
   array de cualquier tipo, incluyendo `char nombre[50]`.

--- ESTRUCTURAS CON CAMPOS DE BITS ---

C permite especificar el numero exacto de bits para un miembro:

```c
#include <stdio.h>

typedef struct {
    unsigned int activo : 1;     // 1 bit: 0 o 1
    unsigned int permiso : 3;    // 3 bits: 0-7
    unsigned int tipo : 4;       // 4 bits: 0-15
    unsigned int id : 8;         // 8 bits: 0-255
} CampoBits;

int main() {
    CampoBits cb = {1, 5, 10, 200};

    printf("sizeof(CampoBits) = %zu bytes\n", sizeof(CampoBits));
    printf("activo=%u, permiso=%u, tipo=%u, id=%u\n",
           cb.activo, cb.permiso, cb.tipo, cb.id);

    // Asignacion directa
    cb.permiso = 7;  // maximo 7 (3 bits = 111 binario)
    printf("permiso actualizado: %u\n", cb.permiso);

    return 0;
}
```

--- ESTRUCTURAS CON FUNCIONES (PUNTEROS A FUNCION) ---

Las estructuras no pueden tener funciones, pero si punteros a funcion:

```c
#include <stdio.h>

typedef struct {
    int id;
    char nombre[50];
    float saldo;
    void (*mostrar)(void*);  // Puntero a funcion generico
    float (*calcularBono)(float);
} Empleado;

float bonoEstandar(float saldo) {
    return saldo * 0.10f;
}

float bonoEjecutivo(float saldo) {
    return saldo * 0.25f;
}

void mostrarEmpleado(void *ptr) {
    Empleado *e = (Empleado*)ptr;
    printf("ID: %d, Nombre: %s, Saldo: $%.2f\n",
           e->id, e->nombre, e->saldo);
}

int main() {
    Empleado e1 = {1, "Ana", 50000, mostrarEmpleado, bonoEstandar};
    Empleado e2 = {2, "Carlos", 120000, mostrarEmpleado, bonoEjecutivo};

    e1.mostrar(&e1);
    printf("Bono: $%.2f\n\n", e1.calcularBono(e1.saldo));

    e2.mostrar(&e2);
    printf("Bono: $%.2f\n", e2.calcularBono(e2.saldo));

    return 0;
}
```

--- ALINEACION Y PADDING AVANZADO ---

Se puede controlar el padding con directivas del compilador:

```c
#include <stdio.h>

// Sin atributos: con padding
typedef struct {
    char a;
    int b;
    char c;
} Normal;

// Con packing: sin padding
typedef struct __attribute__((packed)) {
    char a;
    int b;
    char c;
} Empaquetado;

int main() {
    printf("Normal:     %zu bytes (con padding)\n", sizeof(Normal));
    printf("Empaquetado: %zu bytes (sin padding)\n", sizeof(Empaquetado));

    // Desplazamiento de cada miembro
    Normal n;
    printf("\nDesplazamientos (Normal):\n");
    printf("  a: %zu\n", (size_t)&n.a - (size_t)&n);
    printf("  b: %zu\n", (size_t)&n.b - (size_t)&n);
    printf("  c: %zu\n", (size_t)&n.c - (size_t)&n);

    Empaquetado e;
    printf("\nDesplazamientos (Empaquetado):\n");
    printf("  a: %zu\n", (size_t)&e.a - (size_t)&e);
    printf("  b: %zu\n", (size_t)&e.b - (size_t)&e);
    printf("  c: %zu\n", (size_t)&e.c - (size_t)&e);

    return 0;
}
```

--- ESTRUCTURAS Y MEMORIA DINAMICA ---

Puedes crear estructuras en memoria dinamica con malloc:

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char *nombre;   // Cadena en heap
    int edad;
    float *calificaciones;  // Array dinamico
    int numCalificaciones;
} Estudiante;

Estudiante* crearEstudiante(const char *nombre, int edad, int numCalif) {
    Estudiante *e = malloc(sizeof(Estudiante));
    if (!e) return NULL;

    e->nombre = malloc(strlen(nombre) + 1);
    if (!e->nombre) { free(e); return NULL; }
    strcpy(e->nombre, nombre);

    e->edad = edad;
    e->numCalificaciones = numCalif;
    e->calificaciones = malloc(numCalif * sizeof(float));
    if (!e->calificaciones) { free(e->nombre); free(e); return NULL; }

    return e;
}

void destruirEstudiante(Estudiante *e) {
    if (e) {
        free(e->nombre);
        free(e->calificaciones);
        free(e);
    }
}

void mostrarEstudiante(const Estudiante *e) {
    printf("Nombre: %s\n", e->nombre);
    printf("Edad: %d\n", e->edad);
    printf("Calificaciones: ");
    for (int i = 0; i < e->numCalificaciones; i++) {
        printf("%.1f ", e->calificaciones[i]);
    }
    printf("\n");
}

int main() {
    Estudiante *e = crearEstudiante("Maria Lopez", 22, 4);
    if (!e) { printf("Error de memoria\n"); return 1; }

    e->calificaciones[0] = 9.0;
    e->calificaciones[1] = 8.5;
    e->calificaciones[2] = 9.5;
    e->calificaciones[3] = 10.0;

    mostrarEstudiante(e);
    printf("Tamanio del struct: %zu bytes\n", sizeof(Estudiante));

    destruirEstudiante(e);
    return 0;
}
```

--- ESTRUCTURAS CON MIEMBROS DE TIPO UNION ---

Combinando struct con union (avance del capitulo 44):

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int tipo;  // 0: entero, 1: flotante, 2: cadena
    union {
        int i;
        float f;
        char s[50];
    } valor;
} DatoFlexible;

void mostrarDato(const DatoFlexible *d) {
    printf("Tipo %d: ", d->tipo);
    switch (d->tipo) {
        case 0: printf("Entero: %d\n", d->valor.i); break;
        case 1: printf("Flotante: %.2f\n", d->valor.f); break;
        case 2: printf("Cadena: %s\n", d->valor.s); break;
        default: printf("Desconocido\n");
    }
}

int main() {
    DatoFlexible datos[3];
    datos[0].tipo = 0; datos[0].valor.i = 42;
    datos[1].tipo = 1; datos[1].valor.f = 3.1416f;
    datos[2].tipo = 2; strcpy(datos[2].valor.s, "Hola struct+union");

    for (int i = 0; i < 3; i++) mostrarDato(&datos[i]);

    return 0;
}
```
