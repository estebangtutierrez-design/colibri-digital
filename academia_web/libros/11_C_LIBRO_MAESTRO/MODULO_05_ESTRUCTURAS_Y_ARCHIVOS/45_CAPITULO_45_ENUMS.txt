En los capitulos anteriores has visto estructuras y uniones. Ahora es el
momento de hablar de los enumeradores: una forma de definir constantes
enteras con nombre que hacen tu codigo mas legible y mantenible.

Los enumeradores ("enum") te permiten reemplazar numeros magicos como
0, 1, 2 con nombres significativos como LUNES, MARTES, MIERCOLES o
ROJO, VERDE, AZUL.

Si alguna vez has visto codigo con "#define ROJO 0", "#define VERDE 1",
preparate para conocer una alternativa mucho mejor.

--- INTRODUCCION ---

Un "enum" (enumeracion) es un tipo definido por el usuario que consiste
en un conjunto de constantes enteras con nombre.

Sintaxis basica:

    enum nombre { CONSTANTE1, CONSTANTE2, CONSTANTE3 };

Por defecto, los valores comienzan en 0 y se incrementan en 1:

    enum Dia { LUN, MAR, MIE, JUE, VIE, SAB, DOM };
    // LUN=0, MAR=1, ..., DOM=6

--- VARIABLES ENUM ---

```c
enum Dia hoy;
hoy = MIE;
if (hoy == SAB || hoy == DOM)
    printf("Fin de semana!\n");
```

--- VALORES EXPLICITOS ---

```c
enum Estado { INACTIVO = 0, ACTIVO = 1, SUSPENDIDO = 2, BANEADO = 3 };

enum Color { ROJO = 0xFF0000, VERDE = 0x00FF00, AZUL = 0x0000FF };

enum Prioridad { BAJA = 1, MEDIA, ALTA, CRITICA };  // 1,2,3,4

enum Error { OK = 0, MEMORIA, ARCHIVO, RED = 10, TIMEOUT, CONEXION };
```

--- ENUM CON TYPEDEF ---

```c
typedef enum { LUN, MAR, MIE, JUE, VIE, SAB, DOM } DiaSemana;

typedef enum { PRIMAVERA, VERANO, OTONIO, INVIERNO } Estacion;

int main() {
    DiaSemana hoy = MIE;
    Estacion actual = OTONIO;
    printf("Hoy: %d, Estacion: %d\n", hoy, actual);
    return 0;
}
```

--- ENUM EN SWITCH ---

```c
typedef enum { SUMAR, RESTAR, MULT, DIV } Op;

int aplicar(int a, int b, Op op) {
    switch (op) {
        case SUMAR: return a + b;
        case RESTAR: return a - b;
        case MULT: return a * b;
        case DIV: return b ? a / b : 0;
        default: return 0;
    }
}
```

--- ENUM COMO TAG EN UNIONES ---

```c
typedef enum { TIPO_INT, TIPO_FLOAT, TIPO_STRING } Tipo;

typedef struct {
    Tipo tipo;
    union { int i; float f; char s[50]; };
} Valor;

void mostrar(Valor *v) {
    switch (v->tipo) {
        case TIPO_INT:    printf("int: %d\n", v->i); break;
        case TIPO_FLOAT:  printf("float: %.2f\n", v->f); break;
        case TIPO_STRING: printf("str: %s\n", v->s); break;
    }
}
```

--- ENUM VS #define ---

+---------------------------+---------------------+---------------------+
| Caracteristica            | enum                | #define             |
+---------------------------+---------------------+---------------------+
| Tipo                      | Si (int subyacente) | No (textual)        |
| Scope                     | Respeta bloques     | Global              |
| Debugger                  | Visible como simbolo| Invisible           |
| Auto-incremento           | Si                  | No                  |
| Agrupacion logica         | Si                  | No                  |
+---------------------------+---------------------+---------------------+

```c
// #define - valores sueltos
#define ROJO 0
#define VERDE 1
#define AZUL 2

// enum - agrupados, auto-incrementados
typedef enum { ROJO, VERDE, AZUL } Color;
```

--- sizeof ENUM ---

El tamano de un enum es el mismo que int (generalmente 4 bytes).

```c
typedef enum {

    ERROR_IO = 2

```

```c
typedef enum {
```

```c
#include <stdio.h>

typedef enum {
    ELECTRONICA, ROPA, ALIMENTOS, LIBROS, JUGUETES, MUEBLES
} Categoria;

typedef enum { PENDIENTE, ENVIADO, ENTREGADO, CANCELADO } Estado;

const char* nomCat(Categoria c) {
    switch (c) {
        case ELECTRONICA: return "Electronica";
        case ROPA: return "Ropa";
        case ALIMENTOS: return "Alimentos";
        case LIBROS: return "Libros";
        case JUGUETES: return "Juguetes";
        case MUEBLES: return "Muebles";
        default: return "?";
    }
}

typedef struct {
    char nombre[50];
    float precio;
    Categoria cat;
} Producto;

int main() {
    Producto prods[] = {
        {"Laptop", 15000, ELECTRONICA},
        {"Camiseta", 299, ROPA},
        {"Libro C", 450, LIBROS}
    };
    for (int i = 0; i < 3; i++)
        printf("%-15s $%8.2f | %s\n",
               prods[i].nombre, prods[i].precio, nomCat(prods[i].cat));
    return 0;
}
```

| Aspecto        | enum                          | const int               |

══════════════════════════════════════════════════════════════════════
                    PREGUNTAS FRECUENTES (FAQ)
══════════════════════════════════════════════════════════════════════


P: Ocupan memoria en runtime?
R: No. Son constantes de compilacion.

P: Puedo mezclar valores de diferentes enums?
R: En C si, porque todos son int. En C++ (enum class) no.

P: Que pasa si no asigno valores?
R: Empiezan en 0, incremento de 1.

P: Valores duplicados?
R: Si, diferentes nombres pueden tener el mismo valor.

P: enum o #define?
R: enum para constantes relacionadas. #define para no-enteros o macros.

--- ENUM ANIDADO EN STRUCT ---

Los enums se usan frecuentemente dentro de estructuras:

```c
#include <stdio.h>

typedef enum { CREDITO, DEBITO, TRANSFERENCIA, EFECTIVO } MetodoPago;
typedef enum { PENDIENTE, APROBADO, RECHAZADO, REEMBOLSADO } EstadoPago;

typedef struct {
    int id;
    double monto;
    MetodoPago metodo;
    EstadoPago estado;
    char descripcion[100];
} Pago;

const char* metodoTexto(MetodoPago m) {
    switch (m) {
        case CREDITO:      return "Credito";
        case DEBITO:       return "Debito";
        case TRANSFERENCIA: return "Transferencia";
        case EFECTIVO:     return "Efectivo";
        default:           return "Desconocido";
    }
}

const char* estadoTexto(EstadoPago e) {
    switch (e) {
        case PENDIENTE:   return "Pendiente";
        case APROBADO:    return "Aprobado";
        case RECHAZADO:   return "Rechazado";
        case REEMBOLSADO: return "Reembolsado";
        default:          return "Desconocido";
    }
}

void mostrarPago(const Pago *p) {
    printf("Pago #%d: $%.2f\n", p->id, p->monto);
    printf("  Metodo: %s\n", metodoTexto(p->metodo));
    printf("  Estado: %s\n", estadoTexto(p->estado));
    printf("  Concepto: %s\n", p->descripcion);
}

int main() {
    Pago pagos[] = {
        {1, 1500.00, TRANSFERENCIA, APROBADO, "Pago de nomina"},
        {2, 299.99, CREDITO, PENDIENTE, "Suscripcion mensual"},
        {3, 50.00, EFECTIVO, APROBADO, "Compra en tienda"},
        {4, 1200.00, DEBITO, RECHAZADO, "Fondos insuficientes"}
    };

    int n = sizeof(pagos) / sizeof(pagos[0]);
    for (int i = 0; i < n; i++) {
        mostrarPago(&pagos[i]);
        printf("\n");
    }

    printf("Tamanio de Pago: %zu bytes\n", sizeof(Pago));

    return 0;
}
```

--- ENUM CON MASCARAS DE BITS ---

Los enums son ideales para definir banderas combinables:

```c
#include <stdio.h>

typedef enum {
    PERMISO_NINGUNO = 0,
    PERMISO_LECTURA = 1 << 0,    // 1
    PERMISO_ESCRITURA = 1 << 1,  // 2
    PERMISO_EJECUCION = 1 << 2,  // 4
    PERMISO_BORRAR = 1 << 3,     // 8
    PERMISO_ADMIN = 1 << 4       // 16
} Permiso;

const char* permisos_str(int permisos) {
    static char buf[100];
    buf[0] = '\0';

    if (permisos == 0) return "Ninguno";

    if (permisos & PERMISO_LECTURA)   strcat(buf, "Lectura ");
    if (permisos & PERMISO_ESCRITURA) strcat(buf, "Escritura ");
    if (permisos & PERMISO_EJECUCION) strcat(buf, "Ejecucion ");
    if (permisos & PERMISO_BORRAR)    strcat(buf, "Borrar ");
    if (permisos & PERMISO_ADMIN)     strcat(buf, "Admin ");

    return buf;
}

int main() {
    int permisos_usuario = PERMISO_LECTURA | PERMISO_ESCRITURA;
    int permisos_admin = PERMISO_LECTURA | PERMISO_ESCRITURA |
                         PERMISO_EJECUCION | PERMISO_BORRAR | PERMISO_ADMIN;

    printf("Usuario:  0x%02X -> %s\n", permisos_usuario, permisos_str(permisos_usuario));
    printf("Admin:    0x%02X -> %s\n", permisos_admin, permisos_str(permisos_admin));

    // Verificar permisos especificos
    if (permisos_usuario & PERMISO_ESCRITURA)
        printf("El usuario puede escribir.\n");

    if (!(permisos_usuario & PERMISO_ADMIN))
        printf("El usuario NO es admin.\n");

    return 0;
}
```

--- ENUM CON NOMBRES DE MESES ---

Ejemplo completo con meses del anio:

```c
#include <stdio.h>

typedef enum {
    ENERO = 1, FEBRERO, MARZO, ABRIL, MAYO, JUNIO,
    JULIO, AGOSTO, SEPTIEMBRE, OCTUBRE, NOVIEMBRE, DICIEMBRE
} Mes;

typedef enum {
    INVIERNO, PRIMAVERA, VERANO, OTONIO
} Estacion;

const char* nombreMes(Mes m) {
    static const char *meses[] = {
        "", "Enero", "Febrero", "Marzo", "Abril", "Mayo", "Junio",
        "Julio", "Agosto", "Septiembre", "Octubre", "Noviembre", "Diciembre"
    };
    return (m >= 1 && m <= 12) ? meses[m] : "Invalido";
}

Estacion estacionDelMes(Mes m) {
    switch (m) {
        case DICIEMBRE: case ENERO: case FEBRERO:
            return INVIERNO;
        case MARZO: case ABRIL: case MAYO:
            return PRIMAVERA;
        case JUNIO: case JULIO: case AGOSTO:
            return VERANO;
        default:
            return OTONIO;
    }
}

const char* nombreEstacion(Estacion e) {
    switch (e) {
        case INVIERNO:  return "Invierno";
        case PRIMAVERA: return "Primavera";
        case VERANO:    return "Verano";
        case OTONIO:    return "Otonio";
        default:        return "?";
    }
}

int diasDelMes(Mes m, int anio) {
    switch (m) {
        case ENERO: case MARZO: case MAYO: case JULIO:
        case AGOSTO: case OCTUBRE: case DICIEMBRE:
            return 31;
        case ABRIL: case JUNIO: case SEPTIEMBRE: case NOVIEMBRE:
            return 30;
        case FEBRERO:
            return (anio % 4 == 0 && (anio % 100 != 0 || anio % 400 == 0)) ? 29 : 28;
        default:
            return 0;
    }
}

int main() {
    int anio = 2025;

    printf("Calendario %d:\n\n", anio);

    for (Mes m = ENERO; m <= DICIEMBRE; m++) {
        int dias = diasDelMes(m, anio);
        Estacion est = estacionDelMes(m);

        printf("%-12s | %2d dias | %s\n",
               nombreMes(m), dias, nombreEstacion(est));
    }

    printf("\nTamanio de enum Mes: %zu bytes\n", sizeof(Mes));

    return 0;
}
```

--- ENUM VS #define EN LA PRACTICA ---

```c
// Enfoque con #define (propenso a errores)
#define ESTADO_ACTIVO 1
#define ESTADO_INACTIVO 0
#define ESTADO_SUSPENDIDO 2
// Los valores no estan agrupados, pueden colisionar con otros #define

// Enfoque con enum (agrupado, seguro, auto-incrementable)
typedef enum { INACTIVO = 0, ACTIVO, SUSPENDIDO } Estado;

// Ventaja: Puedes tener funciones que reciban el tipo
void procesarEstado(Estado e) {
    switch (e) {
        case ACTIVO:     printf("Procesando activo...\n"); break;
        case INACTIVO:   printf("Ignorando inactivo.\n"); break;
        case SUSPENDIDO: printf("Revisando suspension...\n"); break;
    }
}
```

--- EJEMPLO CON IA (INTELIGENCIA ARTIFICIAL SIMPLE) ---

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

typedef enum { PIEDRA, PAPEL, TIJERA } Jugada;

typedef enum { GANA_J1, GANA_J2, EMPATE } Resultado;

const char* jugadaStr(Jugada j) {
    switch (j) {
        case PIEDRA: return "Piedra";
        case PAPEL:  return "Papel";
        case TIJERA: return "Tijera";
        default:     return "?";
    }
}

Resultado determinarGanador(Jugada j1, Jugada j2) {
    if (j1 == j2) return EMPATE;
    if ((j1 == PIEDRA && j2 == TIJERA) ||
        (j1 == PAPEL && j2 == PIEDRA) ||
        (j1 == TIJERA && j2 == PAPEL))
        return GANA_J1;
    return GANA_J2;
}

typedef struct {
    int partidasGanadas;
    int partidasPerdidas;
    int empates;
    Jugada ultimaJugada;
} Estadisticas;

void mostrarEstadisticas(const Estadisticas *e) {
    int total = e->partidasGanadas + e->partidasPerdidas + e->empates;
    printf("\n=== ESTADISTICAS ===\n");
    printf("Partidas: %d\n", total);
    printf("Ganadas:  %d (%.1f%%)\n", e->partidasGanadas,
           total > 0 ? 100.0 * e->partidasGanadas / total : 0);
    printf("Perdidas: %d (%.1f%%)\n", e->partidasPerdidas,
           total > 0 ? 100.0 * e->partidasPerdidas / total : 0);
    printf("Empates:  %d (%.1f%%)\n", e->empates,
           total > 0 ? 100.0 * e->empates / total : 0);
}

int main() {
    srand(time(NULL));

    Estadisticas est = {0, 0, 0, PIEDRA};
    int opcion;

    printf("=== PIEDRA, PAPEL O TIJERA ===\n");
    printf("Usando enums para modelar el juego!\n\n");

    do {
        printf("\nTu turno (1=Piedra, 2=Papel, 3=Tijera, 0=Salir): ");
        scanf("%d", &opcion);

        if (opcion == 0) break;

        Jugada jugadaUsuario;
        switch (opcion) {
            case 1: jugadaUsuario = PIEDRA; break;
            case 2: jugadaUsuario = PAPEL;  break;
            case 3: jugadaUsuario = TIJERA; break;
            default: printf("Opcion invalida.\n"); continue;
        }

        Jugada jugadaCPU = rand() % 3;
        Resultado res = determinarGanador(jugadaUsuario, jugadaCPU);

        printf("Tu: %s vs CPU: %s -> ",
               jugadaStr(jugadaUsuario), jugadaStr(jugadaCPU));

        switch (res) {
            case GANA_J1:
                printf("GANASTE!\n");
                est.partidasGanadas++;
                break;
            case GANA_J2:
                printf("Perdiste :(\n");
                est.partidasPerdidas++;
                break;
            case EMPATE:
                printf("Empate!\n");
                est.empates++;
                break;
        }

        est.ultimaJugada = jugadaCPU;

    } while (opcion != 0);

    mostrarEstadisticas(&est);
    printf("\nGracias por jugar!\n");

    return 0;
}
```
