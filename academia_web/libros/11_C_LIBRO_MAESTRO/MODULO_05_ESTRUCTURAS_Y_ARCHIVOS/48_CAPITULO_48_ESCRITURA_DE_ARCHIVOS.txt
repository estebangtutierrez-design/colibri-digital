Ya sabes leer archivos. Ahora aprenderas a escribir en ellos.

C ofrece varias funciones para escribir datos en archivos:

- fputc()  -> Escribe un caracter.
- fputs()  -> Escribe una cadena.
- fprintf()-> Escribe datos formateados.
- fwrite() -> Escribe un bloque de bytes (datos binarios).

Cada una tiene su uso especifico. Aprenderas cuando usar cada una.

--- 1. fputc(): ESCRIBIR UN CARACTER ---

    int fputc(int c, FILE *fp);

Retorna el caracter escrito, o EOF en error.

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("salida.txt", "w");
    if (!fp) { printf("Error.\n"); return 1; }

    for (char c = 'A'; c <= 'Z'; c++) {
        fputc(c, fp);
    }
    fputc('\n', fp);

    fclose(fp);
    printf("Alfabeto escrito en salida.txt\n");
    return 0;
}
```

--- 2. fputs(): ESCRIBIR UNA CADENA ---

    int fputs(const char *s, FILE *fp);

Retorna un valor no negativo en exito, o EOF en error.
NO anade automaticamente \n al final.

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("versos.txt", "w");
    if (!fp) { printf("Error.\n"); return 1; }

    fputs("En un lugar de la Mancha,\n", fp);
    fputs("de cuyo nombre no quiero acordarme,\n", fp);
    fputs("no ha mucho tiempo que vivia\n", fp);
    fputs("un hidalgo de los de lanza en astillero...\n", fp);

    fclose(fp);
    printf("Versos escritos.\n");
    return 0;
}
```

--- 3. fprintf(): ESCRITURA FORMATEADA ---

    int fprintf(FILE *fp, const char *formato, ...);

Retorna el numero de caracteres escritos, o negativo en error.
Funciona como printf() pero escribe en un archivo.

```c
#include <stdio.h>

typedef struct {
    int id;
    char nombre[50];
    float saldo;
} Cliente;

int main() {
    FILE *fp = fopen("clientes.txt", "w");
    if (!fp) { printf("Error.\n"); return 1; }

    Cliente clientes[] = {
        {101, "Ana Lopez", 1500.50},
        {102, "Carlos Ruiz", 3200.00},
        {103, "Elena Gomez", 850.75}
    };

    int n = sizeof(clientes) / sizeof(clientes[0]);
    fprintf(fp, "%-5s %-20s %10s\n", "ID", "NOMBRE", "SALDO");
    fprintf(fp, "%s\n", "------------------------------");

    for (int i = 0; i < n; i++) {
        fprintf(fp, "%-5d %-20s %10.2f\n",
                clientes[i].id, clientes[i].nombre, clientes[i].saldo);
    }

    fclose(fp);
    printf("Reporte generado en clientes.txt\n");
    return 0;
}
```

--- 4. fwrite(): ESCRITURA DE BLOQUES ---

    size_t fwrite(const void *buf, size_t tam_elem,
                  size_t num_elem, FILE *fp);

Retorna el numero de elementos escritos.

```c
#include <stdio.h>

typedef struct {
    int x;
    int y;
} Punto;

int main() {
    FILE *fp = fopen("puntos.bin", "wb");
    if (!fp) { printf("Error.\n"); return 1; }

    Punto puntos[] = {{10, 20}, {30, 40}, {50, 60}, {70, 80}};
    int n = sizeof(puntos) / sizeof(puntos[0]);

    size_t escritos = fwrite(puntos, sizeof(Punto), n, fp);
    printf("Elementos escritos: %zu de %d\n", escritos, n);

    fclose(fp);
    return 0;
}
```

--- COMPARACION DE FUNCIONES ---

+-----------+-------------------------------------+-------------------+
| Funcion   | Uso                                 | Formato           |
+-----------+-------------------------------------+-------------------+
| fputc     | Un caracter                         | int               |
| fputs     | Cadena (debe incluir \n)           | char*             |
| fprintf   | Datos formateados (como printf)     | formato + args    |
| fwrite    | Bloques binarios                    | ptr, tam, num     |
+-----------+-------------------------------------+-------------------+

--- EJEMPLO: GENERADOR DE REPORTES ---

```c
#include <stdio.h>
#include <time.h>

typedef struct {
    int id;
    char producto[50];
    int cantidad;
    float precio;
} Venta;

void generarReporte(const char *ruta, Venta *ventas, int n) {
    FILE *fp = fopen(ruta, "w");
    if (!fp) { printf("Error.\n"); return; }

    time_t t = time(NULL);
    fprintf(fp, "REPORTE DE VENTAS\n");
    fprintf(fp, "Generado: %s\n", ctime(&t));
    fprintf(fp, "========================================\n\n");
    fprintf(fp, "%-5s %-20s %8s %8s %10s\n",
            "ID", "Producto", "Cant.", "Precio", "Total");

    float granTotal = 0;
    for (int i = 0; i < n; i++) {
        float total = ventas[i].cantidad * ventas[i].precio;
        fprintf(fp, "%-5d %-20s %8d %8.2f %10.2f\n",
                ventas[i].id, ventas[i].producto,
                ventas[i].cantidad, ventas[i].precio, total);
        granTotal += total;
    }

    fprintf(fp, "\n========================================\n");
    fprintf(fp, "GRAN TOTAL: $%.2f\n", granTotal);

    fclose(fp);
}

int main() {
    Venta ventas[] = {
        {1, "Laptop", 2, 15000.00},
        {2, "Mouse", 10, 850.50},
        {3, "Monitor", 3, 8500.00},
        {4, "Teclado", 5, 1200.00}
    };

    generarReporte("reporte.txt", ventas, 4);
    printf("Reporte generado.\n");
    return 0;
}
```

--- EJEMPLO: COPIA DE ARCHIVOS (fread/fwrite) ---

```c
#include <stdio.h>
#include <stdlib.h>

int copiarArchivo(const char *origen, const char *destino) {
    FILE *src = fopen(origen, "rb");
    FILE *dst = fopen(destino, "wb");
    if (!src || !dst) {
        printf("Error abriendo archivos.\n");
        if (src) fclose(src);
        if (dst) fclose(dst);
        return 0;
    }

    unsigned char buffer[4096];
    size_t bytes;
    size_t total = 0;

    while ((bytes = fread(buffer, 1, sizeof(buffer), src)) > 0) {
        fwrite(buffer, 1, bytes, dst);
        total += bytes;
    }

    printf("Copiados %zu bytes.\n", total);

    fclose(src);
    fclose(dst);
    return 1;
}

int main(int argc, char *argv[]) {
    if (argc != 3) {
        printf("Uso: %s origen destino\n", argv[0]);
        return 1;
    }
    copiarArchivo(argv[1], argv[2]);
    return 0;
}
```

--- fflush(): FORZAR ESCRITURA ---

Los datos se almacenan en buffer hasta que se llena o se cierra el archivo.
fflush() fuerza la escritura inmediata:

```c
fprintf(fp, "Progreso: 50%%\n");
fflush(fp);  // Escribir inmediatamente
```

--- VERIFICACION DE ERRORES ---

Siempre verifica que la escritura fue exitosa:

```c
if (fputs("texto", fp) == EOF) {
    printf("Error al escribir.\n");
}

if (fprintf(fp, "dato %d", valor) < 0) {
    printf("Error en fprintf.\n");
}

size_t escritos = fwrite(datos, tam, n, fp);
if (escritos < n) {
    printf("Error: solo se escribieron %zu de %zu elementos.\n",
           escritos, (size_t)n);
}
```

══════════════════════════════════════════════════════════════════════
                    PREGUNTAS FRECUENTES (FAQ)
══════════════════════════════════════════════════════════════════════


P: Diferencia entre fputs y fprintf?
R: fputs solo escribe cadenas. fprintf escribe formateado con %d, %s, etc.

P: fputs anade un salto de linea?
R: No. Debes agregar \n explicitamente si lo necesitas.

P: fwrite es para archivos binarios?
R: Principalmente, aunque tambien funciona con texto. Usa "wb" o "ab".

P: Que pasa si el disco esta lleno?
R: fwrite/fprintf pueden fallar. Verifica siempre los valores de retorno.

P: fprintf retorna algo util?
R: Retorna el numero de caracteres escritos, o negativo en error.

--- GENERACION DE ARCHIVO HTML CON FPRINTF ---

```c
#include <stdio.h>
#include <time.h>
#include <string.h>

void generarPaginaHTML(const char *ruta, const char *titulo,
                       const char *autor, const char **parrafos, int n) {
    FILE *fp = fopen(ruta, "w");
    if (!fp) { printf("Error.\n"); return; }

    time_t ahora = time(NULL);

    fprintf(fp, "<!DOCTYPE html>\n");
    fprintf(fp, "<html lang="es">\n");
    fprintf(fp, "<head>\n");
    fprintf(fp, "  <meta charset="UTF-8">\n");
    fprintf(fp, "  <title>%s</title>\n", titulo);
    fprintf(fp, "  <style>\n");
    fprintf(fp, "    body { font-family: Arial; margin: 40px; }\n");
    fprintf(fp, "    h1 { color: #333; }\n");
    fprintf(fp, "    p { line-height: 1.6; }\n");
    fprintf(fp, "    .footer { margin-top: 50px; color: #999; font-size: 0.8em; }\n");
    fprintf(fp, "  </style>\n");
    fprintf(fp, "</head>\n");
    fprintf(fp, "<body>\n");
    fprintf(fp, "  <h1>%s</h1>\n", titulo);
    fprintf(fp, "  <p class="author">Por: %s</p>\n", autor);
    fprintf(fp, "  <hr>\n");

    for (int i = 0; i < n; i++) {
        fprintf(fp, "  <p>%s</p>\n", parrafos[i] ? parrafos[i] : "");
    }

    fprintf(fp, "  <hr>\n");
    fprintf(fp, "  <div class="footer">\n");
    fprintf(fp, "    Generado el %s", ctime(&ahora));
    fprintf(fp, "    por C MASTER BOOK\n");
    fprintf(fp, "  </div>\n");
    fprintf(fp, "</body>\n");
    fprintf(fp, "</html>\n");

    fclose(fp);
    printf("Pagina HTML generada: %s\n", ruta);
}

int main() {
    const char *parrafos[] = {
        "Este es el primer parrafo del documento.",
        "Aqui va el segundo parrafo con mas contenido.",
        "Y este es el tercer y ultimo parrafo por ahora.",
        "C MASTER BOOK - Aprendiendo C desde cero."
    };

    generarPaginaHTML("pagina.html", "Mi Pagina Generada",
                      "Estudiante C", parrafos, 4);

    return 0;
}
```

--- GENERACION DE ARCHIVO CSV CON FPUTS ---

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int id;
    char producto[50];
    int cantidad;
    float precio;
    char categoria[30];
} Inventario;

void generarCSV(const char *ruta, Inventario *items, int n) {
    FILE *fp = fopen(ruta, "w");
    if (!fp) return;

    // Cabecera con fputs
    fputs("id,producto,cantidad,precio,categoria\n", fp);

    // Datos con fprintf

                items[i].id, items[i].producto,
                items[i].cantidad, items[i].precio,
                items[i].categoria);
    }

    fclose(fp);
}

int main() {
    Inventario items[] = {
        {1, "Laptop Gamer", 5, 15000.00, "Electronica"},
        {2, "Mouse Inalambrico", 20, 850.50, "Electronica"},
        {3, "Libro C Avanzado", 15, 450.00, "Libros"},
        {4, "Camiseta Algodon", 30, 299.00, "Ropa"},
        {5, "Mesa Escritorio", 3, 2500.00, "Muebles"}
    };

    int n = sizeof(items) / sizeof(items[0]);

    generarCSV("inventario.csv", items, n);
    printf("CSV generado con %d productos.\n", n);
```

--- ESCRITURA ATOMICA CON FLUSH ---

Para aplicaciones criticas (logs, transacciones), fuerza escritura inmediata:

```c
#include <stdio.h>
#include <time.h>

void escribirLog(const char *ruta, const char *nivel, const char *mensaje) {
    FILE *fp = fopen(ruta, "a");
    if (!fp) return;

    time_t ahora = time(NULL);
    struct tm *t = localtime(&ahora);

    fprintf(fp, "[%04d-%02d-%02d %02d:%02d:%02d] [%-7s] %s\n",
            t->tm_year + 1900, t->tm_mon + 1, t->tm_mday,
            t->tm_hour, t->tm_min, t->tm_sec,
            nivel, mensaje);

    fflush(fp);  // Forzar escritura inmediata a disco
    fclose(fp);
}

int main() {
    escribirLog("aplicacion.log", "INFO", "Programa iniciado");
    escribirLog("aplicacion.log", "INFO", "Cargando configuracion");

    for (int i = 0; i < 3; i++) {
        char msg[100];
        sprintf(msg, "Procesando lote %d...", i + 1);
        escribirLog("aplicacion.log", "DEBUG", msg);
    }

    escribirLog("aplicacion.log", "INFO", "Programa finalizado");
    printf("Log generado en aplicacion.log\n");

    return 0;
}
```

--- COMPARACION DE RENDIMIENTO: FPUTS VS FPRINTF ---

Para cadenas simples, fputs es mas rapido que fprintf:

```c
#include <stdio.h>
#include <time.h>

int main() {
    FILE *fp1 = fopen("test_fputs.txt", "w");
    FILE *fp2 = fopen("test_fprintf.txt", "w");

    if (!fp1 || !fp2) { printf("Error\n"); return 1; }

    clock_t inicio, fin;

    // Prueba con fputs
    inicio = clock();
    for (int i = 0; i < 100000; i++) {
        fputs("Linea de prueba para medir rendimiento\n", fp1);
    }
    fin = clock();
    double tiempo_fputs = (double)(fin - inicio) / CLOCKS_PER_SEC;

    // Prueba con fprintf
    inicio = clock();
    for (int i = 0; i < 100000; i++) {
        fprintf(fp2, "%s\n", "Linea de prueba para medir rendimiento");
    }
    fin = clock();
    double tiempo_fprintf = (double)(fin - inicio) / CLOCKS_PER_SEC;

    fclose(fp1);
    fclose(fp2);

    printf("Tiempo fputs:   %.4f segundos\n", tiempo_fputs);
    printf("Tiempo fprintf: %.4f segundos\n", tiempo_fprintf);
    printf("Diferencia: %.2fx\n", tiempo_fprintf / tiempo_fputs);

    return 0;
}
```

--- ESCRITURA DE ARCHIVOS CON CODIFICACION ---

Para escribir texto con caracteres especiales (UTF-8):

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("utf8.txt", "w");
    if (!fp) return 1;

    // BOM (Byte Order Mark) UTF-8: EF BB BF
    fputs("\xEF\xBB\xBF", fp);  // Opcional

    fputs("Caracteres especiales UTF-8:\n", fp);
    fputs("- Acentos: a, e, i, o, u, n\n", fp);
    fputs("- Simbolos: €, $, (, ), @\n", fp);
    fputs("- Matematicos: +, -, *, /, =, <, >\n", fp);

    fclose(fp);
    printf("Archivo UTF-8 creado.\n");
    return 0;
}
```
