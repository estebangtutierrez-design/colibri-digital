Bienvenido al mundo del manejo de archivos en C.

Hasta ahora todos los programas trabajan con datos en memoria: variables,
arrays, estructuras. Pero cuando el programa termina, los datos se pierden.

Para conservar datos entre ejecuciones necesitas ARCHIVOS. El manejo de
archivos te permite:
- Guardar configuraciones.
- Almacenar resultados de ejecuciones.
- Procesar grandes volumenes de datos.
- Crear bases de datos simples.
- Leer y escribir formatos estandar.

En este capitulo aprenderas "fopen()", la puerta de entrada a toda
operacion con archivos.

--- INTRODUCCION A FILE* ---

En C, los archivos se manejan con un puntero a la estructura FILE,
definida en <stdio.h>. Este puntero actua como identificador del archivo.

    FILE *fp;

- FILE* representa un archivo abierto.
- Lo obtienes llamando a fopen().
- Lo usas en todas las funciones de E/S.
- Lo liberas con fclose().

--- fopen() ---

Prototipo:

    FILE *fopen(const char *nombre_archivo, const char *modo);

Parametros:
- nombre_archivo: ruta al archivo.
- modo: cadena que indica como abrir el archivo.

Retorna:
- FILE* si tiene exito.
- NULL si ocurre un error.

--- MODOS DE APERTURA ---

+--------+-----------------------------------------------------+
| Modo   | Significado                                         |
+--------+-----------------------------------------------------+
| "r"    | Lectura. El archivo debe existir.                   |
| "w"    | Escritura. Crea o sobrescribe.                      |
| "a"    | Anadir (append). Crea si no existe.                 |
| "r+"   | Lectura y escritura. El archivo debe existir.       |
| "w+"   | Lectura y escritura. Crea o sobrescribe.            |
| "a+"   | Lectura y anadidura. Crea si no existe.             |
+--------+-----------------------------------------------------+

Modificador "b": modo binario (ej: "rb", "wb").

--- DESCRIPCION DETALLADA ---

"r" (READ):
- Abre archivo existente para lectura.
- Puntero al inicio.
- Si no existe, fopen retorna NULL.

"w" (WRITE):
- Abre para escritura.
- Si existe, se TRUNCA (vacia).
- Si no existe, se CREA.

"a" (APPEND):
- Abre para escritura al final.
- Si existe, agrega al final.
- Si no existe, se CREA.
- Puntero al final.

"r+" (READ/WRITE):
- Abre existente para lectura y escritura.
- Debe existir.
- Puntero al inicio.

"w+" (WRITE/READ):
- Lectura y escritura.
- Si existe, trunca.
- Si no, crea.

"a+" (APPEND/READ):
- Lectura y escritura al final.
- Si existe, agrega al final.
- Si no, crea.

--- CODIGO BASICO ---

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    FILE *fp;

    fp = fopen("datos.txt", "r");
    if (fp == NULL) {
        printf("Error: No se pudo abrir el archivo.\n");
        return 1;
    }

    printf("Archivo abierto exitosamente.\n");
    fclose(fp);
    return 0;
}
```

--- VERIFICACION DE ERRORES ---

Siempre verifica si fopen retorna NULL:

```c
FILE *fp = fopen("archivo.txt", "r");
if (fp == NULL) {
    perror("Error al abrir archivo");
    return 1;
}
```

La funcion perror() imprime un mensaje de error del sistema.

--- ERRORES COMUNES ---

```c
// Error 1: archivo no existe en modo r
fp = fopen("no_existe.txt", "r");  // NULL

// Error 2: permisos
fp = fopen("/etc/shadow", "r");    // NULL (permiso denegado)

// Error 3: ruta invalida
fp = fopen("/ruta/inexistente/f.txt", "w");  // NULL
```

--- CERRAR ARCHIVOS ---

Siempre cierra los archivos. Si no:

```c
if (fclose(fp) != 0) {
    printf("Error al cerrar el archivo.\n");
}
```

- Los buffers pueden no vaciarse (pierdes datos).
- Se agotan descriptores de archivo.
- fclose() retorna 0 si exito, EOF si error.

--- BUFFERING ---

Los datos no se escriben inmediatamente en disco. Usa fflush():

```c
fprintf(fp, "Dato importante\n");
fflush(fp);  // Forzar escritura inmediata
```

--- EJEMPLO: SISTEMA DE CONFIGURACION ---

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char usuario[50];
    int volumen;
    int ancho;
    int alto;
    char idioma[10];
} Config;

void guardar(const Config *cfg, const char *ruta) {
    FILE *fp = fopen(ruta, "w");
    if (!fp) { printf("Error al guardar.\n"); return; }
    fprintf(fp, "usuario=%s\n", cfg->usuario);
    fprintf(fp, "volumen=%d\n", cfg->volumen);
    fprintf(fp, "ancho=%d\n", cfg->ancho);
    fprintf(fp, "alto=%d\n", cfg->alto);
    fprintf(fp, "idioma=%s\n", cfg->idioma);
    fclose(fp);
}

void cargar(Config *cfg, const char *ruta) {
    FILE *fp = fopen(ruta, "r");
    if (!fp) {
        strcpy(cfg->usuario, "Invitado");
        cfg->volumen = 50;
        cfg->ancho = 800;
        cfg->alto = 600;
        strcpy(cfg->idioma, "es");
        return;
    }
    char clave[30], valor[50];
    while (fscanf(fp, "%[^=]=%s\n", clave, valor) == 2) {
        if (strcmp(clave, "usuario")==0) strcpy(cfg->usuario, valor);
        else if (strcmp(clave, "volumen")==0) cfg->volumen = atoi(valor);
        else if (strcmp(clave, "ancho")==0) cfg->ancho = atoi(valor);
        else if (strcmp(clave, "alto")==0) cfg->alto = atoi(valor);
        else if (strcmp(clave, "idioma")==0) strcpy(cfg->idioma, valor);
    }
    fclose(fp);
}

int main() {
    Config cfg;
    cargar(&cfg, "config.txt");
    printf("Usuario: %s, Vol: %d\n", cfg.usuario, cfg.volumen);
    strcpy(cfg.usuario, "Admin");
    guardar(&cfg, "config.txt");
    return 0;
}
```

--- MODO BINARIO ---

En Windows, "b" desactiva traduccion de caracteres:
- \n -> \r\n en modo texto.
- \n -> \n en modo binario.
En Linux no hay diferencia.

══════════════════════════════════════════════════════════════════════
                    PREGUNTAS FRECUENTES (FAQ)
══════════════════════════════════════════════════════════════════════


P: Que pasa si olvido cerrar un archivo?
R: Datos pueden no escribirse (buffers). Fuga de recursos.

P: Cuantos archivos abiertos simultaneamente?
R: Depende del sistema. FOPEN_MAX en stdio.h.

P: Diferencia entre w y a?
R: "w" borra contenido y empieza de cero. "a" conserva y escribe al final.

P: fopen retorna NULL?
R: Si: archivo no existe (modo r), permisos, ruta invalida, limite alcanzado.

P: Cambiar de modo con archivo abierto?
R: No. Debes cerrar y reabrir.

P: Manejo de errores?
R: Usa perror() para imprimir error del sistema.

--- ERRORES CON FOPEN Y PERROR ---

perror() es una funcion util para imprimir el error del sistema:

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("/no/existe/archivo.txt", "r");

    if (fp == NULL) {
        perror("Error al abrir archivo");
        // Imprime algo como:
        // "Error al abrir archivo: No such file or directory"
        return 1;
    }

    fclose(fp);
    return 0;
}
```

--- DIFERENCIA ENTRE "w" Y "w+" ---

```c
#include <stdio.h>

int main() {
    // "w": solo escritura, no puedes leer
    FILE *fp = fopen("test_w.txt", "w");
    fprintf(fp, "Hola\n");
    // fgetc(fp);  // ERROR: modo solo escritura
    fclose(fp);

    // "w+": lectura y escritura
    fp = fopen("test_w+.txt", "w+");
    fprintf(fp, "Mundo\n");
    rewind(fp);  // Volver al inicio
    char c = fgetc(fp);  // OK: podemos leer
    printf("Leido: %c\n", c);
    fclose(fp);

    return 0;
}
```

--- CREAR DIRECTORIO ANTES DE FOPEN ---

Si intentas abrir un archivo en un directorio que no existe, fopen falla:

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    // Esto fallara si el directorio "datos" no existe
    FILE *fp = fopen("datos/salida.txt", "w");
    if (fp == NULL) {
        perror("Error");
        printf("Crea el directorio 'datos' primero.\n");

        // En Linux, puedes crearlo con:
        system("mkdir -p datos");

        fp = fopen("datos/salida.txt", "w");
        if (fp) {
            fprintf(fp, "Archivo creado en subdirectorio.\n");
            fclose(fp);
            printf("Archivo creado exitosamente.\n");
        }
    } else {
        fclose(fp);
    }

    return 0;
}
```

--- TECNICA: ABRIR ARCHIVO MULTIPLES VECES ---

Puedes abrir el mismo archivo varias veces para diferentes propositos:

```c
#include <stdio.h>

int main() {
    FILE *lectura = fopen("datos.txt", "r");
    FILE *escritura = fopen("datos.txt", "a");

    if (!lectura || !escritura) {
        printf("Error abriendo archivo.\n");
        return 1;
    }

    // Leer del inicio mientras se escribe al final
    char linea[100];
    while (fgets(linea, sizeof(linea), lectura)) {
        printf("Leido: %s", linea);
    }

    fprintf(escritura, "Nueva linea anadida\n");

    fclose(lectura);
    fclose(escritura);

    return 0;
}
```

--- SEGURIDAD CON ARCHIVOS ---

Siempre valida que las operaciones sean exitosas:

```c
#include <stdio.h>
#include <errno.h>
#include <string.h>

int main() {
    FILE *fp = fopen("importante.txt", "w");
    if (fp == NULL) {
        printf("Error %d: %s\n", errno, strerror(errno));
        return 1;
    }

    // Verificar cada escritura
    if (fprintf(fp, "Datos criticos\n") < 0) {
        printf("Error al escribir. Disco lleno?\n");
    }

    if (fclose(fp) == EOF) {
        printf("Error al cerrar archivo.\n");
        return 1;
    }

    printf("Operacion completada con exito.\n");
    return 0;
}
```

--- EJEMPLO: PROCESAMIENTO DE ARCHIVO POR LOTES ---

```c
#include <stdio.h>
#include <stdlib.h>

#define TAM_BLOQUE 1024

int procesarArchivoEnBloques(const char *ruta) {
    FILE *fp = fopen(ruta, "rb");
    if (!fp) {
        perror("Error");
        return -1;
    }

    unsigned char buffer[TAM_BLOQUE];
    size_t bytesLeidos;
    int numBloque = 0;
    size_t totalBytes = 0;
    unsigned long checksum = 0;

    // Leer y procesar en bloques de TAM_BLOQUE bytes
    while ((bytesLeidos = fread(buffer, 1, TAM_BLOQUE, fp)) > 0) {
        numBloque++;
        totalBytes += bytesLeidos;

        // Calcular checksum simple (XOR de todos los bytes)
        for (size_t i = 0; i < bytesLeidos; i++) {
            checksum ^= buffer[i];
        }

        printf("Bloque %d: %zu bytes\n", numBloque, bytesLeidos);
    }

    printf("\nResumen:\n");
    printf("  Bloques procesados: %d\n", numBloque);
    printf("  Bytes totales:      %zu\n", totalBytes);
    printf("  Checksum (XOR):     0x%08lX\n", checksum);

    if (ferror(fp)) {
        printf("Error de lectura en el bloque %d.\n", numBloque);
        fclose(fp);
        return -1;
    }

    if (feof(fp)) {
        printf("Archivo procesado completamente.\n");
    }

    fclose(fp);
    return 0;
}

int main(int argc, char *argv[]) {
    if (argc != 2) {
        printf("Uso: %s <archivo>\n", argv[0]);
        return 1;
    }

    return procesarArchivoEnBloques(argv[1]);
}
```

--- COMBINANDO FOPEN CON STRUCTS ---

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    char nombre[50];
    int edad;
    float altura;
} Persona;

int guardarPersonas(const char *ruta, Persona *personas, int n) {
    FILE *fp = fopen(ruta, "w");
    if (!fp) return 0;

    for (int i = 0; i < n; i++) {
        fprintf(fp, "%s;%d;%.2f\n",
                personas[i].nombre,
                personas[i].edad,
                personas[i].altura);
    }

    fclose(fp);
    return 1;
}

int cargarPersonas(const char *ruta, Persona *personas, int max) {
    FILE *fp = fopen(ruta, "r");
    if (!fp) return 0;

    int n = 0;
    while (n < max &&
           fscanf(fp, "%49[^;];%d;%f\n",
                  personas[n].nombre,
                  &personas[n].edad,
                  &personas[n].altura) == 3) {
        n++;
    }

    fclose(fp);
    return n;
}

int main() {
    Persona lista[] = {
        {"Ana", 28, 1.65},
        {"Luis", 35, 1.80},
        {"Elena", 42, 1.70}
    };

    if (guardarPersonas("personas.txt", lista, 3)) {
        printf("Guardadas 3 personas.\n");
    }

    Persona cargadas[10];
    int n = cargarPersonas("personas.txt", cargadas, 10);

    printf("\nCargadas %d personas:\n", n);
    for (int i = 0; i < n; i++) {
        printf("  %s, %d anios, %.2f m\n",
               cargadas[i].nombre, cargadas[i].edad, cargadas[i].altura);
    }

    return 0;
}
```
