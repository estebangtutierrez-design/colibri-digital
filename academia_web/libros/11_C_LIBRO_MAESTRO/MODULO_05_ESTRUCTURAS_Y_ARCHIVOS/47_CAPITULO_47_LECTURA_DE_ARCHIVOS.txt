                    BIENVENIDA
══════════════════════════════════════════════════════════════════════

Ahora que sabes abrir y cerrar archivos, es momento de aprender a leer
su contenido.

C ofrece varias funciones para leer archivos:

- fgetc()  -> Lee un caracter a la vez.
- fgets()  -> Lee una linea completa.
- fread()  -> Lee un bloque de bytes (datos binarios).
- fscanf() -> Lee datos formateados.
- feof()   -> Verifica si se alcanzo el final del archivo.

--- 1. fgetc(): LEER CARACTER POR CARACTER ---

    int fgetc(FILE *fp);

Retorna el caracter como unsigned char convertido a int, o EOF al final.

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("archivo.txt", "r");
    if (fp == NULL) { printf("Error.\n"); return 1; }

    int c;
    int cont = 0;
    while ((c = fgetc(fp)) != EOF) {
        putchar(c);
        cont++;
    }
    printf("\nTotal caracteres: %d\n", cont);
    fclose(fp);
    return 0;
}
```

--- 2. fgets(): LEER LINEA POR LINEA ---

    char *fgets(char *buf, int tam, FILE *fp);

Retorna el buffer o NULL si EOF/error. Lee hasta \n o tam-1 caracteres.

```c
#include <stdio.h>
#include <string.h>

int main() {
    FILE *fp = fopen("archivo.txt", "r");
    if (fp == NULL) { printf("Error.\n"); return 1; }

    char linea[256];
    int n = 0;
    while (fgets(linea, sizeof(linea), fp) != NULL) {
        n++;
        size_t len = strlen(linea);
        if (len > 0 && linea[len-1] == '\n')
            linea[len-1] = '\0';
        printf("%3d: %s\n", n, linea);
    }
    fclose(fp);
    return 0;
}
```

--- 3. fread(): LEER BLOQUES DE BYTES ---

    size_t fread(void *buf, size_t tam_elem, size_t num_elem, FILE *fp);

Retorna el numero de elementos leidos (puede ser menor al solicitado).

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("datos.bin", "rb");
    if (fp == NULL) { printf("Error.\n"); return 1; }

    unsigned char buf[1024];
    size_t bytes;
    size_t total = 0;

    while ((bytes = fread(buf, 1, sizeof(buf), fp)) > 0)
        total += bytes;

    printf("Total bytes: %zu\n", total);
    fclose(fp);
    return 0;
}
```

--- 4. fscanf(): LECTURA FORMATEADA ---

    int fscanf(FILE *fp, const char *formato, ...);

Retorna el numero de asignaciones, o EOF en error.

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("datos.txt", "r");
    if (fp == NULL) { printf("Error.\n"); return 1; }

    int id; char nombre[50]; float prom;
    while (fscanf(fp, "%d,%49[^,],%f\n", &id, nombre, &prom) == 3)
        printf("%03d | %-20s | %.2f\n", id, nombre, prom);

    fclose(fp);
    return 0;
}
```

--- 5. feof() Y ferror() ---

feof() retorna verdadero si se alcanzo el final del archivo.
ferror() retorna verdadero si ocurrio un error de E/S.

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("datos.txt", "r");
    if (fp == NULL) { printf("Error.\n"); return 1; }

    char linea[100];
    while (fgets(linea, sizeof(linea), fp)) {
        printf("%s", linea);
    }

    if (feof(fp))
        printf("\n--- Fin del archivo ---\n");
    if (ferror(fp))
        printf("\n--- Error de lectura ---\n");

    fclose(fp);
    return 0;
}
```

--- DIFERENCIA: feof() vs EOF ---

fgetc retorna EOF (-1) cuando no hay mas datos. feof() verifica el
indicador de fin de archivo. No uses feof() como condicion de bucle;
es mejor verificar el valor de retorno de la funcion de lectura.

```c
// MAL: no funciona como esperas
while (!feof(fp)) {
    fgets(linea, sizeof(linea), fp);
    printf("%s", linea);  // Se imprime dos veces la ultima linea
}

// BIEN: verifica el retorno
while (fgets(linea, sizeof(linea), fp) != NULL) {
    printf("%s", linea);
}
```

--- LEER ARCHIVO COMPLETO EN MEMORIA ---

```c
#include <stdio.h>
#include <stdlib.h>

char* leerArchivo(const char *ruta, long *tam) {
    FILE *fp = fopen(ruta, "rb");
    if (!fp) return NULL;

    fseek(fp, 0, SEEK_END);
    *tam = ftell(fp);
    rewind(fp);

    char *buffer = malloc(*tam + 1);
    if (!buffer) { fclose(fp); return NULL; }

    fread(buffer, 1, *tam, fp);
    buffer[*tam] = '\0';
    fclose(fp);
    return buffer;
}

int main() {
    long tam;
    char *contenido = leerArchivo("archivo.txt", &tam);
    if (contenido) {
        printf("Tamano: %ld bytes\n", tam);
        printf("Contenido:\n%s\n", contenido);
        free(contenido);
    }
    return 0;
}
```

--- LEER CSV ---

```c
#include <stdio.h>
#include <string.h>

int main() {
    FILE *fp = fopen("datos.csv", "r");
    if (!fp) { printf("Error.\n"); return 1; }

    char linea[512];
    char *campo;
    int lineaNum = 0;

    while (fgets(linea, sizeof(linea), fp)) {
        lineaNum++;
        printf("Linea %d: ", lineaNum);

        campo = strtok(linea, ",\n");
        while (campo) {
            printf("[%s] ", campo);
            campo = strtok(NULL, ",\n");
        }
        printf("\n");
    }

    fclose(fp);
    return 0;
}
```

--- EJEMPLO: CONTADOR DE PALABRAS ---

```c
#include <stdio.h>
#include <ctype.h>

int main() {
    FILE *fp = fopen("texto.txt", "r");
    if (!fp) { printf("Error.\n"); return 1; }

    int c;
    int palabras = 0, lineas = 0, caracteres = 0;
    int enPalabra = 0;

    while ((c = fgetc(fp)) != EOF) {
        caracteres++;
        if (c == '\n') lineas++;
        if (isspace(c)) {
            enPalabra = 0;
        } else if (!enPalabra) {
            enPalabra = 1;
            palabras++;
        }
    }

    printf("Palabras: %d\n", palabras);
    printf("Lineas:   %d\n", lineas);
    printf("Caracteres: %d\n", caracteres);

    fclose(fp);
    return 0;
}
```

--- EJEMPLO: PROCESADOR DE CONFIGURACION ---

```c
#include <stdio.h>
#include <string.h>

typedef struct { char clave[30]; char valor[50]; } ConfigItem;

int leerConfig(const char *ruta, ConfigItem *items, int max) {
    FILE *fp = fopen(ruta, "r");
    if (!fp) return 0;

    char linea[100];
    int n = 0;

    while (fgets(linea, sizeof(linea), fp) && n < max) {
        if (linea[0] == '#' || linea[0] == '\n') continue;
        if (sscanf(linea, "%29[^=]=%49s", items[n].clave, items[n].valor) == 2)
            n++;
    }

    fclose(fp);
    return n;
}

int main() {
    ConfigItem items[20];
    int n = leerConfig("config.txt", items, 20);

    for (int i = 0; i < n; i++)
        printf("%s = %s\n", items[i].clave, items[i].valor);

    return 0;
}
```

══════════════════════════════════════════════════════════════════════
                    PREGUNTAS FRECUENTES (FAQ)
══════════════════════════════════════════════════════════════════════


P: Diferencia entre fgets y fscanf?
R: fgets lee una linea completa. fscanf lee datos formateados.

P: Por que fgetc retorna int y no char?
R: Para poder retornar EOF (-1), que no es un char valido.

P: feof() en el bucle causa problemas?
R: Si. No la uses como condicion. Verifica el retorno de la funcion de
   lectura directamente.

P: fread retorna menos elementos de los solicitados?
R: Si, al final del archivo o si hay error. Compara el retorno.

P: Como leer archivos grandes?
R: Lee en bloques (buffers) con fread, no linea por linea.

--- FGETC PARA CONTAR LINEAS ---

Un uso practico de fgetc() es contar elementos en un archivo:

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("texto.txt", "r");
    if (!fp) { printf("Error.\n"); return 1; }

    int c;
    int lineas = 0;
    int palabras = 0;
    int caracteres = 0;
    int enPalabra = 0;

    while ((c = fgetc(fp)) != EOF) {
        caracteres++;

        if (c == '\n') lineas++;

        if (c == ' ' || c == '\n' || c == '\t') {
            enPalabra = 0;
        } else if (!enPalabra) {
            enPalabra = 1;
            palabras++;
        }
    }

    // Si el archivo no termina en nueva linea, contarla igual
    if (caracteres > 0 && c != '\n') lineas++;

    printf("Estadisticas:\n");
    printf("  Caracteres: %d\n", caracteres);
    printf("  Palabras:   %d\n", palabras);
    printf("  Lineas:     %d\n", lineas);

    fclose(fp);
    return 0;
}
```

--- FSCANF CON FORMATOS AVANZADOS ---

fscanf permite patrones de busqueda complejos:

```c
#include <stdio.h>

int main() {
    FILE *fp = fopen("datos.txt", "r");
    if (!fp) { printf("Error.\n"); return 1; }

    char nombre[50];
    int edad;
    float altura;
    char email[80];
    int items = 0;

    // Formato: Nombre, Edad, Altura, Email
    while (fscanf(fp, "%49[^,],%d,%f,%79s\n",
                  nombre, &edad, &altura, email) == 4) {
        items++;
        printf("[%d] %s, %d anios, %.2f m, %s\n",
               items, nombre, edad, altura, email);
    }

    printf("\nTotal registros: %d\n", items);

    if (ferror(fp)) printf("Error de lectura.\n");

    fclose(fp);
    return 0;
}
```

--- FGETS PARA PROCESAR ARCHIVOS CON FORMATO ESPECIFICO ---

A veces es mejor leer linea por linea y luego analizar:

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

typedef struct {
    char ciudad[50];
    float temperatura;
    float humedad;
    float viento;
} RegistroClima;

int main() {
    FILE *fp = fopen("clima.csv", "r");
    if (!fp) { printf("Error.\n"); return 1; }

    RegistroClima registros[100];
    int n = 0;
    char linea[256];

    // Saltar cabecera
    fgets(linea, sizeof(linea), fp);

    while (fgets(linea, sizeof(linea), fp) && n < 100) {
        // Eliminar salto de linea
        linea[strcspn(linea, "\n")] = 0;

        char *token = strtok(linea, ",");
        if (!token) continue;
        strcpy(registros[n].ciudad, token);

        token = strtok(NULL, ",");
        registros[n].temperatura = token ? atof(token) : 0;

        token = strtok(NULL, ",");
        registros[n].humedad = token ? atof(token) : 0;

        token = strtok(NULL, ",");
        registros[n].viento = token ? atof(token) : 0;

        n++;
    }

    fclose(fp);

    printf("Datos climaticos cargados: %d registros\n\n", n);
    printf("%-20s %10s %10s %10s\n", "Ciudad", "Temp", "Humedad", "Viento");
    for (int i = 0; i < n; i++) {
        printf("%-20s %10.1f %10.1f%% %10.1f\n",
               registros[i].ciudad,
               registros[i].temperatura,
               registros[i].humedad,
               registros[i].viento);
    }

    return 0;
}
```

--- LECTURA DE ARCHIVOS GRANDES CON BUFFER ---

Para archivos muy grandes, lee en bloques grandes:

```c
#include <stdio.h>
#include <stdlib.h>

#define BUFFER_SIZE 65536  // 64 KB

int procesarArchivoGrande(const char *ruta) {
    FILE *fp = fopen(ruta, "rb");
    if (!fp) { perror("Error"); return 0; }

    // Obtener tamano del archivo
    fseek(fp, 0, SEEK_END);
    long tamano = ftell(fp);
    rewind(fp);

    printf("Archivo: %s (%ld bytes)\n", ruta, tamano);

    // Buffer de 64 KB
    unsigned char *buffer = malloc(BUFFER_SIZE);
    if (!buffer) { fclose(fp); return 0; }

    size_t bytesLeidos;
    size_t totalLeido = 0;
    int numBloques = 0;

    while ((bytesLeidos = fread(buffer, 1, BUFFER_SIZE, fp)) > 0) {
        totalLeido += bytesLeidos;
        numBloques++;

        // Procesar bloque (ej: contar bytes con valor > 127)
        int altos = 0;
        for (size_t i = 0; i < bytesLeidos; i++) {
            if (buffer[i] > 127) altos++;
        }

        printf("Bloque %d: %zu bytes (%d no-ASCII)\n",
               numBloques, bytesLeidos, altos);
    }

    free(buffer);
    fclose(fp);

    printf("\nTotal: %zu bytes en %d bloques\n", totalLeido, numBloques);
    return 1;
}

int main(int argc, char *argv[]) {
    if (argc != 2) {
        printf("Uso: %s <archivo>\n", argv[0]);
        return 1;
    }

    return procesarArchivoGrande(argv[1]) ? 0 : 1;
}
```

--- DETECTAR TIPO DE ARCHIVO (MAGIC NUMBERS) ---

Los primeros bytes de un archivo indican su tipo:

```c
#include <stdio.h>

typedef struct {
    const char *extension;
    const char *descripcion;
    unsigned char magic[4];
    int numBytes;
} TipoArchivo;

int main() {
    TipoArchivo tipos[] = {
        {".png", "PNG Image", {0x89, 0x50, 0x4E, 0x47}, 4},
        {".jpg", "JPEG Image", {0xFF, 0xD8, 0xFF, 0xE0}, 4},
        {".pdf", "PDF Document", {0x25, 0x50, 0x44, 0x46}, 4},
        {".gif", "GIF Image", {0x47, 0x49, 0x46, 0x38}, 4},
        {".zip", "ZIP Archive", {0x50, 0x4B, 0x03, 0x04}, 4},
        {".elf", "ELF Binary", {0x7F, 0x45, 0x4C, 0x46}, 4},
        {".mp3", "MP3 Audio", {0x49, 0x44, 0x33, 0x00}, 4},
        {0, 0, {0}, 0}
    };

    char nombre[256];
    printf("Ingresa el nombre del archivo: ");
    scanf("%255s", nombre);

    FILE *fp = fopen(nombre, "rb");
    if (!fp) {
        printf("No se puede abrir el archivo.\n");
        return 1;
    }

    unsigned char cabecera[4];
    fread(cabecera, 1, 4, fp);
    fclose(fp);

    printf("\nCabecera (hex): ");
    for (int i = 0; i < 4; i++)
        printf("%02X ", cabecera[i]);
    printf("\n");

    int encontrado = 0;
    for (int t = 0; tipos[t].descripcion; t++) {
        int coincide = 1;
        for (int i = 0; i < tipos[t].numBytes; i++) {
            if (cabecera[i] != tipos[t].magic[i]) {
                coincide = 0;
                break;
            }
        }
        if (coincide) {
            printf("Tipo detectado: %s (%s)\n",
                   tipos[t].descripcion, tipos[t].extension);
            encontrado = 1;
            break;
        }
    }

    if (!encontrado)
        printf("Tipo de archivo desconocido.\n");

    return 0;
}
```
