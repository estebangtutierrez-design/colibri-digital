En el capitulo anterior aprendiste a declarar estructuras con `struct`.
Pero escribir `struct NombreEstructura` cada vez que declaras una variable
puede volverse tedioso rapidamente.

Imagina tener que escribir esto cien veces en tu codigo:

    struct ListaEnlazada nodo;
    struct PuntoGeometrico centro;
    struct ProductoAlmacen producto;

La palabra clave `struct` se repite innecesariamente. En este capitulo
aprenderas a usar `typedef` para crear alias de tipos, eliminando esta
repeticion y haciendo tu codigo mas legible, profesional y facil de mantener.

En muchos codigos profesionales de C, especialmente en librerias y sistemas
embebidos, `typedef` es omnipresente. Dominarlo es esencial para leer y
escribir codigo C moderno.

--- INTRODUCCION A TYPEDEF ---

`typedef` es una palabra clave de C que permite crear un alias (un nombre
alternativo) para un tipo de datos existente. No crea un nuevo tipo, solo
un nuevo nombre para referirse al mismo tipo.

    typedef tipo_existente nuevo_nombre;

--- SINTAXIS BASICA ---

    typedef unsigned long ulong;

A partir de esta linea, `ulong` es un alias de `unsigned long int`.

    ulong distancia = 1234567890UL;

--- TYPEDEF CON TIPOS PRIMITIVOS ---

Puedes crear alias para cualquier tipo, incluyendo los primitivos:

```c
#include <stdio.h>

typedef unsigned int uint;
typedef unsigned char byte;
typedef long long int llong;

int main() {
    uint contador = 100;
    byte flag = 0xFF;
    llong grande = 9223372036854775807LL;

    printf("uint  : %u\n", contador);
    printf("byte  : 0x%X\n", flag);
    printf("llong : %lld\n", grande);

    printf("sizeof(uint)  = %zu bytes\n", sizeof(uint));
    printf("sizeof(byte)  = %zu bytes\n", sizeof(byte));
    printf("sizeof(llong) = %zu bytes\n", sizeof(llong));

    return 0;
}
```

--- TYPEDEF CON ESTRUCTURAS ---

Aqui es donde `typedef` realmente brilla. Hay dos enfoques principales:

1. Estructura con nombre + typedef:

```c
struct Punto {
    float x;
    float y;
};

typedef struct Punto Punto;
```

Ahora puedes declarar variables asi:

    Punto p1;  // En lugar de struct Punto p1;

2. Estructura anonima + typedef (recomendado):

```c
typedef struct {
    float x;
    float y;
} Punto;
```

Es mas conciso y es el estilo mas usado en la industria.

--- EJEMPLO COMPLETO ---

```c
#include <stdio.h>
#include <string.h>
#include <math.h>

typedef unsigned int uint;
typedef unsigned char byte;

typedef struct {
    float x;
    float y;
} Punto;

typedef struct {
    char nombre[50];
    uint edad;
    float saldo;
    uint activo;
} Cliente;

float distancia(Punto a, Punto b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return sqrtf(dx * dx + dy * dy);
}

void mostrarCliente(Cliente c) {
    printf("Cliente: %s\n", c.nombre);
    printf("  Edad  : %u\n", c.edad);
    printf("  Saldo : $%.2f\n", c.saldo);
    printf("  Estado: %s\n", c.activo ? "Activo" : "Inactivo");
}

int main() {
    Punto origen = {0.0f, 0.0f};
    Punto destino = {3.0f, 4.0f};

    printf("Distancia: %.2f\n", distancia(origen, destino));

    Cliente clientes[3] = {
        {"Ana Lopez", 28, 1500.50, 1},
        {"Carlos Ruiz", 35, 200.00, 1},
        {"Elena Gomez", 42, 0.00, 0}
    };

    printf("\n=== LISTA DE CLIENTES ===\n\n");
    for (uint i = 0; i < 3; i++) {
        printf("--- Cliente %u ---\n", i + 1);
        mostrarCliente(clientes[i]);
        printf("\n");
    }

    return 0;
}
```

--- TYPEDEF CON PUNTEROS A ESTRUCTURAS ---

Es muy comun crear typedef para punteros a estructuras:

```c
typedef struct {
    int dato;
} Nodo;

typedef Nodo* NodoPtr;
```

Tambien puedes hacerlo en una sola linea:

```c
typedef struct Nodo {
    int dato;
    struct Nodo* siguiente;
} Nodo, *NodoPtr;
```

--- TYPEDEF CON ARRAYS ---

Puedes crear alias para arrays:

```c
typedef int Vector5[5];
typedef char Cadena[100];

int main() {
    Vector5 v = {1, 2, 3, 4, 5};
    Cadena nombre = "Hola Mundo";

    printf("Primer elemento: %d\n", v[0]);
    printf("Cadena: %s\n", nombre);

    return 0;
}
```

--- TYPEDEF CON FUNCIONES (punteros a funcion) ---

Uno de los usos mas avanzados de `typedef` es con punteros a funcion:

```c
// Sin typedef
int (*operacion)(int, int);
operacion = suma;

// Con typedef
typedef int (*OperacionMatematica)(int, int);
OperacionMatematica op = suma;
```

Ejemplo completo:

```c
#include <stdio.h>

typedef int (*Operacion)(int, int);

int suma(int a, int b) { return a + b; }
int resta(int a, int b) { return a - b; }
int multiplica(int a, int b) { return a * b; }
int divide(int a, int b) { return b != 0 ? a / b : 0; }

int calcular(int a, int b, Operacion op) {
    return op(a, b);
}

int main() {
    Operacion ops[] = {suma, resta, multiplica, divide};
    char* nombres[] = {"suma", "resta", "multiplica", "divide"};

    int a = 20, b = 4;

    printf("Operaciones con %d y %d:\n", a, b);
    for (int i = 0; i < 4; i++) {
        printf("  %s: %d\n", nombres[i], calcular(a, b, ops[i]));
    }

    return 0;
}
```

--- TYPEDEF Y VISIBILIDAD ---

Puedes poner `typedef` en:
- Ambito global (fuera de funciones): visible en todo el archivo.
- Ambito local (dentro de funciones): visible solo en esa funcion.
- En archivos de cabecera (.h): compartido entre multiples archivos.

--- TYPEDEF VS #define ---

`typedef` y `#define` pueden parecer similares, pero son fundamentalmente
diferentes:

+-----------------------------------+---------------------------+---------------------------+
| Caracteristica                    | typedef                   | #define                   |
+-----------------------------------+---------------------------+---------------------------+
| Procesamiento                     | Compilador                | Preprocesador             |
| Creacion de tipo                  | Si (alias real)           | No (sustitucion txt)      |
| Scope                             | Respeta bloques           | Global (archivo)          |
| Punteros                          | Correcto                  | Problemas con *           |
| Arrays                            | Correcto                  | Correcto                  |
| Debugging                         | Visible en debugger       | Invisible                 |
+-----------------------------------+---------------------------+---------------------------+

Ejemplo de la diferencia:

```c
#define PTR_INT int*
typedef int* TYPEDEF_INT_PTR;

PTR_INT a, b;           // a es int*, b es int (error de concepto!)
TYPEDEF_INT_PTR c, d;   // c y d son int* (correcto)
```

--- CONVENCIONES DE NOMBRES ---

No hay una convencion universal, pero estas son las mas usadas:

1. Mayuscula inicial (recomendado para tipos):
   typedef struct { ... } Punto, Cliente, Producto;

2. _t al final (estilo POSIX, comun en sistemas):
   typedef unsigned int size_t;
   typedef struct { ... } punto_t;

3. Prefijo (estilo Windows / Hungaro):
   typedef struct { ... } PT_Punto;

4. Todo mayusculas (para tipos que parecen constantes, poco comun):
   typedef unsigned int UINT;

--- CUANDO USAR TYPEDEF ---

Usa typedef cuando:
✓ Quieras ocultar detalles de implementacion.
✓ Trabajes con estructuras complejas.
✓ Necesites tipos portables (p.ej. uint8_t, int16_t).
✓ Definas punteros a funcion (callback).

NO uses typedef cuando:
✗ El tipo sea obvio y ampliamente conocido (int, char, float).
✗ Quieras mantener la transparencia del tipo subyacente.
✗ Estes escribiendo codigo que otros leeran y necesitan ver el tipo real.

--- EJEMPLO PRACTICO: SISTEMA DE VENTAS ---

```c
#include <stdio.h>
#include <string.h>

typedef unsigned int uint;
typedef unsigned char byte;

typedef struct {
    uint dia;
    uint mes;
    uint anio;
} Fecha;

typedef struct {
    char producto[50];
    uint cantidad;
    float precioUnitario;
    Fecha fechaVenta;
} Venta;

void mostrarVenta(Venta v) {
    printf("  %s | Cant: %u | $%.2f c/u | ",
           v.producto, v.cantidad, v.precioUnitario);
    printf("Fecha: %02u/%02u/%04u\n",
           v.fechaVenta.dia, v.fechaVenta.mes, v.fechaVenta.anio);
}

float totalVenta(Venta v) {
    return v.cantidad * v.precioUnitario;
}

int main() {
    Venta ventas[] = {
        {"Laptop Gamer", 2, 15000.00, {15, 3, 2025}},
        {"Mouse RGB", 5, 850.50, {16, 3, 2025}},
        {"Teclado Mecanico", 3, 1200.00, {17, 3, 2025}},
        {"Monitor 27\"", 1, 8500.00, {18, 3, 2025}},
        {"Audifonos BT", 4, 950.00, {19, 3, 2025}}
    };

    uint numVentas = sizeof(ventas) / sizeof(ventas[0]);

    printf("=== REPORTE DE VENTAS ===\n\n");

    float granTotal = 0.0f;
    for (uint i = 0; i < numVentas; i++) {
        printf("Venta #%u:\n", i + 1);
        mostrarVenta(ventas[i]);
        float tv = totalVenta(ventas[i]);
        printf("  Total: $%.2f\n\n", tv);
        granTotal += tv;
    }

    printf("GRAN TOTAL: $%.2f\n", granTotal);
    printf("VENTAS REALIZADAS: %u\n", numVentas);

    return 0;
}
```

═══════════════════════════════════════════════════════════════════════════════
                    PREGUNTAS FRECUENTES (FAQ)
═══════════════════════════════════════════════════════════════════════════════

P: ¿`typedef` crea un nuevo tipo?
R: No. Crea un alias, un nombre alternativo para un tipo existente.
   Sigue siendo el mismo tipo subyacente.

P: ¿Puedo usar `typedef` dentro de una funcion?
R: Si. El typedef sera visible solo dentro de esa funcion (ambito local).

P: ¿Es mejor `typedef struct` o `struct` normal?
R: `typedef` es mas limpio y profesional. En proyectos grandes, el uso
   de `typedef` para estructuras es practicamente universal.

P: ¿Por que algunas librerias usan sufijo `_t`?
R: Es una convencion POSIX. Por ejemplo `size_t`, `time_t`, `int32_t`.
   Indica que es un typedef, no una variable.

P: ¿Puedo hacer typedef de un typedef?
R: Si. Puedes crear alias sobre alias, aunque no es recomendable porque
   reduce la claridad.

P: ¿Hay algun riesgo con `typedef` y punteros?
R: Si, ocultar un puntero detras de un typedef puede ser confuso.
   `typedef char* Cadena;` — quien lee el codigo no sabe que es un
   puntero. Uralo con criterio.

P: ¿`typedef` afecta al rendimiento?
R: No. Es puramente una abstraccion en tiempo de compilacion. Genera
   el mismo codigo maquina.

--- TYPEDEF CON ESTRUCTURAS AUTO-REFERENCIADAS ---

Un caso comun son las listas enlazadas:

```c
#include <stdio.h>
#include <stdlib.h>

// La estructura necesita nombre para auto-referenciarse
typedef struct Nodo {
    int dato;
    struct Nodo *siguiente;
} Nodo;

typedef Nodo* Lista;

// Funcion para agregar al inicio
void agregarInicio(Lista *lista, int valor) {
    Nodo *nuevo = malloc(sizeof(Nodo));
    if (!nuevo) return;
    nuevo->dato = valor;
    nuevo->siguiente = *lista;
    *lista = nuevo;
}

void imprimirLista(Lista lista) {
    printf("Lista: ");
    while (lista) {
        printf("%d", lista->dato);
        if (lista->siguiente) printf(" -> ");
        lista = lista->siguiente;
    }
    printf("\n");
}

void liberarLista(Lista *lista) {
    Nodo *actual = *lista;
    while (actual) {
        Nodo *temp = actual;
        actual = actual->siguiente;
        free(temp);
    }
    *lista = NULL;
}

int main() {
    Lista milista = NULL;

    agregarInicio(&milista, 30);
    agregarInicio(&milista, 20);
    agregarInicio(&milista, 10);

    imprimirLista(milista);
    liberarLista(&milista);

    return 0;
}
```

--- TYPEDEF PARA FUNCIONES DE COMPARACION (qsort) ---

La funcion qsort de la biblioteca estandar recibe un puntero a funcion
de comparacion. Con typedef se hace mas claro:

```c
#include <stdio.h>
#include <stdlib.h>

// Definimos el tipo de funcion de comparacion
typedef int (*Comparador)(const void*, const void*);

typedef struct {
    char nombre[50];
    int edad;
    float promedio;
} Estudiante;

// Funciones de comparacion
int porEdad(const void *a, const void *b) {
    return ((Estudiante*)a)->edad - ((Estudiante*)b)->edad;
}

int porPromedio(const void *a, const void *b) {
    float diff = ((Estudiante*)a)->promedio - ((Estudiante*)b)->promedio;
    return (diff > 0) ? 1 : (diff < 0) ? -1 : 0;
}

int porNombre(const void *a, const void *b) {
    return strcmp(((Estudiante*)a)->nombre, ((Estudiante*)b)->nombre);
}

// Funcion generica de ordenamiento
void ordenarEstudiantes(Estudiante *arr, int n, Comparador cmp) {
    qsort(arr, n, sizeof(Estudiante), cmp);
}

void mostrarEstudiantes(Estudiante *arr, int n) {
    printf("%-20s %4s %8s\n", "Nombre", "Edad", "Promedio");
    printf("----------------------------\n");
    for (int i = 0; i < n; i++) {
        printf("%-20s %4d %8.2f\n",
               arr[i].nombre, arr[i].edad, arr[i].promedio);
    }
    printf("\n");
}

int main() {
    Estudiante grupo[] = {
        {"Carlos Ruiz", 22, 8.5},
        {"Ana Lopez", 19, 9.2},
        {"Luis Torres", 25, 7.8},
        {"Elena Gomez", 21, 9.0}
    };
    int n = sizeof(grupo) / sizeof(grupo[0]);

    printf("Original:\n");
    mostrarEstudiantes(grupo, n);

    ordenarEstudiantes(grupo, n, porNombre);
    printf("Ordenado por nombre:\n");
    mostrarEstudiantes(grupo, n);

    ordenarEstudiantes(grupo, n, porPromedio);
    printf("Ordenado por promedio (ascendente):\n");
    mostrarEstudiantes(grupo, n);

    return 0;
}
```

--- TYPEDEF PARA CALLBACKS ---

Los callbacks son funciones que se pasan como argumento:

```c
#include <stdio.h>

typedef void (*CallbackEvento)(int codigo, const char *mensaje);

typedef struct {
    int id;
    char nombre[50];
    CallbackEvento onError;
    CallbackEvento onSuccess;
} Operacion;

void manejadorError(int codigo, const char *msg) {
    printf("[ERROR %d] %s\n", codigo, msg);
}

void manejadorExito(int codigo, const char *msg) {
    printf("[OK %d] %s\n", codigo, msg);
}

void ejecutarOperacion(Operacion *op, int parametro) {
    if (parametro < 0) {
        op->onError(400, "Parametro negativo no valido");
    } else {
        op->onSuccess(200, "Operacion completada");
    }
}

int main() {
    Operacion op = {1, "Prueba", manejadorError, manejadorExito};

    printf("Ejecutando con parametro -5:\n");
    ejecutarOperacion(&op, -5);

    printf("\nEjecutando con parametro 42:\n");
    ejecutarOperacion(&op, 42);

    return 0;
}
```

```c
// -------- persona.c --------
#include "persona.h"
#include <stdio.h>
#include <stdlib.h>

Persona* crearPersona(int id, const char *nombre, int edad, const char *email) {
    Persona *p = malloc(sizeof(Persona));
    if (p) {
        p->id = id;
        strcpy(p->nombre, nombre);
        p->edad = edad;
        strcpy(p->email, email);
    }
    return p;
}

void mostrarPersona(const Persona *p) {
    printf("[%d] %s, %d anios - %s\n",
           p->id, p->nombre, p->edad, p->email);
}
```

--- TYPEDEF Y DEFINICION MULTIPLE ---

Puedes definir multiples alias de una vez:

```c
typedef unsigned char byte, uint8, uchar;
// byte, uint8, uchar son todos alias de unsigned char
```

--- TYPEDEF CON UNIONES ---

```c
typedef union {
    int entero;
    float flotante;
    char *cadena;
} ValorFlexible;
```

--- CUANDO USAR TYPEDEF EN PROYECTOS REALES ---

En la practica profesional, el uso de typedef es casi universal para:

1. Tipos de datos del dominio (Cliente, Producto, Factura)
2. Tipos portables (int32_t, uint16_t)
3. Callbacks y punteros a funcion
4. Tipos opacos (punteros a structs no definidos)

Ejemplo de tipo opaco (encapsulacion):

```c
// En el .h:
typedef struct Banco Banco;  // Tipo opaco

Banco* banco_crear(const char *nombre);
void banco_destruir(Banco *b);
int banco_depositar(Banco *b, int cuenta, double monto);

// En el .c:
struct Banco {
    char nombre[100];
    double fondos[1000];
    int numCuentas;
};
```
