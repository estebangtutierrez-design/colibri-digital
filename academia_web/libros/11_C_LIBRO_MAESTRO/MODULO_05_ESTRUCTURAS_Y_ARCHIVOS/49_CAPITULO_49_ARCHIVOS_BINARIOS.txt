Hasta ahora hemos trabajado con archivos de texto, donde los datos se
almacenan como caracteres legibles. Pero existe otro mundo: los archivos
binarios.

En los archivos binarios, los datos se almacenan exactamente como estan
en memoria, byte por byte. Esto tiene ventajas importantes:
- Ocupan menos espacio (un int ocupa 4 bytes, no 11 como "2147483647").
- Son mas rapidos de leer/escribir.
- Preservan la estructura exacta de los datos (structs completos).
- No hay conversion de formato.

En este capitulo aprenderas a trabajar con archivos binarios en C.

--- TEXTO VS BINARIO ---

+-----------------------------+----------------------------+----------------------------+
| Caracteristica              | Texto                      | Binario                    |
+-----------------------------+----------------------------+----------------------------+
| Legible por humanos         | Si                         | No                         |
| Tamano                      | Mayor                      | Menor                      |
| Velocidad E/S               | Menor (conversion)         | Mayor (copia directa)     |
| Precision numerica          | Puede perder (decimales)   | Exacta                     |
| Portable entre sistemas     | Si (ASCII/UTF-8)           | Depende (endianness)       |
| Editable con bloc de notas | Si                         | No                         |
| Estructuras complejas       | Dificil (parsear)          | Directo (fread/fwrite)    |
+-----------------------------+----------------------------+----------------------------+

--- MODO BINARIO EN FOPEN ---

Para trabajar con archivos binarios, agrega "b" al modo:

    fopen("datos.bin", "rb");   // Lectura binaria
    fopen("datos.bin", "wb");   // Escritura binaria
    fopen("datos.bin", "ab");   // Append binario
    fopen("datos.bin", "r+b");  // Lectura/escritura binaria

En Linux/Unix no hay diferencia practica entre texto y binario.
En Windows, "b" evita la conversion de \r\n a \n.

--- ESCRITURA DE STRUCTS CON FWRITE ---

La gran ventaja de los binarios es poder escribir structs completos:

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int id;
    char nombre[50];
    float salario;
    int activo;
} Empleado;

int main() {
    FILE *fp = fopen("empleados.bin", "wb");
    if (!fp) { printf("Error.\n"); return 1; }

    Empleado emp[] = {
        {101, "Ana Lopez", 45000.50, 1},
        {102, "Carlos Ruiz", 52000.00, 1},
        {103, "Elena Gomez", 38000.75, 0},
        {104, "Luis Torres", 61000.00, 1}
    };

    int n = sizeof(emp) / sizeof(emp[0]);
    fwrite(emp, sizeof(Empleado), n, fp);

    printf("Escritos %d empleados en empleados.bin\n", n);
    fclose(fp);
    return 0;
}
```

--- LECTURA DE STRUCTS CON FREAD ---

Para leer los datos de vuelta:

```c
#include <stdio.h>

typedef struct {
    int id;
    char nombre[50];
    float salario;
    int activo;
} Empleado;

int main() {
    FILE *fp = fopen("empleados.bin", "rb");
    if (!fp) { printf("Error.\n"); return 1; }

    Empleado emp[10];
    int n = fread(emp, sizeof(Empleado), 10, fp);

    printf("Leidos %d empleados:\n", n);
    for (int i = 0; i < n; i++) {
        printf("%03d | %-20s | $%8.2f | %s\n",
               emp[i].id, emp[i].nombre, emp[i].salario,
               emp[i].activo ? "Activo" : "Inactivo");
    }

    fclose(fp);
    return 0;
}
```

--- FSEEK: POSICIONARSE EN EL ARCHIVO ---

    int fseek(FILE *fp, long offset, int origen);

Origen puede ser:
- SEEK_SET: desde el inicio del archivo.
- SEEK_CUR: desde la posicion actual.
- SEEK_END: desde el final del archivo.

```c
typedef struct { int id; char nombre[50]; float salario; } Empleado;

// Ir al tercer registro (indice 2)
fseek(fp, 2 * sizeof(Empleado), SEEK_SET);
fread(&emp, sizeof(Empleado), 1, fp);

// Ir 100 bytes hacia adelante desde la posicion actual
fseek(fp, 100, SEEK_CUR);

// Ir al final
fseek(fp, 0, SEEK_END);
```

--- FTELL: SABER LA POSICION ACTUAL ---

    long ftell(FILE *fp);

Retorna la posicion actual en bytes desde el inicio.

```c
long pos = ftell(fp);
printf("Posicion actual: %ld\n", pos);
```

--- REWIND: VOLVER AL INICIO ---

    void rewind(FILE *fp);

Equivalente a fseek(fp, 0, SEEK_SET).

--- ACCESO ALEATORIO: LEER UN REGISTRO ESPECIFICO ---

```c
#include <stdio.h>

typedef struct {
    int id;
    char nombre[50];
    float salario;
} Empleado;

int leerEmpleado(FILE *fp, int indice, Empleado *emp) {
    fseek(fp, indice * sizeof(Empleado), SEEK_SET);
    return fread(emp, sizeof(Empleado), 1, fp);
}

int main() {
    FILE *fp = fopen("empleados.bin", "rb");
    if (!fp) { printf("Error.\n"); return 1; }

    Empleado e;
    int indices[] = {2, 0, 3};

    for (int i = 0; i < 3; i++) {
        if (leerEmpleado(fp, indices[i], &e)) {
            printf("Registro %d: %s - $%.2f\n",
                   indices[i], e.nombre, e.salario);
        }
    }

    fclose(fp);
    return 0;
}
```

--- MODIFICAR UN REGISTRO ESPECIFICO ---

```c
#include <stdio.h>

typedef struct {
    int id;
    char nombre[50];
    float salario;
    int activo;
} Empleado;

int main() {
    FILE *fp = fopen("empleados.bin", "r+b");
    if (!fp) { printf("Error.\n"); return 1; }

    Empleado e;

    // Leer el registro 1 (indice 1)
    fseek(fp, 1 * sizeof(Empleado), SEEK_SET);
    fread(&e, sizeof(Empleado), 1, fp);
    printf("Antes: %s - $%.2f\n", e.nombre, e.salario);

    // Modificar salario
    e.salario = 55000.00;

    // Volver a la posicion y escribir
    fseek(fp, 1 * sizeof(Empleado), SEEK_SET);
    fwrite(&e, sizeof(Empleado), 1, fp);

    printf("Salario actualizado.\n");

    fclose(fp);
    return 0;
}
```

--- CONTAR REGISTROS EN UN ARCHIVO BINARIO ---

```c
long contarRegistros(FILE *fp, size_t tamRegistro) {
    fseek(fp, 0, SEEK_END);
    long tamArchivo = ftell(fp);
    rewind(fp);
    return tamArchivo / tamRegistro;
}

int main() {
    FILE *fp = fopen("empleados.bin", "rb");
    if (!fp) { printf("Error.\n"); return 1; }

    long n = contarRegistros(fp, sizeof(Empleado));
    printf("Total de empleados: %ld\n", n);

    fclose(fp);
    return 0;
}
```

--- SERIALIZACION: STRUCT A BYTES ---

Los structs con padding pueden no ser portables. Para mayor control,
puedes serializar campo por campo:

```c
void escribirEmpleado(FILE *fp, const Empleado *e) {
    fwrite(&e->id, sizeof(e->id), 1, fp);
    fwrite(e->nombre, sizeof(e->nombre), 1, fp);
    fwrite(&e->salario, sizeof(e->salario), 1, fp);
    fwrite(&e->activo, sizeof(e->activo), 1, fp);
}

void leerEmpleado(FILE *fp, Empleado *e) {
    fread(&e->id, sizeof(e->id), 1, fp);
    fread(e->nombre, sizeof(e->nombre), 1, fp);
    fread(&e->salario, sizeof(e->salario), 1, fp);
    fread(&e->activo, sizeof(e->activo), 1, fp);
}
```

--- EJEMPLO: AGENDA BINARIA ---

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

typedef struct {
    int id;
    char nombre[50];
    char telefono[15];
    char email[60];
} Contacto;

const char *ARCHIVO = "agenda.bin";

int agregarContacto(Contacto c) {
    FILE *fp = fopen(ARCHIVO, "ab");
    if (!fp) return 0;
    int ok = fwrite(&c, sizeof(Contacto), 1, fp);
    fclose(fp);
    return ok;
}

int listarContactos() {
    FILE *fp = fopen(ARCHIVO, "rb");
    if (!fp) return 0;

    Contacto c;
    int n = 0;
    while (fread(&c, sizeof(Contacto), 1, fp)) {
        printf("%03d | %-20s | %-12s | %s\n",
               c.id, c.nombre, c.telefono, c.email);
        n++;
    }

    fclose(fp);
    return n;
}

Contacto* buscarPorNombre(const char *nombre, int *n) {
    FILE *fp = fopen(ARCHIVO, "rb");
    if (!fp) { *n = 0; return NULL; }

    Contacto buffer[100];
    int encontrados = 0;

    Contacto c;
    while (fread(&c, sizeof(Contacto), 1, fp)) {
        if (strstr(c.nombre, nombre)) {
            buffer[encontrados++] = c;
        }
    }

    fclose(fp);

    if (encontrados > 0) {
        Contacto *resultados = malloc(encontrados * sizeof(Contacto));
        memcpy(resultados, buffer, encontrados * sizeof(Contacto));
        *n = encontrados;
        return resultados;
    }

    *n = 0;
    return NULL;
}

int eliminarContacto(int id) {
    FILE *fp = fopen(ARCHIVO, "rb");
    if (!fp) return 0;

    Contacto buffer[100];
    int n = 0;

    Contacto c;
    while (fread(&c, sizeof(Contacto), 1, fp)) {
        if (c.id != id) buffer[n++] = c;
    }

    fclose(fp);

    fp = fopen(ARCHIVO, "wb");
    if (!fp) return 0;

    int ok = fwrite(buffer, sizeof(Contacto), n, fp);
    fclose(fp);
    return ok;
}

int main() {
    int op;
    do {
        printf("\n=== AGENDA BINARIA ===\n");
        printf("1. Agregar contacto\n");
        printf("2. Listar contactos\n");
        printf("3. Buscar por nombre\n");
        printf("4. Eliminar contacto\n");
        printf("5. Salir\n");
        printf("Opcion: ");
        scanf("%d", &op);
        getchar();

        if (op == 1) {
            Contacto c;
            printf("ID: "); scanf("%d", &c.id); getchar();
            printf("Nombre: "); fgets(c.nombre, 50, stdin);
            c.nombre[strcspn(c.nombre, "\n")] = 0;
            printf("Telefono: "); fgets(c.telefono, 15, stdin);
            c.telefono[strcspn(c.telefono, "\n")] = 0;
            printf("Email: "); fgets(c.email, 60, stdin);
            c.email[strcspn(c.email, "\n")] = 0;

            if (agregarContacto(c))
                printf("Contacto agregado.\n");
        }
        else if (op == 2) {
            int n = listarContactos();
            printf("Total: %d contactos.\n", n);
        }
        else if (op == 3) {
            char nombre[50];
            printf("Nombre a buscar: "); fgets(nombre, 50, stdin);
            nombre[strcspn(nombre, "\n")] = 0;

            int n;
            Contacto *res = buscarPorNombre(nombre, &n);
            if (res) {
                for (int i = 0; i < n; i++)
                    printf("%03d | %s | %s | %s\n",
                           res[i].id, res[i].nombre,
                           res[i].telefono, res[i].email);
                free(res);
            } else {
                printf("No encontrados.\n");
            }
        }
        else if (op == 4) {
            int id;
            printf("ID a eliminar: "); scanf("%d", &id);
            if (eliminarContacto(id))
                printf("Eliminado (si existia).\n");
        }
    } while (op != 5);

    return 0;
}
```

--- CONSIDERACIONES DE PORTABILIDAD ---

1. ENDIANNESS: El orden de bytes puede variar entre sistemas (little-endian
   vs big-endian). Los archivos binarios no son portables entre sistemas
   con diferente endianness.

2. PADDING: El tamaño de los structs puede variar segun el compilador
   y las opciones de alineacion. Usa __attribute__((packed)) en GCC
   para eliminar padding.

3. TAMAÑO DE TIPOS: int puede ser 2 o 4 bytes segun el sistema. Usa
   tipos de tamaño fijo como int32_t, uint16_t de <stdint.h>.

```c
#include <stdint.h>

typedef struct {
    int32_t id;        // Entero de exactamente 32 bits
    char nombre[50];
    float salario;     // IEEE 754 (estandar)
    int8_t activo;     // Entero de 8 bits
} __attribute__((packed)) EmpleadoPortable;
```

--- EJEMPLO: PACKING DE STRUCTS ---

```c
#include <stdio.h>
#include <stdint.h>

// Con packing: sin padding
typedef struct __attribute__((packed)) {
    uint32_t id;
    char nombre[50];
    float salario;
    uint8_t activo;
} EmpleadoPacked;

int main() {
    printf("Con packing: %zu bytes\n", sizeof(EmpleadoPacked));
    printf("Sin packing: %zu bytes (tipico)\n", sizeof(EmpleadoPacked) + 3);
    // La diferencia se debe al padding
    return 0;
}
```

══════════════════════════════════════════════════════════════════════
                    PREGUNTAS FRECUENTES (FAQ)
══════════════════════════════════════════════════════════════════════


P: Cuando usar texto vs binario?
R: Texto para datos legibles, configuracion, exportacion. Binario para
   rendimiento, estructuras, acceso aleatorio.

P: Los archivos binarios son portables?
R: No completamente. Endianness, padding y tamaño de tipos pueden variar.

P: Que es __attribute__((packed))?
R: Directiva de GCC que elimina el padding de los structs, haciendo que
   ocupen exactamente la suma de sus miembros.

P: fseek con SEEK_END desde el inicio?
R: fseek(fp, 0, SEEK_END) va al final. fseek(fp, -10, SEEK_END) va 10
   bytes antes del final.

P: Puedo mezclar texto y binario?
R: Tecnicamente si, pero no es recomendable. Usa un formato u otro.
