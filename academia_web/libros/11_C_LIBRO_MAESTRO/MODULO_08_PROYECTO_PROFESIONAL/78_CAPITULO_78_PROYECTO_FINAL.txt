=== 2. MÓDULO 1: BUFFER DE TEXTO ===

El buffer es el corazón del editor. Almacena el texto en memoria
y permite operaciones de inserción, eliminación y modificación.


Implementación del buffer usando una estructura tipo gap buffer
(eficiente para inserciones y eliminaciones en el cursor):

```c
// buffer.c (implementación parcial)
#include "buffer.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct Linea {
    char*  texto;
    size_t capacidad;
    size_t longitud;
} Linea;

struct Buffer {
    Linea* lineas;
    size_t num_lineas;
    size_t capacidad;
    bool   modificado;
    // Clipboard
    Linea* clipboard;
    size_t clip_count;
};

static Linea* linea_crear(const char* texto) {
    Linea* l = malloc(sizeof(Linea));
    if (!l) return NULL;

    size_t len = texto ? strlen(texto) : 0;
    l->capacidad = len + 1;
    l->texto = malloc(l->capacidad);
    if (!l->texto) { free(l); return NULL; }

    if (texto) {
        memcpy(l->texto, texto, len);
    }
    l->texto[len] = '\0';
    l->longitud = len;
    return l;
}

static void linea_destruir(Linea* l) {
    if (!l) return;
    free(l->texto);
    free(l);
}

static bool linea_insertar_char(Linea* l, size_t col, char c) {
    if (!l || col > l->longitud) return false;

    if (l->longitud + 1 >= l->capacidad) {
        size_t nueva_cap = l->capacidad * 2;
        if (nueva_cap < 16) nueva_cap = 16;
        char* nuevo = realloc(l->texto, nueva_cap);
        if (!nuevo) return false;
        l->texto = nuevo;
        l->capacidad = nueva_cap;
    }

    memmove(&l->texto[col + 1], &l->texto[col],
            l->longitud - col + 1);
    l->texto[col] = c;
    l->longitud++;
    return true;
}

Buffer* buffer_crear(void) {
    Buffer* b = malloc(sizeof(Buffer));
    if (!b) return NULL;

    b->capacidad = 100;
    b->lineas = calloc(b->capacidad, sizeof(Linea));
    if (!b->lineas) { free(b); return NULL; }

    b->num_lineas = 0;
    b->modificado = false;
    b->clipboard = NULL;
    b->clip_count = 0;

    return b;
}

bool buffer_insertar_linea(Buffer* b, size_t pos, const char* texto) {
    if (!b || pos > b->num_lineas) return false;

    if (b->num_lineas >= b->capacidad) {
        size_t nueva_cap = b->capacidad * 2;
        Linea* nuevas = realloc(b->lineas,
                                nueva_cap * sizeof(Linea));
        if (!nuevas) return false;
        b->lineas = nuevas;
        b->capacidad = nueva_cap;
    }

    Linea* l = linea_crear(texto);
    if (!l) return false;

    memmove(&b->lineas[pos + 1], &b->lineas[pos],
            (b->num_lineas - pos) * sizeof(Linea));
    b->lineas[pos] = *l;
    b->num_lineas++;
    b->modificado = true;
    free(l);  // La estructura se copió, liberar el original
    return true;
}

bool buffer_eliminar_linea(Buffer* b, size_t pos) {
    if (!b || pos >= b->num_lineas) return false;

    free(b->lineas[pos].texto);
    memmove(&b->lineas[pos], &b->lineas[pos + 1],
            (b->num_lineas - pos - 1) * sizeof(Linea));
    b->num_lineas--;
    b->modificado = true;
    return true;
}

const char* buffer_obtener_linea(const Buffer* b, size_t pos) {
    if (!b || pos >= b->num_lineas) return NULL;
    return b->lineas[pos].texto;
}

size_t buffer_num_lineas(const Buffer* b) {
    return b ? b->num_lineas : 0;
}

bool buffer_insertar_char(Buffer* b, size_t linea, size_t col, char c) {
    if (!b || linea >= b->num_lineas) return false;
    bool ok = linea_insertar_char(&b->lineas[linea], col, c);
    if (ok) b->modificado = true;
    return ok;
}

bool buffer_cargar_archivo(Buffer* b, const char* ruta) {
    if (!b || !ruta) return false;

    FILE* f = fopen(ruta, "r");
    if (!f) return false;

    buffer_limpiar(b);

    char* linea = NULL;
    size_t cap = 0;
    ssize_t len;

    while ((len = getline(&linea, &cap, f)) != -1) {
        if (len > 0 && linea[len - 1] == '\n') {
            linea[len - 1] = '\0';
        }
        if (!buffer_agregar_linea(b, linea)) {
            free(linea);
            fclose(f);
            return false;
        }
    }

    free(linea);
    fclose(f);
    b->modificado = false;
    return true;
}

bool buffer_guardar_archivo(const Buffer* b, const char* ruta) {
    if (!b || !ruta) return false;

    FILE* f = fopen(ruta, "w");
    if (!f) return false;

    for (size_t i = 0; i < b->num_lineas; i++) {
        fprintf(f, "%s\n", b->lineas[i].texto);
    }

    fclose(f);
    return true;
}
```

Este módulo maneja la terminal en modo raw para capturar teclas

```c

#ifndef TERMINAL_H
#define TERMINAL_H

#include <stddef.h>

typedef struct {
    int ancho;
    int alto;
} TamanoTerminal;

bool     terminal_entrar_modo_raw(void);
bool     terminal_salir_modo_raw(void);
TamanoTerminal terminal_obtener_tamano(void);
void     terminal_mover_cursor(int fila, int col);
void     terminal_limpiar_pantalla(void);
void     terminal_limpiar_linea(void);
void     terminal_mostrar_texto(int fila, int col,
                                const char* texto, size_t len);
void     terminal_ocultar_cursor(void);
void     terminal_mostrar_cursor(void);

#endif
```

```c
// terminal.c
#include "terminal.h"
#include <stdio.h>
#include <unistd.h>
#include <termios.h>
#include <sys/ioctl.h>

static struct termios term_original;
static bool modo_raw = false;

bool terminal_entrar_modo_raw(void) {
    if (modo_raw) return true;

    if (!isatty(STDIN_FILENO)) return false;

    tcgetattr(STDIN_FILENO, &term_original);
    struct termios raw = term_original;

    raw.c_iflag &= ~(BRKINT | ICRNL | INPCK | ISTRIP | IXON);
    raw.c_oflag &= ~(OPOST);
    raw.c_cflag |= (CS8);
    raw.c_lflag &= ~(ECHO | ICANON | IEXTEN | ISIG);
    raw.c_cc[VMIN] = 0;
    raw.c_cc[VTIME] = 1;

    if (tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw) < 0) {
        return false;
    }

    modo_raw = true;
    return true;
}

bool terminal_salir_modo_raw(void) {
    if (!modo_raw) return true;
    int r = tcsetattr(STDIN_FILENO, TCSAFLUSH, &term_original);
    modo_raw = false;
    return r == 0;
}

TamanoTerminal terminal_obtener_tamano(void) {
    TamanoTerminal t = {80, 24};  // Valores por defecto

    struct winsize ws;
    if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) == 0) {
        t.ancho = ws.ws_col;
        t.alto = ws.ws_row;
    }

    return t;
}

void terminal_mover_cursor(int fila, int col) {
    printf("\033[%d;%dH", fila, col);
}

void terminal_limpiar_pantalla(void) {
    printf("\033[2J\033[H");
}

void terminal_mostrar_texto(int fila, int col,
                            const char* texto, size_t len) {
    terminal_mover_cursor(fila, col);
    fwrite(texto, 1, len, stdout);
}
```

```c
#ifndef INPUT_H

typedef enum {

    TECLA_CTRL_C,    // Copiar
    TECLA_CTRL_X,    // Cortar
    TECLA_CTRL_V,    // Pegar
    TECLA_CTRL_G,    // Ir a línea

} Tecla;

typedef struct {
    Tecla  tecla;
```

```c
// input.c
#include "input.h"
#include <unistd.h>
#include <stdio.h>

EventoTecla input_leer_tecla(void) {
    EventoTecla evento = {TECLA_NULA, 0};
    char c;
    ssize_t n = read(STDIN_FILENO, &c, 1);

    if (n <= 0) return evento;

    if (c == '\033') {
        char seq[3];
        if (read(STDIN_FILENO, &seq[0], 1) != 1) {
            evento.tecla = TECLA_ESC;
            return evento;
        }
        if (read(STDIN_FILENO, &seq[1], 1) != 1) {
            return evento;
        }

        if (seq[0] == '[') {
            switch (seq[1]) {
                case 'A': evento.tecla = TECLA_ARRIBA;     break;
                case 'B': evento.tecla = TECLA_ABAJO;      break;
                case 'C': evento.tecla = TECLA_DERECHA;    break;
                case 'D': evento.tecla = TECLA_IZQUIERDA;  break;
                case 'H': evento.tecla = TECLA_INICIO;     break;
                case 'F': evento.tecla = TECLA_FIN;        break;
                case '~': /* PgUp, PgDn, Delete */         break;
            }
        }

        return evento;
    }

    switch (c) {
        case 13:  evento.tecla = TECLA_ENTER;    break;
        case 127: evento.tecla = TECLA_BACKSPACE; break;
        case 17:  evento.tecla = TECLA_CTRL_Q;   break;
        case 19:  evento.tecla = TECLA_CTRL_S;   break;
        case 6:   evento.tecla = TECLA_CTRL_F;   break;
        case 18:  evento.tecla = TECLA_CTRL_R;   break;
        case 3:   evento.tecla = TECLA_CTRL_C;   break;
        case 24:  evento.tecla = TECLA_CTRL_X;   break;
        case 22:  evento.tecla = TECLA_CTRL_V;   break;
        case 7:   evento.tecla = TECLA_CTRL_G;   break;
        default:
            evento.tecla = TECLA_OTRO;
            evento.caracter = c;
            break;
    }

    return evento;
}
```

```c
// editor.c (implementación parcial)
#include "editor.h"
#include "terminal.h"
#include "input.h"
#include "buffer.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

struct Editor {
    Buffer* buffer;
    char    nombre_archivo[256];
    size_t  cursor_linea;
    size_t  cursor_col;
    size_t  desplazamiento_linea;  // Scroll vertical
    size_t  desplazamiento_col;    // Scroll horizontal
    bool    ejecutando;
};

Editor* editor_crear(void) {
    Editor* e = malloc(sizeof(Editor));
    if (!e) return NULL;

    e->buffer = buffer_crear();
    if (!e->buffer) { free(e); return NULL; }

    e->nombre_archivo[0] = '\0';
    e->cursor_linea = 0;
    e->cursor_col = 0;
    e->desplazamiento_linea = 0;
    e->desplazamiento_col = 0;
    e->ejecutando = true;

    return e;
}

void editor_destruir(Editor* e) {
    if (!e) return;
    buffer_destruir(e->buffer);
    free(e);
}

static void editor_dibujar(Editor* e) {
    TamanoTerminal tam = terminal_obtener_tamano();
    int altura_editor = tam.alto - 1;  // Una línea para barra de estado

    terminal_mover_cursor(0, 0);

    // Dibujar líneas visibles
    for (int fila = 0; fila < altura_editor; fila++) {
        size_t num_linea = e->desplazamiento_linea + fila;

        if (num_linea >= buffer_num_lineas(e->buffer)) {
            terminal_limpiar_linea();
        } else {
            // Mostrar número de línea
            char num_str[8];
            snprintf(num_str, sizeof(num_str), "%4zu ", num_linea + 1);
            printf("\033[90m%s\033[0m", num_str);  // Gris

            const char* linea = buffer_obtener_linea(e->buffer, num_linea);
            size_t len = buffer_longitud_linea(e->buffer, num_linea);

            if (len > e->desplazamiento_col) {
                size_t mostrar = len - e->desplazamiento_col;
                if (mostrar > (size_t)(tam.ancho - 6)) {
                    mostrar = tam.ancho - 6;
                }
                fwrite(linea + e->desplazamiento_col, 1, mostrar, stdout);
            }
        }
        printf("\r\n");
    }

    // Barra de estado
    printf("\033[7m");  // Invertido
    printf(" %s", e->nombre_archivo[0] ? e->nombre_archivo : "[Sin nombre]");
    if (buffer_modificado(e->buffer)) {
        printf(" [MODIFICADO]");
    }
    printf(" | Ln %zu, Col %zu", e->cursor_linea + 1, e->cursor_col + 1);
    printf("\033[0m");  // Reset
}

static void editor_mover_cursor(Editor* e, Tecla tecla) {
    switch (tecla) {
        case TECLA_ARRIBA:
            if (e->cursor_linea > 0) e->cursor_linea--;
            break;
        case TECLA_ABAJO:
            if (e->cursor_linea < buffer_num_lineas(e->buffer) - 1)
                e->cursor_linea++;
            break;
        case TECLA_IZQUIERDA:
            if (e->cursor_col > 0) e->cursor_col--;
            break;
        case TECLA_DERECHA: {
            size_t len = buffer_longitud_linea(e->buffer, e->cursor_linea);
            if (e->cursor_col < len) e->cursor_col++;
            break;
        }
        default:
            break;
    }

    // Ajustar scroll vertical
    TamanoTerminal tam = terminal_obtener_tamano();
    int altura_editor = tam.alto - 1;

    if (e->cursor_linea < e->desplazamiento_linea) {
        e->desplazamiento_linea = e->cursor_linea;
    }
    if (e->cursor_linea >= e->desplazamiento_linea + (size_t)altura_editor) {
        e->desplazamiento_linea = e->cursor_linea - altura_editor + 1;
    }
}

static void editor_insertar_char(Editor* e, char c) {
    if (c == '\n') {
        // Dividir línea en el cursor
        const char* linea = buffer_obtener_linea(e->buffer, e->cursor_linea);
        const char* resto = linea + e->cursor_col;

        buffer_insertar_linea(e->buffer, e->cursor_linea + 1, resto);

        // Truncar línea actual
        // (simplificado: en realidad necesitamos modificar la longitud)
        e->cursor_linea++;
        e->cursor_col = 0;
    } else {
        buffer_insertar_char(e->buffer, e->cursor_linea,
                             e->cursor_col, c);
        e->cursor_col++;
    }
}

bool editor_abrir_archivo(Editor* e, const char* ruta) {
    if (!e || !ruta) return false;
    strncpy(e->nombre_archivo, ruta, sizeof(e->nombre_archivo) - 1);
    return buffer_cargar_archivo(e->buffer, ruta);
}

bool editor_guardar_archivo(Editor* e) {
    if (!e || !e->nombre_archivo[0]) return false;
    return buffer_guardar_archivo(e->buffer, e->nombre_archivo);
}

void editor_ejecutar(Editor* e) {
    if (!e) return;

    if (!terminal_entrar_modo_raw()) {
        fprintf(stderr, "Error: no se pudo entrar en modo raw\n");
        return;
    }

    terminal_ocultar_cursor();

    while (e->ejecutando) {
        editor_dibujar(e);

        EventoTecla ev = input_leer_tecla();

        switch (ev.tecla) {
            case TECLA_CTRL_Q:
                e->ejecutando = false;
                break;
            case TECLA_CTRL_S:
                if (editor_guardar_archivo(e)) {
                    // Mostrar "Guardado" en la barra de estado
                }
                break;
            case TECLA_ARRIBA:
            case TECLA_ABAJO:
            case TECLA_IZQUIERDA:
            case TECLA_DERECHA:
                editor_mover_cursor(e, ev.tecla);
                break;
            case TECLA_ENTER:
                editor_insertar_char(e, '\n');
                break;
            case TECLA_BACKSPACE:
                // Eliminar carácter anterior
                if (e->cursor_col > 0) {
                    e->cursor_col--;
                    buffer_eliminar_char(e->buffer, e->cursor_linea,
                                         e->cursor_col);
                }
                break;
            case TECLA_OTRO:
                if (isprint((unsigned char)ev.caracter)) {
                    editor_insertar_char(e, ev.caracter);
                }
                break;
            default:
                break;
        }
    }

    terminal_mostrar_cursor();
    terminal_salir_modo_raw();
    terminal_limpiar_pantalla();
}
```

=== 6. MÓDULO 5: BÚSQUEDA Y REEMPLAZO ===

```c
// buscar.h
#ifndef BUSCAR_H
#define BUSCAR_H

#include <stdbool.h>
#include <stddef.h>

typedef struct {
    size_t linea;
    size_t col;
    size_t longitud;
} ResultadoBusqueda;

bool buscar_primero(const Buffer* b, const char* patron,
                    ResultadoBusqueda* resultado);
bool buscar_siguiente(const Buffer* b, const char* patron,
                      ResultadoBusqueda* actual,
                      ResultadoBusqueda* siguiente);
size_t buscar_reemplazar(Buffer* b, const char* patron,
                         const char* reemplazo);
size_t buscar_reemplazar_todo(Buffer* b, const char* patron,
                              const char* reemplazo);

#endif
```

=== 7. Makefile ===

```makefile
CC = gcc
CFLAGS = -Wall -Wextra -std=c11 -g -O2
LDFLAGS =
SRCDIR = src
BUILDDIR = build
BINDIR = bin
SRCS = $(wildcard $(SRCDIR)/**/*.c) $(SRCDIR)/main.c
OBJS = $(patsubst $(SRCDIR)/%.c, $(BUILDDIR)/%.o, $(SRCS))
TARGET = $(BINDIR)/cedit

.PHONY: all clean run valgrind

all: $(TARGET)

$(TARGET): $(OBJS)
	@mkdir -p $(BINDIR)
	$(CC) $(CFLAGS) -o $@ $^ $(LDFLAGS)

$(BUILDDIR)/%.o: $(SRCDIR)/%.c
	@mkdir -p $(dir $@)
	$(CC) $(CFLAGS) -Iinclude -c -o $@ $<

run: $(TARGET)
	./$(TARGET)

valgrind: $(TARGET)
	valgrind --leak-check=full --show-leak-kinds=all ./$(TARGET)

clean:
	rm -rf $(BUILDDIR) $(BINDIR)
```

________________________________________________________________________________
  PREGUNTAS FRECUENTES
________________________________________________________________________________

P: ¿Por qué usar un gap buffer en lugar de un array de strings?
R: El gap buffer ofrece mejor rendimiento para inserciones y
eliminaciones cerca del cursor, que es el caso de uso más común
en un editor. Un array de strings simple requiere desplazar
muchos elementos.

P: ¿Cómo manejo archivos con caracteres no ASCII?
R: Para UTF-8, necesitas manejar caracteres multibyte. La función
mbrtowc() y wcwidth() te ayudan a determinar el ancho de cada
carácter en la pantalla.

P: ¿El editor puede abrir archivos binarios?
R: Técnicamente sí, pero no tiene sentido. Un editor de texto
debería detectar archivos binarios y mostrar una advertencia
antes de abrirlos.

P: ¿Cómo implemento la funcionalidad de deshacer?
R: Mantén una pila de operaciones. Cada operación (inserción,
eliminación, reemplazo) guarda suficiente información para
revertirla. La pila puede tener un límite de profundidad.
