SOLUCIÓN 1.3: Contador de frecuencia de palabras

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

#define MAX_PALABRAS 100000
#define MAX_LONG 256

typedef struct {
    char  palabra[MAX_LONG];
    int   frecuencia;
} Frecuencia;

int comparar_frecuencia(const void* a, const void* b) {
    return ((Frecuencia*)b)->frecuencia - ((Frecuencia*)a)->frecuencia;
}

int comparar_palabra(const void* a, const void* b) {
    return strcmp(((Frecuencia*)a)->palabra, ((Frecuencia*)b)->palabra);
}

int main(int argc, char** argv) {
    if (argc < 2) {
        fprintf(stderr, "Uso: %s <archivo>\n", argv[0]);
        return 1;
    }

    FILE* f = fopen(argv[1], "r");
    if (!f) {
        perror("Error al abrir archivo");
        return 1;
    }

    Frecuencia* freqs = calloc(MAX_PALABRAS, sizeof(Frecuencia));
    int total_palabras = 0;
    int num_unicas = 0;

    char palabra[MAX_LONG];
    while (fscanf(f, "%255s", palabra) == 1) {
        // Limpiar puntuación y convertir a minúsculas
        char limpia[MAX_LONG];
        int j = 0;
        for (int i = 0; palabra[i]; i++) {
            if (isalpha((unsigned char)palabra[i])) {
                limpia[j++] = tolower((unsigned char)palabra[i]);
            }
        }
        limpia[j] = '\0';
        if (j == 0) continue;

        total_palabras++;

        // Buscar si ya existe
        Frecuencia clave;
        strcpy(clave.palabra, limpia);
        Frecuencia* encontrada = bsearch(&clave, freqs, num_unicas,
                                          sizeof(Frecuencia), comparar_palabra);
        if (encontrada) {
            encontrada->frecuencia++;
        } else {
            if (num_unicas < MAX_PALABRAS) {
                strcpy(freqs[num_unicas].palabra, limpia);
                freqs[num_unicas].frecuencia = 1;
                num_unicas++;
                qsort(freqs, num_unicas, sizeof(Frecuencia), comparar_palabra);
            }
        }
    }
    fclose(f);

    // Ordenar por frecuencia (las 10 más comunes)
    qsort(freqs, num_unicas, sizeof(Frecuencia), comparar_frecuencia);

    printf("Total de palabras: %d\n", total_palabras);
    printf("Palabras únicas: %d\n", num_unicas);
    printf("\nTop 10 palabras más frecuentes:\n");
    int mostrar = num_unicas < 10 ? num_unicas : 10;
    for (int i = 0; i < mostrar; i++) {
        printf("%3d. %-20s %d\n", i + 1, freqs[i].palabra, freqs[i].frecuencia);
    }

    free(freqs);
    return 0;
}
```

SOLUCIÓN 2.2: Invertir string

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char* invertir_string(const char* original) {
    if (!original) return NULL;

    size_t len = strlen(original);
    char* invertido = malloc(len + 1);
    if (!invertido) return NULL;

    for (size_t i = 0; i < len; i++) {
        invertido[i] = original[len - 1 - i];
    }
    invertido[len] = '\0';

    return invertido;
}
```

SOLUCIÓN 3.3: Algoritmo de Kadane (subarreglo de suma máxima)

```c
#include <stdio.h>
#include <limits.h>

typedef struct {
    int suma_maxima;
    int inicio;
    int fin;
} ResultadoKadane;

ResultadoKadane kadane(int* arr, int n) {
    ResultadoKadane res = {INT_MIN, 0, 0};
    int suma_actual = 0;
    int inicio_temp = 0;

    for (int i = 0; i < n; i++) {
        suma_actual += arr[i];

        if (suma_actual > res.suma_maxima) {
            res.suma_maxima = suma_actual;
            res.inicio = inicio_temp;
            res.fin = i;
        }

        if (suma_actual < 0) {
            suma_actual = 0;
            inicio_temp = i + 1;
        }
    }

    return res;
}
```

SOLUCIÓN 2.4: Memory Pool

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <string.h>

typedef struct Bloque {
    struct Bloque* siguiente;
} Bloque;

typedef struct {
    Bloque* cabeza;
    size_t  tam_bloque;
    size_t  num_bloques;
    size_t  usados;
} Pool;

Pool* pool_crear(size_t tam_bloque, size_t num_bloques) {
    Pool* p = malloc(sizeof(Pool));
    if (!p) return NULL;

    p->tam_bloque = tam_bloque;
    p->num_bloques = num_bloques;
    p->usados = 0;
    p->cabeza = NULL;

    // Pre-asignar todos los bloques
    for (size_t i = 0; i < num_bloques; i++) {
        Bloque* b = malloc(tam_bloque + sizeof(Bloque));
        if (!b) {
            pool_destruir(p);
            return NULL;
        }
        b->siguiente = p->cabeza;
        p->cabeza = b;
    }

    return p;
}

void* pool_asignar(Pool* p) {
    if (!p || !p->cabeza) return NULL;

    Bloque* b = p->cabeza;
    p->cabeza = b->siguiente;
    p->usados++;

    return (void*)((uint8_t*)b + sizeof(Bloque));
}

void pool_liberar(Pool* p, void* ptr) {
    if (!p || !ptr) return;

    Bloque* b = (Bloque*)((uint8_t*)ptr - sizeof(Bloque));
    b->siguiente = p->cabeza;
    p->cabeza = b;
    p->usados--;
}

void pool_destruir(Pool* p) {
    if (!p) return;
    Bloque* actual = p->cabeza;
    while (actual) {
        Bloque* siguiente = actual->siguiente;
        free(actual);
        actual = siguiente;
    }
    free(p);
}
```

```c
typedef struct NodoHash {

    void*           valor;
    struct NodoHash* siguiente;
} NodoHash;

typedef struct {
    NodoHash** tabla;
    size_t     capacidad;
    size_t     count;
} HashTable;

static size_t hash_funcion(const char* clave, size_t capacidad) {
    size_t hash = 5381;
    int c;
    while ((c = *clave++)) {
        hash = ((hash << 5) + hash) + c;
    }
    return hash % capacidad;
}

HashTable* hash_crear(size_t capacidad_inicial) {
    HashTable* ht = malloc(sizeof(HashTable));
    if (!ht) return NULL;

    ht->capacidad = capacidad_inicial;
    ht->count = 0;
    ht->tabla = calloc(capacidad_inicial, sizeof(NodoHash*));
    if (!ht->tabla) { free(ht); return NULL; }

    return ht;
}

static bool hash_redimensionar(HashTable* ht) {
    size_t nueva_cap = ht->capacidad * 2;
    NodoHash** nueva_tabla = calloc(nueva_cap, sizeof(NodoHash*));
    if (!nueva_tabla) return false;

    // Re-hash
    for (size_t i = 0; i < ht->capacidad; i++) {
        NodoHash* actual = ht->tabla[i];
        while (actual) {
            NodoHash* siguiente = actual->siguiente;
            size_t idx = hash_funcion(actual->clave, nueva_cap);
            actual->siguiente = nueva_tabla[idx];
            nueva_tabla[idx] = actual;
            actual = siguiente;
        }
    }

    free(ht->tabla);
    ht->tabla = nueva_tabla;
    ht->capacidad = nueva_cap;
    return true;
}

bool hash_insertar(HashTable* ht, const char* clave, void* valor) {
    if (!ht || !clave) return false;

    // Redimensionar si es necesario
    if ((double)ht->count / ht->capacidad > FACTOR_CARGA_MAX) {
        if (!hash_redimensionar(ht)) return false;
    }

    size_t idx = hash_funcion(clave, ht->capacidad);

    // Verificar si ya existe
    NodoHash* actual = ht->tabla[idx];
    while (actual) {
        if (strcmp(actual->clave, clave) == 0) {
            actual->valor = valor;
            return true;
        }
        actual = actual->siguiente;
    }

    // Insertar nuevo nodo
    NodoHash* nodo = malloc(sizeof(NodoHash));
    if (!nodo) return false;

    nodo->clave = strdup(clave);
    if (!nodo->clave) { free(nodo); return false; }
    nodo->valor = valor;
    nodo->siguiente = ht->tabla[idx];
    ht->tabla[idx] = nodo;
    ht->count++;

    return true;
}

void* hash_buscar(HashTable* ht, const char* clave) {
    if (!ht || !clave) return NULL;

    size_t idx = hash_funcion(clave, ht->capacidad);
    NodoHash* actual = ht->tabla[idx];

    while (actual) {
        if (strcmp(actual->clave, clave) == 0) {
            return actual->valor;
        }
        actual = actual->siguiente;
    }

    return NULL;
}

bool hash_eliminar(HashTable* ht, const char* clave) {
    if (!ht || !clave) return false;

    size_t idx = hash_funcion(clave, ht->capacidad);
    NodoHash* actual = ht->tabla[idx];
    NodoHash* anterior = NULL;

    while (actual) {
        if (strcmp(actual->clave, clave) == 0) {
            if (anterior) {
                anterior->siguiente = actual->siguiente;
            } else {
                ht->tabla[idx] = actual->siguiente;
            }
            free(actual->clave);
            free(actual);
            ht->count--;
            return true;
        }
        anterior = actual;
        actual = actual->siguiente;
    }

    return false;
}

void hash_destruir(HashTable* ht) {
    if (!ht) return;

    for (size_t i = 0; i < ht->capacidad; i++) {
        NodoHash* actual = ht->tabla[i];
        while (actual) {
            NodoHash* siguiente = actual->siguiente;
            free(actual->clave);
            free(actual);
            actual = siguiente;
        }
    }

    free(ht->tabla);
    free(ht);
}
```

SOLUCIÓN 4.5: Sistema de Plugins

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <dlfcn.h>

#define MAX_PLUGINS 32

typedef int (*PluginInitFunc)(void);
typedef int (*PluginExecFunc)(const char* args);
typedef void (*PluginFiniFunc)(void);

typedef struct {
    void*   handle;
    char    nombre[64];
    char    descripcion[256];
    PluginInitFunc init;
    PluginExecFunc exec;
    PluginFiniFunc fini;
    bool    cargado;
} Plugin;

typedef struct {
    Plugin plugins[MAX_PLUGINS];
    int    num_plugins;
} PluginManager;

PluginManager* pm_crear(void) {
    PluginManager* pm = malloc(sizeof(PluginManager));
    if (pm) {
        pm->num_plugins = 0;
        memset(pm->plugins, 0, sizeof(pm->plugins));
    }
    return pm;
}

bool pm_cargar(PluginManager* pm, const char* ruta_so) {
    if (!pm || pm->num_plugins >= MAX_PLUGINS) return false;

    Plugin* p = &pm->plugins[pm->num_plugins];

    p->handle = dlopen(ruta_so, RTLD_LAZY);
    if (!p->handle) return false;

    // Obtener funciones del plugin
    p->init = (PluginInitFunc)dlsym(p->handle, "plugin_init");
    p->exec = (PluginExecFunc)dlsym(p->handle, "plugin_exec");
    p->fini = (PluginFiniFunc)dlsym(p->handle, "plugin_fini");

    // Obtener metadatos
    const char** nombre = (const char**)dlsym(p->handle, "plugin_nombre");
    const char** desc = (const char**)dlsym(p->handle, "plugin_descripcion");

    if (nombre) strncpy(p->nombre, *nombre, sizeof(p->nombre) - 1);
    if (desc) strncpy(p->descripcion, *desc, sizeof(p->descripcion) - 1);

    if (!p->nombre[0]) {
        snprintf(p->nombre, sizeof(p->nombre), "%s", ruta_so);
    }

    p->cargado = true;
    pm->num_plugins++;

    // Llamar init si existe
    if (p->init) p->init();

    return true;
}

int pm_ejecutar(PluginManager* pm, const char* nombre, const char* args) {
    for (int i = 0; i < pm->num_plugins; i++) {
        if (strcmp(pm->plugins[i].nombre, nombre) == 0) {
            if (pm->plugins[i].exec) {
                return pm->plugins[i].exec(args);
            }
        }
    }
    return -1;
}

void pm_descargar(PluginManager* pm, const char* nombre) {
    for (int i = 0; i < pm->num_plugins; i++) {
        if (strcmp(pm->plugins[i].nombre, nombre) == 0) {
            if (pm->plugins[i].fini) pm->plugins[i].fini();
            dlclose(pm->plugins[i].handle);
            pm->plugins[i].cargado = false;
            break;
        }
    }
}

void pm_destruir(PluginManager* pm) {
    if (!pm) return;
    for (int i = 0; i < pm->num_plugins; i++) {
        if (pm->plugins[i].cargado) {
            if (pm->plugins[i].fini) pm->plugins[i].fini();
            dlclose(pm->plugins[i].handle);
        }
    }
    free(pm);
}
```

=== SECCIÓN 12: SOLUCIONES A EJERCICIOS DE SECCIÓN 1 ===

RESPUESTA 1.1: Declaración vs Definición

Declaración: Le dice al compilador "esto existe en alguna parte".
```c
extern int contador;       // Declaración de variable
void procesar(int x);       // Declaración de función
typedef struct Nodo Nodo;   // Declaración forward de struct
```

Definición: Asigna memoria y proporciona implementación.
```c
int contador = 0;           // Definición de variable
void procesar(int x) {      // Definición de función
    printf("%d\n", x);
}
struct Nodo {               // Definición de struct
    int dato;
    struct Nodo* sig;
};
```

RESPUESTA 1.2: Calificadores const con punteros

```c
const int* p;        // Puntero a int constante (no se puede modificar *p)
int* const p;        // Puntero constante a int (no se puede modificar p)
const int* const p;  // Puntero constante a int constante (ni *p ni p)
```

RESPUESTA 1.6: Comportamiento indefinido

```c
printf("%d %d %d\n", x++, x++, x++);
```
El orden de evaluación de los argumentos no está especificado.
Diferentes compiladores pueden producir diferentes resultados.
Con GCC típicamente imprime "7 6 5" (evaluación de derecha a izquierda).
