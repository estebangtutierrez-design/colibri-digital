=== 1. ESTILO DE CÓDIGO: K&R ===

El estilo K&R (Kernighan & Ritchie) es el estilo histórico de C,
utilizado en el libro "The C Programming Language". Es el más
ampliamente adoptado en la comunidad C.

REGLAS DEL ESTILO K&R:

```c
// 1. LLAVES: La llave de apertura va en la misma línea
int main(int argc, char** argv) {
    // ...
}

// 2. SANGRÍA: 4 espacios por nivel (sin tabs)
void funcion_ejemplo(void) {
    if (condicion) {
        for (int i = 0; i < 10; i++) {
            procesar(datos[i]);
        }
    }
}

// 3. ELSE EN LA MISMA LÍNEA QUE LA LLAVE DE CIERRE
if (x > 0) {
    positivo();
} else {
    negativo();
}

// 4. ESPACIOS ALREDEDOR DE OPERADORES
int suma = a + b;
if (x == 0 && y != 0) {
    // ...
}

// 5. NOMBRES DE FUNCIONES: minúsculas con guiones bajos
void procesar_datos(int* datos, size_t n);
int  calcular_imc(float peso, float altura);

// 6. NOMBRES DE CONSTANTES Y MACROS: MAYÚSCULAS
#define MAX_BUFFER 4096
#define VERSION "1.0.0"

// 7. NOMBRES DE TIPOS: Capitalizado o con _t
typedef struct ListaEnlazada ListaEnlazada;
typedef size_t tamano_t;
```

VARIACIONES COMUNES:

```c
// Estilo Allman (llaves en línea separada)
int main(int argc, char** argv)
{
    // ...
}

// Estilo GNU (sangría de 2 espacios, llaves en K&R)
int main (int argc, char **argv)
{
  // ...
}
```

RECOMENDACIÓN: Usa K&R. Es el estándar de facto en C y el que
encontrarás en la mayoría de proyectos profesionales.

=== 2. CONVENCIONES DE NOMBRES ===

Un buen nombre comunica la intención del código sin necesidad
de comentarios.

REGLAS PARA NOMBRES:

  - VARIABLES: Sustantivos descriptivos en minúsculas.
    int contador, total_usuarios;
    char* nombre_completo;

  - FUNCIONES: Verbos + sustantivos, prefijo del módulo.
    int  usuario_validar(const Usuario* u);
    void usuario_guardar(Usuario* u);

  - ESTRUCTURAS: Sustantivos, capitalizados o con sufijo _t.
    typedef struct { ... } Configuracion;
    typedef struct { ... } config_t;

  - ENUMERACIONES: Tipo capitalizado, valores en MAYÚSCULAS.
    typedef enum {
        COLOR_ROJO,
        COLOR_VERDE,
        COLOR_AZUL
    } Color;

  - CONSTANTES Y MACROS: MAYÚSCULAS con guiones bajos.
    #define PI 3.1415926535
    #define TAMANO_MAXIMO 1024

  - TIPOS OPACOS: Prefijo del módulo + nombre.
    typedef struct CacheLRU CacheLRU;

PREFIJOS DE MÓDULOS:

```c
// Módulo json
JSONDoc*    json_parsear(const char* texto);
JSONError   json_get_int(const JSONDoc* doc, const char* clave, int* valor);

// Módulo socket
Socket*     socket_crear(int dominio, int tipo);
SocketError socket_conectar(Socket* s, const char* host, int puerto);

// Módulo config
Config*     config_cargar(const char* ruta);
int         config_get_int(const Config* cfg, const char* clave, int def);
```

=== 3. ORGANIZACIÓN DE ARCHIVOS ===

CADA ARCHIVO .C DEBE TENER ESTA ESTRUCTURA:

```c
/*
 * nombre_modulo.c - Breve descripción del módulo
 * Autor: Tu Nombre
 * Fecha: 2026
 */

// 1. Include propio (primero, para verificar que es autocontenido)
#include "nombre_modulo.h"

// 2. Headers del sistema
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 3. Headers del proyecto
#include "otro_modulo.h"
#include "tipos_comunes.h"

// 4. Constantes privadas
#define MAX_INTERNO 1024
#define TAM_BUFFER 256

// 5. Tipos privados
typedef struct {
    int x;
    int y;
} PuntoInterno;

// 6. Variables static (estado interno del módulo)
static int contador_interno = 0;
static Error ultimo_error;

// 7. Funciones privadas (static)
static bool validar_parametros(const char* entrada) {
    return entrada && strlen(entrada) > 0;
}

// 8. Implementación de funciones públicas
int modulo_funcion(const char* entrada) {
    if (!validar_parametros(entrada)) {
        return -1;
    }
    // ... implementación ...
    return 0;
}
```

CADA ARCHIVO .H DEBE TENER:

```c
/*
 * nombre_modulo.h - Interfaz del módulo
 * Autor: Tu Nombre
 * Fecha: 2026
 *
 * Este módulo proporciona funciones para...
 *
 * Ejemplo de uso:
 * @code
 *   Contexto* ctx = modulo_crear();
 *   modulo_procesar(ctx, datos);
 *   modulo_destruir(ctx);
 * @endcode
 */

#ifndef NOMBRE_MODULO_H
#define NOMBRE_MODULO_H

// Includes necesarios
#include <stddef.h>

// Tipos públicos
typedef struct Contexto Contexto;

// Constantes públicas
#define MODULO_VERSION "1.0.0"

// Funciones públicas
Contexto* modulo_crear(size_t capacidad);
void      modulo_destruir(Contexto* ctx);
int       modulo_procesar(Contexto* ctx, const int* datos, size_t n);

#endif /* NOMBRE_MODULO_H */
```

=== 4. COMENTARIOS Y DOCUMENTACIÓN ===

TIPOS DE COMENTARIOS:

```c
// 1. Comentarios de línea - para explicar decisiones locales
int x = calcular_algo();  // No usar float aquí por precisión

// 2. Comentarios de bloque - para secciones de código
/*
 * Este bloque implementa el algoritmo de ordenamiento QuickSort
 * con optimización de selección de pivote mediana-de-tres.
 */

// 3. Comentarios Doxygen - para documentación de APIs
/**
 * @brief  Calcula el IMC (Índice de Masa Corporal).
 * @param  peso   Peso en kilogramos.
 * @param  altura Altura en metros.
 * @return IMC calculado, o -1.0 si los parámetros son inválidos.
 *
 * La fórmula utilizada es: IMC = peso / (altura ^ 2)
 * Los valores válidos son: peso > 0, altura > 0
 */
float calcular_imc(float peso, float altura);

// 4. Comentarios TODO/FIXME/HACK
// TODO: optimizar este bucle con SIMD
// FIXME: esta función falla con valores negativos
// HACK: solución temporal hasta que se corrija el driver
```

CUÁNDO COMENTAR:

  - INTERFACES PÚBLICAS (.h): Siempre documenta cada función.
  - CÓDIGO COMPLEJO: Cuando la lógica no es obvia.
  - DECISIONES DE DISEÑO: Explica POR QUÉ, no qué.
  - ADVERTENCIAS: Si el código tiene side effects o limitaciones.

CUÁNDO NO COMENTAR:

  - CÓDIGO AUTODOCUMENTADO: No comentes lo obvio.
    // MAL: Suma a más b
    int resultado = a + b;
    
  - CÓDIGO QUE CAMBIA FRECUENTEMENTE: Los comentarios se vuelven
    mentira. Mejor código claro.

=== 5. GESTIÓN DE ERRORES ===

Un patrón consistente para manejo de errores:

```c
// PATRÓN 1: Códigos de retorno
typedef enum {
    OK = 0,
    ERROR_MEMORIA = -1,
    ERROR_PARAMETRO = -2,
    ERROR_IO = -3,
    ERROR_ESTADO = -4
} Resultado;

// PATRÓN 2: Parámetro de salida para el resultado
Resultado procesar_datos(const char* entrada, int* resultado) {
    if (!entrada || !resultado) return ERROR_PARAMETRO;

    int* buffer = malloc(1024 * sizeof(int));
    if (!buffer) return ERROR_MEMORIA;

    // ... procesar ...
    *resultado = 42;

    free(buffer);
    return OK;
}

// PATRÓN 3: goto para limpieza (válido en C)
Resultado operacion_compleja(const char* ruta) {
    FILE* f = NULL;
    int* datos = NULL;
    Resultado res = OK;

    f = fopen(ruta, "r");
    if (!f) { res = ERROR_IO; goto limpiar; }

    datos = malloc(1024 * sizeof(int));
    if (!datos) { res = ERROR_MEMORIA; goto limpiar; }

    if (fread(datos, sizeof(int), 1024, f) != 1024) {
        res = ERROR_IO; goto limpiar;
    }

    // ... operación exitosa ...

limpiar:
    if (f) fclose(f);
    if (datos) free(datos);
    return res;
}
```

=== 6. PRUEBAS ===

BUENAS PRÁCTICAS PARA PRUEBAS:

```c
// 1. Prueba cada función públicamente
void test_contador_crear(void) {
    Contador* c = contador_crear(0);
    assert(c != NULL);
    assert(contador_valor(c) == 0);
    contador_destruir(c);
}

void test_contador_incrementar(void) {
    Contador* c = contador_crear(0);
    contador_incrementar(c, 5);
    assert(contador_valor(c) == 5);
    contador_incrementar(c, 3);
    assert(contador_valor(c) == 8);
    contador_destruir(c);
}

// 2. Prueba casos límite
void test_contador_limite(void) {
    Contador* c = contador_crear(INT_MAX);
    contador_incrementar(c, 1);  // No debe desbordar
    assert(contador_valor(c) == INT_MAX);
    contador_destruir(c);
}

// 3. Prueba valores nulos
void test_contador_nulo(void) {
    assert(contador_valor(NULL) == 0);
    contador_destruir(NULL);  // No debe crash
}

// 4. Prueba fugas de memoria (valgrind)
void test_contador_memoria(void) {
    for (int i = 0; i < 1000; i++) {
        Contador* c = contador_crear(i);
        contador_destruir(c);
    }
}
```

=== 8. FLUJO DE TRABAJO PROFESIONAL ===

EL CICLO DE DESARROLLO:

1. PLANIFICAR: Entender el problema antes de escribir código.
2. DISEÑAR: Definir interfaces y estructura de datos.
3. IMPLEMENTAR: Escribir código claro y probado.
4. PROBAR: Unitarias, integración, memoria (valgrind).
5. DOCUMENTAR: Comentarios, README, ejemplos.
6. REVISAR: Code review con colegas.
7. INTEGRAR: Merge a la rama principal.
8. DESPLEGAR: Compilar, empaquetar, distribuir.

LISTA DE VERIFICACIÓN PRE-COMMIT:

[ ] El código compila sin warnings (-Wall -Wextra -Werror).
[ ] Pasa Valgrind sin errores de memoria.
[ ] Las pruebas unitarias pasan.
[ ] El código sigue el estilo del proyecto.
[ ] Las funciones nuevas tienen documentación.
[ ] No hay código comentado o muerto.
[ ] Los mensajes de commit son descriptivos.
[ ] No hay fugas de memoria.
[ ] Los recursos se liberan en todos los caminos.
[ ] Los nombres de variables/funciones son claros.

```c
    if (!origen) return NULL;
```

```c
void operacion_segura(const char* ruta) {
    FILE* f = NULL;
    char* buf = NULL;
    int* datos = NULL;
    int error = 0;

    f = fopen(ruta, "r");
    if (!f) { error = 1; goto cleanup; }

    buf = malloc(1024);
    if (!buf) { error = 1; goto cleanup; }

    datos = malloc(4096);
    if (!datos) { error = 1; goto cleanup; }

    // ... operaciones ...

cleanup:
    if (datos) free(datos);
    if (buf) free(buf);
    if (f) fclose(f);
    if (error) /* manejar error */;
}
```

```c
// ============================================================================
// calculadora.c - Calculadora científica por línea de comandos
// Autor: Lucas Ega
// Fecha: 2026
//
// Esta calculadora soporta operaciones básicas y avanzadas:
//   +, -, *, /, ^, sqrt, sin, cos, tan, log, ln
//
// Ejemplo de uso:
//   $ ./calculadora "2 + 3 * 4"
//   Resultado: 14
// ============================================================================

#include "calculadora.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <ctype.h>

// ---------------------------------------------------------------------------
// Constantes privadas
// ---------------------------------------------------------------------------
#define MAX_EXPRESION 1024
#define MAX_TOKEN 64

// ---------------------------------------------------------------------------
// Tipos privados
// ---------------------------------------------------------------------------
typedef enum {
    TOKEN_NUMERO,
    TOKEN_OPERADOR,
    TOKEN_PAREN_IZQ,
    TOKEN_PAREN_DER,
    TOKEN_FUNCION,
    TOKEN_ERROR
} TipoToken;

typedef struct {
    TipoToken tipo;
    double    valor_num;
    char      texto[MAX_TOKEN];
} Token;

// ---------------------------------------------------------------------------
// Funciones privadas
// ---------------------------------------------------------------------------
static bool es_operador(char c) {
    return c == '+' || c == '-' || c == '*' || c == '/' || c == '^';
}

static int precedencia(char op) {
    switch (op) {
        case '+': case '-': return 1;
        case '*': case '/': return 2;
        case '^':           return 3;
        default:            return 0;
    }
}

// ---------------------------------------------------------------------------
// Implementación de funciones públicas
// ---------------------------------------------------------------------------
double calcular(const char* expresion, ErrorCalc* error) {
    if (!expresion || !error) {
        if (error) *error = ERROR_PARAMETRO;
        return 0.0;
    }

    if (strlen(expresion) == 0) {
        *error = ERROR_EXPRESION_VACIA;
        return 0.0;
    }

    if (strlen(expresion) >= MAX_EXPRESION) {
        *error = ERROR_EXPRESION_LARGA;
        return 0.0;
    }

    // TODO: Implementar parser Shunting Yard de Dijkstra
    // Por ahora, operación simple de ejemplo
    char* end;
    double resultado = strtod(expresion, &end);

    if (end == expresion) {
        *error = ERROR_SINTAXIS;
        return 0.0;
    }

    *error = ERROR_OK;
    return resultado;
}

const char* calc_error_str(ErrorCalc error) {
    switch (error) {
        case ERROR_OK:            return "Operación exitosa";
        case ERROR_PARAMETRO:     return "Parámetro inválido";
        case ERROR_EXPRESION_VACIA: return "Expresión vacía";
        case ERROR_EXPRESION_LARGA: return "Expresión demasiado larga";
        case ERROR_SINTAXIS:      return "Error de sintaxis";
        case ERROR_DIV_CERO:      return "División por cero";
        case ERROR_MEMORIA:       return "Error de memoria";
        default:                  return "Error desconocido";
    }
}
```

=== 14. REFACTORIZACIÓN ===

SEÑALES DE QUE NECESITAS REFACTORIZAR:

  1. FUNCIÓN DEMASIADO LARGA (> 50 líneas).
  2. FUNCIÓN CON DEMASIADOS PARÁMETROS (> 4).
  3. CÓDIGO DUPLICADO (DRY: Don't Repeat Yourself).
  4. ANIDAMIENTO EXCESIVO (> 3 niveles).
  5. VARIABLES CON NOMBRES POCO CLAROS (a, b, tmp, foo).
  6. COMENTARIOS QUE EXPLICAN QUÉ EN LUGAR DE POR QUÉ.
  7. MÓDULO QUE HACE DEMASIADAS COSAS.

TÉCNICAS DE REFACTORIZACIÓN:

```c
// ANTES: Función larga con muchas responsabilidades
void procesar_archivo(const char* ruta) {
    FILE* f = fopen(ruta, "r");
    if (!f) { printf("Error\\n"); return; }

    int suma = 0, count = 0;
    char linea[1024];
    while (fgets(linea, sizeof(linea), f)) {
        int valor = atoi(linea);
        suma += valor;
        count++;
        printf("Leído: %d\\n", valor);
    }
    printf("Suma: %d, Promedio: %f\\n", suma, (double)suma/count);
    fclose(f);
}

// DESPUÉS: Funciones pequeñas con responsabilidad única
static int* leer_numeros(const char* ruta, int* n) {
    FILE* f = fopen(ruta, "r");
    if (!f) return NULL;

    int capacidad = 100;
    int* datos = malloc(capacidad * sizeof(int));
    *n = 0;

    char linea[1024];
    while (fgets(linea, sizeof(linea), f)) {
        if (*n >= capacidad) {
            capacidad *= 2;
            int* tmp = realloc(datos, capacidad * sizeof(int));
            if (!tmp) { free(datos); fclose(f); return NULL; }
            datos = tmp;
        }
        datos[(*n)++] = atoi(linea);
    }

    fclose(f);
    return datos;
}

static int calcular_suma(const int* datos, int n) {
    int suma = 0;
    for (int i = 0; i < n; i++) suma += datos[i];
    return suma;
}

static void imprimir_estadisticas(int suma, int n) {
    printf("Suma: %d, Promedio: %f\\n", suma, (double)suma / n);
}

Resultado procesar_archivo(const char* ruta) {
    int n;
    int* datos = leer_numeros(ruta, &n);
    if (!datos) return ERROR_IO;

    int suma = calcular_suma(datos, n);
    imprimir_estadisticas(suma, n);

    free(datos);
    return OK;
}
```

________________________________________________________________________________
  PREGUNTAS FRECUENTES
________________________________________________________________________________

P: ¿Debo usar tabs o espacios para la sangría?
R: Espacios (4 por nivel). Los tabs se ven diferentes en cada
editor y causan conflictos de formato innecesarios.

P: ¿Cuánto debe medir una función?
R: Idealmente menos de 40 líneas. Si una función pasa de 100
líneas, casi seguro necesita dividirse.

P: ¿Comento todo mi código?
R: No. Comenta el POR QUÉ, no el QUÉ. El código debe ser
auto-documentado con buenos nombres de variables y funciones.

P: ¿Qué hago con el código legacy lleno de malas prácticas?
R: Refactoriza gradualmente. Primero agrega pruebas, luego
mejora el código. No intentes arreglar todo de una vez.

P: ¿Debo usar goto?
R: En C, goto es aceptable para limpiar recursos en funciones
con múltiples puntos de salida. En otros contextos, evítalo.
