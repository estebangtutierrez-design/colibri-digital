```c
double sumar(double a, double b) {
    return a + b;
}

double restar(double a, double b) {
    return a - b;
}

double multiplicar(double a, double b) {
    return a * b;
}

double dividir(double a, double b) {
    if (b == 0.0) {
        mostrar_error("Division por cero no permitida.");
        return 0.0;
    }
    return a / b;
}

unsigned long long factorial(int n) {
    if (n < 0) {
        mostrar_error("Factorial de numero negativo no definido.");
        return 0;
    }
    unsigned long long resultado = 1;
    for (int i = 2; i <= n; i++) {
        resultado *= i;
    }
    return resultado;
}

double potencia(double a, double b) {
    return pow(a, b);
}

double raiz_cuadrada(double a) {
    if (a < 0.0) {
        mostrar_error("Raiz cuadrada de numero negativo no definida.");
        return 0.0;
    }
    return sqrt(a);
}
```

```c
#include "utilidades.h"
#include "interfaz.h"
#include "config.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

static char historial[MAX_HISTORIAL][MAX_LINEA];
static int num_historial = 0;

double leer_double(const char* mensaje) {
    char buffer[MAX_LINEA];
    double valor;
    char extra;

    while (1) {
        printf("%s", mensaje);
        if (fgets(buffer, sizeof(buffer), stdin) == NULL) {
            continue;
        }

        if (sscanf(buffer, "%lf %c", &valor, &extra) == 1) {
            return valor;
        }
        mostrar_error("Entrada invalida. Ingresa un numero real.");
    }
}

int leer_entero(const char* mensaje) {
    char buffer[MAX_LINEA];
    int valor;
    char extra;

    while (1) {
        printf("%s", mensaje);
        if (fgets(buffer, sizeof(buffer), stdin) == NULL) {
            continue;
        }

        if (sscanf(buffer, "%d %c", &valor, &extra) == 1) {
            return valor;
        }
        mostrar_error("Entrada invalida. Ingresa un numero entero.");
    }
}

int es_numero_valido(const char* str) {
    if (str == NULL || *str == '\0') return 0;

    int i = 0;
    if (str[i] == '-' || str[i] == '+') i++;
    if (str[i] == '\0') return 0;

    int punto = 0;
    while (str[i] != '\0') {
        if (str[i] == '.') {
            if (punto) return 0;
            punto = 1;
        } else if (!isdigit(str[i])) {
            return 0;
        }
        i++;
    }
    return 1;
}

void agregar_historial(const char* entrada) {
    if (num_historial < MAX_HISTORIAL) {
        strncpy(historial[num_historial], entrada, MAX_LINEA - 1);
        historial[num_historial][MAX_LINEA - 1] = '\0';
        num_historial++;
    } else {
        // Desplazar historial hacia arriba (eliminar el mas antiguo)
        for (int i = 1; i < MAX_HISTORIAL; i++) {
            strcpy(historial[i - 1], historial[i]);
        }
        strncpy(historial[MAX_HISTORIAL - 1], entrada, MAX_LINEA - 1);
        historial[MAX_HISTORIAL - 1][MAX_LINEA - 1] = '\0';
    }
}

void mostrar_historial(void) {
    limpiar_pantalla();
    mostrar_linea('=', 50);
    printf("          HISTORIAL DE OPERACIONES
");
    mostrar_linea('=', 50);

    if (num_historial == 0) {
        printf("
  No hay operaciones en el historial.
");
    } else {
        printf("
");
        for (int i = 0; i < num_historial; i++) {
            printf("  %2d. %s
", i + 1, historial[i]);
        }
    }

    printf("
  Total de operaciones: %d
", num_historial);
    mostrar_linea('-', 50);
    pausa();
}
```

```c
/**
 * @brief Ejecuta el modo de calculadora basica.
 * Permite al usuario realizar operaciones aritmeticas simples.
 */
void modo_calculadora_basica(void) {
    limpiar_pantalla();
    mostrar_linea('=', 50);
    printf("        CALCULADORA BASICA
");
    mostrar_linea('=', 50);
    printf("
  Operaciones: +, -, *, /
");
    printf("  Ingresa 'q' para volver al menu.

");

    double a, b;
    char op;
    char buffer[100];

    while (1) {
        printf("  > ");
        if (fgets(buffer, sizeof(buffer), stdin) == NULL) break;

        if (buffer[0] == 'q' || buffer[0] == 'Q') break;

        if (sscanf(buffer, "%lf %c %lf", &a, &op, &b) != 3) {
            mostrar_error("Formato invalido. Usa: numero operador numero");
            continue;
        }

        double resultado;
        char desc[100];
        int valido = 1;

        switch (op) {
            case '+': resultado = sumar(a, b); break;
            case '-': resultado = restar(a, b); break;
            case '*': resultado = multiplicar(a, b); break;
            case '/':
                if (b == 0.0) {
                    mostrar_error("Division por cero");
                    valido = 0;
                } else {
                    resultado = dividir(a, b);
                }
                break;
            default:
                mostrar_error("Operador no valido. Usa: + - * /");
                valido = 0;
        }

        if (valido) {
            snprintf(desc, sizeof(desc), "%.2f %c %.2f = %.4f", a, op, b, resultado);
            agregar_historial(desc);
            mostrar_resultado(desc, resultado);
        }
    }
}

/**

/**
 * @brief Calcula promedios de notas escolares.
 *
 * Permite ingresar varias notas y calcular el promedio.
 */
void modo_promedios(void) {
    limpiar_pantalla();
    mostrar_linea('=', 50);
    printf("        CALCULO DE PROMEDIOS
");
    mostrar_linea('=', 50);

    int num_notas = leer_entero("  Numero de notas: ");
    if (num_notas <= 0) {
        mostrar_error("Debe ingresar al menos una nota");
        pausa();
        return;
    }

    double suma = 0.0;
    double nota;
    double max_nota = 0.0;
    double min_nota = 100.0;

    printf("
  Ingresa las %d notas:
", num_notas);
    for (int i = 0; i < num_notas; i++) {
        char prompt[50];
        snprintf(prompt, sizeof(prompt), "  Nota %d: ", i + 1);
        nota = leer_double(prompt);

        if (nota < 0 || nota > 20) {
            mostrar_error("La nota debe estar entre 0 y 20");
            i--;
            continue;
        }

        suma += nota;
        if (nota > max_nota) max_nota = nota;
        if (nota < min_nota) min_nota = nota;
    }

    double promedio = suma / num_notas;

    printf("
");

/**

 *

    mostrar_bienvenida();

    int opcion;

        mostrar_menu_principal();
        opcion = leer_entero("");

            case 1: modo_calculadora_basica(); break;
            case 2: modo_cientifico(); break;
            case 3: modo_conversor(); break;
            case 4: mostrar_historial(); break;
            case 5: modo_promedios(); break;

}
```

7. MAKEFILE DEL PROYECTO

```makefile
CC = gcc
CFLAGS = -Wall -Wextra -std=c11 -g
LDFLAGS = -lm
TARGET = calculadora
OBJS = main.o calculos.o interfaz.o utilidades.o

all: $(TARGET)

$(TARGET): $(OBJS)
	$(CC) $(OBJS) $(LDFLAGS) -o $(TARGET)

main.o: main.c config.h interfaz.h calculos.h utilidades.h
	$(CC) $(CFLAGS) -c main.c -o main.o

calculos.o: calculos.c calculos.h utilidades.h
	$(CC) $(CFLAGS) -c calculos.c -o calculos.o

interfaz.o: interfaz.c interfaz.h config.h
	$(CC) $(CFLAGS) -c interfaz.c -o interfaz.o

utilidades.o: utilidades.c utilidades.h interfaz.h config.h
	$(CC) $(CFLAGS) -c utilidades.c -o utilidades.o

clean:
	rm -f $(OBJS) $(TARGET)

run: $(TARGET)
	./$(TARGET)

debug: $(TARGET)
	gdb ./$(TARGET)

valgrind: $(TARGET)
	valgrind --leak-check=full ./$(TARGET)

.PHONY: all clean run debug valgrind
```
