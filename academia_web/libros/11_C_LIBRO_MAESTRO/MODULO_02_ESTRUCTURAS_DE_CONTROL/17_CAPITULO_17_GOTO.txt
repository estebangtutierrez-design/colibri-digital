  Pero en C, goto tiene usos legitimos.

  -------------------------------------------------------------------------
  17.1  QUE ES goto?
  -------------------------------------------------------------------------


  goto transfiere el control incondicionalmente a
  una etiqueta dentro de la misma funcion.

  Sintaxis:

    goto etiqueta;
    ...
    etiqueta:
        // codigo

  La etiqueta es un identificador seguido de dos puntos.
  Debe estar en la misma funcion que el goto.

  EJEMPLO 1: GOTO BASICO

```c
#include <stdio.h>

int main() {
    int x = 5;

    if (x > 0) {
        goto salto;
    }

    printf("Esto no se ejecuta\n");

salto:
    printf("Salto ejecutado! x=%d\n", x);

    return 0;
}```

  Salida:
    Salto ejecutado! x=5

  -------------------------------------------------------------------------
  17.2  GOTO PARA SALIR DE BUCLES ANIDADOS
  -------------------------------------------------------------------------


  Este es el uso MAS ACEPTADO de goto.
  break solo sale del bucle mas interno.
  goto puede salir de todos a la vez.

```c
#include <stdio.h>

int main() {
    int matriz[3][3] = {{1,2,3},{4,5,6},{7,8,9}};
    int buscado = 5;

    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            if (matriz[i][j] == buscado) {
                printf("Encontrado en [%d][%d]\n", i, j);
                goto encontrado;
            }
        }
    }
    printf("No encontrado\n");
    goto fin;

encontrado:
    printf("Busqueda exitosa\n");

fin:
    printf("Fin del programa\n");
    return 0;
}```

  -------------------------------------------------------------------------
  17.3  GOTO PARA MANEJO DE ERRORES
  -------------------------------------------------------------------------


  En C sin excepciones, goto es util para
  centralizar la limpieza de recursos.

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    int *datos = NULL;
    FILE *archivo = NULL;
    int exito = 0;

    datos = (int*)malloc(10 * sizeof(int));
    if (!datos) {
        printf("Error: no hay memoria\n");
        goto limpiar;
    }

    archivo = fopen("datos.txt", "r");
    if (!archivo) {
        printf("Error: no se pudo abrir archivo\n");
        goto limpiar;
    }

    // Procesamiento...
    printf("Todo OK\n");
    exito = 1;

limpiar:
    if (datos) free(datos);
    if (archivo) fclose(archivo);

    if (exito) {
        printf("Programa completado\n");
    } else {
        printf("Programa fallido\n");
    }
    return exito ? 0 : 1;
}```

  -------------------------------------------------------------------------
  17.4  GOTO vs break/continue
  -------------------------------------------------------------------------


  break: sale del bucle mas interno.
  continue: salta a la siguiente iteracion.
  goto: salta a cualquier etiqueta.

  goto es mas potente pero mas peligroso.

  EJEMPLO DE EQUIVALENCIA:

  CON BREAK (solo sale del interno):
```c
for (int i = 0; i < 5; i++) {
    for (int j = 0; j < 5; j++) {
        if (condicion) break; // solo sale de j
    }
    // sigue aqui
}```

  CON GOTO (sale de ambos):
```c
for (int i = 0; i < 5; i++) {
    for (int j = 0; j < 5; j++) {
        if (condicion) goto salida;
    }
}
salida:
    // continua aqui```

  -------------------------------------------------------------------------
  17.5  LA POLEMICA DE goto
  -------------------------------------------------------------------------


  1968: Edsger Dijkstra publica "Go To Statement
  Considered Harmful", argumentando que goto
  dificulta la legibilidad y el mantenimiento.

  Problemas de goto:
  * Crea codigo espagueti (flujo dificil de seguir).
  * Dificulta la verificacion formal del programa.
  * Puede saltar a lugares inesperados.

  Usos aceptados de goto:
  * Salir de bucles anidados profundos.
  * Manejo centralizado de errores (clean-up).
  * Maquinas de estado (state machines).

  -------------------------------------------------------------------------
  17.6  RESTRICCIONES DE goto
  -------------------------------------------------------------------------


  * No puedes saltar fuera de una funcion.
  * No puedes saltar a una etiqueta en otro archivo.
  * No puedes saltar a una etiqueta dentro de un
    array variable-length (VLA) si esta en ambito.
  * En C++ no puedes saltar sobre inicializaciones.

  -------------------------------------------------------------------------
  17.7  EJEMPLO: MAQUINA DE ESTADOS CON GOTO
  -------------------------------------------------------------------------


```c
#include <stdio.h>

int main() {
    int comando;

inicio:
    printf("\nEstado: INICIO\n");
    printf("1: avanzar  2: error  3: salir\n");
    printf("Comando: ");
    scanf("%d", &comando);

    if (comando == 1) goto procesando;
    if (comando == 2) goto error;
    if (comando == 3) goto fin;
    goto inicio;

procesando:
    printf("\nEstado: PROCESANDO\n");
    printf("1: completar  2: error\n");
    printf("Comando: ");
    scanf("%d", &comando);
    if (comando == 1) goto completado;
    if (comando == 2) goto error;
    goto procesando;

completado:
    printf("\nEstado: COMPLETADO\n");
    printf("Tarea finalizada con exito.\n");
    goto fin;

error:
    printf("\nEstado: ERROR\n");
    printf("Ocurrio un error. Recuperando...\n");
    goto inicio;

fin:
    printf("\nPrograma terminado.\n");
    return 0;
}```

  -------------------------------------------------------------------------
  17.8  ALTERNATIVAS A goto
  -------------------------------------------------------------------------


  En lugar de goto para salir de bucles:
  * Usar banderas (flags) en la condicion.
  * Extraer la logica a una funcion con return.
  * Reestructurar el codigo.

  En lugar de goto para errores:
  * Funciones con codigos de error.
  * Structs con resultado y error.
  * setjmp/longjmp (para saltos no locales).

  -------------------------------------------------------------------------
  17.9  GOTO CON ARREGLOS
  -------------------------------------------------------------------------


```c
#include <stdio.h>

int main() {
    int arr[10] = {0};
    // Inicializar con goto (NO recomendado, solo ejemplo)
    int i = 0;
inicio_bucle:
    if (i >= 10) goto fin_bucle;
    arr[i] = i * 2;
    i++;
    goto inicio_bucle;
fin_bucle:

    for (int j = 0; j < 10; j++) {
        printf("arr[%d] = %d\n", j, arr[j]);
    }
    return 0;
}```

  Este ejemplo muestra que goto puede reemplazar
  bucles, pero NO es recomendable.

  -------------------------------------------------------------------------
  17.10  CUANDO USAR Y CUANDO EVITAR goto
  -------------------------------------------------------------------------


  USA goto CUANDO:
  * Necesitas salir de bucles anidados (>2 niveles).
  * Implementas clean-up centralizado de errores.
  * Construyes maquinas de estado.
  * El codigo sin goto es mas complejo.

  EVITA goto CUANDO:
  * Puedes usar if/else, while, for, break, continue.
  * El goto salta hacia atras (bucle oculto).
  * El goto cruza ambitos de variables.
  * Solo quieres evitar un nivel de indentacion.

  RESUMEN DEL CAPITULO
===============================================================================

  * goto transfiere el control a una etiqueta
  * Es util para salir de bucles anidados

  * En C moderno, goto se usa casi solo para cleanup

===============================================================================
  PREGUNTAS FRECUENTES (FAQ)
===============================================================================

  P: Es verdad que goto es siempre malo?
  R: No. En C tiene usos legitimos como cleanup y salida de bucles anidados. El problema es el abuso.

  P: Puedo saltar a una etiqueta en otra funcion?
  R: No. goto solo funciona dentro de la misma funcion. Para saltos entre funciones usa setjmp/longjmp.

  P: goto es mas rapido que break?
  R: La diferencia es insignificante. No sacrifiques legibilidad por micro-optimizaciones.

  P: Que es codigo espagueti?
  R: Es codigo con flujo de control enredado, tipicamente por uso excesivo de goto, dificil de leer y mantener.

  P: Linux usa goto?
  R: Si. El kernel de Linux usa goto extensivamente para manejo de errores y cleanup.

  P: Que pasa si dos etiquetas tienen el mismo nombre?
  R: Error de compilacion por etiqueta duplicada.

  P: Puedo usar goto dentro de un switch?
  R: Si, pero cuidado con saltar a etiquetas dentro del switch.

  P: goto puede saltar hacia atras?
  R: Si, y eso crea bucles implicitos que son dificiles de detectar.

  -------------------------------------------------------------------------
  17.11  goto EN MANEJO DE RECURSOS MULTIPLES
  -------------------------------------------------------------------------

    #include <stdio.h>
    #include <stdlib.h>

    int main() {
        int *datos1 = NULL;
        int *datos2 = NULL;
        int *datos3 = NULL;
        FILE *f1 = NULL;
        FILE *f2 = NULL;
        int exito = 0;

        printf("Asignando recursos...\n");

        datos1 = (int*)malloc(100 * sizeof(int));
        if (!datos1) goto limpieza;

        datos2 = (int*)malloc(200 * sizeof(int));
        if (!datos2) goto limpieza;

        datos3 = (int*)malloc(300 * sizeof(int));
        if (!datos3) goto limpieza;

        f1 = fopen("archivo1.txt", "w");
        if (!f1) goto limpieza;

        f2 = fopen("archivo2.txt", "w");
        if (!f2) goto limpieza;

        printf("Todos los recursos asignados con exito.\n");
        printf("Procesando datos...\n");
        exito = 1;

    limpieza:
        printf("\nLimpiando recursos...\n");
        if (datos1) { free(datos1); printf("  datos1 liberado\n"); }
        if (datos2) { free(datos2); printf("  datos2 liberado\n"); }
        if (datos3) { free(datos3); printf("  datos3 liberado\n"); }
        if (f1)     { fclose(f1);   printf("  f1 cerrado\n");     }
        if (f2)     { fclose(f2);   printf("  f2 cerrado\n");     }

        if (exito) {
            printf("Programa completado exitosamente.\n");
        } else {
            printf("Programa fallido, recursos liberados.\n");
        }

        return exito ? 0 : 1;
    }

  -------------------------------------------------------------------------
  17.12  goto PARA REINTENTAR OPERACIONES
  -------------------------------------------------------------------------

    #include <stdio.h>
    #include <stdlib.h>
    #include <time.h>

    int main() {
        int intentos = 0;
        int maxIntentos = 3;
        int exito = 0;

        srand(time(NULL));

    intentar:
        if (intentos >= maxIntentos) goto fallo;

        intentos++;
        printf("Intento %d: ", intentos);

        int resultado = rand() % 10;
        printf("numero = %d ", resultado);

        if (resultado >= 8) {
            printf("-> EXITO!\n");
            exito = 1;
            goto fin;
        } else {
            printf("-> fallo, reintentando...\n");
            goto intentar;
        }

    fallo:
        printf("\nSe agotaron los intentos.\n");

    fin:
        if (exito) {
            printf("Operacion completada.\n");
        } else {
            printf("Operacion fallida.\n");
        }

        return 0;
    }

  -------------------------------------------------------------------------
  17.15  COMBINACION DE goto CON switch
  -------------------------------------------------------------------------

    #include <stdio.h>

    int main() {
        int opcion;
        int estado = 0;

    menu:
        printf("\nEstado: %d\n", estado);
        printf("1. Incrementar  2. Decrementar  3. Reset  4. Salir\n");
        printf("Opcion: ");
        scanf("%d", &opcion);

        switch (opcion) {
            case 1:
                estado++;
                printf("Estado incrementado\n");
                goto menu;
            case 2:
                estado--;
                printf("Estado decrementado\n");
                goto menu;
            case 3:
                estado = 0;
                printf("Estado reseteado\n");
                goto menu;
            case 4:
                printf("Saliendo...\n");
                break;
            default:
                printf("Opcion invalida\n");
                goto menu;
        }

        return 0;
    }

===============================================================================
  RESPUESTAS A PREGUNTAS FRECUENTES ADICIONALES
===============================================================================

  P: Es cierto que Linux usa goto?
  R: Si. El kernel de Linux usa goto principalmente para
     manejo de errores y limpieza de recursos. Es un uso
     aceptado y controlado.

  P: Puedo usar goto en C++?
  R: Si, pero con restricciones. No puedes saltar sobre
     declaraciones de variables que tienen constructores.

  P: goto afecta el rendimiento?
  R: No significativamente. Es una instruccion de salto
     condicional como cualquier otra.

  P: Como se documenta un goto?
  R: Con comentarios claros: /* goto necesario para salir
     de bucles anidados */ o /* goto para cleanup */.

  P: Cual es la alternativa mas comun a goto?
  R: Para bucles anidados: banderas. Para errores:
     funciones separadas con codigos de retorno.
