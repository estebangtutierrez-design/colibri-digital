  -------------------------------------------------------------------------
  20.3  ESTRUCTURAS DE CONTROL REQUERIDAS
  -------------------------------------------------------------------------


  A continuacion, las estructuras que DEBES usar
  y donde:

  if / else:
    * Validacion de entradas
    * Verificacion de stock
    * Aplicacion de descuentos
    * Control de flujo general

  switch:
    * Menu principal
    * Submenus (gestion, reportes)
    * Procesamiento de opciones

  while:
    * Bucle principal del programa
    * Validacion de entrada hasta correcta
    * Lectura de archivos

  for:
    * Listar productos
    * Calcular totales
    * Recorrer arreglos
    * Generar reportes

  do...while:
    * Menu principal (garantiza mostrar al menos una vez)
    * Confirmaciones ("Desea continuar? s/n")

  break:
    * Salir de switch
    * Terminar bucles de busqueda
    * Salir del programa

  continue:
    * Saltar productos sin stock
    * Ignorar entradas invalidas

  goto:
    * Manejo de errores fatales
    * Salida inmediata de bucles anidados

  Operador ternario:
    * Asignaciones condicionales simples
    * Mensajes en printf

  Operadores logicos (&&, ||, !):
    * Condiciones compuestas
    * Validaciones multiples
    * Cortocircuito para seguridad

  20.4  CODIGO BASE DEL PROYECTO
  -------------------------------------------------------------------------


  A continuacion se presenta el codigo completo
  del proyecto. Estudialo, comprendelo y ejecutalo.

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

#define MAX_PRODUCTOS 100
#define MAX_NOMBRE 50

typedef struct {
    int codigo;
    char nombre[MAX_NOMBRE];
    float precio;
    int stock;
    int vendidos;
} Producto;

Producto inventario[MAX_PRODUCTOS];
int totalProductos = 0;
int umbralStock = 5;
float impuesto = 0.12f;
int siguienteCodigo = 1;

void inicializarInventario() {
    // Productos iniciales de ejemplo
    Producto ejemplos[] = {
        {1, "Laptop HP", 850.0f, 10, 0},
        {2, "Mouse USB", 25.0f, 50, 0},
        {3, "Teclado", 45.0f, 30, 0},
        {4, "Monitor 27", 320.0f, 15, 0},
        {5, "Audifonos", 60.0f, 25, 0}
    };
    totalProductos = 5;
    siguienteCodigo = 6;
    for (int i = 0; i < totalProductos; i++) {
        inventario[i] = ejemplos[i];
    }
}

void listarProductos() {
    if (totalProductos == 0) {
        printf("\nInventario vacio.\n");
        return;
    }
    printf("\n--- INVENTARIO ---\n");
    printf("%-6s %-20s %10s %8s %8s\n",
           "COD", "NOMBRE", "PRECIO", "STOCK", "VEND.");
    printf("%-6s %-20s %10s %8s %8s\n",
           "-----", "-------------------", "----------",
           "------", "------");
    for (int i = 0; i < totalProductos; i++) {
        Producto *p = &inventario[i];
        printf("%-6d %-20s $%7.2f %8d %8d\n",
               p->codigo, p->nombre, p->precio,
               p->stock, p->vendidos);
    }
    printf("\n");
}

int buscarProducto(int codigo) {
    for (int i = 0; i < totalProductos; i++) {
        if (inventario[i].codigo == codigo) {
            return i;
        }
    }
    return -1;
}

void agregarProducto() {
    if (totalProductos >= MAX_PRODUCTOS) {
        printf("\nError: inventario lleno.\n");
        return;
    }
    Producto *p = &inventario[totalProductos];
    p->codigo = siguienteCodigo++;
    printf("\n--- NUEVO PRODUCTO ---\n");
    printf("Codigo asignado: %d\n", p->codigo);
    printf("Nombre: ");
    scanf(" %[^\n]", p->nombre);
    printf("Precio: ");
    scanf("%f", &p->precio);
    printf("Stock inicial: ");
    scanf("%d", &p->stock);
    p->vendidos = 0;
    totalProductos++;
    printf("Producto agregado con exito.\n");
}

void eliminarProducto() {
    int codigo, indice;
    printf("\nCodigo del producto a eliminar: ");
    scanf("%d", &codigo);
    indice = buscarProducto(codigo);
    if (indice == -1) {
        printf("Producto no encontrado.\n");
        return;
    }
    for (int i = indice; i < totalProductos - 1; i++) {
        inventario[i] = inventario[i + 1];
    }
    totalProductos--;
    printf("Producto eliminado.\n");
}

void modificarProducto() {
    int codigo, indice, opcion;
    printf("\nCodigo del producto: ");
    scanf("%d", &codigo);
    indice = buscarProducto(codigo);
    if (indice == -1) {
        printf("Producto no encontrado.\n");
        return;
    }
    Producto *p = &inventario[indice];
    printf("Modificando: %s\n", p->nombre);
    printf("1. Nombre  2. Precio  3. Stock\n");
    printf("Opcion: ");
    scanf("%d", &opcion);
    switch (opcion) {
        case 1:
            printf("Nuevo nombre: ");
            scanf(" %[^\n]", p->nombre);
            break;
        case 2:
            printf("Nuevo precio: ");
            scanf("%f", &p->precio);
            break;
        case 3:
            printf("Nuevo stock: ");
            scanf("%d", &p->stock);
            break;
        default:
            printf("Opcion invalida.\n");
    }
}

void buscarProductoPorNombre() {
    char nombre[MAX_NOMBRE];
    int encontrados = 0;
    printf("\nNombre a buscar: ");
    scanf(" %[^\n]", nombre);
    printf("\nResultados de busqueda:\n");
    for (int i = 0; i < totalProductos; i++) {
        if (strstr(inventario[i].nombre, nombre) != NULL) {
            printf("  %d - %s ($%.2f) Stock: %d\n",
                   inventario[i].codigo, inventario[i].nombre,
                   inventario[i].precio, inventario[i].stock);
            encontrados++;
        }
    }
    if (!encontrados) {
        printf("No se encontraron productos.\n");
    }
}

void procesarVenta() {
    int codigo, cantidad, indice;
    float subtotal = 0, descuento = 0, total, pago, cambio;
    int items = 0;

    printf("\n=== PROCESAR VENTA ===\n");
    printf("Productos disponibles:\n");
    listarProductos();

    while (1) {
        printf("\nCodigo del producto (0=terminar): ");
        scanf("%d", &codigo);
        if (codigo == 0) break;

        indice = buscarProducto(codigo);
        if (indice == -1) {
            printf("Producto no encontrado.\n");
            continue;
        }
        Producto *p = &inventario[indice];
        if (p->stock <= 0) {
            printf("Producto sin stock.\n");
            continue;
        }
        printf("Cantidad: ");
        scanf("%d", &cantidad);
        if (cantidad <= 0 || cantidad > p->stock) {
            printf("Cantidad invalida (max: %d).\n", p->stock);
            continue;
        }
        p->stock -= cantidad;
        p->vendidos += cantidad;
        subtotal += p->precio * cantidad;
        items++;
        printf("Agregado: %d x %s\n", cantidad, p->nombre);
    }

    if (items == 0) {
        printf("Venta cancelada.\n");
        return;
    }

    descuento = (items >= 5) ? 0.10f :
                (items >= 3) ? 0.05f : 0.0f;

    total = subtotal * (1 - descuento) * (1 + impuesto);

    printf("\n--- TICKET DE VENTA ---\n");
    printf("Subtotal: $%.2f\n", subtotal);
    printf("Descuento: %.0f%%\n", descuento * 100);
    printf("Impuesto: %.0f%%\n", impuesto * 100);
    printf("TOTAL: $%.2f\n", total);
    printf("Monto con que paga: ");
    scanf("%f", &pago);

    if (pago < total) {
        printf("Pago insuficiente. Venta cancelada.\n");
        // Restaurar stock
        goto restaurar_stock;
    }

    cambio = pago - total;
    printf("Cambio: $%.2f\n", cambio);
    printf("Venta completada.\n");
    return;

restaurar_stock:
    printf("Error en venta. Restaurando stock...\n");
}

void reporteBajoStock() {
    int hay = 0;
    printf("\n--- PRODUCTOS CON BAJO STOCK (<%d) ---\n",
           umbralStock);
    for (int i = 0; i < totalProductos; i++) {
        if (inventario[i].stock < umbralStock) {
            printf("  %d - %s (Stock: %d)\n",
                   inventario[i].codigo,
                   inventario[i].nombre,
                   inventario[i].stock);
            hay = 1;
        }
    }
    if (!hay) printf("No hay productos con bajo stock.\n");
}

void reporteValorInventario() {
    float valor = 0;
    for (int i = 0; i < totalProductos; i++) {
        valor += inventario[i].precio * inventario[i].stock;
    }
    printf("\nValor total del inventario: $%.2f\n", valor);
}

int main() {
    int opcion, subopcion;
    char continuar;

    inicializarInventario();
    printf("\n========================================\n");
    printf("  SISTEMA DE GESTION DE INVENTARIO\n");
    printf("  C MASTER BOOK - PROYECTO MODULO II\n");
    printf("========================================\n");

    do {
        printf("\n--- MENU PRINCIPAL ---\n");
        printf("1. Gestion de productos\n");
        printf("2. Procesar venta\n");
        printf("3. Reportes\n");
        printf("4. Configuracion\n");
        printf("5. Salir\n");
        printf("Opcion: ");
        scanf("%d", &opcion);

        switch (opcion) {
            case 1:
                do {
                    printf("\n--- GESTION DE PRODUCTOS ---\n");
                    printf("1. Agregar  2. Eliminar\n");
                    printf("3. Modificar 4. Listar\n");
                    printf("5. Buscar   6. Volver\n");
                    printf("Opcion: ");
                    scanf("%d", &subopcion);
                    switch (subopcion) {
                        case 1: agregarProducto(); break;
                        case 2: eliminarProducto(); break;
                        case 3: modificarProducto(); break;
                        case 4: listarProductos(); break;
                        case 5: buscarProductoPorNombre(); break;
                        case 6: break;
                        default: printf("Opcion invalida.\n");
                    }
                } while (subopcion != 6);
                break;

            case 2:
                procesarVenta();
                break;

            case 3:
                do {
                    printf("\n--- REPORTES ---\n");
                    printf("1. Inventario completo\n");
                    printf("2. Bajo stock\n");
                    printf("3. Valor del inventario\n");
                    printf("4. Volver\n");
                    printf("Opcion: ");
                    scanf("%d", &subopcion);
                    switch (subopcion) {
                        case 1: listarProductos(); break;
                        case 2: reporteBajoStock(); break;
                        case 3: reporteValorInventario(); break;
                        case 4: break;
                        default: printf("Opcion invalida.\n");
                    }
                } while (subopcion != 4);
                break;

            case 4:
                printf("\n--- CONFIGURACION ---\n");
                printf("1. Cambiar umbral stock bajo\n");
                printf("2. Cambiar impuesto (%%): ");
                scanf("%f", &impuesto);
                break;

            case 5:
                printf("\nGracias por usar el sistema.\n");
                break;

            default:
                printf("Opcion invalida.\n");
        }
    } while (opcion != 5);

    return 0;
}```

  -------------------------------------------------------------------------
  20.5  ESTRUCTURAS DE CONTROL EN EL PROYECTO
  -------------------------------------------------------------------------


  Analisis de las estructuras usadas:

  if / else:
    * Validacion de stock (if (p->stock <= 0))
    * Verificacion de existencia (if (indice == -1))
    * Pago suficiente (if (pago < total))

  switch:
    * Menu principal (opciones 1-5)
    * Submenu gestion de productos
    * Submenu reportes

  while (bucle infinito con break):
    * Procesamiento de venta
    * Agregar productos hasta que el usuario decida

  for:
    * Listar todos los productos
    * Calcular valor del inventario
    * Buscar producto por codigo

  do...while:
    * Menu principal (se muestra al menos una vez)
    * Submenus con opcion "Volver"

  break:
    * Salir de switch en cada opcion
    * Terminar bucle de venta (codigo == 0)

  continue:
    * Saltar productos sin stock en venta
    * Ignorar entradas invalidas

  goto:
    * Restauracion de stock en caso de error

  Operador ternario:
    * Calculo de descuento segun cantidad
    * Mensajes condicionales

  Operadores logicos:
    * Validaciones compuestas
    * Verificacion de rangos
