===============================================================================
1. FUNDAMENTOS DE ARGC Y ARGV
===============================================================================

Cuando ejecutamos un programa en la terminal, el sistema operativo (a través
del shell) prepara los argumentos y se los pasa al programa. En C, esto
ocurre a través de los parámetros de main().

1.1 La firma de main()

```c
// Forma estándar (mínima)
int main(int argc, char *argv[]);

// Forma equivalente
int main(int argc, char **argv);

// Forma con entorno (extension POSIX)
int main(int argc, char *argv[], char *envp[]);
```

- argc (argument count): Número de argumentos, incluyendo el nombre del
  programa. Siempre es >= 1.
- argv (argument vector): Array de strings terminados en NULL. argv[0] es
  el nombre del programa (o una string vacía si no está disponible).
  argv[argc] es NULL (centinela).

1.2 Primer programa: eco de argumentos

```c
#include <stdio.h>

int main(int argc, char *argv[])
{
    printf("Numero de argumentos: %d\n", argc);

    for (int i = 0; i < argc; i++)
    {
        printf("argv[%d] = \"%s\"\n", i, argv[i]);
    }

    // El centinela NULL
    printf("argv[%d] = %s\n", argc, argv[argc] == NULL ? "NULL" : "NO NULL");

    return 0;
}
```

Compile y ejecute:
```
$ gcc eco.c -o eco
$ ./eco hola mundo 42
Numero de argumentos: 4
argv[0] = "./eco"
argv[1] = "hola"
argv[2] = "mundo"
argv[3] = "42"
argv[4] = NULL
```

1.3 argv[0] no siempre es el nombre del programa

```c
#include <stdio.h>

int main(int argc, char *argv[])
{
    printf("argv[0] = \"%s\"\n", argv[0]);

    // Diferentes formas de ejecutar el mismo programa
    // $ ./programa       -> argv[0] = "./programa"
    // $ /ruta/completa/programa -> argv[0] = "/ruta/completa/programa"
    // $ programa         -> argv[0] = "programa" (si esta en PATH)
    // $ ln -s programa otro && ./otro -> argv[0] = "./otro"

    return 0;
}
```

1.4 Modificando argv (no recomendado, pero posible)

```c
#include <stdio.h>
#include <string.h>

int main(int argc, char *argv[])
{
    printf("argv[0] original: %s\n", argv[0]);

    // argv es modificable (es un array de punteros a strings)
    // argv[1] = "modificado";  // Esto es legal pero confuso para el usuario

    // Sin embargo, las strings individuales pueden modificarse
    if (argc > 1)
    {
        // Convertir a mayusculas el primer argumento
        for (char *p = argv[1]; *p; p++)
            *p = (*p >= 'a' && *p <= 'z') ? *p - 32 : *p;
        printf("argv[1] en mayusculas: %s\n", argv[1]);
    }

    return 0;
}
```

2. PARSEO BÁSICO DE ARGUMENTOS
===============================================================================

La forma más simple de procesar argumentos es iterar sobre argv y decidir
qué hacer según el contenido.

2.1 Argumentos posicionales

```c
#include <stdio.h>
#include <string.h>

int main(int argc, char *argv[])
{
    // Programa que espera: programa <nombre> <edad> <ciudad>
    if (argc != 4)
    {
        printf("Uso: %s <nombre> <edad> <ciudad>\n", argv[0]);
        return 1;
    }

    printf("Nombre: %s\n", argv[1]);
    printf("Edad:   %s\n", argv[2]);
    printf("Ciudad: %s\n", argv[3]);

    return 0;
}
```

2.2 Argumentos con flags (estilo -v, -o archivo)

```c
#include <stdio.h>
#include <string.h>

int main(int argc, char *argv[])
{
    int verbose = 0;
    char *output = NULL;
    int mostrar_ayuda = 0;

    // Saltamos argv[0] (nombre del programa)
    for (int i = 1; i < argc; i++)
    {
        if (strcmp(argv[i], "-v") == 0)
        {
            verbose = 1;
        }
        else if (strcmp(argv[i], "-o") == 0)
        {
            if (i + 1 < argc)
                output = argv[++i];
            else
            {
                fprintf(stderr, "Error: -o requiere un argumento\n");
                return 1;
            }
        }
        else if (strcmp(argv[i], "-h") == 0 ||
                 strcmp(argv[i], "--help") == 0)
        {
            mostrar_ayuda = 1;
        }
        else
        {
            fprintf(stderr, "Argumento desconocido: %s\n", argv[i]);
            fprintf(stderr, "Use -h para ayuda.\n");
            return 1;
        }
    }

    if (mostrar_ayuda)
    {
        printf("Uso: %s [-v] [-o archivo] [-h]\n", argv[0]);
        printf("  -v          Modo verbose\n");
        printf("  -o archivo  Archivo de salida\n");
        printf("  -h          Muestra esta ayuda\n");
        return 0;
    }

    printf("Verbose: %s\n", verbose ? "Si" : "No");
    printf("Output:  %s\n", output ? output : "(ninguno)");

    return 0;
}
```

2.3 Combinación de argumentos posicionales y flags

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
    // Calculadora simple: calcula [-v] <operacion> <a> <b>
    // operacion: suma, resta, mult, div
    if (argc < 4)
    {
        fprintf(stderr, "Uso: %s [-v] <operacion> <a> <b>\n", argv[0]);
        return 1;
    }

    int verbose = 0;
    int idx_operacion = 1;

    if (strcmp(argv[1], "-v") == 0)
    {
        verbose = 1;
        idx_operacion = 2;
    }

    if (argc - idx_operacion < 3)
    {
        fprintf(stderr, "Error: pocos argumentos\n");
        return 1;
    }

    const char *op = argv[idx_operacion];
    double a = atof(argv[idx_operacion + 1]);
    double b = atof(argv[idx_operacion + 2]);
    double resultado;

    if (strcmp(op, "suma") == 0)
        resultado = a + b;
    else if (strcmp(op, "resta") == 0)
        resultado = a - b;
    else if (strcmp(op, "mult") == 0)
        resultado = a * b;
    else if (strcmp(op, "div") == 0)
    {
        if (b == 0)
        {
            fprintf(stderr, "Error: division por cero\n");
            return 1;
        }
        resultado = a / b;
    }
    else
    {
        fprintf(stderr, "Operacion desconocida: %s\n", op);
        fprintf(stderr, "Opciones: suma, resta, mult, div\n");
        return 1;
    }

    if (verbose)
        printf("%.2f %s %.2f = ", a, op, b);

    printf("%.2f\n", resultado);

    return 0;
}
```

===============================================================================
3. CONVERSIÓN DE ARGUMENTOS
===============================================================================

Los argumentos siempre llegan como strings. Para usarlos como números,
debemos convertirlos explícitamente.

3.1 atoi() y familia

```c
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
    if (argc < 2)
    {
        printf("Uso: %s <numero>\n", argv[0]);
        return 1;
    }

    // atoi - simple pero sin deteccion de errores
    int valor = atoi(argv[1]);
    printf("atoi: \"%s\" -> %d\n", argv[1], valor);

    // atol - long
    long valor_l = atol(argv[1]);
    printf("atol: \"%s\" -> %ld\n", argv[1], valor_l);

    // atof - double
    double valor_f = atof(argv[1]);
    printf("atof: \"%s\" -> %f\n", argv[1], valor_f);

    // Problema: atoi("abc") -> 0 (sin error!)
    // Problema: atoi("9999999999") -> undefined behavior (overflow)
    printf("atoi(\"abc\") = %d (error silencioso!)\n", atoi("abc"));

    return 0;
}
```

3.2 strtol() y strtoul() - conversión segura

```c
#include <stdio.h>
#include <stdlib.h>
#include <errno.h>
#include <limits.h>

int main(int argc, char *argv[])
{
    if (argc < 2)
    {
        printf("Uso: %s <numero> [base]\n", argv[0]);
        return 1;
    }

    char *endptr;
    int base = (argc > 2) ? atoi(argv[2]) : 0; // 0 = auto-detect

    errno = 0;  // Limpiar errno antes de la llamada
    long valor = strtol(argv[1], &endptr, base);

    // Verificar errores
    if (errno == ERANGE)
    {
        if (valor == LONG_MAX)
            fprintf(stderr, "Overflow: numero muy grande\n");
        else if (valor == LONG_MIN)
            fprintf(stderr, "Underflow: numero muy pequenio\n");
        return 1;
    }

    if (endptr == argv[1])
    {
        fprintf(stderr, "Error: no se pudo convertir \"%s\"\n", argv[1]);
        return 1;
    }

    if (*endptr != '\0')
    {
        fprintf(stderr, "Advertencia: caracteres extra: \"%s\"\n", endptr);
    }

    printf("Valor convertido: %ld (base ", valor);
    if (base == 0)
    {
        if (argv[1][0] == '0' && (argv[1][1] == 'x' || argv[1][1] == 'X'))
            printf("16");
        else if (argv[1][0] == '0')
            printf("8");
        else
            printf("10");
    }
    else
        printf("%d", base);
    printf(")\n");

    // unsigned version
    unsigned long uvalor = strtoul(argv[1], &endptr, base);
    printf("Valor unsigned: %lu\n", uvalor);

    return 0;
}
```

3.3 strtod() para punto flotante

```c
#include <stdio.h>
#include <stdlib.h>
#include <errno.h>
#include <math.h>

int main(int argc, char *argv[])
{
    if (argc < 2)
    {
        printf("Uso: %s <numero>\n", argv[0]);
        return 1;
    }

    char *endptr;
    errno = 0;
    double valor = strtod(argv[1], &endptr);

    if (errno == ERANGE)
    {
        if (fabs(valor) == HUGE_VAL)
            fprintf(stderr, "Overflow\n");
        else
            fprintf(stderr, "Underflow\n");
        return 1;
    }

    if (endptr == argv[1])
    {
        fprintf(stderr, "Error: no se pudo convertir \"%s\"\n", argv[1]);
        return 1;
    }

    printf("Double: %f (hex: %A)\n", valor, valor);

    return 0;
}
```

3.4 Función de conversión robusta

```c
#include <stdio.h>
#include <stdlib.h>
#include <errno.h>
#include <limits.h>

typedef enum {
    CONV_OK,
    CONV_OVERFLOW,
    CONV_UNDERFLOW,
    CONV_INVALIDA,
    CONV_VACIO
} ConversionResult;

ConversionResult convertir_long(const char *str, long *resultado, int base)
{
    if (str == NULL || *str == '\0')
        return CONV_VACIO;

    char *endptr;
    errno = 0;
    long val = strtol(str, &endptr, base);

    if (errno == ERANGE)
    {
        if (val == LONG_MAX) return CONV_OVERFLOW;
        if (val == LONG_MIN) return CONV_UNDERFLOW;
    }

    if (endptr == str)
        return CONV_INVALIDA;

    if (*endptr != '\0')
    {
        // Caracteres extra no fatales, pero reportamos
        fprintf(stderr, "Advertencia: ignorando \"%s\"\n", endptr);
    }

    *resultado = val;
    return CONV_OK;
}

int main(int argc, char *argv[])
{
    if (argc < 2)
    {
        printf("Uso: %s <numero>\n", argv[0]);
        return 1;
    }

    long valor;
    ConversionResult r = convertir_long(argv[1], &valor, 0);

    switch (r)
    {
        case CONV_OK:
            printf("OK: %ld\n", valor);
            break;
        case CONV_OVERFLOW:
            fprintf(stderr, "Overflow\n");
            return 1;
        case CONV_UNDERFLOW:
            fprintf(stderr, "Underflow\n");
            return 1;
        case CONV_INVALIDA:
            fprintf(stderr, "Argumento invalido: %s\n", argv[1]);
            return 1;
        case CONV_VACIO:
            fprintf(stderr, "Argumento vacio\n");
            return 1;
    }

    return 0;
}
```

4. GETOPT(): PARSEO PROFESIONAL DE OPCIONES
===============================================================================

La biblioteca estándar de POSIX proporciona getopt() para procesar opciones
de línea de comandos de forma consistente.

4.1 Uso básico de getopt()

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>  // getopt(), optarg, optind, opterr, optopt

int main(int argc, char *argv[])
{
    int opt;
    int verbose = 0;
    char *output = NULL;
    int modo = 0;  // 0=default, 1=fast, 2=slow

    // La cadena de opciones especifica que opciones aceptamos
    // "vo:m" significa:
    //   v: opcion sin argumento (-v)
    //   o: opcion CON argumento (-o archivo)
    //   m: opcion sin argumento (-m)
    // El ':' despues de 'o' indica que espera un argumento

    while ((opt = getopt(argc, argv, "vo:mh")) != -1)
    {
        switch (opt)
        {
            case 'v':
                verbose = 1;
                break;
            case 'o':
                output = optarg;
                break;
            case 'm':
                modo = 1;
                break;
            case 'h':
                printf("Uso: %s [-v] [-m] [-o archivo] [-h]\n", argv[0]);
                return 0;
            case '?':  // Opcion no reconocida
                fprintf(stderr, "Uso: %s [-v] [-m] [-o archivo] [-h]\n",
                        argv[0]);
                return 1;
        }
    }

    printf("Verbose: %d\n", verbose);
    printf("Output:  %s\n", output ? output : "(ninguno)");
    printf("Modo:    %s\n", modo ? "fast" : "normal");

    // Argumentos no-opcion (posicionales) despues del parseo
    printf("Argumentos posicionales (%d):\n", argc - optind);
    for (int i = optind; i < argc; i++)
        printf("  argv[%d] = %s\n", i, argv[i]);

    return 0;
}
```

4.2 getopt_long(): opciones largas (--help)

```c
#include <stdio.h>
#include <stdlib.h>
#include <getopt.h>  // getopt_long()

int main(int argc, char *argv[])
{
    int verbose = 0;
    char *output = NULL;
    int modo = 0;

    // Definir opciones largas
    static struct option opciones_largas[] = {
        {"verbose",    no_argument,       &verbose, 1},
        {"output",     required_argument, 0,       'o'},
        {"fast",       no_argument,       &modo,   1},
        {"help",       no_argument,       0,       'h'},
        {"version",    no_argument,       0,       'v'},
        {0, 0, 0, 0}  // Terminador
    };

    int opt;
    int idx_opcion = 0;

    while ((opt = getopt_long(argc, argv, "vo:mh",
                              opciones_largas, &idx_opcion)) != -1)
    {
        switch (opt)
        {
            case 0:
                // Si la opcion larga setea una variable directamente
                break;
            case 'v':
                printf("Version 1.0\n");
                return 0;
            case 'o':
                output = optarg;
                break;
            case 'm':
                modo = 1;
                break;
            case 'h':
                printf("Uso: %s [opciones]\n", argv[0]);
                printf("  -v, --version      Muestra version\n");
                printf("  -o, --output ARCH  Archivo de salida\n");
                printf("  --verbose          Modo verbose\n");
                printf("  --fast             Modo rapido\n");
                printf("  -h, --help         Muestra esta ayuda\n");
                return 0;
            case '?':
                return 1;
        }
    }

    printf("Verbose: %d\n", verbose);
    printf("Output:  %s\n", output);
    printf("Modo:    %s\n", modo ? "fast" : "normal");

    return 0;
}
```

4.3 Manejo de errores con opterr

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

int main(int argc, char *argv[])
{
    // Suprimir mensajes de error automaticos de getopt
    opterr = 0;

    int opt;
    while ((opt = getopt(argc, argv, "a:b:")) != -1)
    {
        switch (opt)
        {
            case 'a':
                printf("a = %s\n", optarg);
                break;
            case 'b':
                printf("b = %s\n", optarg);
                break;
            case ':':
                fprintf(stderr, "Error: -%c requiere un argumento\n",
                        optopt);
                break;
            case '?':
                fprintf(stderr, "Error: opcion -%c desconocida\n",
                        optopt);
                break;
        }
    }

    return 0;
}
```

5.1 Subcomandos (estilo git, docker)

```c
#include <stdio.h>
#include <string.h>

int cmd_init(int argc, char *argv[])
{
    printf("Comando: init\n");
    for (int i = 1; i < argc; i++)
        printf("  arg: %s\n", argv[i]);
    return 0;
}

int cmd_add(int argc, char *argv[])
{
    printf("Comando: add\n");
    for (int i = 1; i < argc; i++)
        printf("  arg: %s\n", argv[i]);
    return 0;
}

int cmd_commit(int argc, char *argv[])
{
    int opt;
    char *mensaje = NULL;
    while ((opt = getopt(argc, argv, "m:")) != -1)
    {
        if (opt == 'm') mensaje = optarg;
    }
    printf("Commit: %s\n", mensaje ? mensaje : "(sin mensaje)");
    return 0;
}

int main(int argc, char *argv[])
{
    if (argc < 2)
    {
        printf("Uso: %s <comando> [opciones]\n", argv[0]);
        printf("Comandos:\n");
        printf("  init        Inicializar repositorio\n");
        printf("  add         Aniadir archivos\n");
        printf("  commit      Hacer commit\n");
        return 1;
    }

    const char *comando = argv[1];

    if (strcmp(comando, "init") == 0)
        return cmd_init(argc - 1, argv + 1);
    else if (strcmp(comando, "add") == 0)
        return cmd_add(argc - 1, argv + 1);
    else if (strcmp(comando, "commit") == 0)
        return cmd_commit(argc - 1, argv + 1);
    else
    {
        fprintf(stderr, "Comando desconocido: %s\n", comando);
        return 1;
    }
}
```

5.2 Argumentos con valores por defecto

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char *host;
    int puerto;
    int timeout;
    int verbose;
    char *log_file;
} Config;

Config config_default()
{
    Config cfg = {
        .host = "localhost",
        .puerto = 8080,
        .timeout = 30,
        .verbose = 0,
        .log_file = NULL
    };
    return cfg;
}

void mostrar_uso(const char *prog)
{
    printf("Uso: %s [opciones]\n", prog);
    printf("  --host HOST        Host (default: localhost)\n");
    printf("  -p, --puerto PORT  Puerto (default: 8080)\n");
    printf("  -t, --timeout T    Timeout segundos (default: 30)\n");
    printf("  -v, --verbose      Modo verbose\n");
    printf("  -l, --log ARCH     Archivo de log\n");
    printf("  -h, --help         Ayuda\n");
}

int parsear_argumentos(int argc, char *argv[], Config *cfg)
{
    static struct option opciones[] = {
        {"host",    required_argument, 0,  'H'},
        {"puerto",  required_argument, 0,  'p'},
        {"timeout", required_argument, 0,  't'},
        {"verbose", no_argument,       0,  'v'},
        {"log",     required_argument, 0,  'l'},
        {"help",    no_argument,       0,  'h'},
        {0, 0, 0, 0}
    };

    int opt;
    while ((opt = getopt_long(argc, argv, "p:t:vl:h", opciones, NULL)) != -1)
    {
        switch (opt)
        {
            case 'H':
                cfg->host = optarg;
                break;
            case 'p':
                cfg->puerto = atoi(optarg);
                if (cfg->puerto < 1 || cfg->puerto > 65535)
                {
                    fprintf(stderr, "Puerto invalido: %d\n", cfg->puerto);
                    return -1;
                }
                break;
            case 't':
                cfg->timeout = atoi(optarg);
                if (cfg->timeout < 0)
                {
                    fprintf(stderr, "Timeout invalido\n");
                    return -1;
                }
                break;
            case 'v':
                cfg->verbose = 1;
                break;
            case 'l':
                cfg->log_file = optarg;
                break;
            case 'h':
                mostrar_uso(argv[0]);
                return 1;
            case '?':
                return -1;
        }
    }
    return optind;  // Indice del primer argumento no-opcion
}

int main(int argc, char *argv[])
{
    Config cfg = config_default();

    int resto = parsear_argumentos(argc, argv, &cfg);
    if (resto < 0)
        return 1;
    if (resto == 1)
        return 0;

    printf("Configuracion:\n");
    printf("  Host:    %s\n", cfg.host);
    printf("  Puerto:  %d\n", cfg.puerto);
    printf("  Timeout: %d\n", cfg.timeout);
    printf("  Verbose: %s\n", cfg.verbose ? "Si" : "No");
    printf("  Log:     %s\n", cfg.log_file ? cfg.log_file : "(ninguno)");

    // Argumentos posicionales restantes
    if (optind < argc)
    {
        printf("Argumentos:\n");
        while (optind < argc)
            printf("  %s\n", argv[optind++]);
    }

    return 0;
}
```

```c
int leer_rango(const char *str, long *valor, long min, long max)
{
    char *endptr;
    errno = 0;
    long v = strtol(str, &endptr, 10);

    if (errno || endptr == str || *endptr)
    {
        fprintf(stderr, "Error: '%s' no es un numero valido\n", str);
        return -1;

    {
        fprintf(stderr, "Error: %ld fuera de rango [%ld, %ld]\n",

        return -1;
    }

    *valor = v;
    return 0;
}

int leer_entero_positivo(const char *str, int *valor)
{
    long v;
    if (leer_rango(str, &v, 1, INT_MAX) < 0)
        return -1;
    *valor = (int)v;
    return 0;
}
```

8. EJEMPLO COMPLETO: HERRAMIENTA PROFESIONAL

Construyamos una herramienta completa de línea de comandos que integre
todas las técnicas aprendidas:

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <getopt.h>
#include <errno.h>
#include <limits.h>
#include <time.h>

#define MAX_LINEA 1024
#define MAX_ARCHIVOS 256

typedef struct {
    char *archivos[MAX_ARCHIVOS];
    int n_archivos;
    int ordenar;
    int unico;
    int contar;
    int ignorar_mayusculas;
    int salida_numerica;
    char *output;
    int verbose;
} Opciones;

void mostrar_version()
{
    printf("tool v1.0.0\n");
    printf("Copyright (c) 2024 Lucas EGA Academy\n");
    printf("Licencia MIT\n");
}

void mostrar_ayuda(const char *prog)
{
    printf("Uso: %s [OPCIONES] <archivo>...\n\n", prog);
    printf("Procesa archivos de texto y muestra estadisticas.\n\n");
    printf("Opciones:\n");
    printf("  -o, --output ARCH    Escribe salida a ARCH\n");
    printf("  -s, --sort           Ordenar resultados\n");
    printf("  -u, --unique         Mostrar solo valores unicos\n");
    printf("  -c, --count          Mostrar conteo de ocurrencias\n");
    printf("  -i, --ignore-case    Ignorar mayusculas/minusculas\n");
    printf("  -n, --numeric        Orden numerico\n");
    printf("  -v, --verbose        Informacion detallada\n");
    printf("  -V, --version        Mostrar version\n");
    printf("  -h, --help           Mostrar ayuda\n");
}

int parsear_opciones(int argc, char *argv[], Opciones *opts)
{
    memset(opts, 0, sizeof(*opts));

    static struct option opciones_largas[] = {
        {"output",       required_argument, 0, 'o'},
        {"sort",         no_argument,       0, 's'},
        {"unique",       no_argument,       0, 'u'},
        {"count",        no_argument,       0, 'c'},
        {"ignore-case",  no_argument,       0, 'i'},
        {"numeric",      no_argument,       0, 'n'},
        {"verbose",      no_argument,       0, 'v'},
        {"version",      no_argument,       0, 'V'},
        {"help",         no_argument,       0, 'h'},
        {0, 0, 0, 0}
    };

    int opt;
    while ((opt = getopt_long(argc, argv, "o:sucinvVh",
                              opciones_largas, NULL)) != -1)
    {
        switch (opt)
        {
            case 'o': opts->output = optarg; break;
            case 's': opts->ordenar = 1; break;
            case 'u': opts->unico = 1; break;
            case 'c': opts->contar = 1; break;
            case 'i': opts->ignorar_mayusculas = 1; break;
            case 'n': opts->salida_numerica = 1; break;
            case 'v': opts->verbose = 1; break;
            case 'V': mostrar_version(); exit(0);
            case 'h': mostrar_ayuda(argv[0]); exit(0);
            case '?': exit(1);
        }
    }

    // Archivos restantes
    while (optind < argc && opts->n_archivos < MAX_ARCHIVOS)
    {
        opts->archivos[opts->n_archivos++] = argv[optind++];
    }

    if (opts->n_archivos == 0)
    {
        fprintf(stderr, "Error: debe especificar al menos un archivo\n");
        fprintf(stderr, "Use '%s --help' para ayuda.\n", argv[0]);
        return -1;
    }

    return 0;
}

int procesar_archivo(const char *ruta, FILE *salida, Opciones *opts)
{
    if (opts->verbose)
        fprintf(stderr, "Procesando: %s\n", ruta);

    FILE *f = fopen(ruta, "r");
    if (!f)
    {
        fprintf(stderr, "Error: no se pudo abrir '%s': %s\n",
                ruta, strerror(errno));
        return -1;
    }

    char linea[MAX_LINEA];
    int n_lineas = 0;
    int n_palabras = 0;
    int n_caracteres = 0;

    while (fgets(linea, sizeof(linea), f))
    {
        n_lineas++;
        n_caracteres += strlen(linea);

        // Contar palabras (separadas por espacios)
        char *p = linea;
        int en_palabra = 0;
        while (*p)
        {
            if (*p == ' ' || *p == '\t' || *p == '\n')
                en_palabra = 0;
            else if (!en_palabra)
            {
                en_palabra = 1;
                n_palabras++;
            }
            p++;
        }
    }

    fclose(f);

    fprintf(salida, "%-20s: %4d lineas, %4d palabras, %6d caracteres\n",
            ruta, n_lineas, n_palabras, n_caracteres);

    return 0;
}

int main(int argc, char *argv[])
{
    Opciones opts;
    if (parsear_opciones(argc, argv, &opts) < 0)
        return 1;

    FILE *salida = stdout;
    if (opts.output)
    {
        salida = fopen(opts.output, "w");
        if (!salida)
        {
            fprintf(stderr, "Error: no se pudo crear '%s': %s\n",
                    opts.output, strerror(errno));
            return 1;
        }
    }

    if (opts.verbose)
    {
        fprintf(stderr, "Opciones:\n");
        fprintf(stderr, "  Archivos: %d\n", opts.n_archivos);
        fprintf(stderr, "  Ordenar: %s\n", opts.ordenar ? "Si" : "No");
        fprintf(stderr, "  Unico: %s\n", opts.unico ? "Si" : "No");
        fprintf(stderr, "  Contar: %s\n", opts.contar ? "Si" : "No");
    }

    fprintf(salida, "=== Reporte de Archivos ===\n\n");
    int exit_code = 0;

    for (int i = 0; i < opts.n_archivos; i++)
    {
        if (procesar_archivo(opts.archivos[i], salida, &opts) < 0)
            exit_code = 1;
    }

    fprintf(salida, "\nTotal: %d archivos procesados\n", opts.n_archivos);

    if (opts.output)
    {
        fclose(salida);
        if (opts.verbose)
            fprintf(stderr, "Reporte escrito en: %s\n", opts.output);
    }

    return exit_code;
}
```

10.2 Parsing manual con punteros

```c
#include <stdio.h>

int es_prefijo(const char *prefijo, const char *str)
{
    while (*prefijo && *str && *prefijo == *str)
    {
        prefijo++;
        str++;
    }
    return *prefijo == '\0';
}

int main(int argc, char *argv[])
{
    // Parseo manual de --opciones=valor
    for (int i = 1; i < argc; i++)
    {
        const char *arg = argv[i];

        if (es_prefijo("--", arg))
        {
            arg += 2;

            // Buscar =
            const char *igual = arg;
            while (*igual && *igual != '=')
                igual++;

            if (*igual == '=')
            {
                int len_opcion = igual - arg;
                char opcion[64];
                snprintf(opcion, sizeof(opcion), "%.*s", len_opcion, arg);
                const char *valor = igual + 1;
                printf("Opcion larga: %s = %s\n", opcion, valor);
            }
            else
            {
                printf("Flag larga: %s\n", arg);
            }
        }
        else if (arg[0] == '-' && arg[1] && arg[2] == '\0')
        {
            printf("Flag corta: -%c\n", arg[1]);
        }
        else
        {
            printf("Posicional: %s\n", arg);
        }
    }

    return 0;
}
```

RESUMEN DEL CAPÍTULO
===============================================================================

- argc cuenta los argumentos (argv[0] es el nombre del programa)
- argv es un array de strings terminado en NULL
- argv[0] puede ser el nombre del programa, ruta, o alias
- Los argumentos siempre son strings; deben convertirse explicitamente
- strtol()/strtoul() son seguras con deteccion de errores; atoi() no
- getopt() procesa opciones cortas (-v); getopt_long() soporta --help
- optind indica la posicion del primer argumento no-opcion
- Validar rangos, existencia de archivos, y tipos antes de usar
- Mensajes de error informativos: "programa: error: mensaje"
- Los codigos de retorno (exit status) son esenciales para scripting
- getopt reordena argv: opciones al inicio, posicionales al final

===============================================================================
PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Qué pasa si argc es 0?** En C estándar, argc siempre es al menos 1.
argv[0] es el nombre del programa (o string vacía). No debería ocurrir
argc == 0 en sistemas normales.

❓ **¿Cómo se diferencian opciones de argumentos posicionales?** Por
convención, las opciones empiezan con - (cortas) o -- (largas). Todo lo
demás son argumentos posicionales. getopt() maneja esta distinción.

❓ **¿Puedo modificar argv?** Sí, argv y sus strings son modificables.
Sin embargo, no se recomienda por claridad y compatibilidad.

❓ **¿Qué es optarg, optind, opterr, optopt?** optarg apunta al argumento
de la opción actual. optind es el índice del siguiente elemento a procesar.
opterr controla mensajes de error automáticos. optopt almacena la opción
desconocida.

❓ **¿Cómo manejo argumentos con espacios?** Use comillas dobles en el shell:
"./prog" "argumento con espacios". El shell pasa el string completo como
un solo argv.

❓ **¿getopt() es estándar C o POSIX?** getopt() es parte de POSIX, no de
C estándar. En Windows, use getopt() de bibliotecas como mingw o
implemente su propio parser.

❓ **¿Cómo hago que una opción sea obligatoria?** getopt() no soporta
