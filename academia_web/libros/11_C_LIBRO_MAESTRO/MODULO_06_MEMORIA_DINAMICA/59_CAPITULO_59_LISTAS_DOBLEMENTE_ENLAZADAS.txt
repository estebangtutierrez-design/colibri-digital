===============================================================================
1. ESTRUCTURA DEL NODO DOBLE
===============================================================================

La diferencia fundamental con la lista simple es que cada nodo tiene
un puntero al nodo anterior (prev) además del puntero al siguiente
(next).

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Nodo {
    int dato;
    struct Nodo *anterior;  // ← Nuevo puntero
    struct Nodo *siguiente;
} Nodo;

typedef struct {
    Nodo *cabeza;  // Primer nodo (NULL si vacía)
    Nodo *cola;    // Último nodo (NULL si vacía)
    int tam;       // Número de elementos
} ListaDoble;

// Crear un nuevo nodo
Nodo *nodo_crear(int valor) {
    Nodo *nuevo = (Nodo *)malloc(sizeof(Nodo));
    if (nuevo == NULL) return NULL;
    nuevo->dato = valor;
    nuevo->anterior = NULL;
    nuevo->siguiente = NULL;
    return nuevo;
}

// Inicializar la lista
void ld_init(ListaDoble *lista) {
    lista->cabeza = NULL;
    lista->cola = NULL;
    lista->tam = 0;
}

// Recorrido hacia adelante
void ld_imprimir_adelante(ListaDoble *lista) {
    printf("Adelante: ");
    Nodo *actual = lista->cabeza;
    while (actual != NULL) {
        printf("%d -> ", actual->dato);
        actual = actual->siguiente;
    }
    printf("NULL\n");
}

// Recorrido hacia atrás
void ld_imprimir_atras(ListaDoble *lista) {
    printf("Atrás:    ");
    Nodo *actual = lista->cola;
    while (actual != NULL) {
        printf("%d -> ", actual->dato);
        actual = actual->anterior;
    }
    printf("NULL\n");
}

int main() {
    // Construir manualmente una lista doble con 3 nodos
    Nodo *n1 = nodo_crear(10);
    Nodo *n2 = nodo_crear(20);
    Nodo *n3 = nodo_crear(30);

    if (!n1 || !n2 || !n3) return 1;

    // Conectar hacia adelante
    n1->siguiente = n2;
    n2->siguiente = n3;

    // Conectar hacia atrás
    n2->anterior = n1;
    n3->anterior = n2;

    ListaDoble lista;
    lista.cabeza = n1;
    lista.cola = n3;
    lista.tam = 3;

    ld_imprimir_adelante(&lista);
    ld_imprimir_atras(&lista);

    printf("Cabeza: %d, Cola: %d, Tamaño: %d\n",
           lista.cabeza->dato, lista.cola->dato, lista.tam);

    // Liberar
    free(n1); free(n2); free(n3);
    return 0;
}
```

Cada nodo ahora sabe quién está antes y quién después. Esto permite
recorridos bidireccionales y simplifica la eliminación (no necesitamos
buscar el nodo anterior).

===============================================================================
2. INSERCIÓN EN LISTA DOBLEMENTE ENLAZADA
===============================================================================

2.1. Inserción al inicio

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Nodo {
    int dato;
    struct Nodo *anterior;
    struct Nodo *siguiente;
} Nodo;

typedef struct {
    Nodo *cabeza;
    Nodo *cola;
    int tam;
} ListaDoble;

Nodo *nodo_crear(int valor) {
    Nodo *n = (Nodo *)malloc(sizeof(Nodo));
    if (n) {
        n->dato = valor;
        n->anterior = NULL;
        n->siguiente = NULL;
    }
    return n;
}

void ld_init(ListaDoble *lista) {
    lista->cabeza = NULL;
    lista->cola = NULL;
    lista->tam = 0;
}

// Insertar al inicio
void ld_insertar_inicio(ListaDoble *lista, int valor) {
    Nodo *nuevo = nodo_crear(valor);
    if (nuevo == NULL) return;

    if (lista->cabeza == NULL) {
        // Lista vacía
        lista->cabeza = nuevo;
        lista->cola = nuevo;
    } else {
        nuevo->siguiente = lista->cabeza;
        lista->cabeza->anterior = nuevo;
        lista->cabeza = nuevo;
    }
    lista->tam++;
}

// Insertar al final
void ld_insertar_final(ListaDoble *lista, int valor) {
    Nodo *nuevo = nodo_crear(valor);
    if (nuevo == NULL) return;

    if (lista->cola == NULL) {
        lista->cabeza = nuevo;
        lista->cola = nuevo;
    } else {
        nuevo->anterior = lista->cola;
        lista->cola->siguiente = nuevo;
        lista->cola = nuevo;
    }
    lista->tam++;
}

void ld_imprimir_adelante(ListaDoble *lista) {
    printf("Adelante: ");
    for (Nodo *n = lista->cabeza; n != NULL; n = n->siguiente)
        printf("%d -> ", n->dato);
    printf("NULL\n");
}

void ld_imprimir_atras(ListaDoble *lista) {
    printf("Atrás:    ");
    for (Nodo *n = lista->cola; n != NULL; n = n->anterior)
        printf("%d -> ", n->dato);
    printf("NULL\n");
}

int main() {
    ListaDoble lista;
    ld_init(&lista);

    ld_insertar_final(&lista, 10);
    ld_insertar_final(&lista, 20);
    ld_insertar_final(&lista, 30);
    printf("Después de insertar al final (10, 20, 30):\n");
    ld_imprimir_adelante(&lista);
    ld_imprimir_atras(&lista);

    ld_insertar_inicio(&lista, 5);
    ld_insertar_inicio(&lista, 1);
    printf("\nDespués de insertar al inicio (5, 1):\n");
    ld_imprimir_adelante(&lista);
    ld_imprimir_atras(&lista);

    printf("Cabeza: %d, Cola: %d, Tamaño: %d\n",
           lista.cabeza->dato, lista.cola->dato, lista.tam);

    return 0;
}
```

Tanto la inserción al inicio como al final son O(1) porque mantenemos
punteros tanto a la cabeza como a la cola.

===============================================================================
3. INSERCIÓN EN POSICIÓN ARBITRARIA
===============================================================================

```c
#include <stdio.h>
#include <stdlib.h>

// (Estructuras y funciones básicas como antes)

// Insertar en una posición específica (0 = inicio)
int ld_insertar_en(ListaDoble *lista, int valor, int posicion) {
    if (posicion < 0 || posicion > lista->tam) return -1;

    if (posicion == 0) {
        ld_insertar_inicio(lista, valor);
        return 0;
    }

    if (posicion == lista->tam) {
        ld_insertar_final(lista, valor);
        return 0;
    }

    Nodo *nuevo = nodo_crear(valor);
    if (nuevo == NULL) return -1;

    // Decidir por dónde empezar a buscar (optimización)
    Nodo *actual;
    if (posicion <= lista->tam / 2) {
        // Empezar desde la cabeza
        actual = lista->cabeza;
        for (int i = 0; i < posicion; i++) {
            actual = actual->siguiente;
        }
    } else {
        // Empezar desde la cola (más rápido si posición está al final)
        actual = lista->cola;
        for (int i = lista->tam - 1; i > posicion; i--) {
            actual = actual->anterior;
        }
    }

    // actual es el nodo que actualmente está en 'posicion'
    nuevo->anterior = actual->anterior;
    nuevo->siguiente = actual;
    actual->anterior->siguiente = nuevo;
    actual->anterior = nuevo;

    lista->tam++;
    return 0;
}

void ld_liberar(ListaDoble *lista) {
    Nodo *actual = lista->cabeza;
    while (actual != NULL) {
        Nodo *temp = actual;
        actual = actual->siguiente;
        free(temp);
    }
    lista->cabeza = NULL;
    lista->cola = NULL;
    lista->tam = 0;
}

int main() {
    ListaDoble lista;
    ld_init(&lista);

    for (int i = 0; i < 5; i++) {
        ld_insertar_final(&lista, (i + 1) * 10);
    }
    printf("Original: ");
    ld_imprimir_adelante(&lista);

    ld_insertar_en(&lista, 25, 2);
    printf("Insertar 25 en posición 2: ");
    ld_imprimir_adelante(&lista);

    ld_insertar_en(&lista, 5, 0);
    printf("Insertar 5 en posición 0: ");
    ld_imprimir_adelante(&lista);

    ld_insertar_en(&lista, 99, lista.tam);
    printf("Insertar 99 al final: ");
    ld_imprimir_adelante(&lista);

    printf("Atrás:    ");
    ld_imprimir_atras(&lista);

    ld_liberar(&lista);
    return 0;
}
```

La optimización de decidir si empezar desde cabeza o cola reduce el
tiempo de búsqueda a O(min(posición, n - posición)) en lugar de O(n).

===============================================================================
4. ELIMINACIÓN EN LISTA DOBLEMENTE ENLAZADA
===============================================================================

La eliminación es mucho más simple que en la lista simple porque
tenemos acceso directo al nodo anterior mediante el puntero prev.

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Nodo {
    int dato;
    struct Nodo *anterior;
    struct Nodo *siguiente;
} Nodo;

typedef struct {
    Nodo *cabeza;
    Nodo *cola;
    int tam;
} ListaDoble;

// Eliminar el primer nodo
int ld_eliminar_inicio(ListaDoble *lista) {
    if (lista->cabeza == NULL) return -1;

    Nodo *temp = lista->cabeza;
    int valor = temp->dato;

    lista->cabeza = temp->siguiente;
    if (lista->cabeza != NULL) {
        lista->cabeza->anterior = NULL;
    } else {
        lista->cola = NULL;  // La lista quedó vacía
    }

    free(temp);
    lista->tam--;
    return valor;
}

// Eliminar el último nodo
int ld_eliminar_final(ListaDoble *lista) {
    if (lista->cola == NULL) return -1;

    Nodo *temp = lista->cola;
    int valor = temp->dato;

    lista->cola = temp->anterior;
    if (lista->cola != NULL) {
        lista->cola->siguiente = NULL;
    } else {
        lista->cabeza = NULL;  // La lista quedó vacía
    }

    free(temp);
    lista->tam--;
    return valor;
}

// Eliminar por valor (primera ocurrencia)
int ld_eliminar_valor(ListaDoble *lista, int valor) {
    Nodo *actual = lista->cabeza;
    while (actual != NULL) {
        if (actual->dato == valor) {
            // Ajustar punteros de los nodos vecinos
            if (actual->anterior != NULL) {
                actual->anterior->siguiente = actual->siguiente;
            } else {
                lista->cabeza = actual->siguiente;  // Era el primero
            }

            if (actual->siguiente != NULL) {
                actual->siguiente->anterior = actual->anterior;
            } else {
                lista->cola = actual->anterior;  // Era el último
            }

            int valor_elim = actual->dato;
            free(actual);
            lista->tam--;
            return valor_elim;
        }
        actual = actual->siguiente;
    }
    return -1;  // No encontrado
}

// Eliminar en posición específica
int ld_eliminar_en(ListaDoble *lista, int posicion) {
    if (posicion < 0 || posicion >= lista->tam) return -1;

    if (posicion == 0) return ld_eliminar_inicio(lista);
    if (posicion == lista->tam - 1) return ld_eliminar_final(lista);

    // Buscar el nodo a eliminar (empezando por el lado más cercano)
    Nodo *actual;
    if (posicion <= lista->tam / 2) {
        actual = lista->cabeza;
        for (int i = 0; i < posicion; i++) actual = actual->siguiente;
    } else {
        actual = lista->cola;
        for (int i = lista->tam - 1; i > posicion; i--) actual = actual->anterior;
    }

    // Ajustar punteros (actual no es ni cabeza ni cola)
    actual->anterior->siguiente = actual->siguiente;
    actual->siguiente->anterior = actual->anterior;

    int valor = actual->dato;
    free(actual);
    lista->tam--;
    return valor;
}

int main() {
    ListaDoble lista;
    ld_init(&lista);

    for (int i = 1; i <= 8; i++) ld_insertar_final(&lista, i * 10);

    printf("Original: ");
    ld_imprimir_adelante(&lista);

    printf("Eliminar inicio: %d\n", ld_eliminar_inicio(&lista));
    ld_imprimir_adelante(&lista);

    printf("Eliminar final: %d\n", ld_eliminar_final(&lista));
    ld_imprimir_adelante(&lista);

    printf("Eliminar valor 40: %d\n", ld_eliminar_valor(&lista, 40));
    ld_imprimir_adelante(&lista);

    printf("Eliminar en posición 2: %d\n", ld_eliminar_en(&lista, 2));
    ld_imprimir_adelante(&lista);

    printf("Atrás después de eliminaciones: ");
    ld_imprimir_atras(&lista);

    return 0;
}
```

En la lista simple, eliminar un nodo requería encontrar el anterior.
En la lista doble, el nodo mismo tiene el puntero al anterior, lo que
hace la eliminación O(1) una vez que tienes el nodo.

5. LISTA DOBLEMENTE ENLAZADA CIRCULAR
===============================================================================

La lista doblemente enlazada circular conecta la cabeza con la cola,
formando un anillo. El último nodo apunta al primero como siguiente,
y el primero apunta al último como anterior.

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Nodo {
    int dato;
    struct Nodo *anterior;
    struct Nodo *siguiente;
} Nodo;

typedef struct {
    Nodo *cabeza;
    int tam;
} ListaDCircular;

void ldc_init(ListaDCircular *lista) {
    lista->cabeza = NULL;
    lista->tam = 0;
}

void ldc_insertar(ListaDCircular *lista, int valor) {
    Nodo *nuevo = (Nodo *)malloc(sizeof(Nodo));
    if (nuevo == NULL) return;
    nuevo->dato = valor;

    if (lista->cabeza == NULL) {
        // Primer nodo: apunta a sí mismo
        nuevo->siguiente = nuevo;
        nuevo->anterior = nuevo;
        lista->cabeza = nuevo;
    } else {
        // Insertar después de la cabeza
        Nodo *cola = lista->cabeza->anterior;

        nuevo->siguiente = lista->cabeza;
        nuevo->anterior = cola;

        lista->cabeza->anterior = nuevo;
        cola->siguiente = nuevo;
    }
    lista->tam++;
}

void ldc_imprimir_adelante(ListaDCircular *lista, int vueltas) {
    if (lista->cabeza == NULL) return;

    printf("Adelante (%d vueltas): ", vueltas);
    Nodo *actual = lista->cabeza;
    for (int i = 0; i < lista->tam * vueltas; i++) {
        printf("%d -> ", actual->dato);
        actual = actual->siguiente;
    }
    printf("(vuelta al inicio: %d)\n", actual->dato);
}

void ldc_imprimir_atras(ListaDCircular *lista, int vueltas) {
    if (lista->cabeza == NULL) return;

    printf("Atrás (%d vueltas):   ", vueltas);
    Nodo *actual = lista->cabeza->anterior;  // Empezar desde la cola
    for (int i = 0; i < lista->tam * vueltas; i++) {
        printf("%d -> ", actual->dato);
        actual = actual->anterior;
    }
    printf("(vuelta al final: %d)\n", actual->dato);
}

int ldc_eliminar_cabeza(ListaDCircular *lista) {
    if (lista->cabeza == NULL) return -1;

    Nodo *temp = lista->cabeza;
    int valor = temp->dato;

    if (lista->tam == 1) {
        lista->cabeza = NULL;
    } else {
        Nodo *cola = lista->cabeza->anterior;
        lista->cabeza = temp->siguiente;
        lista->cabeza->anterior = cola;
        cola->siguiente = lista->cabeza;
    }

    free(temp);
    lista->tam--;
    return valor;
}

void ldc_liberar(ListaDCircular *lista) {
    if (lista->cabeza == NULL) return;

    Nodo *actual = lista->cabeza;
    for (int i = 0; i < lista->tam; i++) {
        Nodo *temp = actual;
        actual = actual->siguiente;
        free(temp);
    }
    lista->cabeza = NULL;
    lista->tam = 0;
}

int main() {
    ListaDCircular lista;
    ldc_init(&lista);

    ldc_insertar(&lista, 10);
    ldc_insertar(&lista, 20);
    ldc_insertar(&lista, 30);
    ldc_insertar(&lista, 40);

    printf("Tamaño: %d\n", lista.tam);
    ldc_imprimir_adelante(&lista, 2);
    ldc_imprimir_atras(&lista, 2);

    printf("\nEliminar cabeza: %d\n", ldc_eliminar_cabeza(&lista));
    ldc_imprimir_adelante(&lista, 2);

    ldc_liberar(&lista);
    return 0;
}
```

Las listas circulares son útiles para:
- Round-robin scheduling
- Buffers circulares
- Juegos de turnos
- Cachés LRU

6. LISTA DOBLE CON FUNCIONES DE BÚSQUEDA
===============================================================================

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Nodo {
    int dato;
    struct Nodo *anterior;
    struct Nodo *siguiente;
} Nodo;

typedef struct {
    Nodo *cabeza;
    Nodo *cola;
    int tam;
} ListaDoble;

// Búsqueda hacia adelante
Nodo *ld_buscar_adelante(ListaDoble *lista, int valor) {
    Nodo *actual = lista->cabeza;
    while (actual != NULL) {
        if (actual->dato == valor) return actual;
        actual = actual->siguiente;
    }
    return NULL;
}

// Búsqueda hacia atrás
Nodo *ld_buscar_atras(ListaDoble *lista, int valor) {
    Nodo *actual = lista->cola;
    while (actual != NULL) {
        if (actual->dato == valor) return actual;
        actual = actual->anterior;
    }
    return NULL;
}

// Búsqueda bidireccional simultánea (dos punteros)
Nodo *ld_buscar_bidireccional(ListaDoble *lista, int valor) {
    Nodo *izq = lista->cabeza;
    Nodo *der = lista->cola;

    while (izq != NULL && der != NULL && izq != der && izq->anterior != der) {
        if (izq->dato == valor) return izq;
        if (der->dato == valor) return der;
        izq = izq->siguiente;
        der = der->anterior;
    }

    // Caso en que queda un solo nodo por revisar
    if (izq != NULL && izq->dato == valor) return izq;

    return NULL;
}

// Obtener elemento en posición (empezando por el extremo más cercano)
int ld_obtener(ListaDoble *lista, int posicion) {
    if (posicion < 0 || posicion >= lista->tam) return -1;

    Nodo *actual;
    if (posicion <= lista->tam / 2) {
        actual = lista->cabeza;
        for (int i = 0; i < posicion; i++) actual = actual->siguiente;
    } else {
        actual = lista->cola;
        for (int i = lista->tam - 1; i > posicion; i--) actual = actual->anterior;
    }
    return actual->dato;
}

// Obtener el nodo del medio (sin recorrer toda la lista dos veces)
Nodo *ld_medio(ListaDoble *lista) {
    if (lista->cabeza == NULL) return NULL;

    Nodo *rapido = lista->cabeza;
    Nodo *lento = lista->cabeza;

    while (rapido != NULL && rapido->siguiente != NULL) {
        rapido = rapido->siguiente->siguiente;
        lento = lento->siguiente;
    }
    return lento;
}

void demo_busqueda() {
    ListaDoble lista;
    ld_init(&lista);

    for (int i = 1; i <= 10; i++) ld_insertar_final(&lista, i * 10);

    printf("Lista: ");
    ld_imprimir_adelante(&lista);

    int buscar = 50;
    Nodo *enc = ld_buscar_adelante(&lista, buscar);
    printf("Búsqueda adelante %d: %s\n", buscar, enc ? "encontrado" : "no encontrado");

    enc = ld_buscar_atras(&lista, buscar);
    printf("Búsqueda atrás %d: %s\n", buscar, enc ? "encontrado" : "no encontrado");

    buscar = 99;
    enc = ld_buscar_bidireccional(&lista, buscar);
    printf("Búsqueda bidireccional %d: %s\n", buscar, enc ? "encontrado" : "no encontrado");

    printf("Elemento en posición 3: %d\n", ld_obtener(&lista, 3));
    printf("Elemento en posición 7: %d\n", ld_obtener(&lista, 7));

    Nodo *medio = ld_medio(&lista);
    printf("Elemento del medio: %d\n", medio ? medio->dato : -1);
}
```

===============================================================================
7. APLICACIÓN: NAVEGADOR CON HISTORIAL
===============================================================================

Un uso clásico de lista doblemente enlazada es el historial de un
navegador web, donde puedes ir adelante y atrás.

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct Pagina {
    char url[256];
    struct Pagina *anterior;
    struct Pagina *siguiente;
} Pagina;

typedef struct {
    Pagina *actual;
    Pagina *inicio;
    Pagina *fin;
} Historial;

void historial_init(Historial *h) {
    h->actual = NULL;
    h->inicio = NULL;
    h->fin = NULL;
}

void historial_visitar(Historial *h, const char *url) {
    Pagina *nueva = (Pagina *)malloc(sizeof(Pagina));
    if (nueva == NULL) return;

    strncpy(nueva->url, url, 255);
    nueva->url[255] = '\0';
    nueva->siguiente = NULL;

    // Si estamos en medio del historial, eliminar páginas adelante
    if (h->actual != NULL) {
        Pagina *temp = h->actual->siguiente;
        while (temp != NULL) {
            Pagina *sig = temp->siguiente;
            free(temp);
            temp = sig;
        }
        h->actual->siguiente = nueva;
        nueva->anterior = h->actual;
        h->fin = nueva;
    } else {
        // Historial vacío
        nueva->anterior = NULL;
        h->inicio = nueva;
        h->fin = nueva;
    }

    h->actual = nueva;
}

const char *historial_atras(Historial *h) {
    if (h->actual == NULL || h->actual->anterior == NULL) return NULL;
    h->actual = h->actual->anterior;
    return h->actual->url;
}

const char *historial_adelante(Historial *h) {
    if (h->actual == NULL || h->actual->siguiente == NULL) return NULL;
    h->actual = h->actual->siguiente;
    return h->actual->url;
}

void historial_mostrar(Historial *h) {
    printf("\n=== HISTORIAL COMPLETO ===\n");
    Pagina *p = h->inicio;
    while (p != NULL) {
        printf("  %s", p->url);
        if (p == h->actual) printf(" ← ACTUAL");
        printf("\n");
        p = p->siguiente;
    }
    printf("==========================\n");
}

void historial_liberar(Historial *h) {
    Pagina *p = h->inicio;
    while (p != NULL) {
        Pagina *temp = p;
        p = p->siguiente;
        free(temp);
    }
    h->actual = NULL;
    h->inicio = NULL;
    h->fin = NULL;
}

int main() {
    Historial h;
    historial_init(&h);

    historial_visitar(&h, "google.com");
    historial_visitar(&h, "github.com");
    historial_visitar(&h, "stackoverflow.com");
    historial_visitar(&h, "opencode.ai");

    historial_mostrar(&h);

    printf("\n← Atrás: %s\n", historial_atras(&h) ?: "no hay");
    printf("← Atrás: %s\n", historial_atras(&h) ?: "no hay");
    printf("→ Adelante: %s\n", historial_adelante(&h) ?: "no hay");

    // Visitar nueva página (borra el historial adelante)
    historial_visitar(&h, "chatgpt.com");

    historial_mostrar(&h);

    historial_liberar(&h);
    return 0;
}
```

8. COMPARATIVA: LISTA SIMPLE VS LISTA DOBLE
===============================================================================

| Operación              | Lista Simple    | Lista Doble     |
|------------------------|-----------------|-----------------|
| Inserción al inicio    | O(1)            | O(1)            |
| Inserción al final     | O(n) / O(1)*    | O(1)            |
| Inserción en posición  | O(n)            | O(n) / O(min(p,n-p))|
| Eliminación al inicio  | O(1)            | O(1)            |
| Eliminación al final   | O(n)            | O(1)            |
| Eliminación por valor  | O(n)            | O(n)            |
| Búsqueda               | O(n)            | O(n)            |
| Recorrido atrás        | No posible      | O(n)            |
| Memoria por nodo       | 1 puntero       | 2 punteros      |
| Complejidad de código  | Baja            | Media           |

* O(1) si mantenemos puntero al último nodo.

La lista doble es más versátil pero usa más memoria (un puntero extra
por nodo). Elige según las necesidades:

- ¿Necesitas recorrer hacia atrás? → Lista doble.
- ¿Eliminas frecuentemente al final? → Lista doble.
- ¿La memoria es crítica? → Lista simple.
- ¿Necesitas máxima simplicidad? → Lista simple.

9. PARSER DE LISTA DOBLE: CONVERSIÓN A ARRAY
===============================================================================

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Nodo {
    int dato;
    struct Nodo *anterior;
    struct Nodo *siguiente;
} Nodo;

typedef struct {
    Nodo *cabeza;
    Nodo *cola;
    int tam;
} ListaDoble;

// Convertir lista doble a array
int *ld_to_array(ListaDoble *lista, int *n) {
    if (lista == NULL || lista->tam == 0) {
        *n = 0;
        return NULL;
    }

    int *arr = (int *)malloc(lista->tam * sizeof(int));
    if (arr == NULL) {
        *n = 0;
        return NULL;
    }

    Nodo *actual = lista->cabeza;
    for (int i = 0; i < lista->tam; i++) {
        arr[i] = actual->dato;
        actual = actual->siguiente;
    }

    *n = lista->tam;
    return arr;
}

// Convertir array a lista doble
ListaDoble *array_to_ld(int *arr, int n) {
    ListaDoble *lista = (ListaDoble *)malloc(sizeof(ListaDoble));
    if (lista == NULL) return NULL;
    ld_init(lista);

    for (int i = 0; i < n; i++) {
        ld_insertar_final(lista, arr[i]);
    }

    return lista;
}

// Verificar que la lista sea consistente (depuración)
int ld_verificar(ListaDoble *lista) {
    if (lista->cabeza == NULL && lista->cola == NULL && lista->tam == 0)
        return 1;  // Vacía OK
    if (lista->cabeza == NULL || lista->cola == NULL || lista->tam == 0)
        return 0;  // Inconsistente

    // Verificar que cabeza->anterior sea NULL
    if (lista->cabeza->anterior != NULL) return 0;
    // Verificar que cola->siguiente sea NULL
    if (lista->cola->siguiente != NULL) return 0;

    // Contar nodos hacia adelante
    int count = 0;
    Nodo *actual = lista->cabeza;
    while (actual != NULL) {
        count++;
        if (actual->siguiente != NULL) {
            if (actual->siguiente->anterior != actual) return 0;
        }
        actual = actual->siguiente;
    }

    return count == lista->tam;
}

int main() {
    // Array a lista doble
    int arr[] = {1, 2, 3, 4, 5};
    int n = 5;

    ListaDoble *lista = array_to_ld(arr, n);
    printf("Lista desde array: ");
    ld_imprimir_adelante(lista);

    printf("Lista atrás: ");
    ld_imprimir_atras(lista);

    printf("¿Lista consistente? %s\n", ld_verificar(lista) ? "Sí" : "No");

    // Lista doble a array
    int m;
    int *arr2 = ld_to_array(lista, &m);
    printf("Array desde lista: [");
    for (int i = 0; i < m; i++) {
        printf("%d", arr2[i]);
        if (i < m - 1) printf(", ");
    }
    printf("]\n");

    free(arr2);
    ld_liberar(lista);
    free(lista);

    return 0;
}
```

RESUMEN DEL CAPÍTULO
===============================================================================

- Cada nodo en una lista doblemente enlazada tiene punteros al
  siguiente (next) y al anterior (prev).
- La lista se representa con punteros a cabeza y cola, más el tamaño.
- La inserción y eliminación al inicio y al final son O(1).
- La eliminación de un nodo es más simple que en lista simple porque
  se tiene acceso directo al nodo anterior.
- La lista circular doblemente enlazada conecta cabeza y cola, formando
  un anillo, permitiendo recorridos infinitos.
- La búsqueda se puede optimizar empezando por cabeza o cola según
  la posición.
- La lista doble usa más memoria (2 punteros por nodo) pero ofrece
  más flexibilidad que la lista simple.
- Aplicaciones: navegador web, editor de texto, LRU cache, juegos.

===============================================================================
PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Cuándo debo usar lista doble en lugar de lista simple?**
Usa lista doble cuando necesites: (1) eliminar el último nodo
eficientemente, (2) recorrer hacia atrás, (3) eliminar un nodo
dado sin conocer el anterior. Para todo lo demás, la simple es
más ligera.

❓ **¿La lista doble es siempre mejor?**
No. Tiene el doble de memoria por nodo (el puntero anterior) y el
código es más complejo. Si las operaciones son principalmente al
inicio de la lista, la simple es suficiente.

❓ **¿Cómo libero correctamente una lista doble?**
Recorre desde cabeza hacia adelante, guardando el siguiente antes
de liberar cada nodo. La función ld_liberar presentada en este
capítulo lo hace correctamente.

❓ **¿El centinela (dummy node) es útil en listas dobles?**
Sí. Un nodo centinela elimina los casos especiales de cabeza/cola
NULL, simplificando las operaciones de inserción y eliminación.
Es común en implementaciones profesionales.
