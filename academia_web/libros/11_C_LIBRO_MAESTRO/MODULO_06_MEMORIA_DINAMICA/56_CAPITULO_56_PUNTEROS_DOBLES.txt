1. FUNDAMENTOS DEL PUNTERO DOBLE
===============================================================================

Un puntero doble es una variable que almacena la dirección de otra
variable que ya es un puntero. Es decir, un puntero que apunta a otro
puntero.

Sintaxis:

  tipo **nombre;

Donde:
  - tipo: el tipo de dato final al que se apunta
  - *: indica que es un puntero (el primero)
  - *: indica que es un puntero a un puntero (el segundo)

Gráficamente:

  int x = 42;
  int *p = &x;    // p apunta a x
  int **pp = &p;  // pp apunta a p

  pp  →  p  →  x
  **pp = *p = x = 42

Ejemplo básico:

```c
#include <stdio.h>

int main() {
    int x = 42;        // Variable normal
    int *p = &x;       // Puntero a int (apunta a x)
    int **pp = &p;     // Puntero a puntero a int (apunta a p)

    printf("x = %d\n", x);
    printf("&x = %p\n", (void*)&x);
    printf("\n");

    printf("p = %p\n", (void*)p);
    printf("&p = %p\n", (void*)&p);
    printf("*p = %d\n", *p);
    printf("\n");

    printf("pp = %p\n", (void*)pp);
    printf("&pp = %p\n", (void*)&pp);
    printf("*pp = %p\n", (void*)*pp);       // Esto es p (la dirección de x)
    printf("**pp = %d\n", **pp);            // Esto es x (el valor)
    printf("\n");

    // Modificamos x a través de pp
    **pp = 100;
    printf("Después de **pp = 100:\n");
    printf("x = %d\n", x);
    printf("*p = %d\n", *p);

    // Modificamos p a través de pp
    int y = 999;
    *pp = &y;  // Ahora p apunta a y
    printf("\nDespués de *pp = &y:\n");
    printf("*p = %d (p apunta a y)\n", *p);
    printf("**pp = %d\n", **pp);

    return 0;
}
```

La clave está en entender los niveles de indirección:

  - **pp  →  valor final (int)
  - *pp   →  puntero intermedio (int*)
  - pp    →  puntero doble (int**)

Cada asterisco adicional elimina un nivel de indirección.
Cada & añade un nivel de indirección.

===============================================================================
2. SINTAXIS Y DECLARACIONES
===============================================================================

2.1. Declaración de punteros dobles

```c
int **pp;            // Puntero a puntero a int
char **ss;           // Puntero a puntero a char (típico para strings)
float ***ppp;        // Puntero a puntero a puntero a float (poco común)
void **vp;           // Puntero a puntero a void (genérico)
```

2.2. Asignación básica

```c
#include <stdio.h>

int main() {
    int valor = 10;
    int *ptr = &valor;
    int **pptr = &ptr;

    // Leyendo valores:
    printf("Valor directo: %d\n", valor);
    printf("Valor via *ptr: %d\n", *ptr);
    printf("Valor via **pptr: %d\n", **pptr);

    // Escribiendo valores:
    **pptr = 20;
    printf("Valor después de **pptr = 20: %d\n", valor);

    // Cambiando a qué apunta ptr:
    int otro_valor = 30;
    *pptr = &otro_valor;
    printf("Valor via *ptr después de cambiar: %d\n", *ptr);

    // Punteros anidados (tres niveles):
    int ***ppptr = &pptr;
    printf("Valor via ***ppptr: %d\n", ***ppptr);

    return 0;
}
```

2.3. Declaración vs desreferencia

Es importante distinguir entre la declaración y la desreferencia:

// Declaración:
int **pp;    // "pp es un puntero a puntero a int"

// Desreferencia:
*pp          // "el valor al que apunta pp" (tipo: int*)
**pp         // "el valor al que apunta *pp" (tipo: int)

// Operador & con punteros dobles:
int *p;
int **pp = &p;   // &p es de tipo int** (dirección de un puntero)

// Matrices de punteros:
char *nombres[5];     // Array de 5 punteros a char
char **p = nombres;   // p apunta al primer elemento del array

===============================================================================
3. MODIFICAR EL PUNTERO DEL LLAMANTE
===============================================================================

Una de las aplicaciones más importantes de los punteros dobles es
permitir que una función modifique un puntero en el llamante.

En C, los parámetros se pasan por valor. Si pasamos un puntero simple,
la función puede modificar el valor apuntado, pero no el puntero mismo.
Para modificar el puntero, necesitamos un puntero al puntero.

3.1. malloc dentro de una función

```c
#include <stdio.h>
#include <stdlib.h>

// ❌ MAL: No modifica el puntero del llamante
void asignar_mal(int *p) {
    p = (int *)malloc(sizeof(int));  // Modifica la copia local
    if (p != NULL) *p = 42;
    // Al salir, la copia se destruye. El llamante no ve el cambio.
}

// ✅ BIEN: Modifica el puntero del llamante
void asignar_bien(int **pp) {
    *pp = (int *)malloc(sizeof(int));  // Modifica el puntero original
    if (*pp != NULL) **pp = 42;
}

int main() {
    int *ptr = NULL;

    asignar_mal(ptr);
    if (ptr == NULL) {
        printf("asignar_mal NO modificó ptr (ptr sigue NULL)\n");
    }

    asignar_bien(&ptr);
    if (ptr != NULL) {
        printf("asignar_bien SÍ modificó ptr: *ptr = %d\n", *ptr);
        free(ptr);
        ptr = NULL;
    }

    return 0;
}
```

3.2. Función de liberación segura

```c
#include <stdio.h>
#include <stdlib.h>

void liberar_puntero(void **pp) {
    if (pp != NULL && *pp != NULL) {
        free(*pp);
        *pp = NULL;
    }
}

#define liberar(ptr) liberar_puntero((void**)&(ptr))

int main() {
    int *datos = (int *)malloc(10 * sizeof(int));
    if (datos == NULL) return 1;

    for (int i = 0; i < 10; i++) datos[i] = i;

    printf("Antes de liberar: datos = %p\n", (void*)datos);
    liberar(datos);
    printf("Después de liberar: datos = %p\n", (void*)datos);

    // Es seguro llamar de nuevo
    liberar(datos);

    return 0;
}
```

3.3. Insertar nodo en lista enlazada

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Nodo {
    int dato;
    struct Nodo *siguiente;
} Nodo;

// Necesitamos ** para modificar la cabeza de la lista
void insertar_al_inicio(Nodo **cabecera, int valor) {
    Nodo *nuevo = (Nodo *)malloc(sizeof(Nodo));
    if (nuevo == NULL) return;

    nuevo->dato = valor;
    nuevo->siguiente = *cabecera;  // El nuevo apunta al antiguo primero
    *cabecera = nuevo;             // La cabecera ahora apunta al nuevo
}

void imprimir_lista(Nodo *cabecera) {
    printf("Lista: ");
    while (cabecera != NULL) {
        printf("%d ", cabecera->dato);
        cabecera = cabecera->siguiente;
    }
    printf("\n");
}

void liberar_lista(Nodo **cabecera) {
    Nodo *actual = *cabecera;
    while (actual != NULL) {
        Nodo *temp = actual;
        actual = actual->siguiente;
        free(temp);
    }
    *cabecera = NULL;
}

int main() {
    Nodo *lista = NULL;  // Inicialmente vacía

    insertar_al_inicio(&lista, 30);
    insertar_al_inicio(&lista, 20);
    insertar_al_inicio(&lista, 10);

    imprimir_lista(lista);

    liberar_lista(&lista);
    printf("Lista después de liberar: %p\n", (void*)lista);

    return 0;
}
```

Este patrón es fundamental en estructuras de datos (listas, árboles,
grafos) donde necesitamos modificar la raíz o cabeza desde funciones.

4. ARGV Y ENVIRON: PUNTEROS DOBLES EN LA PRÁCTICA
===============================================================================

El ejemplo más ubicuo de punteros dobles en C es el parámetro argv de
main():

  int main(int argc, char **argv)

argv es un puntero doble a char. Es decir, un array de strings.

También existe environ:

  extern char **environ;

```c
#include <stdio.h>

// argv como puntero doble
int main(int argc, char **argv) {
    printf("Argumentos del programa:\n");
    printf("  argc = %d\n\n", argc);

    // argv[0] es el nombre del programa
    // argv[1..argc-1] son los argumentos
    // argv[argc] es NULL

    for (int i = 0; i < argc; i++) {
        printf("  argv[%d] = %s\n", i, argv[i]);
    }

    printf("\n--- Recorriendo argv con puntero doble ---\n");
    char **p = argv;
    int idx = 0;
    while (*p != NULL) {
        printf("  Argumento %d: %s\n", idx++, *p);
        p++;
    }

    return 0;
}
```

También podemos modificar argumentos (con cuidado):

```c
#include <stdio.h>
#include <string.h>

int main(int argc, char **argv) {
    printf("Argumentos originales:\n");
    for (int i = 0; i < argc; i++) {
        printf("  argv[%d] = %s\n", i, argv[i]);
    }

    // Modificar un argumento (si es lo suficientemente largo)
    if (argc > 1 && strlen(argv[1]) >= 5) {
        // argv apunta a memoria modificable
        argv[1][0] = 'M';
        argv[1][1] = 'O';
        argv[1][2] = 'D';
        argv[1][3] = '!';
        argv[1][4] = '\0';
    }

    printf("\nArgumentos después de modificar argv[1]:\n");
    for (int i = 0; i < argc; i++) {
        printf("  argv[%d] = %s\n", i, argv[i]);
    }

    return 0;
}
```

Las variables de entorno también se acceden con puntero doble:

```c
#include <stdio.h>

extern char **environ;

int main() {
    printf("Variables de entorno:\n");

    char **env = environ;
    int count = 0;

    while (*env != NULL) {
        printf("  %s\n", *env);
        env++;
        count++;
        if (count >= 10) break;  // Solo mostrar las primeras 10
    }

    printf("\nTotal de variables: %d\n", count);

    // Buscar una variable específica
    env = environ;
    while (*env != NULL) {
        // Buscamos "PATH="
        char *var = *env;
        if (var[0] == 'P' && var[1] == 'A' && var[2] == 'T' &&
            var[3] == 'H' && var[4] == '=') {
            printf("\nPATH encontrado: %s\n", var);
            break;
        }
        env++;
    }

    return 0;
}
```

5. ARRAYS 2D CON PUNTEROS DOBLES
===============================================================================

Los punteros dobles son la base para crear arrays multidimensionales
dinámicos. Un array de 2D se representa como un puntero a puntero.

5.1. Asignación de matriz 2D dinámica

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
    int filas = 3;
    int columnas = 4;

    // Paso 1: Asignar array de punteros (filas)
    int **matriz = (int **)malloc(filas * sizeof(int *));
    if (matriz == NULL) return 1;

    // Paso 2: Asignar cada fila
    for (int i = 0; i < filas; i++) {
        matriz[i] = (int *)malloc(columnas * sizeof(int));
        if (matriz[i] == NULL) {
            // Liberar filas ya asignadas en caso de error
            for (int j = 0; j < i; j++) free(matriz[j]);
            free(matriz);
            return 1;
        }
    }

    // Inicializar e imprimir
    for (int i = 0; i < filas; i++) {
        for (int j = 0; j < columnas; j++) {
            matriz[i][j] = i * columnas + j;
            printf("%3d ", matriz[i][j]);
        }
        printf("\n");
    }

    // Liberar: orden inverso a la asignación
    for (int i = 0; i < filas; i++) {
        free(matriz[i]);  // Liberar cada fila
    }
    free(matriz);  // Liberar el array de punteros

    return 0;
}
```

5.2. Acceso por filas vs columnas

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    int filas = 1000, columnas = 1000;

    // Asignar matriz
    int **m = (int **)malloc(filas * sizeof(int *));
    for (int i = 0; i < filas; i++) {
        m[i] = (int *)malloc(columnas * sizeof(int));
    }

    // Inicializar
    for (int i = 0; i < filas; i++) {
        for (int j = 0; j < columnas; j++) {
            m[i][j] = rand() % 100;
        }
    }

    // Acceso por filas (rápido: localidad de referencia)
    clock_t inicio = clock();
    long suma_filas = 0;
    for (int i = 0; i < filas; i++) {
        for (int j = 0; j < columnas; j++) {
            suma_filas += m[i][j];  // m[i][j] = *(*(m+i)+j)
        }
    }
    clock_t fin = clock();
    double tiempo_filas = (double)(fin - inicio) / CLOCKS_PER_SEC;

    // Acceso por columnas (lento: saltos en memoria)
    inicio = clock();
    long suma_columnas = 0;
    for (int j = 0; j < columnas; j++) {
        for (int i = 0; i < filas; i++) {
            suma_columnas += m[i][j];
        }
    }
    fin = clock();
    double tiempo_columnas = (double)(fin - inicio) / CLOCKS_PER_SEC;

    printf("Suma por filas:    %ld (tiempo: %.3f seg)\n", suma_filas, tiempo_filas);
    printf("Suma por columnas: %ld (tiempo: %.3f seg)\n", suma_columnas, tiempo_columnas);
    printf("Diferencia: %.2fx más rápido por filas\n",
           tiempo_columnas / tiempo_filas);

    for (int i = 0; i < filas; i++) free(m[i]);
    free(m);

    return 0;
}
```

El acceso por filas es mucho más rápido porque los elementos de una
misma fila están contiguos en memoria (localidad espacial).

6. MATRICES DINÁMICAS AVANZADAS
===============================================================================

6.1. Matriz triangular

```c
#include <stdio.h>
#include <stdlib.h>

int **crear_matriz_triangular(int n) {
    int **m = (int **)malloc(n * sizeof(int *));
    if (m == NULL) return NULL;

    for (int i = 0; i < n; i++) {
        m[i] = (int *)malloc((i + 1) * sizeof(int));
        if (m[i] == NULL) {
            for (int j = 0; j < i; j++) free(m[j]);
            free(m);
            return NULL;
        }
        for (int j = 0; j <= i; j++) {
            m[i][j] = i * (i + 1) / 2 + j + 1;
        }
    }
    return m;
}

void imprimir_matriz_triangular(int **m, int n) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j <= i; j++) {
            printf("%3d ", m[i][j]);
        }
        printf("\n");
    }
}

void liberar_matriz_triangular(int **m, int n) {
    for (int i = 0; i < n; i++) free(m[i]);
    free(m);
}

int main() {
    int n = 6;
    int **triangular = crear_matriz_triangular(n);

    if (triangular == NULL) return 1;

    printf("Matriz triangular de %dx%d:\n\n", n, n);
    imprimir_matriz_triangular(triangular, n);

    liberar_matriz_triangular(triangular, n);
    return 0;
}
```

6.2. Funciones genéricas con void**

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void **crear_matriz_generica(int filas, int columnas, size_t tam_elem) {
    void **matriz = (void **)malloc(filas * sizeof(void *));
    if (matriz == NULL) return NULL;

    for (int i = 0; i < filas; i++) {
        matriz[i] = malloc(columnas * tam_elem);
        if (matriz[i] == NULL) {
            for (int j = 0; j < i; j++) free(matriz[j]);
            free(matriz);
            return NULL;
        }
    }
    return matriz;
}

void liberar_matriz_generica(void **matriz, int filas) {
    if (matriz == NULL) return;
    for (int i = 0; i < filas; i++) {
        free(matriz[i]);
    }
    free(matriz);
}

int main() {
    // Matriz de enteros
    int **enteros = (int **)crear_matriz_generica(3, 4, sizeof(int));
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 4; j++)
            enteros[i][j] = i * 10 + j;

    printf("Matriz de enteros:\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 4; j++)
            printf("%3d ", enteros[i][j]);
        printf("\n");
    }

    liberar_matriz_generica((void **)enteros, 3);

    // Matriz de doubles
    double **reales = (double **)crear_matriz_generica(2, 3, sizeof(double));
    for (int i = 0; i < 2; i++)
        for (int j = 0; j < 3; j++)
            reales[i][j] = i * 1.5 + j * 0.5;

    printf("\nMatriz de doubles:\n");
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++)
            printf("%.1f ", reales[i][j]);
        printf("\n");
    }

    liberar_matriz_generica((void **)reales, 2);

    return 0;
}
```

7.1. Array dinámico de estructuras

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    int id;
    char nombre[50];
    float promedio;
} Estudiante;

Estudiante **crear_grupo(int num) {
    Estudiante **grupo = (Estudiante **)malloc(num * sizeof(Estudiante *));
    if (grupo == NULL) return NULL;

    for (int i = 0; i < num; i++) {
        grupo[i] = (Estudiante *)malloc(sizeof(Estudiante));
        if (grupo[i] == NULL) {
            for (int j = 0; j < i; j++) free(grupo[j]);
            free(grupo);
            return NULL;
        }
        grupo[i]->id = i + 1;
        snprintf(grupo[i]->nombre, 50, "Estudiante_%d", i + 1);
        grupo[i]->promedio = 5.0 + (rand() % 51) / 10.0;
    }
    return grupo;
}

void imprimir_grupo(Estudiante **grupo, int num) {
    printf("=== GRUPO DE ESTUDIANTES ===\n");
    for (int i = 0; i < num; i++) {
        printf("ID: %d | Nombre: %s | Promedio: %.1f\n",
               grupo[i]->id, grupo[i]->nombre, grupo[i]->promedio);
    }
}

void liberar_grupo(Estudiante **grupo, int num) {
    for (int i = 0; i < num; i++) free(grupo[i]);
    free(grupo);
}

int main() {
    Estudiante **grupo = crear_grupo(5);
    if (grupo == NULL) return 1;

    imprimir_grupo(grupo, 5);
    liberar_grupo(grupo, 5);

    return 0;
}
```

7.2. Modificar estructura desde función

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char *titulo;
    int paginas;
} Libro;

// Necesitamos ** para modificar el puntero del llamante
int libro_init(Libro **libro, const char *titulo, int paginas) {
    *libro = (Libro *)malloc(sizeof(Libro));
    if (*libro == NULL) return -1;

    (*libro)->titulo = (char *)malloc(strlen(titulo) + 1);
    if ((*libro)->titulo == NULL) {
        free(*libro);
        *libro = NULL;
        return -1;
    }

    strcpy((*libro)->titulo, titulo);
    (*libro)->paginas = paginas;
    return 0;
}

void libro_destroy(Libro **libro) {
    if (libro == NULL || *libro == NULL) return;
    free((*libro)->titulo);
    free(*libro);
    *libro = NULL;
}

int main() {
    Libro *libro = NULL;

    if (libro_init(&libro, "C Master Book", 999) == 0) {
        printf("Libro: %s (%d páginas)\n", libro->titulo, libro->paginas);
        libro_destroy(&libro);
    }

    if (libro == NULL) {
        printf("El libro fue destruido correctamente.\n");
    }

    return 0;
}
```

8. PUNTEROS TRIPLES Y MÁS ALLÁ
===============================================================================

Aunque los punteros dobles son los más comunes después de los simples,
ocasionalmente te encontrarás con punteros triples (***) o más.

8.1. Puntero triple para matriz 3D

```c
#include <stdio.h>
#include <stdlib.h>

int ***crear_matriz_3d(int x, int y, int z) {
    int ***m = (int ***)malloc(x * sizeof(int **));
    if (m == NULL) return NULL;

    for (int i = 0; i < x; i++) {
        m[i] = (int **)malloc(y * sizeof(int *));
        if (m[i] == NULL) {
            for (int k = 0; k < i; k++) {
                for (int j = 0; j < y; j++) free(m[k][j]);
                free(m[k]);
            }
            free(m);
            return NULL;
        }
        for (int j = 0; j < y; j++) {
            m[i][j] = (int *)malloc(z * sizeof(int));
            if (m[i][j] == NULL) {
                for (int k = 0; k < j; k++) free(m[i][k]);
                for (int k = 0; k < i; k++) {
                    for (int j2 = 0; j2 < y; j2++) free(m[k][j2]);
                    free(m[k]);
                }
                free(m);
                return NULL;
            }
        }
    }
    return m;
}

void liberar_matriz_3d(int ***m, int x, int y) {
    if (m == NULL) return;
    for (int i = 0; i < x; i++) {
        for (int j = 0; j < y; j++) {
            free(m[i][j]);
        }
        free(m[i]);
    }
    free(m);
}

int main() {
    int x = 3, y = 4, z = 5;

    int ***cubo = crear_matriz_3d(x, y, z);
    if (cubo == NULL) return 1;

    for (int i = 0; i < x; i++)
        for (int j = 0; j < y; j++)
            for (int k = 0; k < z; k++)
                cubo[i][j][k] = i * 100 + j * 10 + k;

    printf("Elemento [1][2][3] = %d\n", cubo[1][2][3]);
    printf("Elemento [2][0][4] = %d\n", cubo[2][0][4]);

    liberar_matriz_3d(cubo, x, y);
    return 0;
}
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef int (*Comparador)(const void *, const void *);

void ordenar_filas(int **matriz, int filas, int columnas, Comparador cmp) {
    qsort(matriz, filas, sizeof(int *), cmp);
}

int comparar_por_suma(const void *a, const void *b) {
    int *fila_a = *(int **)a;
    int *fila_b = *(int **)b;
    // Comparamos por primer elemento (simplificado)
    return fila_a[0] - fila_b[0];
}

int main() {
    int **m = (int **)malloc(3 * sizeof(int *));
    for (int i = 0; i < 3; i++) {
        m[i] = (int *)malloc(4 * sizeof(int));
        for (int j = 0; j < 4; j++)
            m[i][j] = (3 - i) * 10 + j;  // Filas inversas
    }

    printf("Antes de ordenar:\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 4; j++)
            printf("%3d ", m[i][j]);
        printf("\n");
    }

    ordenar_filas(m, 3, 4, comparar_por_suma);

    printf("Después de ordenar (por primer elemento):\n");
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 4; j++)
            printf("%3d ", m[i][j]);
        printf("\n");
    }

    for (int i = 0; i < 3; i++) free(m[i]);
    free(m);
    return 0;
}
```

RESUMEN DEL CAPÍTULO
===============================================================================

- Un puntero doble (**ptr) almacena la dirección de otro puntero.
- **ptr accede al valor final, *ptr accede al puntero intermedio.
- Los punteros dobles permiten modificar punteros desde funciones.
- La función main() recibe char **argv como puntero doble.
- Para crear matrices 2D dinámicas, se asigna primero un array de
  punteros y luego cada fila individualmente.
- El acceso por filas es más rápido que por columnas por localidad
  de referencia.
- Los punteros triples (***) existen pero son raramente necesarios.
- Siempre inicializa los punteros dobles y verifica NULL.

===============================================================================
PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Cuándo necesito un puntero doble en lugar de un puntero simple?**
Cuando necesitas que una función modifique el puntero mismo (no solo
el valor apuntado). También para arrays 2D dinámicos, argv, y para
implementar estructuras de datos como listas enlazadas.

❓ **¿Puedo tener punteros con más de dos niveles?**
Sí, int ***p es válido. Pero más de 2 niveles rara vez son necesarios
y hacen el código difícil de leer. Si necesitas más, considera usar
estructuras o typedef.

❓ **¿Cómo sé cuántos asteriscos poner en una declaración?**
Cada nivel de indirección añade un *. Si declaras un array de strings
(como argv), es char**. Si declaras un array de arrays de strings,
sería char***, y así sucesivamente.

❓ **¿Es seguro castear int** a void**?**
No directamente. void** y int** son tipos diferentes. Sin embargo,
puedes castear int* a void* sin problemas. Para matrices genéricas,
usa void* como base, no void**.
