1. DESCRIPCIÓN DEL PROYECTO
===============================================================================

Vamos a construir un **Sistema de Gestión Académica** que permita:

  - Registrar estudiantes con: ID, nombre, edad, y calificaciones.
  - Almacenar las calificaciones en una matriz dinámica.
  - Mantener los estudiantes en una lista doblemente enlazada.
  - Cargar y guardar los datos en un archivo binario.
  - Buscar estudiantes por ID o por nombre.
  - Calcular estadísticas: promedio general, mejor/peor estudiante.
  - Ordenar la lista por diferentes criterios.
  - Liberar toda la memoria correctamente.

Estructuras de datos utilizadas:

  1. **Calificaciones**: matriz dinámica contigua (double *).
  2. **Estudiante**: estructura con campos dinámicos (nombre, califs).
  3. **ListaDoble**: lista doblemente enlazada de estudiantes.

Flujo del programa:

  1. Inicializar lista vacía.
  2. Cargar datos de archivo (si existe).
  3. Mostrar menú interactivo.
  4. Ejecutar operaciones según entrada del usuario.
  5. Guardar datos al salir.
  6. Liberar toda la memoria.

```c
int califs_init(Calificaciones *c, int num_materias) {
    c->notas = (double *)malloc(num_materias * sizeof(double));
    if (c->notas == NULL) return -1;
    c->num_materias = num_materias;
    for (int i = 0; i < num_materias; i++) {
        c->notas[i] = 0.0;
    }
    return 0;
}

void califs_liberar(Calificaciones *c) {
    free(c->notas);
    c->notas = NULL;
    c->num_materias = 0;
}

double califs_promedio(Calificaciones *c) {
    if (c == NULL || c->num_materias == 0) return 0.0;
    double suma = 0.0;
    for (int i = 0; i < c->num_materias; i++) {
        suma += c->notas[i];
    }
    return suma / c->num_materias;
}

void califs_imprimir(Calificaciones *c) {
    printf("[");
    for (int i = 0; i < c->num_materias; i++) {
        printf("%.1f", c->notas[i]);
        if (i < c->num_materias - 1) printf(", ");
    }
    printf("] (prom: %.2f)", califs_promedio(c));
}
```

```c
Estudiante *estudiante_crear(int id, const char *nombre, int edad,
                              int num_materias) {
    Estudiante *e = (Estudiante *)malloc(sizeof(Estudiante));
    if (e == NULL) return NULL;

    e->id = id;
    e->nombre = (char *)malloc(strlen(nombre) + 1);
    if (e->nombre == NULL) {
        free(e);
        return NULL;
    }
    strcpy(e->nombre, nombre);
    e->edad = edad;

    if (califs_init(&e->califs, num_materias) != 0) {
        free(e->nombre);
        free(e);
        return NULL;
    }

    e->anterior = NULL;
    e->siguiente = NULL;
    return e;
}

void estudiante_imprimir(Estudiante *e) {
    if (e == NULL) return;
    printf("ID: %d | Nombre: %s | Edad: %d | ", e->id, e->nombre, e->edad);
    califs_imprimir(&e->califs);
    printf("\n");
}

void estudiante_liberar(Estudiante *e) {
    if (e == NULL) return;
    califs_liberar(&e->califs);
    free(e->nombre);
    e->nombre = NULL;
    free(e);
}
```

```c
void lista_init(ListaEstudiantes *lista) {
    lista->cabeza = NULL;
    lista->cola = NULL;
    lista->total = 0;
}

// Insertar al final
int lista_insertar(ListaEstudiantes *lista, Estudiante *e) {
    if (e == NULL) return -1;

    if (lista->cola == NULL) {
        lista->cabeza = e;
        lista->cola = e;
    } else {
        e->anterior = lista->cola;
        lista->cola->siguiente = e;
        lista->cola = e;
    }
    lista->total++;
    return 0;
}

// Eliminar estudiante por ID
int lista_eliminar(ListaEstudiantes *lista, int id) {
    Estudiante *actual = lista->cabeza;
    while (actual != NULL) {
        if (actual->id == id) {
            // Ajustar punteros de vecinos
            if (actual->anterior != NULL) {
                actual->anterior->siguiente = actual->siguiente;
            } else {
                lista->cabeza = actual->siguiente;
            }

            if (actual->siguiente != NULL) {
                actual->siguiente->anterior = actual->anterior;
            } else {
                lista->cola = actual->anterior;
            }

            estudiante_liberar(actual);
            lista->total--;
            return 0;
        }
        actual = actual->siguiente;
    }
    return -1;  // No encontrado
}

// Buscar por ID
Estudiante *lista_buscar_id(ListaEstudiantes *lista, int id) {
    Estudiante *actual = lista->cabeza;
    while (actual != NULL) {
        if (actual->id == id) return actual;
        actual = actual->siguiente;
    }
    return NULL;
}

// Buscar por nombre (primera coincidencia)
Estudiante *lista_buscar_nombre(ListaEstudiantes *lista, const char *nombre) {
    Estudiante *actual = lista->cabeza;
    while (actual != NULL) {
        if (strcmp(actual->nombre, nombre) == 0) return actual;
        actual = actual->siguiente;
    }
    return NULL;
}

// Mostrar todos los estudiantes
void lista_mostrar(ListaEstudiantes *lista) {
    if (lista->cabeza == NULL) {
        printf("No hay estudiantes registrados.\n");
        return;
    }
    printf("\n=== LISTA DE ESTUDIANTES (%d) ===\n", lista->total);
    Estudiante *actual = lista->cabeza;
    while (actual != NULL) {
        estudiante_imprimir(actual);
        actual = actual->siguiente;
    }
    printf("================================\n");
}

// Mostrar en orden inverso
void lista_mostrar_inverso(ListaEstudiantes *lista) {
    if (lista->cola == NULL) {
        printf("No hay estudiantes registrados.\n");
        return;
    }
    printf("\n=== LISTA INVERSA (%d) ===\n", lista->total);
    Estudiante *actual = lista->cola;
    while (actual != NULL) {
        estudiante_imprimir(actual);
        actual = actual->anterior;
    }
    printf("============================\n");
}

// Liberar toda la lista
void lista_liberar(ListaEstudiantes *lista) {
    Estudiante *actual = lista->cabeza;
    while (actual != NULL) {
        Estudiante *temp = actual;
        actual = actual->siguiente;
        estudiante_liberar(temp);
    }
    lista->cabeza = NULL;
    lista->cola = NULL;
    lista->total = 0;
}
```

```c
int guardar_datos(ListaEstudiantes *lista) {
    FILE *f = fopen(ARCHIVO_DATOS, "wb");
    if (f == NULL) {
        perror("Error al guardar");
        return -1;
    }

    // Escribir metadatos: número de estudiantes y materias
    // Asumimos mismo número de materias para todos
    int num_materias = 0;
    if (lista->cabeza != NULL) {
        num_materias = lista->cabeza->califs.num_materias;
    }

    int total = lista->total;
    fwrite(&total, sizeof(int), 1, f);
    fwrite(&num_materias, sizeof(int), 1, f);

    // Escribir cada estudiante
    Estudiante *actual = lista->cabeza;
    while (actual != NULL) {
        // Escribir campos fijos
        fwrite(&actual->id, sizeof(int), 1, f);
        int len_nombre = strlen(actual->nombre) + 1;
        fwrite(&len_nombre, sizeof(int), 1, f);
        fwrite(actual->nombre, sizeof(char), len_nombre, f);
        fwrite(&actual->edad, sizeof(int), 1, f);

        // Escribir calificaciones
        fwrite(actual->califs.notas, sizeof(double), num_materias, f);

        actual = actual->siguiente;
    }

    fclose(f);
    printf("Datos guardados en '%s' (%d estudiantes).\n",
           ARCHIVO_DATOS, total);
    return 0;
}

int cargar_datos(ListaEstudiantes *lista) {
    FILE *f = fopen(ARCHIVO_DATOS, "rb");
    if (f == NULL) return -1;  // Archivo no existe (no es error)

    int total, num_materias;
    fread(&total, sizeof(int), 1, f);
    fread(&num_materias, sizeof(int), 1, f);

    for (int i = 0; i < total; i++) {
        int id, edad, len_nombre;
        fread(&id, sizeof(int), 1, f);
        fread(&len_nombre, sizeof(int), 1, f);

        char *nombre = (char *)malloc(len_nombre);
        if (nombre == NULL) {
            fclose(f);
            return -1;
        }
        fread(nombre, sizeof(char), len_nombre, f);
        fread(&edad, sizeof(int), 1, f);

        Estudiante *e = estudiante_crear(id, nombre, edad, num_materias);
        free(nombre);

        if (e == NULL) {
            fclose(f);
            return -1;
        }

        fread(e->califs.notas, sizeof(double), num_materias, f);
        lista_insertar(lista, e);
    }

    fclose(f);
    printf("Datos cargados desde '%s' (%d estudiantes).\n",
           ARCHIVO_DATOS, total);
    return 0;
}
```

```c
void lista_ordenar_por_nombre(ListaEstudiantes *lista) {
    if (lista->total <= 1) return;

    int intercambiado;
    do {
        intercambiado = 0;
        Estudiante *actual = lista->cabeza;
        while (actual != NULL && actual->siguiente != NULL) {
            if (strcmp(actual->nombre, actual->siguiente->nombre) > 0) {
                // Intercambiar datos (no punteros)
                // Intercambiar IDs
                int temp_id = actual->id;
                actual->id = actual->siguiente->id;
                actual->siguiente->id = temp_id;

                // Intercambiar nombres
                char *temp_nom = actual->nombre;
                actual->nombre = actual->siguiente->nombre;
                actual->siguiente->nombre = temp_nom;

                // Intercambiar edades
                int temp_edad = actual->edad;
                actual->edad = actual->siguiente->edad;
                actual->siguiente->edad = temp_edad;

                // Intercambiar calificaciones
                Calificaciones temp_cal = actual->califs;
                actual->califs = actual->siguiente->califs;
                actual->siguiente->califs = temp_cal;

                intercambiado = 1;
            }
            actual = actual->siguiente;
        }
    } while (intercambiado);
}

void lista_ordenar_por_promedio(ListaEstudiantes *lista) {
    if (lista->total <= 1) return;

    int intercambiado;
    do {
        intercambiado = 0;
        Estudiante *actual = lista->cabeza;
        while (actual != NULL && actual->siguiente != NULL) {
            double p1 = califs_promedio(&actual->califs);
            double p2 = califs_promedio(&actual->siguiente->califs);
            if (p1 < p2) {
                int temp_id = actual->id;
                actual->id = actual->siguiente->id;
                actual->siguiente->id = temp_id;

                char *temp_nom = actual->nombre;
                actual->nombre = actual->siguiente->nombre;
                actual->siguiente->nombre = temp_nom;

                int temp_edad = actual->edad;
                actual->edad = actual->siguiente->edad;
                actual->siguiente->edad = temp_edad;

                Calificaciones temp_cal = actual->califs;
                actual->califs = actual->siguiente->califs;
                actual->siguiente->califs = temp_cal;

                intercambiado = 1;
            }
            actual = actual->siguiente;
        }
    } while (intercambiado);
}

// ESTADÍSTICAS

void calcular_estadisticas(ListaEstudiantes *lista) {
    if (lista->cabeza == NULL) {
        printf("No hay estudiantes para calcular estadísticas.\n");
        return;
    }

    double suma_total = 0.0;
    double max_prom = -1.0;
    double min_prom = 101.0;
    Estudiante *mejor = NULL;
    Estudiante *peor = NULL;

    Estudiante *actual = lista->cabeza;
    while (actual != NULL) {
        double prom = califs_promedio(&actual->califs);
        suma_total += prom;

        if (prom > max_prom) {
            max_prom = prom;
            mejor = actual;
        }
        if (prom < min_prom) {
            min_prom = prom;
            peor = actual;
        }

        actual = actual->siguiente;
    }

    double promedio_general = suma_total / lista->total;

    printf("\n=== ESTADÍSTICAS ACADÉMICAS ===\n");
    printf("Total estudiantes: %d\n", lista->total);
    printf("Promedio general:  %.2f\n", promedio_general);
    if (mejor) printf("Mejor estudiante:   %s (%.2f)\n", mejor->nombre, max_prom);
    if (peor)  printf("Peor estudiante:    %s (%.2f)\n", peor->nombre, min_prom);
    printf("================================\n");
}
```

```c
    int c;
    while ((c = getchar()) != '\n' && c != EOF);

void agregar_estudiante(ListaEstudiantes *lista) {
    int id, edad, num_materias;
    char nombre[MAX_NOMBRE];

    printf("ID: ");
    scanf("%d", &id);
    limpiar_buffer();

    // Verificar que el ID no exista
    if (lista_buscar_id(lista, id) != NULL) {
        printf("Error: ya existe un estudiante con ID %d.\n", id);
        return;
    }

    printf("Nombre: ");
    fgets(nombre, MAX_NOMBRE, stdin);
    nombre[strcspn(nombre, "\n")] = '\0';

    printf("Edad: ");
    scanf("%d", &edad);
    limpiar_buffer();

    printf("Número de materias: ");
    scanf("%d", &num_materias);
    limpiar_buffer();

    if (num_materias <= 0 || num_materias > MAX_MATERIAS) {
        printf("Error: número de materias inválido (1-%d).\n", MAX_MATERIAS);
        return;
    }

    Estudiante *e = estudiante_crear(id, nombre, edad, num_materias);
    if (e == NULL) {
        printf("Error: no se pudo crear el estudiante.\n");
        return;
    }

    // Ingresar calificaciones iniciales
    printf("Ingresa las calificaciones (%d materias):\n", num_materias);
    for (int i = 0; i < num_materias; i++) {
        printf("  Materia %d: ", i + 1);
        double nota;
        scanf("%lf", &nota);
        limpiar_buffer();
        if (nota < 0.0) nota = 0.0;
        if (nota > 10.0) nota = 10.0;
        e->califs.notas[i] = nota;
    }

    lista_insertar(lista, e);
    printf("Estudiante '%s' agregado correctamente.\n", nombre);
}

void eliminar_estudiante(ListaEstudiantes *lista) {
    int id;
    printf("ID del estudiante a eliminar: ");
    scanf("%d", &id);
    limpiar_buffer();

    if (lista_eliminar(lista, id) == 0) {
        printf("Estudiante con ID %d eliminado.\n", id);
    } else {
        printf("No se encontró estudiante con ID %d.\n", id);
    }
}

    printf("ID a buscar: ");

void asignar_calificacion(ListaEstudiantes *lista) {
    int id, materia;
    double nota;

    printf("ID del estudiante: ");
    scanf("%d", &id);
    limpiar_buffer();

    Estudiante *e = lista_buscar_id(lista, id);
    if (e == NULL) {
        printf("Estudiante no encontrado.\n");
        return;
    }

    printf("Número de materia (1-%d): ", e->califs.num_materias);

        printf("Materia inválida.\n");
        return;

    printf("Nueva calificación (0-10): ");
    scanf("%lf", &nota);
    limpiar_buffer();

    if (nota < 0.0) nota = 0.0;
    if (nota > 10.0) nota = 10.0;

    e->califs.notas[materia - 1] = nota;
    printf("Calificación actualizada.\n");
}
```

```c
int main() {

    do {

                break;

                lista_mostrar_inverso(&lista);
                break;

                lista_mostrar(&lista);
                break;
            case 8:
                lista_ordenar_por_promedio(&lista);
                printf("Lista ordenada por promedio descendente.\n");
                lista_mostrar(&lista);
                break;
            case 9:
                calcular_estadisticas(&lista);
                break;
            case 10:
                asignar_calificacion(&lista);
                break;
            case 0:
                guardar_datos(&lista);
                printf("¡Hasta luego!\n");
                break;
            default:
                printf("Opción inválida.\n");
                break;
        }

        if (opcion != 0) pausa();

    } while (opcion != 0);

    lista_liberar(&lista);
    return 0;
}
```

===============================================================================
PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Cómo sé si mi programa tiene memory leaks?**
Usa Valgrind: `valgrind --leak-check=full ./programa`. Si dice
"definitely lost: 0 bytes in 0 blocks", no hay leaks. Si muestra
pérdidas, revisa cada malloc/free.

❓ **¿Por qué se intercambian los datos del nodo en lugar de los punteros?**
Intercambiar datos es más simple para este proyecto. Intercambiar
punteros es más eficiente para nodos grandes, pero requiere manejar
cuidadosamente los punteros anterior/siguiente.

❓ **¿Puedo usar el programa con cientos de estudiantes?**
Sí. La lista doblemente enlazada maneja cualquier cantidad (dentro
de la memoria disponible). El archivo binario es eficiente para
grandes volúmenes.

❓ **¿Qué hago si el archivo de datos se corrompe?**
El programa intenta cargar pero si falla (fopen devuelve NULL),
simplemente comienza con lista vacía. Puedes eliminar el archivo
manualmente para reiniciar.
