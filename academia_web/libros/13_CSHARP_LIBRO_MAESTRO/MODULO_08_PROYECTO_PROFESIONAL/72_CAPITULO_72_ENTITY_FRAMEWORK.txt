  1. ¿QUÉ ES ENTITY FRAMEWORK CORE?
  ──────────────────────────────────
  EF Core es un ORM (Object-Relational Mapper) que permite a los desarrolladores
  trabajar con bases de datos usando objetos .NET. Elimina la necesidad de
  escribir SQL manual para la mayoría de las operaciones.

  Características principales:
    • Multiplataforma (Windows, Linux, macOS).
    • Soporte para múltiples proveedores: SQL Server, PostgreSQL, SQLite, etc.
    • Migraciones automáticas.
    • LINQ para consultas.
    • Seguimiento de cambios (Change Tracker).
    • Lazy, Eager y Explicit loading.

  2. DBCONTEXT Y DBSET
  ─────────────────────
  DbContext representa la sesión con la base de datos. Es la clase principal
  para interactuar con EF Core.
    • DbSet<T> representa una tabla en la base de datos.
    • DbContextOptions configuran el proveedor y la cadena de conexión.

  3. MIGRACIONES
  ──────────────
  Las migraciones son el mecanismo de EF Core para sincronizar el modelo de
  datos con el esquema de la base de datos. Cada migración representa un
  cambio incremental.

  Comandos principales (Package Manager Console):
    Add-Migration NombreMigracion
    Update-Database
    Remove-Migration
    Script-Migration

  4. RELACIONES
  ─────────────
  EF Core soporta tres tipos de relaciones:
    • Uno a uno (1:1) — ej. Persona → Pasaporte.
    • Uno a muchos (1:N) — ej. Categoría → Productos.
    • Muchos a muchos (N:N) — ej. Estudiante ←→ Curso.

  5. LINQ TO SQL
  ──────────────
  LINQ (Language Integrated Query) permite escribir consultas directamente
  en C#. EF Core traduce estas consultas a SQL.

  Métodos comunes: Where, Select, Include, OrderBy, GroupBy, Join, Any, All.

  6. CICLO DE VIDA DE UNA ENTIDAD
  ────────────────────────────────
  Estados en el Change Tracker:
    • Added — nueva entidad.
    • Unchanged — sin cambios desde la carga.
    • Modified — una propiedad cambió.
    • Deleted — marcada para eliminación.
    • Detached — no está siendo rastreada.

  using Microsoft.EntityFrameworkCore;
  using MiApp.Domain.Entities;

  namespace MiApp.Infrastructure.Data
  {
      public class BibliotecaDbContext : DbContext
      {
          public BibliotecaDbContext(DbContextOptions<BibliotecaDbContext> options)
              : base(options) { }

          public DbSet<Libro> Libros { get; set; }
          public DbSet<Autor> Autores { get; set; }
          public DbSet<Categoria> Categorias { get; set; }
          public DbSet<Prestamo> Prestamos { get; set; }
          public DbSet<Usuario> Usuarios { get; set; }

          protected override void OnModelCreating(ModelBuilder modelBuilder)
          {
              base.OnModelCreating(modelBuilder);

              modelBuilder.ApplyConfigurationsFromAssembly(
                  typeof(BibliotecaDbContext).Assembly);
          }
      }
  }

  Ejemplo 3: Configuración con Fluent API (IEntityTypeConfiguration)
  ──────────────────────────────────────────────────────────────────────────────

  // ============================
  // Infrastructure/Data/Configurations/LibroConfiguration.cs
  // ============================

  using Microsoft.EntityFrameworkCore;
  using Microsoft.EntityFrameworkCore.Metadata.Builders;
  using MiApp.Domain.Entities;

  namespace MiApp.Infrastructure.Data.Configurations
  {
      public class LibroConfiguration : IEntityTypeConfiguration<Libro>
      {
          public void Configure(EntityTypeBuilder<Libro> builder)
          {
              builder.ToTable("Libros");

              builder.HasKey(l => l.Id);

              builder.Property(l => l.Titulo)
                     .IsRequired()
                     .HasMaxLength(300);

              builder.Property(l => l.ISBN)
                     .IsRequired()
                     .HasMaxLength(17);

              builder.Property(l => l.AnioPublicacion)
                     .IsRequired();

              builder.Property(l => l.Paginas)
                     .IsRequired();

              builder.Property(l => l.Disponible)
                     .HasDefaultValue(true);

              builder.Property(l => l.FechaRegistro)
                     .HasDefaultValueSql("GETUTCDATE()");

              // Relación 1:N con Categoria
              builder.HasOne(l => l.Categoria)
                     .WithMany(c => c.Libros)
                     .HasForeignKey(l => l.CategoriaId)
                     .OnDelete(DeleteBehavior.Restrict);

              // Relación N:N con Autor (tabla intermedia)
              builder.HasMany(l => l.Autores)
                     .WithMany(a => a.Libros)
                     .UsingEntity(j => j.ToTable("LibrosAutores"));
          }
      }
  }

  // ============================
  // Infrastructure/Data/Configurations/PrestamoConfiguration.cs
  // ============================

  using Microsoft.EntityFrameworkCore;
  using Microsoft.EntityFrameworkCore.Metadata.Builders;
  using MiApp.Domain.Entities;

  namespace MiApp.Infrastructure.Data.Configurations
  {
      public class PrestamoConfiguration : IEntityTypeConfiguration<Prestamo>
      {
          public void Configure(EntityTypeBuilder<Prestamo> builder)
          {
              builder.ToTable("Prestamos");

              builder.HasKey(p => p.Id);

              builder.Property(p => p.FechaPrestamo)
                     .HasDefaultValueSql("GETUTCDATE()");

              builder.Property(p => p.FechaMaximaDevolucion)
                     .IsRequired();

              builder.Property(p => p.Devuelto)
                     .HasDefaultValue(false);

              builder.HasOne(p => p.Libro)
                     .WithMany(l => l.Prestamos)
                     .HasForeignKey(p => p.LibroId);

              builder.HasOne(p => p.Usuario)
                     .WithMany(u => u.Prestamos)
                     .HasForeignKey(p => p.UsuarioId);
          }
      }
  }

  Ejemplo 6: CRUD completo con EF Core
  ──────────────────────────────────────────────────────────────────────────────

  // ============================
  // Application/Servicios/LibroService.cs
  // ============================

  using System;
  using System.Collections.Generic;
  using System.Linq;
  using System.Threading.Tasks;
  using Microsoft.EntityFrameworkCore;
  using MiApp.Application.DTOs;
  using MiApp.Domain.Entities;
  using MiApp.Domain.Interfaces;
  using MiApp.Infrastructure.Data;

  namespace MiApp.Application.Servicios
  {
      public class LibroService : ILibroService
      {
          private readonly BibliotecaDbContext _context;

          public LibroService(BibliotecaDbContext context)
          {
              _context = context;
          }

          // CREATE
          public async Task<LibroDTO> CrearAsync(CrearLibroDTO dto)
          {
              var libro = new Libro
              {
                  Id = Guid.NewGuid(),
                  Titulo = dto.Titulo,
                  ISBN = dto.ISBN,
                  AnioPublicacion = dto.AnioPublicacion,
                  Paginas = dto.Paginas,
                  Disponible = true,
                  CategoriaId = dto.CategoriaId,
                  FechaRegistro = DateTime.UtcNow
              };

              _context.Libros.Add(libro);
              await _context.SaveChangesAsync();

              return await ObtenerPorIdAsync(libro.Id);
          }

          // READ (todos)
          public async Task<IEnumerable<LibroDTO>> ObtenerTodosAsync()
          {
              return await _context.Libros
                  .Include(l => l.Categoria)
                  .Include(l => l.Autores)
                  .Select(l => new LibroDTO
                  {
                      Id = l.Id,
                      Titulo = l.Titulo,
                      ISBN = l.ISBN,
                      AnioPublicacion = l.AnioPublicacion,
                      Paginas = l.Paginas,
                      Disponible = l.Disponible,
                      CategoriaNombre = l.Categoria.Nombre,
                      Autores = l.Autores.Select(a => $"{a.Nombre} {a.Apellido}").ToList()
                  })
                  .ToListAsync();
          }

          // READ (por Id)
          public async Task<LibroDTO> ObtenerPorIdAsync(Guid id)
          {
              var libro = await _context.Libros
                  .Include(l => l.Categoria)
                  .Include(l => l.Autores)
                  .FirstOrDefaultAsync(l => l.Id == id);

              if (libro == null)
                  throw new KeyNotFoundException("Libro no encontrado.");

              return new LibroDTO
              {
                  Id = libro.Id,
                  Titulo = libro.Titulo,
                  ISBN = libro.ISBN,
                  AnioPublicacion = libro.AnioPublicacion,
                  Paginas = libro.Paginas,
                  Disponible = libro.Disponible,
                  CategoriaNombre = libro.Categoria.Nombre,
                  Autores = libro.Autores.Select(a => $"{a.Nombre} {a.Apellido}").ToList()
              };
          }

          // UPDATE
          public async Task<LibroDTO> ActualizarAsync(ActualizarLibroDTO dto)
          {
              var libro = await _context.Libros.FindAsync(dto.Id);
              if (libro == null)
                  throw new KeyNotFoundException("Libro no encontrado.");

              libro.Titulo = dto.Titulo;
              libro.ISBN = dto.ISBN;
              libro.AnioPublicacion = dto.AnioPublicacion;
              libro.Paginas = dto.Paginas;
              libro.CategoriaId = dto.CategoriaId;

              await _context.SaveChangesAsync();

              return await ObtenerPorIdAsync(libro.Id);
          }

          // DELETE
          public async Task<bool> EliminarAsync(Guid id)
          {
              var libro = await _context.Libros.FindAsync(id);
              if (libro == null) return false;

              _context.Libros.Remove(libro);
              await _context.SaveChangesAsync();
              return true;
          }
      }
  }

  public class ConsultasAvanzadas
  {
      private readonly BibliotecaDbContext _context;

      public ConsultasAvanzadas(BibliotecaDbContext context)
      {
          _context = context;
      }

      // Buscar libros por título (contains)
      public async Task<List<Libro>> BuscarPorTituloAsync(string termino)
      {
          return await _context.Libros
              .Where(l => l.Titulo.Contains(termino))
              .OrderBy(l => l.Titulo)
              .ToListAsync();
      }

      // Libros disponibles con sus autores
      public async Task<List<Libro>> LibrosDisponiblesAsync()
      {
          return await _context.Libros
              .Where(l => l.Disponible)
              .Include(l => l.Autores)
              .Include(l => l.Categoria)
              .ToListAsync();
      }

      // Préstamos vencidos
      public async Task<List<Prestamo>> PrestamosVencidosAsync()
      {
          return await _context.Prestamos
              .Where(p => !p.Devuelto && p.FechaMaximaDevolucion < DateTime.UtcNow)
              .Include(p => p.Libro)
              .Include(p => p.Usuario)
              .ToListAsync();
      }

      // Top 5 libros más prestados
      public async Task<List<LibroMasPrestadoDTO>> TopLibrosMasPrestadosAsync()
      {
          return await _context.Prestamos
              .GroupBy(p => p.LibroId)
              .Select(g => new
              {
                  LibroId = g.Key,
                  Total = g.Count()
              })
              .OrderByDescending(x => x.Total)
              .Take(5)
              .Join(_context.Libros,
                    p => p.LibroId,
                    l => l.Id,
                    (p, l) => new LibroMasPrestadoDTO
                    {
                        Titulo = l.Titulo,
                        TotalPrestamos = p.Total
                    })
              .ToListAsync();
      }

      // Usuarios con préstamos activos
      public async Task<List<UsuarioConPrestamosDTO>> UsuariosConPrestamosActivosAsync()
      {
          return await _context.Usuarios
              .Where(u => u.Prestamos.Any(p => !p.Devuelto))
              .Select(u => new UsuarioConPrestamosDTO
              {
                  UsuarioNombre = u.Nombre,
                  Email = u.Email,
                  PrestamosActivos = u.Prestamos.Count(p => !p.Devuelto)
              })
              .OrderByDescending(u => u.PrestamosActivos)
              .ToListAsync();
      }

      // Reporte de morosidad
      public async Task<ReporteMorosidadDTO> ObtenerReporteMorosidadAsync()
      {
          var totalPrestamos = await _context.Prestamos.CountAsync();
          var prestamosVencidos = await _context.Prestamos
              .CountAsync(p => !p.Devuelto && p.FechaMaximaDevolucion < DateTime.UtcNow);

          return new ReporteMorosidadDTO
          {
              TotalPrestamos = totalPrestamos,
              PrestamosVencidos = prestamosVencidos,
              PorcentajeMorosidad = totalPrestamos > 0
                  ? (double)prestamosVencidos / totalPrestamos * 100
                  : 0
          };
      }
  }

  // DTOs para consultas
  public class LibroMasPrestadoDTO

  }

  public class UsuarioConPrestamosDTO

  public class ReportesService
  {
      private readonly BibliotecaDbContext _context;

      public ReportesService(BibliotecaDbContext context)
      {
          _context = context;
      }

      // Ejecutar Stored Procedure
      public async Task<List<Libro>> LibrosPorCategoriaAsync(Guid categoriaId)
      {
          return await _context.Libros
              .FromSqlRaw("EXEC ObtenerLibrosPorCategoria @p0", categoriaId)
              .ToListAsync();
      }

      // Consulta SQL directa
      public async Task<int> ContarPrestamosActivosAsync()
      {
          var count = await _context.Database
              .ExecuteSqlRawAsync("SELECT COUNT(*) FROM Prestamos WHERE Devuelto = 0");
          return count;
      }

      // Insert con SQL
      public async Task InsertarLogAsync(string accion, string detalle)
      {
          await _context.Database
              .ExecuteSqlInterpolatedAsync(
                  $"INSERT INTO Logs (Id, Accion, Detalle, Fecha) VALUES (NEWID(), {accion}, {detalle}, GETUTCDATE())");
      }
  }

  public async Task<bool> RealizarPrestamoAsync(Guid libroId, Guid usuarioId)
  {
      using var transaction = await _context.Database.BeginTransactionAsync();

      try
      {
          var libro = await _context.Libros.FindAsync(libroId);
          if (libro == null || !libro.Disponible)
              return false;

          var usuario = await _context.Usuarios.FindAsync(usuarioId);
          if (usuario == null)
              return false;

          // Crear préstamo
          var prestamo = new Prestamo
          {
              Id = Guid.NewGuid(),
              LibroId = libroId,
              UsuarioId = usuarioId,
              FechaPrestamo = DateTime.UtcNow,
              FechaMaximaDevolucion = DateTime.UtcNow.AddDays(14),
              Devuelto = false
          };

          _context.Prestamos.Add(prestamo);

          // Marcar libro como no disponible
          libro.Disponible = false;

          await _context.SaveChangesAsync();

          // Confirmar transacción
          await transaction.CommitAsync();
          return true;
      }
      catch
      {
          // Revertir cambios
          await transaction.RollbackAsync();
          throw;
      }
  }

  Ejemplo 10: Shadow properties y filtros globales
  ──────────────────────────────────────────────────────────────────────────────

  // Shadow properties son propiedades que existen en la BD pero no en la entidad.

  public class BibliotecaDbContext : DbContext
  {
      // ...

      protected override void OnModelCreating(ModelBuilder modelBuilder)
      {
          // Shadow property "FechaEliminacion" en todas las entidades
          modelBuilder.Entity<Libro>().Property<DateTime?>("FechaEliminacion");

          // Filtro global: solo entidades no eliminadas
          modelBuilder.Entity<Libro>().HasQueryFilter(l => EF.Property<DateTime?>(l, "FechaEliminacion") == null);
      }

      public override async Task<int> SaveChangesAsync(CancellationToken ct = default)
      {
          // Soft delete: en lugar de eliminar, marcar fecha
          foreach (var entry in ChangeTracker.Entries())
          {
              if (entry.State == EntityState.Deleted)
              {
                  entry.State = EntityState.Modified;
                  entry.Property("FechaEliminacion").CurrentValue = DateTime.UtcNow;
              }
          }

          return await base.SaveChangesAsync(ct);
      }
  }

  [PREGUNTAS FRECUENTES — FAQ]
================================================================================

  P: ¿EF Core es más lento que ADO.NET?
  R: EF Core tiene overhead, pero para el 95% de los casos su productividad
     supera la diferencia de rendimiento. Para consultas críticas puedes
     usar SQL raw con FromSqlRaw.

  P: ¿Cómo manejar N+1 queries?
  R: Usa .Include() para Eager Loading. Nunca accedas a propiedades de
     navegación dentro de un bucle sin haberlas cargado antes.

  P: ¿Debo usar Repository con EF Core?
  R: Es un debate abierto. EF Core ya es un patrón Repository y Unit of Work.
     Si tu capa de negocio necesita pruebas unitarias aisladas, puede tener
     sentido. Para proyectos pequeños, usa EF Core directamente.

  P: ¿Qué es AsNoTracking()?
  R: Desactiva el Change Tracker para consultas de solo lectura. Mejora el
     rendimiento significativamente.

  P: ¿Cómo migrar de una BD existente?
  R: Usa Scaffold-DbContext o dotnet ef dbcontext scaffold para generar
     el modelo desde la BD existente.
