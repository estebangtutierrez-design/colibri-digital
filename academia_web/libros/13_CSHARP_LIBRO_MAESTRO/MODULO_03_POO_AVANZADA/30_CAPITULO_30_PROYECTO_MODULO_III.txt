```csharp
// Models/Dinero.cs (Value Object)

namespace Inventario.Models;

public readonly record struct Dinero(decimal Cantidad, string Moneda)
{
    public static readonly Dinero Zero = new(0, "USD");

    public Dinero Sumar(Dinero otro)
    {
        ValidarMismaMoneda(otro);
        return this with { Cantidad = Cantidad + otro.Cantidad };
    }

    public Dinero Restar(Dinero otro)
    {
        ValidarMismaMoneda(otro);
        if (otro.Cantidad > Cantidad)
            throw new InvalidOperationException("Saldo insuficiente");
        return this with { Cantidad = Cantidad - otro.Cantidad };
    }

    public Dinero Multiplicar(decimal factor) =>
        this with { Cantidad = Cantidad * factor };

    public Dinero Dividir(decimal divisor)
    {
        if (divisor == 0) throw new DivideByZeroException();
        return this with { Cantidad = Cantidad / divisor };
    }

    private void ValidarMismaMoneda(Dinero otro)
    {
        if (!Moneda.Equals(otro.Moneda, StringComparison.OrdinalIgnoreCase))
            throw new InvalidOperationException(
                $"Monedas incompatibles: {Moneda} y {otro.Moneda}");
    }

    // Operadores
    public static Dinero operator +(Dinero a, Dinero b) => a.Sumar(b);
    public static Dinero operator -(Dinero a, Dinero b) => a.Restar(b);
    public static Dinero operator *(Dinero a, decimal f) => a.Multiplicar(f);
    public static Dinero operator /(Dinero a, decimal d) => a.Dividir(d);
    public static bool operator >(Dinero a, Dinero b) => a.Comparar(b) > 0;
    public static bool operator <(Dinero a, Dinero b) => a.Comparar(b) < 0;
    public static bool operator >=(Dinero a, Dinero b) => a.Comparar(b) >= 0;
    public static bool operator <=(Dinero a, Dinero b) => a.Comparar(b) <= 0;

    private int Comparar(Dinero otro)
    {
        ValidarMismaMoneda(otro);
        return Cantidad.CompareTo(otro.Cantidad);
    }

    public override string ToString() => $"{Moneda} {Cantidad:N2}";
}
```

```csharp
    Dinero PrecioVenta,
    AtributosProducto Atributos,
    EstadoProducto Estado,
    DateTime FechaCreacion
)

    public bool EsFragil => Atributos.HasFlag(AtributosProducto.Fragil);
    public bool EstaEnOferta => Atributos.HasFlag(AtributosProducto.EnOferta);
    public decimal MargenGanancia =>
        PrecioCompra.Cantidad > 0
        ? ((PrecioVenta.Cantidad - PrecioCompra.Cantidad) / PrecioCompra.Cantidad) * 100
        : 0;

    public Dinero CalcularPrecioConDescuento(decimal porcentaje) =>
        PrecioVenta.Multiplicar(1 - porcentaje / 100);
```

```csharp
// ============================================================
// Services/InMemoryRepository.cs
// ============================================================
namespace Inventario.Services;

public class InMemoryRepository<T, TId> : IRepository<T, TId>
    where T : class
    where TId : notnull
{
    protected readonly Dictionary<TId, T> _items = new();

    public virtual T? GetById(TId id) =>
        _items.TryGetValue(id, out var item) ? item : null;

    public virtual IEnumerable<T> GetAll() => _items.Values;

    public virtual void Add(T entity)
    {
        var id = ObtenerId(entity);
        if (_items.ContainsKey(id))
            throw new InvalidOperationException($"Ya existe una entidad con ID {id}");
        _items[id] = entity;
    }

    public virtual void Update(T entity)
    {
        var id = ObtenerId(entity);
        if (!_items.ContainsKey(id))
            throw new KeyNotFoundException($"No se encontró entidad con ID {id}");
        _items[id] = entity;
    }

    public virtual void Delete(TId id) => _items.Remove(id);

    public bool Exists(TId id) => _items.ContainsKey(id);

    public int Count => _items.Count;

    protected virtual TId ObtenerId(T entity) =>
        (TId)typeof(T).GetProperty("Id")?.GetValue(entity)
        ?? throw new InvalidOperationException("La entidad no tiene propiedad Id");
}
```

```csharp
namespace Inventario.Services;

using Inventario.Models;

public class ProductoRepository : InMemoryRepository<Producto, ProductoId>
{
    private int _siguienteId = 1;

    public Producto CrearProducto(
        string nombre,
        string? descripcion,
        CategoriaProducto categoria,
        Dinero precioCompra,
        Dinero precioVenta,
        AtributosProducto atributos = AtributosProducto.Ninguno)
    {
        var id = new ProductoId(_siguienteId++);
        var producto = new Producto(
            id, nombre, descripcion, categoria,
            precioCompra, precioVenta, atributos,
            EstadoProducto.Activo, DateTime.UtcNow
        );
        Add(producto);
        return producto;
    }

    public IEnumerable<Producto> GetByCategoria(CategoriaProducto categoria) =>
        _items.Values.Where(p => p.Categoria == categoria);

    public IEnumerable<Producto> GetByEstado(EstadoProducto estado) =>
        _items.Values.Where(p => p.Estado == estado);

    public IEnumerable<Producto> Search(string query) =>
        _items.Values.Where(p =>
            p.Nombre.Contains(query, StringComparison.OrdinalIgnoreCase) ||
            (p.Descripcion?.Contains(query, StringComparison.OrdinalIgnoreCase) ?? false));

    public IEnumerable<Producto> GetEnOferta() =>
        _items.Values.Where(p => p.EstaEnOferta);
}
```

```csharp
// ============================================================
// Services/InventarioService.cs
// ============================================================
namespace Inventario.Services;

using Inventario.Models;

public class InventarioService
{
    private readonly Dictionary<int, List<StockItem>> _stockPorAlmacen = new();
    private readonly ProductoRepository _productos = new();
    private readonly List<MovimientoStock> _movimientos = new();
    private int _siguienteMovId = 1;
    private int _siguienteAlmacenId = 1;

    public ProductoRepository Productos => _productos;

    // Indexador por almacén
    public List<StockItem> this[int almacenId]
    {
        get => _stockPorAlmacen.TryGetValue(almacenId, out var items)
            ? items
            : new List<StockItem>();
    }

    // Indexador por producto (stock en todos los almacenes)
    public List<(int AlmacenId, StockItem Item)> this[ProductoId productoId]
    {
        get
        {
            var result = new List<(int, StockItem)>();
            foreach (var (almacenId, items) in _stockPorAlmacen)
            {
                var item = items.FirstOrDefault(i => i.ProductoId == productoId);
                if (item != default)
                    result.Add((almacenId, item));
            }
            return result;
        }
    }

    // Indexador combinado (almacén y producto)
    public StockItem? this[int almacenId, ProductoId productoId]
    {
        get
        {
            if (!_stockPorAlmacen.TryGetValue(almacenId, out var items))
                return null;
            return items.FirstOrDefault(i => i.ProductoId == productoId);
        }
    }

    public Almacen CrearAlmacen(string nombre, string direccion)
    {
        var id = _siguienteAlmacenId++;
        var almacen = new Almacen(id, nombre, direccion, true);
        _stockPorAlmacen[id] = new List<StockItem>();
        return almacen;
    }

    public StockItem RegistrarEntrada(
        ProductoId productoId, int almacenId,
        int cantidad, Dinero costoUnitario,
        TipoMovimiento tipo, string? referencia = null,
        string? observaciones = null, string usuario = "sistema")
    {
        if (!_productos.Exists(productoId))
            throw new KeyNotFoundException($"Producto {productoId} no encontrado");

        var ubicacion = new Ubicacion("GEN", "A", "1", "1");
        var item = new StockItem(productoId, cantidad, ubicacion, DateTime.UtcNow);

        if (!_stockPorAlmacen.ContainsKey(almacenId))
            _stockPorAlmacen[almacenId] = new List<StockItem>();

        var existente = _stockPorAlmacen[almacenId]
            .FirstOrDefault(i => i.ProductoId == productoId);

        if (existente != default)
        {
            var idx = _stockPorAlmacen[almacenId].IndexOf(existente);
            _stockPorAlmacen[almacenId][idx] = existente.ActualizarCantidad(
                existente.Cantidad + cantidad);
        }
        else
        {
            _stockPorAlmacen[almacenId].Add(item);
        }

        var movimiento = new MovimientoStock(
            _siguienteMovId++, productoId, almacenId, tipo,
            cantidad, costoUnitario, referencia, observaciones,
            DateTime.UtcNow, usuario
        );
        _movimientos.Add(movimiento);

        return _stockPorAlmacen[almacenId]
            .First(i => i.ProductoId == productoId);
    }

    public StockItem RegistrarSalida(
        ProductoId productoId, int almacenId,
        int cantidad, Dinero costoUnitario,
        TipoMovimiento tipo, string? referencia = null,
        string? observaciones = null, string usuario = "sistema")
    {
        if (!_stockPorAlmacen.TryGetValue(almacenId, out var items))
            throw new KeyNotFoundException($"Almacén {almacenId} no encontrado");

        var existente = items.FirstOrDefault(i => i.ProductoId == productoId);
        if (existente == default)
            throw new InvalidOperationException($"Producto {productoId} sin stock en almacén {almacenId}");

        if (existente.Cantidad < cantidad)
            throw new InvalidOperationException(
                $"Stock insuficiente. Disponible: {existente.Cantidad}, solicitado: {cantidad}");

        var idx = items.IndexOf(existente);
        _stockPorAlmacen[almacenId][idx] = existente.ActualizarCantidad(
            existente.Cantidad - cantidad);

        var movimiento = new MovimientoStock(
            _siguienteMovId++, productoId, almacenId, tipo,
            cantidad, costoUnitario, referencia, observaciones,
            DateTime.UtcNow, usuario
        );
        _movimientos.Add(movimiento);

        return _stockPorAlmacen[almacenId][idx];
    }

    public IEnumerable<Producto> GetStockBajo(int limite = 5)
    {
        var productosConStock = _stockPorAlmacen.Values
            .SelectMany(items => items)
            .GroupBy(i => i.ProductoId)
            .Select(g => new { ProductoId = g.Key, Total = g.Sum(i => i.Cantidad) })
            .Where(x => x.Total < limite && x.Total > 0);

        foreach (var x in productosConStock)
        {
            var producto = _productos.GetById(x.ProductoId);
            if (producto != null)
                yield return producto;
        }
    }

    public IEnumerable<Producto> GetSinStock()
    {
        var productosConStock = _stockPorAlmacen.Values
            .SelectMany(items => items)
            .GroupBy(i => i.ProductoId)
            .Select(g => g.Key)
            .ToHashSet();

        return _productos.GetAll()
            .Where(p => !productosConStock.Contains(p.Id) && p.Estado == EstadoProducto.Activo);
    }

    public Dinero ValorTotalInventario(int? almacenId = null)
    {
        var items = almacenId.HasValue
            ? _stockPorAlmacen.GetValueOrDefault(almacenId.Value, new())
            : _stockPorAlmacen.Values.SelectMany(i => i);

        Dinero total = Dinero.Zero;
        foreach (var item in items)
        {
            var producto = _productos.GetById(item.ProductoId);
            if (producto != null)
            {
                total += producto.PrecioCompra.Multiplicar(item.Cantidad);
            }
        }
        return total;
    }

    public IEnumerable<MovimientoStock> GetMovimientos(
        ProductoId? productoId = null,
        int? almacenId = null,
        DateTime? desde = null,
        DateTime? hasta = null,
        TipoMovimiento? tipo = null)
    {
        IEnumerable<MovimientoStock> query = _movimientos;

        if (productoId.HasValue)
            query = query.Where(m => m.ProductoId == productoId.Value);
        if (almacenId.HasValue)
            query = query.Where(m => m.AlmacenId == almacenId.Value);
        if (desde.HasValue)
            query = query.Where(m => m.Fecha >= desde.Value);
        if (hasta.HasValue)
            query = query.Where(m => m.Fecha <= hasta.Value);
        if (tipo.HasValue)
            query = query.Where(m => m.Tipo == tipo.Value);

        return query.OrderByDescending(m => m.Fecha);
    }

    public void MostrarResumenInventario()
    {
        Console.WriteLine($"\n=== RESUMEN DE INVENTARIO ===");
        Console.WriteLine($"Total productos: {_productos.Count}");
        Console.WriteLine($"Total almacenes: {_stockPorAlmacen.Count}");
        Console.WriteLine($"Total movimientos: {_movimientos.Count}");
        Console.WriteLine($"Valor total: {ValorTotalInventario()}");

        Console.WriteLine($"\nProductos con stock bajo:");
        foreach (var p in GetStockBajo(10))
        {
            var stock = this[p.Id];
            int total = stock.Sum(s => s.Item.Cantidad);
            Console.WriteLine($"  {p.Nombre}: {total} unidades");
        }

        Console.WriteLine($"\nProductos sin stock:");
        foreach (var p in GetSinStock())
            Console.WriteLine($"  {p.Nombre}");
    }
}
```

```csharp
namespace Inventario.Services;

using Inventario.Models;

public class ReporteService
{
    private readonly InventarioService _inventario;

    public ReporteService(InventarioService inventario)
    {
        _inventario = inventario;
    }

    public void ReporteProductosPorCategoria()
    {
        Console.WriteLine("\n=== PRODUCTOS POR CATEGORÍA ===");
        var grupos = _inventario.Productos.GetAll()
            .GroupBy(p => p.Categoria)
            .OrderBy(g => g.Key);

        foreach (var grupo in grupos)
        {
            Console.WriteLine($"\n{grupo.Key}: {grupo.Count()} productos");
            Console.WriteLine(new string('-', 40));
            foreach (var p in grupo.OrderBy(p => p.Nombre))
            {
                var stock = _inventario[p.Id];
                int total = stock.Sum(s => s.Item.Cantidad);
                Console.WriteLine($"  {p.Nombre,-25} Stock: {total,5}  Precio: {p.PrecioVenta}");
            }
        }
    }

    public void ReporteMovimientosRecientes(int dias = 7)
    {
        var desde = DateTime.UtcNow.AddDays(-dias);
        Console.WriteLine($"\n=== MOVIMIENTOS ÚLTIMOS {dias} DÍAS ===");

        var movs = _inventario.GetMovimientos(desde: desde).Take(20);

        foreach (var m in movs)
        {
            var tipoIcono = m.EsEntrada ? "⬆" : "⬇";
            var prod = _inventario.Productos.GetById(m.ProductoId);
            string nombre = prod?.Nombre ?? "(eliminado)";
            Console.WriteLine($"{tipoIcono} [{m.Fecha:g}] {m.Tipo,-25} {nombre,-20} x{m.Cantidad,5} {m.CostoTotal}");
        }
    }

                // Almacén no existe

    public void ReporteProveedores()
```

================================================================================
  FAQ — PREGUNTAS FRECUENTES DEL PROYECTO
================================================================================

P: ¿Cómo ejecuto el proyecto?
R: Necesitas .NET 8 SDK. Crea un proyecto console: dotnet new console -n Inventario
   y copia los archivos en las carpetas correspondientes.

P: ¿Por qué usamos record en lugar de class para los modelos?
R: Porque los records proporcionan igualdad estructural, inmutabilidad,
   y sintaxis concisa. Son ideales para modelos de datos y value objects.

P: ¿Puedo usar este proyecto en producción?
R: Es una base excelente, pero necesitarás añadir persistencia real (BD),
   autenticación, logging, y más validaciones.

P: ¿Por qué el indexador puede devolver null?
R: Usamos nullable para indicar que un producto puede no tener stock
   en un almacén específico. Es un uso correcto de null.

P: ¿Cómo añado una nueva categoría de producto?
R: Solo añade un nuevo valor al enum CategoriaProducto. Los switch
   expressions con exhaustiveness checking te recordarán actualizarlo.

P: ¿Los métodos de extensión afectan el rendimiento?
R: No significativamente. Son llamadas estáticas compiladas.

P: ¿Qué debo estudiar después de este proyecto?
R: El Módulo IV cubre LINQ avanzado, expresiones lambda, y programación
   funcional en C#. También recomendamos estudiar ASP.NET Core para
   convertir este proyecto en una API web.
