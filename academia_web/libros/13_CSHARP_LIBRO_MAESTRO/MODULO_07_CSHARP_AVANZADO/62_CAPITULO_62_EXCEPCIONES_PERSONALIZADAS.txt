--- 1. ¿POR QUÉ EXCEPCIONES PERSONALIZADAS? ---

Las excepciones built-in de .NET cubren situaciones generales, pero en
aplicaciones reales necesitamos excepciones que reflejen el dominio del
negocio. Beneficios:

- Semántica clara: DomainException, PaymentFailedException
- Datos contextuales: ID de transacción, nombre de usuario
- Manejo selectivo: catch (SaldoInsuficienteException)
- Serialización para跨 procesos (remoting, WCF)

--- 2. ESTRUCTURA BÁSICA ---

Una excepción personalizada debe:
1. Heredar de Exception (o una subclase apropiada).
2. Tener al menos los tres constructores estándar.
3. Opcionalmente, agregar propiedades con datos adicionales.
4. Marcar con [Serializable] si se necesita serialización.

```csharp
using System;
using System.Runtime.Serialization;

[Serializable]
public class SaldoInsuficienteException : Exception
{
    // Constructores estándar
    public SaldoInsuficienteException() 
        : base("Saldo insuficiente para realizar la operación.") { }
    
    public SaldoInsuficienteException(string message) 
        : base(message) { }
    
    public SaldoInsuficienteException(string message, Exception inner) 
        : base(message, inner) { }
    
    // Constructor para serialización
    protected SaldoInsuficienteException(
        SerializationInfo info, StreamingContext context) 
        : base(info, context) { }
}
```

--- 5. USO DE EXCEPCIONES PERSONALIZADAS ---

```csharp
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        Console.WriteLine("=== EXCEPCIONES PERSONALIZADAS EN ACCIÓN ===\n");
        
        var banco = new Banco();
        
        // Registrar cuentas
        banco.RegistrarCuenta("12345", 1000m);
        banco.RegistrarCuenta("67890", 500m);
        
        // Probar operaciones
        ProbarOperacion(banco, "12345", 200m, "deposito");
        ProbarOperacion(banco, "12345", 1500m, "retiro");
        ProbarOperacion(banco, "99999", 100m, "retiro");
        ProbarOperacion(banco, "67890", 600m, "retiro");
        ProbarOperacion(banco, "67890", 50m, "retiro");
    }
    
    static void ProbarOperacion(Banco banco, string cuenta, decimal monto, string tipo)
    {
        try
        {
            Console.WriteLine($"\nOperación: {tipo} - Cuenta: {cuenta} - Monto: {monto:C}");
            
            if (tipo == "deposito")
                banco.Depositar(cuenta, monto);
            else if (tipo == "retiro")
                banco.Retirar(cuenta, monto);
            
            var info = banco.ObtenerCuenta(cuenta);
            Console.WriteLine($"  Éxito. Saldo actual: {info.Saldo:C}");
        }
        catch (CuentaNoEncontradaException ex)
        {
            Console.WriteLine($"  ERROR [{ex.CodigoError}]: {ex.Message}");
        }
        catch (SaldoInsuficienteException ex)
        {
            Console.WriteLine($"  ERROR [{ex.CodigoError}]: {ex.Message}");
            Console.WriteLine($"    Saldo: {ex.SaldoActual:C}, Solicitado: {ex.MontoSolicitado:C}");
        }
        catch (LimiteTransaccionException ex)
        {
            Console.WriteLine($"  ERROR [{ex.CodigoError}]: {ex.Message}");
            Console.WriteLine($"    Límite: {ex.Limite:C}, Intentado: {ex.MontoIntentado:C}");
        }
        catch (DominioException ex)
        {
            Console.WriteLine($"  ERROR DE DOMINIO [{ex.CodigoError}]: {ex.Message}");
        }
    }
}

public class CuentaInfo
{
    public string Numero { get; set; }
    public decimal Saldo { get; set; }
}

public class Banco
{
    private readonly Dictionary<string, decimal> _cuentas = new();
    private const decimal LIMITE_RETIRO = 1000m;
    
    public void RegistrarCuenta(string numero, decimal saldoInicial)
    {
        _cuentas[numero] = saldoInicial;
        Console.WriteLine($"Cuenta {numero} registrada con saldo {saldoInicial:C}");
    }
    
    public CuentaInfo ObtenerCuenta(string numero)
    {
        if (!_cuentas.ContainsKey(numero))
            throw new CuentaNoEncontradaException(numero);
        
        return new CuentaInfo 
        { 
            Numero = numero, 
            Saldo = _cuentas[numero] 
        };
    }
    
    public void Depositar(string numeroCuenta, decimal monto)
    {
        if (!_cuentas.ContainsKey(numeroCuenta))
            throw new CuentaNoEncontradaException(numeroCuenta);
        
        if (monto <= 0)
            throw new ArgumentException("El monto debe ser positivo.", nameof(monto));
        
        _cuentas[numeroCuenta] += monto;
    }
    
    public void Retirar(string numeroCuenta, decimal monto)
    {
        if (!_cuentas.ContainsKey(numeroCuenta))
            throw new CuentaNoEncontradaException(numeroCuenta);
        
        if (monto <= 0)
            throw new ArgumentException("El monto debe ser positivo.", nameof(monto));
        
        if (monto > LIMITE_RETIRO)
            throw new LimiteTransaccionException("retiro diario", LIMITE_RETIRO, monto);
        
        if (_cuentas[numeroCuenta] < monto)
            throw new SaldoInsuficienteException(
                numeroCuenta, _cuentas[numeroCuenta], monto);
        
        _cuentas[numeroCuenta] -= monto;
    }
}
```

```csharp
[Serializable]
public class ValidacionException : Exception
{
    public IReadOnlyList<string> Errores { get; }
    public string Campo { get; }
    public string ValorRecibido { get; }
    
    public ValidacionException() 
        : base("Error de validación.") { }
    
    public ValidacionException(string message) : base(message) { }
    
    public ValidacionException(
        string campo, string valor, string mensaje)
        : base(mensaje)
    {
        Campo = campo;
        ValorRecibido = valor;
        Errores = new List<string> { mensaje };
    }
    
    public ValidacionException(
        string campo, string valor, List<string> errores)
        : base($"Error de validación en campo '{campo}': {errores.Count} error(es).")
    {
        Campo = campo;
        ValorRecibido = valor;
        Errores = errores.AsReadOnly();
    }
    
    protected ValidacionException(
        SerializationInfo info, StreamingContext context) 
        : base(info, context)
    {
        Campo = info.GetString(nameof(Campo));
        ValorRecibido = info.GetString(nameof(ValorRecibido));
        Errores = (List<string>)info.GetValue(nameof(Errores), typeof(List<string>));
    }
    
    public override void GetObjectData(
        SerializationInfo info, StreamingContext context)
    {
        base.GetObjectData(info, context);
        info.AddValue(nameof(Campo), Campo);
        info.AddValue(nameof(ValorRecibido), ValorRecibido);
        info.AddValue(nameof(Errores), Errores?.ToList());
    }
}

// Uso
class ValidadorFormulario
{
    public static void ValidarUsuario(string nombre, string email, int edad)
    {
        var errores = new List<string>();
        
        if (string.IsNullOrWhiteSpace(nombre))
            errores.Add("El nombre es obligatorio.");
        else if (nombre.Length < 2)
            errores.Add("El nombre debe tener al menos 2 caracteres.");
        
        if (string.IsNullOrWhiteSpace(email))
            errores.Add("El email es obligatorio.");
        else if (!email.Contains("@"))
            errores.Add("El email debe contener @.");
        
        if (edad < 18)
            errores.Add("Debe ser mayor de 18 años.");
        else if (edad > 120)
            errores.Add("Edad inválida.");
        
        if (errores.Count > 0)
            throw new ValidacionException("usuario", 
                $"nombre={nombre}, email={email}, edad={edad}", errores);
    }
}
```

```csharp
class ManejadorCentralErrores
{
    public void ManejarError(Exception ex)
    {
        if (ex is ErrorAplicacionException appEx)
        {
            Console.WriteLine($"  [{appEx.Severidad}] {appEx.Categoria}: {appEx.Message}");
            if (appEx.Metadatos.Count > 0)
            {
                Console.WriteLine("  Metadatos:");
                foreach (var kvp in appEx.Metadatos)
                    Console.WriteLine($"    {kvp.Key}: {kvp.Value}");
            }
        }
        else
        {
            Console.WriteLine($"  [NO CONTROLADA] {ex.GetType().Name}: {ex.Message}");
        }
        
        if (ex.InnerException != null)
            ManejarError(ex.InnerException);
    }
}
```

```csharp
using System;

// Alternativa: Result<T> en vez de excepciones para flujo esperado
public class Result<T>
{
    public bool IsSuccess { get; }
    public T Value { get; }
    public string ErrorCode { get; }
    public string ErrorMessage { get; }
    
    protected Result(bool isSuccess, T value, string errorCode, string errorMessage)
    {
        IsSuccess = isSuccess;
        Value = value;
        ErrorCode = errorCode;
        ErrorMessage = errorMessage;
    }
    
    public static Result<T> Ok(T value) => 
        new Result<T>(true, value, null, null);
    
    public static Result<T> Fail(string code, string message) => 
        new Result<T>(false, default, code, message);
    
    public void Match(Action<T> onSuccess, Action<string, string> onFailure)
    {
        if (IsSuccess)
            onSuccess(Value);
        else
            onFailure(ErrorCode, ErrorMessage);
    }
}

class ServicioDividir
{
    public Result<decimal> Dividir(decimal a, decimal b)
    {
        if (b == 0)
            return Result<decimal>.Fail("DIV-001", "No se puede dividir por cero.");
        return Result<decimal>.Ok(a / b);
    }
}

class Program
{
    static void Main()
    {
        var servicio = new ServicioDividir();
        
        servicio.Dividir(10, 2).Match(
            onSuccess: valor => Console.WriteLine($"Resultado: {valor}"),
            onFailure: (codigo, msg) => Console.WriteLine($"Error [{codigo}]: {msg}")
        );
        
        servicio.Dividir(10, 0).Match(
            onSuccess: valor => Console.WriteLine($"Resultado: {valor}"),
            onFailure: (codigo, msg) => Console.WriteLine($"Error [{codigo}]: {msg}")
        );
    }
}
```

```csharp
        : base(mensaje)

        Endpoint = endpoint;
        DetallesValidacion = detalles;

        out HttpStatusCode statusCode,

        SerializationInfo info, StreamingContext context) 
        : base(info, context)
    {
        StatusCode = (HttpStatusCode)info.GetInt32(nameof(StatusCode));
        Endpoint = info.GetString(nameof(Endpoint));
        DetallesValidacion = (Dictionary<string, string>)info.GetValue(
            nameof(DetallesValidacion), typeof(Dictionary<string, string>));
    }
    
    public override void GetObjectData(
        SerializationInfo info, StreamingContext context)
    {
        base.GetObjectData(info, context);
        info.AddValue(nameof(StatusCode), (int)StatusCode);
        info.AddValue(nameof(Endpoint), Endpoint);
        info.AddValue(nameof(DetallesValidacion), DetallesValidacion);
```

```csharp
using System;
using System.Transactions;

class ProcesadorTransaccional
{
    public void ProcesarPago(string cuentaOrigen, string cuentaDestino, decimal monto)
    {
        using (var scope = new TransactionScope())
        {
            try
            {
                Debitar(cuentaOrigen, monto);
                Acreditar(cuentaDestino, monto);
                
                scope.Complete();
                Console.WriteLine("Transacción completada exitosamente.");
            }
            catch (SaldoInsuficienteException)
            {
                Console.WriteLine("Transacción revertida por saldo insuficiente.");
                throw;  // Rollback automático
            }
            catch (CuentaNoEncontradaException)
            {
                Console.WriteLine("Transacción revertida por cuenta no encontrada.");
                throw;  // Rollback automático
            }
        }
    }
    
    void Debitar(string cuenta, decimal monto)
    {
        // Simulación
        if (monto > 1000)
            throw new SaldoInsuficienteException(cuenta, 500, monto);
        Console.WriteLine($"  Debitados {monto:C} de {cuenta}");
    }
    
    void Acreditar(string cuenta, decimal monto)
    {
        Console.WriteLine($"  Acreditados {monto:C} a {cuenta}");
    }
}
```

```csharp
using System;

class ExceptionConverter
{
    public static T Convertir<T>(Exception original) where T : Exception
    {
        // Crear nueva excepción del tipo T con la original como InnerException
        var constructor = typeof(T).GetConstructor(
            new[] { typeof(string), typeof(Exception) });
        
        if (constructor != null)
        {
            return (T)constructor.Invoke(
                new object[] { 
                    $"Error convertido desde {original.GetType().Name}: {original.Message}", 
                    original 
                });
        }
        
        throw new InvalidOperationException(
            $"El tipo {typeof(T).Name} no tiene constructor (string, Exception).");
    }
    
    static void Main()
    {
        try
        {
            try
            {
                throw new InvalidOperationException("Error de operación inválida.");
            }
            catch (Exception ex)
            {
                throw ExceptionConverter.Convertir<DataAccessException>(ex);
            }
        }
        catch (DataAccessException ex)
        {
            Console.WriteLine($"Convertida: {ex.Message}");
            Console.WriteLine($"Inner: {ex.InnerException?.Message}");
        }
    }
}
```

```csharp
using System;
using System.IO;
using System.Runtime.Serialization.Formatters.Binary;

class PersistenciaExcepciones
{
    public static void GuardarExcepcion(Exception ex, string ruta)
    {
        try
        {
            var formatter = new BinaryFormatter();
            using (var stream = File.Create(ruta))
            {
                formatter.Serialize(stream, ex);
            }
            Console.WriteLine($"Excepción guardada en {ruta}");
        }
        catch (Exception serialEx)
        {
            Console.WriteLine($"Error al serializar: {serialEx.Message}");
        }
    }
    
    public static Exception CargarExcepcion(string ruta)
    {
        try
        {
            var formatter = new BinaryFormatter();
            using (var stream = File.OpenRead(ruta))
            {
                return (Exception)formatter.Deserialize(stream);
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Error al cargar: {ex.Message}");
            return null;
        }
    }
    
    static void Main()
    {
        var excepcionOriginal = new SaldoInsuficienteException(
            "12345", 100m, 500m);
        
        GuardarExcepcion(excepcionOriginal, "excepcion.bin");
        
        var excepcionRestaurada = CargarExcepcion("excepcion.bin");
        
        if (excepcionRestaurada is SaldoInsuficienteException siEx)
        {
            Console.WriteLine($"\nExcepción restaurada:");
            Console.WriteLine($"  Cuenta: {siEx.NumeroCuenta}");
            Console.WriteLine($"  Saldo: {siEx.SaldoActual:C}");
            Console.WriteLine($"  Monto: {siEx.MontoSolicitado:C}");
            Console.WriteLine($"  Mensaje: {siEx.Message}");
        }
    }
}
```

================================================================================
  PREGUNTAS FRECUENTES (FAQ)
================================================================================

P: ¿Debo heredar de Exception o de ApplicationException?
R: De Exception. Microsoft ya no recomienda ApplicationException. Herede
directamente de Exception para crear una raíz de su jerarquía.

P: ¿Es obligatorio el constructor de serialización?
R: Si su aplicación usa remoting, WCF, procesos跨, o guarda excepciones
en archivos, es obligatorio. En aplicaciones simples puede omitirse, pero
es buena práctica incluirlo siempre.

P: ¿Cuántas excepciones personalizadas debo crear?
R: Las justas. Cree una por cada concepto de error significativo en su
dominio. No cree una excepción por cada método.

P: ¿Puedo tener propiedades mutables en mi excepción?
R: Técnicamente sí, pero no es recomendable. Las excepciones deben ser
inmutables (propiedades get-only).

P: ¿Debo documentar XML mis excepciones personalizadas?
R: Sí, es muy recomendable. Incluya cuándo se lanza, qué propiedades
contiene, y cómo debe manejarse.
