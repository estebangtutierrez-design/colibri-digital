--- 1. SINGLETON ---

Singleton garantiza que una clase tenga solo una instancia y proporciona
un punto de acceso global a ella.

```csharp
using System;

// Version 1: Thread-safe con Lazy<T> (recomendada)
public sealed class ConfiguracionGlobal
{
    private static readonly Lazy<ConfiguracionGlobal> _instancia =
        new Lazy<ConfiguracionGlobal>(() => new ConfiguracionGlobal());

    public static ConfiguracionGlobal Instancia => _instancia.Value;

    private ConfiguracionGlobal()
    {
        Console.WriteLine("  ConfiguracionGlobal inicializada.");
        Inicializar();
    }

    public string Servidor { get; private set; }
    public int Puerto { get; private set; }
    public string CadenaConexion { get; private set; }

    private void Inicializar()
    {
        Servidor = "localhost";
        Puerto = 5432;
        CadenaConexion = $"Server={Servidor};Port={Puerto};Database=master";
    }
}

// Version 2: Thread-safe con lock (clasica)
public sealed class Logger
{
    private static Logger _instancia;
    private static readonly object _lock = new object();

    private readonly string _logFile;

    private Logger()
    {
        _logFile = $"app_{DateTime.Now:yyyyMMdd}.log";
    }

    public static Logger Instancia
    {
        get
        {
            if (_instancia == null) // Double-check locking
            {
                lock (_lock)
                {
                    if (_instancia == null)
                        _instancia = new Logger();
                }
            }
            return _instancia;
        }
    }

    public void Log(string mensaje)
    {
        string linea = $"[{DateTime.Now:HH:mm:ss}] {mensaje}";
        Console.WriteLine($"  [LOG] {linea}");
        // En produccion: File.AppendAllText(_logFile, linea + Environment.NewLine);
    }
}

// Version 3: Inicializacion estatica (simple, thread-safe)
public sealed class ContadorVisitas
{
    public static readonly ContadorVisitas Instancia = new ContadorVisitas();

    private int _visitas;

    static ContadorVisitas() { } // Asegura inicializacion perezosa

    private ContadorVisitas() { }

    public int Incrementar() => ++_visitas;
    public int Total => _visitas;
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== SINGLETON ===\n");

        ConfiguracionGlobal c1 = ConfiguracionGlobal.Instancia;
        ConfiguracionGlobal c2 = ConfiguracionGlobal.Instancia;
        Console.WriteLine($"Misma instancia: {ReferenceEquals(c1, c2)}");
        Console.WriteLine($"Servidor: {c1.Servidor}");

        Logger l1 = Logger.Instancia;
        Logger l2 = Logger.Instancia;
        Console.WriteLine($"\nMismo Logger: {ReferenceEquals(l1, l2)}");
        l1.Log("Aplicacion iniciada");
        l2.Log("Usuario autenticado");

        Console.WriteLine($"\nVisitas: {ContadorVisitas.Instancia.Incrementar()}");
        Console.WriteLine($"Visitas: {ContadorVisitas.Instancia.Incrementar()}");
        Console.WriteLine($"Total: {ContadorVisitas.Instancia.Total}");
    }
}
```

--- 2. FACTORY METHOD ---

Factory Method define una interfaz para crear objetos, pero permite a
las subclases decidir que clase instanciar.

```csharp
using System;

// Producto abstracto
public abstract class Notificador
{
    public abstract void Enviar(string mensaje);
}

// Productos concretos
public class EmailNotificador : Notificador
{
    private readonly string _servidor;

    public EmailNotificador(string servidor)
    {
        _servidor = servidor;
    }

    public override void Enviar(string mensaje)
    {
        Console.WriteLine($"  [EMAIL] ({_servidor}): {mensaje}");
    }
}

public class SMSNotificador : Notificador
{
    private readonly string _numeroOrigen;

    public SMSNotificador(string numeroOrigen)
    {
        _numeroOrigen = numeroOrigen;
    }

    public override void Enviar(string mensaje)
    {
        Console.WriteLine($"  [SMS] ({_numeroOrigen}): {mensaje}");
    }
}

public class PushNotificador : Notificador
{
    public override void Enviar(string mensaje)
    {
        Console.WriteLine($"  [PUSH]: {mensaje}");
    }
}

// Creator abstracto
public abstract class NotificadorFactory
{
    public abstract Notificador CrearNotificador();

    public void EnviarMensaje(string mensaje)
    {
        var notificador = CrearNotificador();
        notificador.Enviar(mensaje);
    }
}

// Creatores concretos
public class EmailFactory : NotificadorFactory
{
    private readonly string _servidor;

    public EmailFactory(string servidor)
    {
        _servidor = servidor;
    }

    public override Notificador CrearNotificador()
    {
        return new EmailNotificador(_servidor);
    }
}

public class SMSFactory : NotificadorFactory
{
    private readonly string _numero;

    public SMSFactory(string numero)
    {
        _numero = numero;
    }

    public override Notificador CrearNotificador()
    {
        return new SMSNotificador(_numero);
    }
}

public class PushFactory : NotificadorFactory
{
    public override Notificador CrearNotificador()
    {
        return new PushNotificador();
    }
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== FACTORY METHOD ===\n");

        var factories = new NotificadorFactory[]
        {
            new EmailFactory("smtp.empresa.com"),
            new SMSFactory("+123456789"),
            new PushFactory()
        };

        foreach (var factory in factories)
        {
            Console.WriteLine($"\nUsando {factory.GetType().Name}:");
            factory.EnviarMensaje("Notificacion importante!");
        }
    }
}
```

--- 3. ABSTRACT FACTORY ---

Abstract Factory proporciona una interfaz para crear familias de objetos
relacionados sin especificar sus clases concretas.

```csharp
using System;

// Productos abstractos
public abstract class Boton
{
    public abstract void Dibujar();
}

public abstract class Ventana
{
    public abstract void Mostrar();
}

// Productos concretos: Windows
public class WindowsBoton : Boton
{
    public override void Dibujar()
        => Console.WriteLine("  [Windows] Boton dibujado con estilo WinUI");
}

public class WindowsVentana : Ventana
{
    public override void Mostrar()
        => Console.WriteLine("  [Windows] Ventana mostrada con borde redondeado");
}

// Productos concretos: macOS
public class MacBoton : Boton
{
    public override void Dibujar()
        => Console.WriteLine("  [macOS] Boton dibujado con estilo Aqua");
}

public class MacVentana : Ventana
{
    public override void Mostrar()
        => Console.WriteLine("  [macOS] Ventana mostrada con sombra paralaje");
}

// Productos concretos: Linux
public class LinuxBoton : Boton
{
    public override void Dibujar()
        => Console.WriteLine("  [Linux] Boton dibujado con estilo GTK");
}

public class LinuxVentana : Ventana
{
    public override void Mostrar()
        => Console.WriteLine("  [Linux] Ventana mostrada con tema Adwaita");
}

// Abstract Factory
public interface IUIFactory
{
    Boton CrearBoton();
    Ventana CrearVentana();
}

// Concrete Factories
public class WindowsFactory : IUIFactory
{
    public Boton CrearBoton() => new WindowsBoton();
    public Ventana CrearVentana() => new WindowsVentana();
}

public class MacFactory : IUIFactory
{
    public Boton CrearBoton() => new MacBoton();
    public Ventana CrearVentana() => new MacVentana();
}

public class LinuxFactory : IUIFactory
{
    public Boton CrearBoton() => new LinuxBoton();
    public Ventana CrearVentana() => new LinuxVentana();
}

public enum SOPlatform { Windows, Mac, Linux }

public static class UIFactoryProvider
{
    public static IUIFactory CrearFactory(SOPlatform platform)
    {
        return platform switch
        {
            SOPlatform.Windows => new WindowsFactory(),
            SOPlatform.Mac => new MacFactory(),
            SOPlatform.Linux => new LinuxFactory(),
            _ => throw new ArgumentException($"Plataforma {platform} no soportada")
        };
    }
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== ABSTRACT FACTORY ===\n");

        var platforms = new[] { SOPlatform.Windows, SOPlatform.Mac, SOPlatform.Linux };

        foreach (var platform in platforms)
        {
            Console.WriteLine($"\nPlataforma: {platform}");
            var factory = UIFactoryProvider.CrearFactory(platform);

            var boton = factory.CrearBoton();
            var ventana = factory.CrearVentana();

            boton.Dibujar();
            ventana.Mostrar();
        }
    }
}
```

--- 4. DEPENDENCY INJECTION (MANUAL) ---

Dependency Injection invierte el control: en lugar de que las clases
creen sus dependencias, estas se inyectan desde el exterior.

```csharp
using System;
using System.Collections.Generic;

// Interfaces
public interface IRepositorioUsuarios
{
    Usuario ObtenerPorId(int id);
    void Guardar(Usuario usuario);
}

public interface IServicioEmail
{
    void Enviar(string destinatario, string asunto, string cuerpo);
}

public interface ILogger
{
    void Info(string mensaje);
    void Error(string mensaje, Exception ex = null);
}

// Modelo
public class Usuario
{
    public int Id { get; set; }
    public string Nombre { get; set; }
    public string Email { get; set; }
}

// Implementaciones concretas
public class RepositorioUsuariosSQL : IRepositorioUsuarios
{
    private readonly string _connectionString;
    private readonly ILogger _logger;

    public RepositorioUsuariosSQL(string connectionString, ILogger logger)
    {
        _connectionString = connectionString;
        _logger = logger;
    }

    public Usuario ObtenerPorId(int id)
    {
        _logger.Info($"Obteniendo usuario #{id}");
        return new Usuario { Id = id, Nombre = "Juan", Email = "juan@email.com" };
    }

    public void Guardar(Usuario usuario)
    {
        _logger.Info($"Guardando usuario {usuario.Nombre}");
    }
}

public class ServicioEmailSMTP : IServicioEmail
{
    private readonly string _smtpServer;
    private readonly ILogger _logger;

    public ServicioEmailSMTP(string smtpServer, ILogger logger)
    {
        _smtpServer = smtpServer;
        _logger = logger;
    }

    public void Enviar(string destinatario, string asunto, string cuerpo)
    {
        _logger.Info($"Enviando email a {destinatario}: {asunto}");
        Console.WriteLine($"  [EMAIL] Para: {destinatario}, Asunto: {asunto}");
    }
}

public class ConsoleLogger : ILogger
{
    public void Info(string mensaje)
        => Console.WriteLine($"  [INFO] {mensaje}");

    public void Error(string mensaje, Exception ex = null)
    {
        Console.WriteLine($"  [ERROR] {mensaje}");
        if (ex != null)
            Console.WriteLine($"    {ex.Message}");
    }
}

// Clase que usa DI
public class RegistroUsuarioService
{
    private readonly IRepositorioUsuarios _repositorio;
    private readonly IServicioEmail _email;
    private readonly ILogger _logger;

    // Todas las dependencias se inyectan por constructor
    public RegistroUsuarioService(
        IRepositorioUsuarios repositorio,
        IServicioEmail email,
        ILogger logger)
    {
        _repositorio = repositorio;
        _email = email;
        _logger = logger;
    }

    public void Registrar(string nombre, string email)
    {
        _logger.Info($"Iniciando registro de {nombre}");

        var usuario = new Usuario
        {
            Nombre = nombre,
            Email = email
        };

        _repositorio.Guardar(usuario);

        _email.Enviar(email, "Bienvenido!",
            $"Hola {nombre}, gracias por registrarte.");

        _logger.Info("Registro completado exitosamente.");
    }
}

// Contenedor DI simple
public class SimpleContainer
{
    private readonly Dictionary<Type, Func<object>> _registrations = new();

    public void Register<TInterface, TImplementation>()
        where TImplementation : TInterface
    {
        _registrations[typeof(TInterface)] = () => Resolve(typeof(TImplementation));
    }

    public void RegisterSingleton<TInterface, TImplementation>()
        where TImplementation : TInterface
    {
        TImplementation instancia = default;
        _registrations[typeof(TInterface)] = () =>
        {
            if (instancia == null)
                instancia = (TImplementation)Resolve(typeof(TImplementation));
            return instancia;
        };
    }

    public void RegisterInstance<TInterface>(TInterface instance)
    {
        _registrations[typeof(TInterface)] = () => instance;
    }

    public T Resolve<T>() => (T)Resolve(typeof(T));

    public object Resolve(Type type)
    {
        if (_registrations.TryGetValue(type, out var factory))
            return factory();

        // Crear automaticamente si es concreto
        if (!type.IsAbstract && !type.IsInterface)
        {
            var ctor = type.GetConstructors()[0];
            var parameters = ctor.GetParameters();
            var args = new object[parameters.Length];

            for (int i = 0; i < parameters.Length; i++)
                args[i] = Resolve(parameters[i].ParameterType);

            return ctor.Invoke(args);
        }

        throw new InvalidOperationException($"No se puede resolver {type.Name}");
    }
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== DEPENDENCY INJECTION ===\n");

        // Configurar contenedor
        var container = new SimpleContainer();
        container.RegisterSingleton<ILogger, ConsoleLogger>();
        container.Register<IRepositorioUsuarios, RepositorioUsuariosSQL>();
        container.Register<IServicioEmail, ServicioEmailSMTP>();
        container.RegisterInstance("connectionString", "Server=.;Database=test");
        container.RegisterInstance("smtpServer", "smtp.empresa.com");

        // Resolver servicio
        var service = container.Resolve<RegistroUsuarioService>();
        service.Registrar("Ana Garcia", "ana@email.com");

        Console.WriteLine("\nRegistro completado!");
    }
}

// Extension para SimpleContainer con valores
public static class SimpleContainerExtensions
{
    private static readonly Dictionary<Type, object> _values = new();

    public static void RegisterInstance<T>(this SimpleContainer container, string name, T value)
    {
        _values[typeof(T)] = value;
    }

    public static T Resolve<T>(this SimpleContainer container, string name)
    {
        return (T)_values[typeof(T)];
    }
}
```

--- 5. IDISPOSABLE PATTERN ---

El patron IDisposable libera recursos no administrados de forma
determinista. Es crucial para archivos, conexiones, etc.

```csharp
using System;
using System.IO;

// Implementacion basica
public class ManejadorArchivo : IDisposable
{
    private FileStream _fileStream;
    private bool _disposed;

    public ManejadorArchivo(string ruta)
    {
        _fileStream = File.OpenRead(ruta);
        Console.WriteLine($"  Archivo abierto: {ruta}");
    }

    public void Dispose()
    {
        Dispose(true);
        GC.SuppressFinalize(this);
    }

    protected virtual void Dispose(bool disposing)
    {
        if (_disposed) return;

        if (disposing)
        {
            // Liberar recursos managed
            _fileStream?.Dispose();
            Console.WriteLine("  Recursos managed liberados.");
        }

        // Liberar recursos no managed (si los hubiera)
        _disposed = true;
    }

    ~ManejadorArchivo()
    {
        Dispose(false); // Solo recursos no managed
    }

    public void Leer()
    {
        if (_disposed)
            throw new ObjectDisposedException(nameof(ManejadorArchivo));

        byte[] buffer = new byte[100];
        _fileStream.Read(buffer, 0, buffer.Length);
    }
}

// Implementacion completa con SafeHandle
public class RecursoSeguro : IDisposable
{
    private IntPtr _recursoNativo;
    private bool _disposed;

    public RecursoSeguro()
    {
        _recursoNativo = System.Runtime.InteropServices.Marshal.AllocHGlobal(1000);
        Console.WriteLine("  Memoria nativa asignada.");
    }

    public void Dispose()
    {
        Dispose(true);
        GC.SuppressFinalize(this);
    }

    protected virtual void Dispose(bool disposing)
    {
        if (_disposed) return;

        if (disposing)
        {
            // Liberar otros recursos managed
        }

        // Liberar recurso nativo
        if (_recursoNativo != IntPtr.Zero)
        {
            System.Runtime.InteropServices.Marshal.FreeHGlobal(_recursoNativo);
            _recursoNativo = IntPtr.Zero;
            Console.WriteLine("  Memoria nativa liberada.");
        }

        _disposed = true;
    }

    ~RecursoSeguro()
    {
        Dispose(false);
    }
}

// Uso con using
class Program
{
    static void Main()
    {
        Console.WriteLine("=== IDISPOSABLE PATTERN ===\n");

        // using statement (recomendado)
        Console.WriteLine("1. Usando 'using':");
        using (var recurso = new RecursoSeguro())
        {
            Console.WriteLine("  Usando el recurso...");
        }
        Console.WriteLine("  Recurso liberado automaticamente.");

        // using declaration (C# 8+)
        Console.WriteLine("\n2. Usando 'using declaration':");
        using var archivo = new ManejadorArchivo("/etc/hosts");
        archivo.Leer();
        Console.WriteLine("  Archivo leido.");
        // Se libera al salir del ambito

        // Try/finally manual
        Console.WriteLine("\n3. Try/finally manual:");
        RecursoSeguro recurso2 = null;
        try
        {
            recurso2 = new RecursoSeguro();
            Console.WriteLine("  Usando recurso en try/finally...");
        }
        finally
        {
            recurso2?.Dispose();
        }

        // Usar con null-conditional
        Console.WriteLine("\n4. Con null-conditional:");
        (null as RecursoSeguro)?.Dispose();
        Console.WriteLine("  Dispose seguro con null.");
    }
}
```

--- 6. STRATEGY PATTERN ---

Strategy permite definir una familia de algoritmos, encapsularlos y
hacerlos intercambiables.

```csharp
using System;
using System.Collections.Generic;

// Strategy interface
public interface ICompresionStrategy
{
    string Comprimir(string datos);
    string Descomprimir(string datos);
    string Tipo { get; }
}

// Concrete strategies
public class CompresionGZip : ICompresionStrategy
{
    public string Tipo => "GZip";

    public string Comprimir(string datos)
    {
        Console.WriteLine($"  [GZip] Comprimiendo {datos.Length} chars...");
        return $"gzip:{datos}";
    }

    public string Descomprimir(string datos)
    {
        Console.WriteLine("  [GZip] Descomprimiendo...");
        return datos.Replace("gzip:", "");
    }
}

public class CompresionBrotli : ICompresionStrategy
{
    public string Tipo => "Brotli";

    public string Comprimir(string datos)
    {
        Console.WriteLine($"  [Brotli] Comprimiendo {datos.Length} chars...");
        return $"br:{datos}";
    }

    public string Descomprimir(string datos)
    {
        Console.WriteLine("  [Brotli] Descomprimiendo...");
        return datos.Replace("br:", "");
    }
}

public class CompresionLZ4 : ICompresionStrategy
{
    public string Tipo => "LZ4";

    public string Comprimir(string datos)
    {
        Console.WriteLine($"  [LZ4] Comprimiendo {datos.Length} chars (alta velocidad)...");
        return $"lz4:{datos}";
    }

    public string Descomprimir(string datos)
    {
        Console.WriteLine("  [LZ4] Descomprimiendo...");
        return datos.Replace("lz4:", "");
    }
}

// Context
public class ProcesadorArchivos
{
    private ICompresionStrategy _strategy;

    public ProcesadorArchivos(ICompresionStrategy strategy)
    {
        _strategy = strategy;
    }

    public void CambiarStrategy(ICompresionStrategy strategy)
    {
        _strategy = strategy;
        Console.WriteLine($"  Strategy cambiada a {strategy.Tipo}");
    }

    public string Procesar(string contenido)
    {
        Console.WriteLine($"\nProcesando con {_strategy.Tipo}:");

        string comprimido = _strategy.Comprimir(contenido);
        Console.WriteLine($"  Comprimido: {comprimido.Substring(0, Math.Min(30, comprimido.Length))}...");

        string descomprimido = _strategy.Descomprimir(comprimido);
        Console.WriteLine($"  Descomprimido: {descomprimido.Substring(0, Math.Min(30, descomprimido.Length))}...");

        return descomprimido;
    }
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== STRATEGY PATTERN ===\n");

        var procesador = new ProcesadorArchivos(new CompresionGZip());
        procesador.Procesar("Este es un texto de ejemplo para compresion...");

        procesador.CambiarStrategy(new CompresionBrotli());
        procesador.Procesar("Otro texto para probar Brotli...");

        procesador.CambiarStrategy(new CompresionLZ4());
        procesador.Procesar("LZ4 es ideal para velocidad...");
    }
}
```

--- 7. OBSERVER PATTERN ---

Observer establece una relacion de uno a muchos donde cuando un objeto
cambia su estado, todos sus dependientes son notificados.

```csharp
using System;
using System.Collections.Generic;

// Event args personalizados
public class CambioPrecioEventArgs : EventArgs
{
    public string Simbolo { get; }
    public decimal PrecioAnterior { get; }
    public decimal PrecioNuevo { get; }
    public decimal CambioPorcentual => ((PrecioNuevo - PrecioAnterior) / PrecioAnterior) * 100;

    public CambioPrecioEventArgs(string simbolo, decimal anterior, decimal nuevo)
    {
        Simbolo = simbolo;
        PrecioAnterior = anterior;
        PrecioNuevo = nuevo;
    }
}

// Sujeto (Observable)
public class Accion
{
    private decimal _precio;
    private readonly List<IObserver<CambioPrecioEventArgs>> _observers = new();

    public string Simbolo { get; }
    public decimal Precio => _precio;

    public Accion(string simbolo, decimal precioInicial)
    {
        Simbolo = simbolo;
        _precio = precioInicial;
    }

    public IDisposable Subscribe(IObserver<CambioPrecioEventArgs> observer)
    {
        _observers.Add(observer);
        return new Unsubscriber(_observers, observer);
    }

    public void ActualizarPrecio(decimal nuevoPrecio)
    {
        if (nuevoPrecio == _precio) return;

        var args = new CambioPrecioEventArgs(Simbolo, _precio, nuevoPrecio);
        _precio = nuevoPrecio;

        foreach (var observer in _observers.ToArray())
            observer.OnNext(args);
    }

    private class Unsubscriber : IDisposable
    {
        private readonly List<IObserver<CambioPrecioEventArgs>> _observers;
        private readonly IObserver<CambioPrecioEventArgs> _observer;

        public Unsubscriber(List<IObserver<CambioPrecioEventArgs>> observers,
                          IObserver<CambioPrecioEventArgs> observer)
        {
            _observers = observers;
            _observer = observer;
        }

        public void Dispose()
        {
            _observers.Remove(_observer);
        }
    }
}

// Observers
public class Inversor : IObserver<CambioPrecioEventArgs>
{
    public string Nombre { get; }

    public Inversor(string nombre)
    {
        Nombre = nombre;
    }

    public void OnNext(CambioPrecioEventArgs value)
    {
        string accion = value.CambioPorcentual >= 0
            ? $"GANANCIA de {value.CambioPorcentual:F2}%"
            : $"PERDIDA de {Math.Abs(value.CambioPorcentual):F2}%";

        Console.WriteLine($"  [Inversor {Nombre}] {value.Simbolo}: " +
            $"${value.PrecioAnterior} -> ${value.PrecioNuevo} ({accion})");
    }

    public void OnError(Exception error)
        => Console.WriteLine($"  [Inversor {Nombre}] Error: {error.Message}");

    public void OnCompleted()
        => Console.WriteLine($"  [Inversor {Nombre}] Mercado cerrado.");
}

public class Analista : IObserver<CambioPrecioEventArgs>
{
    public void OnNext(CambioPrecioEventArgs value)
    {
```

```csharp
public class Producto

    IEnumerable<Producto> ObtenerTodos();
    IEnumerable<Producto> Buscar(Func<Producto, bool> criterio);
    void Agregar(Producto producto);
    void Actualizar(Producto producto);

// Implementacion en memoria
public class ProductoRepositoryMemoria : IProductoRepository
{
    private readonly List<Producto> _productos = new();
    private int _nextId = 1;

    public ProductoRepositoryMemoria()
    {
        // Datos de ejemplo
        Agregar(new Producto { Nombre = "Laptop", Precio = 1500m, Stock = 10 });
        Agregar(new Producto { Nombre = "Mouse", Precio = 25m, Stock = 100 });
        Agregar(new Producto { Nombre = "Teclado", Precio = 80m, Stock = 50 });
    }

    public Producto ObtenerPorId(int id)
        => _productos.FirstOrDefault(p => p.Id == id);

    public IEnumerable<Producto> ObtenerTodos()
        => _productos.Where(p => p.Activo).ToList();

    public IEnumerable<Producto> Buscar(Func<Producto, bool> criterio)
        => _productos.Where(criterio).ToList();

    public void Agregar(Producto producto)
    {
        producto.Id = _nextId++;
        producto.Activo = true;
        _productos.Add(producto);
    }

    public void Actualizar(Producto producto)
    {
        var existente = ObtenerPorId(producto.Id);
        if (existente != null)
        {
            existente.Nombre = producto.Nombre;
            existente.Precio = producto.Precio;
            existente.Stock = producto.Stock;
        }
    }

    public void Eliminar(int id)
    {
        var producto = ObtenerPorId(id);
        if (producto != null)
            producto.Activo = false; // Soft delete
    }

    public bool Existe(int id)
        => _productos.Any(p => p.Id == id);

    public int Contar()
        => _productos.Count(p => p.Activo);
}

// Servicio de dominio que usa el repositorio
public class InventarioService
{
    private readonly IProductoRepository _repository;

    public InventarioService(IProductoRepository repository)
    {
        _repository = repository;
    }

    public void AgregarProducto(string nombre, decimal precio, int stock)
    {
        if (string.IsNullOrWhiteSpace(nombre))
            throw new ArgumentException("El nombre es requerido.");

        if (precio <= 0)
            throw new ArgumentException("El precio debe ser positivo.");

        if (stock < 0)
            throw new ArgumentException("El stock no puede ser negativo.");

        var producto = new Producto
        {
            Nombre = nombre,
            Precio = precio,
            Stock = stock
        };

        _repository.Agregar(producto);
        Console.WriteLine($"  Producto '{nombre}' agregado.");
    }

    public void VenderProducto(int productoId, int cantidad)
    {
        var producto = _repository.ObtenerPorId(productoId);
        if (producto == null)
            throw new InvalidOperationException($"Producto #{productoId} no encontrado.");

        if (producto.Stock < cantidad)
            throw new InvalidOperationException(
                $"Stock insuficiente para #{productoId}. Disponible: {producto.Stock}");

        producto.Stock -= cantidad;
        _repository.Actualizar(producto);
        Console.WriteLine($"  Vendidos {cantidad} de '{producto.Nombre}'. Stock restante: {producto.Stock}");
    }

    public void ListarProductos()
    {
        Console.WriteLine("\nInventario:");
        foreach (var p in _repository.ObtenerTodos())
            Console.WriteLine($"  #{p.Id} {p.Nombre} - ${p.Precio} (stock: {p.Stock})");
    }
}
```

Chain of Responsibility pasa la solicitud a traves de una cadena de

```csharp
using System;

public class SolicitudSoporte
{

    public NivelSeveridad Severidad { get; set; }
    public bool Resuelta { get; set; }

public enum NivelSeveridad
{
    Baja,
    Media,
    Alta,
    Critica
}

// Manejador abstracto
public abstract class ManejadorSoporte
{
    protected ManejadorSoporte _siguiente;

    {
        _siguiente = siguiente;
        return siguiente;

    {

    }
}

    }
```

```csharp
{
    private string _tabla;
    private List<string> _columnas = new() { "*" };
    private List<string> _condiciones = new();
    private List<string> _ordenes = new();
    private int? _limite;

        return this;
    }

        return this;

        return this;
    }

    {
        _ordenes.Add($"{columna} {(ascendente ? "ASC" : "DESC")}");
        return this;
    }

    public ConsultaSQL Limitar(int limite)
    {
        _limite = limite;
        return this;

WHERE {string.Join(" AND ", _condiciones)}";

        if (_ordenes.Count > 0)
            sql += $"
ORDER BY {string.Join(", ", _ordenes)}";

        if (_limite.HasValue)
            sql += $"
LIMIT {_limite.Value}";

        sql += ";";
        return sql;
    }
```

```csharp
using System;

// Gestor de recursos con tracking
public class RecursoRastreable : IDisposable
{
    private static readonly List<RecursoRastreable> _instancias = new();
    private static readonly object _lock = new();

    private readonly string _nombre;
    private bool _disposed;

    public RecursoRastreable(string nombre)
    {
        _nombre = nombre;
        lock (_lock)
        {
            _instancias.Add(this);
        }
        Console.WriteLine($"  [+] Recurso '{_nombre}' creado. Total: {_instancias.Count}");
    }

    public void Dispose()
    {
        Dispose(true);
        GC.SuppressFinalize(this);
    }

    protected virtual void Dispose(bool disposing)
    {
        if (_disposed) return;

        if (disposing)
        {
            lock (_lock)
            {
                _instancias.Remove(this);
            }
            Console.WriteLine($"  [-] Recurso '{_nombre}' liberado. Restantes: {_instancias.Count}");
        }

        _disposed = true;
    }

    ~RecursoRastreable() => Dispose(false);

    public static int InstanciasActivas
    {
        get { lock (_lock) { return _instancias.Count; } }
    }
}

public class RecursoBase : IDisposable

    public void Dispose()
    {
        Dispose(true);
        GC.SuppressFinalize(this);
    }

    protected virtual void Dispose(bool disposing)

            Console.WriteLine("  [Base] Liberando recursos managed.");
        Console.WriteLine("  [Base] Liberando recursos nativos.");

    }

public class RecursoDerivado : RecursoBase

    private FileStream _archivo;

    public RecursoDerivado(string ruta)
    {

        Console.WriteLine($"  [Derivado] Archivo abierto: {ruta}");
    }

    protected override void Dispose(bool disposing)
    {
        if (_disposed) return;

        if (disposing)
        {
            _archivo?.Dispose();
            Console.WriteLine("  [Derivado] Liberando recursos managed.");
        }

        Console.WriteLine("  [Derivado] Liberando recursos nativos.");
        _disposed = true;

        // Llamar al base
        base.Dispose(disposing);
    }

    ~RecursoDerivado() => Dispose(false);
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== IDISPOSABLE AVANZADO ===\n");

        // Tracking de recursos
        using (var r1 = new RecursoRastreable("Recurso-1"))
        using (var r2 = new RecursoRastreable("Recurso-2"))
        {
            Console.WriteLine($"  Instancias activas: {RecursoRastreable.InstanciasActivas}");
            using (var r3 = new RecursoRastreable("Recurso-3"))
            {
                Console.WriteLine($"  Instancias activas: {RecursoRastreable.InstanciasActivas}");
            }
            Console.WriteLine($"  Instancias activas: {RecursoRastreable.InstanciasActivas}");
        }
        Console.WriteLine($"  Instancias activas final: {RecursoRastreable.InstanciasActivas}");

        // Factory con Dispose
        Console.WriteLine("\nFactory con Dispose:");
        using (var db = ConexionBD.Conectar("Server=.;Database=master"))
        {
            db.EjecutarQuery("SELECT * FROM Usuarios");
        }

        // Herencia con Dispose
        Console.WriteLine("\nHerencia con Dispose:");
        using (var recurso = new RecursoDerivado("/etc/hosts"))
        {
            Console.WriteLine("  Usando recurso...");
        }
    }
}
```

================================================================================
  PREGUNTAS FRECUENTES (FAQ)
================================================================================

P: Cual patron debo usar primero? R: Dependency Injection es el mas
importante en C# moderno. Facilita testing y desacoplamiento.

P: Singleton es anti-patron? R: Puede serlo si se usa para estado global
mutable. Use Singleton para configuracion, logging, caching (inmutable).

P: Cuando usar Factory vs Builder? R: Factory para crear objetos en un
solo paso. Builder para construccion paso a paso con muchas opciones.

P: IDisposable siempre necesario? R: Solo si tiene recursos no
administrados (archivos, conexiones, handles nativos). Para recursos
managed use using y confie en el GC.

P: Que es mejor: interfaces o clases abstractas para patrones? R:
Interfaces para contratos (Strategy, Observer). Clases abstractas
cuando hay comportamiento base (Template Method, Decorator).
