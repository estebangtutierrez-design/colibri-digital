--- 1. ESTRUCTURA DE UN ATRIBUTO PERSONALIZADO ---

Un atributo personalizado es una clase que:
1. Hereda de System.Attribute (directa o indirectamente).
2. Tiene el sufijo "Attribute" por convención (el compilador permite omitirlo).
3. Se marca con [AttributeUsage] para especificar destinos válidos.

```csharp
using System;

// Definición básica
[AttributeUsage(AttributeTargets.Class | AttributeTargets.Method)]
public class InformacionAttribute : Attribute
{
    public string Descripcion { get; }
    
    public InformacionAttribute(string descripcion)
    {
        Descripcion = descripcion;
    }
}

// Uso
[Informacion("Esta clase maneja operaciones de usuarios")]
public class UsuarioService
{
    [Informacion("Autentica un usuario con credenciales")]
    public bool Login(string user, string pass) => true;
}
```

--- 4. RECUPERAR ATRIBUTOS CON REFLECTION ---

```csharp
using System;
using System.Reflection;

class Program
{
    static void Main()
    {
        Console.WriteLine("=== RECUPERAR ATRIBUTOS ===\n");
        
        // Obtener atributos de una clase
        Type tipo = typeof(ServicioAutenticacion);
        
        var infoAttr = tipo.GetCustomAttribute<InformacionAttribute>();
        if (infoAttr != null)
        {
            Console.WriteLine($"Clase: {tipo.Name}");
            Console.WriteLine($"  Descripción: {infoAttr.Descripcion}");
        }
        
        // Obtener atributos de métodos
        Console.WriteLine("\nMétodos:");
        foreach (var metodo in tipo.GetMethods(
            BindingFlags.Public | BindingFlags.Instance | BindingFlags.DeclaredOnly))
        {
            var attr = metodo.GetCustomAttribute<InformacionAttribute>();
            Console.WriteLine($"  {metodo.Name}: {(attr?.Descripcion ?? "Sin descripción")}");
        }
        
        // Obtener TODOS los atributos (no solo uno)
        Console.WriteLine("\nTodos los atributos de la clase:");
        foreach (var attr in tipo.GetCustomAttributes(false))
        {
            Console.WriteLine($"  - {attr.GetType().Name}");
        }
        
        // Verificar si existe un atributo
        bool tieneAtributo = tipo.IsDefined(typeof(InformacionAttribute), false);
        Console.WriteLine($"\n¿Tiene InformacionAttribute? {tieneAtributo}");
        
        // GetCustomAttributes con filtro
        var atributos = tipo.GetCustomAttributes(typeof(ObsoletoAttribute), false);
        Console.WriteLine($"¿Es obsoleto? {atributos.Length > 0}");
    }
}

[Informacion("Servicio de autenticación de usuarios")]
[Obsoleto("Usar ServicioAutenticacionV2")]
public class ServicioAutenticacion
{
    [Informacion("Inicia sesión en el sistema")]
    public void Login(string user, string pass) { }
    
    [Informacion("Cierra la sesión actual")]
    public void Logout() { }
    
    public void CambiarPassword(string oldPass, string newPass) { }
}
```

```csharp
using System;
using System.Collections.Generic;

[AttributeUsage(AttributeTargets.All, AllowMultiple = true)]
public class MetadataAttribute : Attribute
{
    public string Clave { get; }
    public object Valor { get; }
    
    // Propiedades con tipos complejos
    public Type TipoAsociado { get; set; }
    public Dictionary<string, object> Datos { get; set; }
    public int[] Numeros { get; set; }
    
    public MetadataAttribute(string clave, object valor)
    {
        Clave = clave;
        Valor = valor;
        Datos = new Dictionary<string, object>();
    }
    
    public T ObtenerValor<T>()
    {
        try { return (T)Valor; }
        catch { return default; }
    }
}

[Metadata("autor", "Juan Pérez", TipoAsociado = typeof(Program))]
[Metadata("version", "2.0.1")]
[Metadata("fecha_creacion", "2024-01-15")]
[Metadata("tags", new[] { "utilidad", "ejemplo", "atributos" })]
class ClaseConMetadata
{
    [Metadata("campo", "importante", Numeros = new[] { 1, 2, 3 })]
    public string Propiedad { get; set; }
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== ATRIBUTOS COMPLEJOS ===\n");
        
        var tipo = typeof(ClaseConMetadata);
        
        foreach (MetadataAttribute meta in tipo.GetCustomAttributes(typeof(MetadataAttribute), false))
        {
            Console.WriteLine($"  {meta.Clave} = {meta.Valor} (tipo: {meta.Valor?.GetType().Name})");
            if (meta.TipoAsociado != null)
                Console.WriteLine($"    Tipo asociado: {meta.TipoAsociado.Name}");
        }
        
        // Leer atributo de propiedad
        var prop = tipo.GetProperty("Propiedad");
        foreach (MetadataAttribute meta in prop.GetCustomAttributes(typeof(MetadataAttribute), false))
        {
            Console.WriteLine($"\nPropiedad.Propiedad:");
            Console.WriteLine($"  {meta.Clave} = {meta.Valor}");
            if (meta.Numeros != null)
                Console.WriteLine($"  Números: [{string.Join(", ", meta.Numeros)}]");
        }
    }
}
```

--- 6. ATRIBUTOS REPETIBLES (AllowMultiple = true) ---

```csharp
using System;
using System.Collections.Generic;
using System.Reflection;

[AttributeUsage(AttributeTargets.Property, AllowMultiple = true)]
public class ReglaValidacionAttribute : Attribute
{
    public string Nombre { get; }
    public object[] Parametros { get; }
    
    public ReglaValidacionAttribute(string nombre, params object[] parametros)
    {
        Nombre = nombre;
        Parametros = parametros;
    }
}

[AttributeUsage(AttributeTargets.Property, AllowMultiple = true)]
public class EtiquetaAttribute : Attribute
{
    public string Texto { get; }
    public EtiquetaAttribute(string texto) => Texto = texto;
}

class UsuarioModelo
{
    [ReglaValidacion("Requerido")]
    [ReglaValidacion("LongitudMaxima", 100)]
    [ReglaValidacion("ExpresionRegular", @"^[a-zA-Z\s]+$")]
    [Etiqueta("Nombre Completo")]
    [Etiqueta("Campo Obligatorio")]
    public string Nombre { get; set; }
    
    [ReglaValidacion("Requerido")]
    [ReglaValidacion("Email")]
    [Etiqueta("Correo Electrónico")]
    public string Email { get; set; }
    
    [ReglaValidacion("Rango", 18, 120)]
    public int Edad { get; set; }
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== ATRIBUTOS REPETIBLES ===\n");
        
        var tipo = typeof(UsuarioModelo);
        
        foreach (var prop in tipo.GetProperties())
        {
            Console.WriteLine($"Propiedad: {prop.Name}");
            
            // Obtener TODAS las instancias del atributo
            var reglas = prop.GetCustomAttributes<ReglaValidacionAttribute>();
            foreach (var regla in reglas)
            {
                string parametros = regla.Parametros.Length > 0
                    ? $" ({string.Join(", ", regla.Parametros)})"
                    : "";
                Console.WriteLine($"  - {regla.Nombre}{parametros}");
            }
            
            var etiquetas = prop.GetCustomAttributes<EtiquetaAttribute>();
            foreach (var etiqueta in etiquetas)
                Console.WriteLine($"  [Etiqueta: {etiqueta.Texto}]");
            
            Console.WriteLine();
        }
    }
}
```

--- 7. ATRIBUTOS Y HERENCIA ---

```csharp
using System;

[AttributeUsage(AttributeTargets.Class, Inherited = true)]
public class VersionableAttribute : Attribute
{
    public int Version { get; }
    public VersionableAttribute(int v) => Version = v;
}

[AttributeUsage(AttributeTargets.Class, Inherited = false)]
public class NoHerableAttribute : Attribute { }

[Versionable(1)]
[NoHerable]
public class Base
{
    public virtual void Metodo() { }
}

// Hereda VersionableAttribute pero NO NoHerableAttribute
[Versionable(2)] // Sobrescribe el valor heredado
public class Derivada : Base
{
    public override void Metodo() { }
}

// Hereda Versionable de Derivada (que a su vez heredó de Base)
public class SubDerivada : Derivada { }

class Program
{
    static void Main()
    {
        Console.WriteLine("=== ATRIBUTOS Y HERENCIA ===\n");
        
        // Base
        Console.WriteLine("Base:");
        var baseAttr1 = typeof(Base).GetCustomAttribute<VersionableAttribute>(false);
        var baseAttr2 = typeof(Base).GetCustomAttribute<VersionableAttribute>(true);
        Console.WriteLine($"  Versionable (no inhert): {baseAttr1?.Version}");
        Console.WriteLine($"  Versionable (inhert): {baseAttr2?.Version}");
        Console.WriteLine($"  NoHerable: {typeof(Base).IsDefined(typeof(NoHerableAttribute), false)}");
        
        // Derivada
        Console.WriteLine("\nDerivada:");
        var derAttr1 = typeof(Derivada).GetCustomAttribute<VersionableAttribute>(false);
        var derAttr2 = typeof(Derivada).GetCustomAttribute<VersionableAttribute>(true);
        Console.WriteLine($"  Versionable (no inhert): {derAttr1?.Version}");
        Console.WriteLine($"  Versionable (inhert): {derAttr2?.Version}");
        Console.WriteLine($"  NoHerable: {typeof(Derivada).IsDefined(typeof(NoHerableAttribute), false)}");
        
        // SubDerivada
        Console.WriteLine("\nSubDerivada:");
        var subAttr1 = typeof(SubDerivada).GetCustomAttribute<VersionableAttribute>(false);
        var subAttr2 = typeof(SubDerivada).GetCustomAttribute<VersionableAttribute>(true);
        Console.WriteLine($"  Versionable (no inhert): {subAttr1?.Version}");
        Console.WriteLine($"  Versionable (inhert): {subAttr2?.Version}");
    }
}
```

--- 8. SISTEMA DE PLUGINS CON ATRIBUTOS ---

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Reflection;

[AttributeUsage(AttributeTargets.Class)]
public class PluginInfoAttribute : Attribute
{
    public string Nombre { get; }
    public string Version { get; }
    public string Autor { get; set; }
    public string Descripcion { get; set; }
    
    public PluginInfoAttribute(string nombre, string version)
    {
        Nombre = nombre;
        Version = version;
    }
}

[AttributeUsage(AttributeTargets.Class)]
public class PluginDependencyAttribute : Attribute
{
    public string NombrePlugin { get; }
    public string VersionMinima { get; }
    
    public PluginDependencyAttribute(string nombre, string version)
    {
        NombrePlugin = nombre;
        VersionMinima = version;
    }
}

// Interfaz que deben implementar los plugins
public interface IPlugin
{
    void Inicializar();
    void Ejecutar();
    void Finalizar();
}

// Plugin de ejemplo 1
[PluginInfo("CompresorImagenes", "1.2.0",
    Autor = "Juan Pérez",
    Descripcion = "Comprime imágenes en formato JPEG y PNG")]
public class CompresorImagenesPlugin : IPlugin
{
    public void Inicializar()
        => Console.WriteLine("  [CompresorImagenes] Inicializado");
    public void Ejecutar()
        => Console.WriteLine("  [CompresorImagenes] Comprimiendo...");
    public void Finalizar()
        => Console.WriteLine("  [CompresorImagenes] Finalizado");
}

// Plugin de ejemplo 2
[PluginInfo("EncriptadorAES", "2.0.1",
    Autor = "María García",
    Descripcion = "Encripta y desencripta archivos con AES-256")]
[PluginDependency("CompresorImagenes", "1.0.0")]
public class EncriptadorAESPlugin : IPlugin
{
    public void Inicializar()
        => Console.WriteLine("  [EncriptadorAES] Inicializado");
    public void Ejecutar()
        => Console.WriteLine("  [EncriptadorAES] Encriptando...");
    public void Finalizar()
        => Console.WriteLine("  [EncriptadorAES] Finalizado");
}

// Plugin de ejemplo 3
[PluginInfo("GeneradorReportes", "1.0.0",
    Autor = "Carlos Ruiz",
    Descripcion = "Genera reportes en PDF y Excel")]
public class GeneradorReportesPlugin : IPlugin
{
    public void Inicializar()
        => Console.WriteLine("  [GeneradorReportes] Inicializado");
    public void Ejecutar()
        => Console.WriteLine("  [GeneradorReportes] Generando reporte...");
    public void Finalizar()
        => Console.WriteLine("  [GeneradorReportes] Finalizado");
}

// Motor de plugins
public class PluginManager
{
    private readonly Dictionary<string, Type> _plugins = new();
    private readonly Dictionary<string, IPlugin> _instancias = new();
    
    public void CargarPlugins(Assembly assembly)
    {
        foreach (var tipo in assembly.GetTypes())
        {
            if (typeof(IPlugin).IsAssignableFrom(tipo) && !tipo.IsInterface)
            {
                var info = tipo.GetCustomAttribute<PluginInfoAttribute>();
                if (info != null)
                {
                    _plugins[info.Nombre] = tipo;
                    Console.WriteLine($"Plugin registrado: {info.Nombre} v{info.Version}");
                }
            }
        }
    }
    
    public void InicializarTodos()
    {
        foreach (var kvp in _plugins)
        {
            var plugin = (IPlugin)Activator.CreateInstance(kvp.Value);
            _instancias[kvp.Key] = plugin;
            plugin.Inicializar();
        }
    }
    
    public void EjecutarPlugin(string nombre)
    {
        if (_instancias.TryGetValue(nombre, out var plugin))
        {
            Console.WriteLine($"\nEjecutando plugin: {nombre}");
            plugin.Ejecutar();
        }
        else
        {
            Console.WriteLine($"Plugin '{nombre}' no encontrado.");
        }
    }
    
    public void ListarPlugins()
    {
        Console.WriteLine("\nPlugins cargados:");
        foreach (var kvp in _plugins)
        {
            var info = kvp.Value.GetCustomAttribute<PluginInfoAttribute>();
            var deps = kvp.Value.GetCustomAttributes<PluginDependencyAttribute>();
            
            Console.WriteLine($"  - {info?.Nombre} v{info?.Version}");
            Console.WriteLine($"    Autor: {info?.Autor}");
            Console.WriteLine($"    Desc: {info?.Descripcion}");
            
            if (deps.Any())
            {
                Console.WriteLine($"    Dependencias:");
                foreach (var dep in deps)
                    Console.WriteLine($"      - {dep.NombrePlugin} >= {dep.VersionMinima}");
            }
        }
    }
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== SISTEMA DE PLUGINS CON ATRIBUTOS ===\n");
        
        var manager = new PluginManager();
        manager.CargarPlugins(Assembly.GetExecutingAssembly());
        manager.ListarPlugins();
        manager.InicializarTodos();
        
        manager.EjecutarPlugin("CompresorImagenes");
        manager.EjecutarPlugin("GeneradorReportes");
        manager.EjecutarPlugin("PluginInexistente");
    }
}
```

--- 9. ATRIBUTOS PARA MAPEO OBJETO-RELACIONAL (ORM) ---

```csharp
using System;
using System.Collections.Generic;
using System.Reflection;
using System.Text;

[AttributeUsage(AttributeTargets.Class)]
public class TablaAttribute : Attribute
{
    public string Nombre { get; }
    public string Esquema { get; set; }
    public TablaAttribute(string nombre) => Nombre = nombre;
}

[AttributeUsage(AttributeTargets.Property)]
public class ColumnaAttribute : Attribute
{
    public string Nombre { get; }
    public bool EsClave { get; set; }
    public bool EsAutoIncrement { get; set; }
    public bool EsNulo { get; set; } = true;
    public int Longitud { get; set; }
    public string TipoSql { get; set; }
    public ColumnaAttribute(string nombre) => Nombre = nombre;
}

[AttributeUsage(AttributeTargets.Property)]
public class RelacionAttribute : Attribute
{
    public string Tipo { get; }
    public string EntidadRelacionada { get; }
    public string ColumnaLocal { get; set; }
    public string ColumnaForanea { get; set; }
    public RelacionAttribute(string tipo, string entidad)
    {
        Tipo = tipo;
        EntidadRelacionada = entidad;
    }
}

// Modelos usando atributos
[Tabla("Categorias", Esquema = "dbo")]
public class Categoria
{
    [Columna("Id", EsClave = true, EsAutoIncrement = true)]
    public int Id { get; set; }
    
    [Columna("Nombre", EsNulo = false, Longitud = 100)]
    public string Nombre { get; set; }
    
    [Relacion("OneToMany", "dbo.Productos", ColumnaLocal = "Id", ColumnaForanea = "CategoriaId")]
    public List<Producto> Productos { get; set; }
}

[Tabla("Productos", Esquema = "dbo")]
public class Producto
{
    [Columna("Id", EsClave = true, EsAutoIncrement = true)]
    public int Id { get; set; }
    
    [Columna("Nombre", EsNulo = false, Longitud = 200)]
    public string Nombre { get; set; }
    
    [Columna("Precio", TipoSql = "DECIMAL(10,2)")]
    public decimal Precio { get; set; }
    
    [Columna("CategoriaId")]
    public int CategoriaId { get; set; }
    
    [Relacion("ManyToOne", "dbo.Categorias", ColumnaLocal = "CategoriaId", ColumnaForanea = "Id")]
    public Categoria Categoria { get; set; }
}

// Generador SQL
public class GeneradorSQL
{
    public string GenerarInsert<T>(T entidad)
    {
        var tipo = typeof(T);
        var propiedades = tipo.GetProperties(BindingFlags.Public | BindingFlags.Instance);
        
        var columnas = new List<string>();
        var valores = new List<string>();
        
        foreach (var prop in propiedades)
        {
            var columna = prop.GetCustomAttribute<ColumnaAttribute>();
            if (columna == null || columna.EsAutoIncrement) continue;
            
            columnas.Add(columna.Nombre);
            object valor = prop.GetValue(entidad);
            
            if (valor == null)
                valores.Add("NULL");
            else if (valor is string || valor is DateTime)
                valores.Add($"'{valor}'");
            else if (valor is bool)
                valores.Add((bool)valor ? "1" : "0");
            else
                valores.Add(valor.ToString());
        }
        
        var tablaAttr = tipo.GetCustomAttribute<TablaAttribute>();
        string nombreTabla = tablaAttr != null
            ? $"{tablaAttr.Esquema}.{tablaAttr.Nombre}"
            : tipo.Name;
        
        return $"INSERT INTO {nombreTabla} ({string.Join(", ", columnas)})\n" +
               $"VALUES ({string.Join(", ", valores)});";
    }
    
    public string GenerarSelect<T>(string condicion = null)
    {
        var tipo = typeof(T);
        var propiedades = tipo.GetProperties(BindingFlags.Public | BindingFlags.Instance);
        
        var columnas = new List<string>();
        foreach (var prop in propiedades)
        {
            var columna = prop.GetCustomAttribute<ColumnaAttribute>();
            if (columna != null)
                columnas.Add(columna.Nombre);
            else
                columnas.Add(prop.Name);
        }
        
        var tablaAttr = tipo.GetCustomAttribute<TablaAttribute>();
        string nombreTabla = tablaAttr != null
            ? $"{tablaAttr.Esquema}.{tablaAttr.Nombre}"
            : tipo.Name;
        
        string sql = $"SELECT {string.Join(", ", columnas)}\nFROM {nombreTabla}";
        
        if (!string.IsNullOrEmpty(condicion))
            sql += $"\nWHERE {condicion}";
        
        // Joins por relaciones
        foreach (var prop in propiedades)
        {
            var relacion = prop.GetCustomAttribute<RelacionAttribute>();
            if (relacion != null && relacion.Tipo == "ManyToOne")
            {
                sql += $"\nINNER JOIN {relacion.EntidadRelacionada} ON " +
                       $"{relacion.EntidadRelacionada}.{relacion.ColumnaForanea} = " +
                       $"{nombreTabla}.{relacion.ColumnaLocal}";
            }
        }
        
        return sql + ";";
    }
}

class Program
{
    static void Main()
    {
        Console.WriteLine("=== ORM CON ATRIBUTOS PERSONALIZADOS ===\n");
        
        var sql = new GeneradorSQL();
        
        var producto = new Producto
        {
            Nombre = "Laptop Gamer",
            Precio = 1500.99m,
            CategoriaId = 1
        };
        
        Console.WriteLine("INSERT generado:");
        Console.WriteLine(sql.GenerarInsert(producto));
        
        Console.WriteLine("\nSELECT generado:");
        Console.WriteLine(sql.GenerarSelect<Producto>("Productos.Precio > 100"));
    }
}
```

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Reflection;
using System.Text.RegularExpressions;

[AttributeUsage(AttributeTargets.Property)]
public class ValidarRequeridoAttribute : Attribute
{
    public string Mensaje { get; set; }
    public ValidarRequeridoAttribute() { Mensaje = "El campo es requerido."; }
    public ValidarRequeridoAttribute(string mensaje) { Mensaje = mensaje; }
}

[AttributeUsage(AttributeTargets.Property)]
public class ValidarRangoAttribute : Attribute
{
    public double Min { get; }

public class ResultadoValidacion
{
    public bool EsValido => Errores.Count == 0;
    public Dictionary<string, List<string>> Errores { get; } = new();
    
    public void AgregarError(string propiedad, string mensaje)
    {
        if (!Errores.ContainsKey(propiedad))
            Errores[propiedad] = new List<string>();
        Errores[propiedad].Add(mensaje);
    }
}

public class ValidadorPorAtributos
{
    public ResultadoValidacion Validar<T>(T objeto)
    {
        var resultado = new ResultadoValidacion();
        var tipo = typeof(T);
        
        foreach (var prop in tipo.GetProperties())
        {
            var valor = prop.GetValue(objeto);
            
            foreach (var atributo in prop.GetCustomAttributes(true))
            {
                if (atributo is ValidarRequeridoAttribute req)
                {
                    if (valor == null || (valor is string s && string.IsNullOrWhiteSpace(s)))
                        resultado.AgregarError(prop.Name, req.Mensaje);
                }
                else if (atributo is ValidarRangoAttribute rango)
                {
                    if (valor is IComparable comparable)
                    {
                        if (comparable.CompareTo(Convert.ChangeType(rango.Min, comparable.GetType())) < 0 ||
                            comparable.CompareTo(Convert.ChangeType(rango.Max, comparable.GetType())) > 0)
                            resultado.AgregarError(prop.Name, rango.Mensaje);
                    }
                }
                else if (atributo is ValidarLongitudAttribute lon)
                {
                    if (valor is string strVal)
                    {
                        if (strVal.Length < lon.Min || strVal.Length > lon.Max)
                            resultado.AgregarError(prop.Name, lon.Mensaje);
                    }
                }
                else if (atributo is ValidarEmailAttribute)
                {
                    if (valor is string email && !string.IsNullOrWhiteSpace(email))
                    {
                        if (!Regex.IsMatch(email, @"^[^@\s]+@[^@\s]+\.[^@\s]+$"))
                            resultado.AgregarError(prop.Name, "El formato del email no es válido.");
                    }
                }
                else if (atributo is ValidarExpresionRegularAttribute regex)
                {
                    if (valor is string str && !string.IsNullOrWhiteSpace(str))
                    {
                        if (!Regex.IsMatch(str, regex.Patron))
                            resultado.AgregarError(prop.Name, regex.Mensaje);
                    }
                }
                else if (atributo is ValidarCompararAttribute comp)
                {
                    var otraProp = tipo.GetProperty(comp.OtraPropiedad);
                    if (otraProp != null)
                    {
                        var otroValor = otraProp.GetValue(objeto);
                        if (!Equals(valor, otroValor))
                            resultado.AgregarError(prop.Name, comp.Mensaje ?? "Los valores no coinciden.");
                    }
                }
            }
        }
        
        return resultado;
    }
}
```

```csharp
public class Interceptor
{
    public static void EjecutarConAtributos(object target, string metodo, params object[] args)
    {
        var tipo = target.GetType();
        var methodInfo = tipo.GetMethod(metodo);
        
        if (methodInfo == null)
        {
            Console.WriteLine($"Método '{metodo}' no encontrado.");
            return;
        }
        
        var timerAttr = methodInfo.GetCustomAttribute<MedirTiempoAttribute>();
        var cacheAttr = methodInfo.GetCustomAttribute<CachearAttribute>();
        var reintentarAttr = methodInfo.GetCustomAttribute<ReintentarAttribute>();
        var logAttr = methodInfo.GetCustomAttribute<RegistrarAttribute>();
        
        if (logAttr != null)
            Console.WriteLine($"[LOG] [{logAttr.Categoria}] Invocando {metodo}");
        
        if (timerAttr != null)
            Console.WriteLine($"[TIMER] Midiendo {timerAttr.NombreOperacion ?? metodo}...");
        
        if (reintentarAttr != null)
            Console.WriteLine($"[RETRY] Máximo {reintentarAttr.Intentos} intentos, espera {reintentarAttr.EsperaMs}ms");
        
        if (cacheAttr != null)
            Console.WriteLine($"[CACHE] Buscando clave '{cacheAttr.Clave ?? metodo}'...");
        
        // Ejecución real
        var sw = System.Diagnostics.Stopwatch.StartNew();
        try
        {
            methodInfo.Invoke(target, args);
            sw.Stop();
            
            if (timerAttr != null)
                Console.WriteLine($"[TIMER] {timerAttr.NombreOperacion ?? metodo}: {sw.ElapsedMilliseconds}ms");
        }
        catch (Exception ex)
        {
            sw.Stop();
            Console.WriteLine($"[ERROR] {metodo}: {ex.InnerException?.Message ?? ex.Message}");
        }
    }
}
```

```csharp
public class ControlSeguridad
{
    private readonly Dictionary<string, List<string>> _rolesPorUsuario = new()
    {
        ["admin"] = new() { "Admin", "Usuario" },
        ["jperez"] = new() { "Usuario" }
    };
    
    private readonly Dictionary<string, int> _contadorLlamadas = new();
    
    public bool EjecutarSiAutorizado(
        object target, string metodo, string usuario, params object[] args)
    {
        var tipo = target.GetType();
        var methodInfo = tipo.GetMethod(metodo);
        
        if (methodInfo == null) return false;
        
        // Verificar autorización
        var autorizar = methodInfo.GetCustomAttribute<AutorizarAttribute>();
        if (autorizar != null && autorizar.Roles.Length > 0)
        {
            if (!_rolesPorUsuario.TryGetValue(usuario, out var roles))
            {
                Console.WriteLine($"  [SEGURIDAD] Usuario '{usuario}' no encontrado.");
                return false;
            }
            
            bool autorizado = false;
            foreach (var rol in autorizar.Roles)
            {
                if (roles.Contains(rol))
                {
                    autorizado = true;
                    break;
                }
            }
            
            if (!autorizado)
            {
                Console.WriteLine($"  [SEGURIDAD] '{usuario}' no tiene rol requerido.");
                return false;
            }
        }
        
        // Verificar tasa límite
        var tasa = methodInfo.GetCustomAttribute<TasaLimitadaAttribute>();
        if (tasa != null)
        {
            string clave = $"{usuario}:{metodo}";
            _contadorLlamadas.TryGetValue(clave, out int llamadas);
            _contadorLlamadas[clave] = ++llamadas;
            
            if (llamadas > tasa.MaxLlamadas)
            {
                Console.WriteLine($"  [TASA] '{usuario}' excedió límite de {metodo}");
                return false;
            }
        }
        
        // Auditoría
        var auditoria = methodInfo.GetCustomAttribute<AuditoriaAttribute>();
        if (auditoria != null)
        {
            Console.WriteLine($"  [AUDITORÍA] {usuario} -> {auditoria.TipoEvento} a las {DateTime.Now:HH:mm:ss}");
        }
        
        // Ejecutar
        methodInfo.Invoke(target, args);
        return true;
    }
}
```

```csharp
public class ConfiguradorAuto
{
    private Dictionary<string, object> _singletons = new();
    
    public void Configurar(object componente)
    {
        var tipo = componente.GetType();
        var compAttr = tipo.GetCustomAttribute<ComponenteAttribute>();
        
        if (compAttr == null) return;
        
        Console.WriteLine($"Configurando componente: {compAttr.Id} ({compAttr.CicloVida})");
        
        // Inyectar dependencias
        foreach (var metodo in tipo.GetMethods())
        {
            var inyectar = metodo.GetCustomAttribute<InyectarAttribute>();
            if (inyectar != null)
            {
                Console.WriteLine($"  Inyectando '{inyectar.NombreServicio}' en {metodo.Name}");
                // En un contenedor real, resolvería la dependencia
                metodo.Invoke(componente, new object[] { $"<servicio:{inyectar.NombreServicio}>" });
            }
        }
        
        // Configurar propiedades
        foreach (var prop in tipo.GetProperties())
        {
            var config = prop.GetCustomAttribute<ConfigurarAttribute>();
            if (config != null)
            {
                string valor = config.ValorDefecto ?? $"<valor para {config.Clave}>";
                Console.WriteLine($"  Configurando {prop.Name} <- {valor}");
                
                if (prop.PropertyType == typeof(string))
                    prop.SetValue(componente, valor);
                else if (prop.PropertyType == typeof(int))
                    prop.SetValue(componente, int.TryParse(valor, out int n) ? n : 0);
            }
        }
    }
}
```

```csharp
    public string GenerarDocumentacion(Type tipo)
    {
        var lines = new List<string>();
        
        var doc = tipo.GetCustomAttribute<DocumentarAttribute>();
        lines.Add($"=== {tipo.Name} ===");
        lines.Add($"Descripción: {doc?.Descripcion ?? "Sin documentación"}");
        
        if (doc?.Categoria != null)
            lines.Add($"Categoría: {doc.Categoria}");
        
        if (doc?.Notas != null)
            lines.Add($"Notas: {doc.Notas}");
        
        if (doc?.Tags != null)
            lines.Add($"Tags: {string.Join(", ", doc.Tags)}");
        
        lines.Add("");
        
        foreach (var prop in tipo.GetProperties())
        {
            var propDoc = prop.GetCustomAttribute<DocumentarAttribute>();
            if (propDoc != null)
            {
                lines.Add($"  [Prop] {prop.Name}: {propDoc.Descripcion}");
                if (propDoc.Ejemplo != null)
                    lines.Add($"    Ej: {propDoc.Ejemplo}");
            }
        }
        
        foreach (var metodo in tipo.GetMethods(BindingFlags.Public | BindingFlags.Instance | BindingFlags.DeclaredOnly))
        {
            var metodoDoc = metodo.GetCustomAttribute<DocumentarAttribute>();
            if (metodoDoc != null)
            {
                var parametros = metodo.GetParameters();
                string paramStr = string.Join(", ",
                    Array.ConvertAll(parametros, p => $"{p.ParameterType.Name} {p.Name}"));
                
                lines.Add($"  [Met] {metodo.ReturnType.Name} {metodo.Name}({paramStr})");
                lines.Add($"    {metodoDoc.Descripcion}");
                
                if (metodoDoc.Ejemplo != null)
                    lines.Add($"    Ej: {metodoDoc.Ejemplo}");
                
                if (metodoDoc.Notas != null)
                    lines.Add($"    Notas: {metodoDoc.Notas}");
            }
        }
        
        return string.Join("\n", lines);
    }
```

================================================================================
  RESUMEN DEL CAPÍTULO
================================================================================

1. Los atributos personalizados heredan de System.Attribute.
2. [AttributeUsage] controla dónde y cómo se aplica el atributo.
3. Los parámetros posicionales van en el constructor (obligatorios).
4. Los parámetros nombrados son propiedades/fields (opcionales).
5. AllowMultiple = true permite múltiples instancias del mismo atributo.
6. Inherited controla si el atributo se hereda a clases derivadas.
7. La recuperación se hace con GetCustomAttribute(s) mediante reflection.
8. Los atributos personalizados son la base de frameworks: validación,
   ORMs, plugins, AOP, seguridad, configuración, etc.

================================================================================
  PREGUNTAS FRECUENTES (FAQ)
================================================================================

P: ¿Los atributos personalizados afectan el rendimiento?
R: Los atributos en sí no. La reflexión para leerlos sí, pero puede
mitigarse cacheando los resultados de GetCustomAttribute.

P: ¿Puedo tener un atributo que ejecute código al aplicarse?
R: No. Los atributos solo almacenan metadatos. El código que responde
a ellos debe ejecutarse explícitamente (por un framework, por reflection,
por un source generator).

P: ¿Debo usar el sufijo "Attribute"?
R: Es la convención. El compilador permite omitirlo al usar el atributo,
pero la clase debe llamarse así. Ej: class MyAttribute : Attribute ->
[My] o [MyAttribute].

P: ¿Puedo tener propiedades de solo lectura en un atributo?
R: Sí, pero solo se pueden establecer mediante parámetros posicionales
del constructor. Los parámetros nombrados requieren setter público.

P: ¿Qué tipos puede tener un parámetro de atributo?
R: bool, byte, short, int, long, float, double, char, string, enum, Type,
y arrays unidimensionales de esos tipos.

  DESAFÍO DEL ARQUITECTO

Diseñe e implemente un framework completo de "Mapeo Objeto-Objeto"
similar a AutoMapper, basado en atributos personalizados:

1. Atributos de configuración:
   - [MapDe(origen)]: en clase destino, indica la clase origen.
   - [Ignorar]: propiedad a ignorar durante el mapeo.
   - [Mapear(origen)]: mapea a propiedad con nombre diferente.
   - [Convertir(origen, tipoConversor)]: aplica un conversor personalizado.
   - [ValorDefecto(valor)]: valor por defecto si origen es null.
   - [DespuesDelMapa(metodo)]: método a ejecutar después del mapeo.

2. Mapeador genérico:
   - TDestino Mapear<TO, TD>(TO origen) donde TD : new()
   - Soporte para mapeo de colecciones (List<T>, arrays, IEnumerable)
   - Soporte para mapeo anidado (propiedades que son objetos)
   - Soporte para mapeo inverso (bidireccional)

3. Conversores incluidos:
   - StringToInt, StringToDateTime, IntToString
   - Conversor de enums (string -> enum, int -> enum)
   - Formato de fechas y números

4. Perfiles de mapeo:
   - Atributo [Perfil(nombre)] para agrupar configuraciones
   - Posibilidad de tener múltiples perfiles para diferentes contextos

5. Validación post-mapeo:
   - Atributo [ValidarDespues] que ejecuta validaciones en el destino
   - Reporte de errores de mapeo

6. Rendimiento:
   - Cache de expresiones de mapeo compiladas (Expression Trees)
   - Cache de PropertyInfo y MethodInfo
   - Uso de CreateDelegate para invocación rápida

Extensión: Integre con el sistema de validación del capítulo anterior
para validar automáticamente los objetos mapeados.
