1. VISIÓN GENERAL DEL PROYECTO
===============================================================================
Vamos a construir un **Sistema Integral de Gestión Educativa** (SIGE) para la
Lucas Ega Academy. Este sistema permitirá gestionar estudiantes, cursos,
calificaciones y generar reportes. Es una aplicación de consola completa que
demuestra todos los conceptos del módulo.

**Funcionalidades del sistema:**

1. **Gestión de Estudiantes**: Agregar, listar, buscar y eliminar estudiantes.
2. **Gestión de Cursos**: Agregar, listar y gestionar cursos disponibles.
3. **Inscripciones**: Inscribir estudiantes en cursos.
4. **Calificaciones**: Asignar y consultar notas de estudiantes.
5. **Reportes**: Generar estadísticas académicas (promedios, aprobados, etc.).
6. **Persistencia**: Guardar y cargar datos (usando archivos de texto).

```csharp
namespace SistemaGestionEducativa.Modelos;

public record Estudiante

    public string Email { get; set; } = "";

    {
        return $"{Id,-4} {Nombre,-20} {Email,-25} {Edad,-5} {FechaRegistro:yyyy-MM-dd}";
    }
    
    public string ToFileString()
    {
        return $"{Id}|{Nombre}|{Email}|{Edad}|{FechaRegistro:yyyy-MM-dd}";

    {
        var partes = linea.Split('|');
        return new Estudiante
        {
            Id = int.Parse(partes[0]),
            Nombre = partes[1],
            Email = partes[2],
            Edad = int.Parse(partes[3]),
            FechaRegistro = DateTime.Parse(partes[4])
        };
```

```csharp
    public int CursoId { get; set; }

        {
            Id = int.Parse(partes[0]),

            CursoId = int.Parse(partes[2]),
            Calificacion = calif,
```

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using SistemaGestionEducativa.Modelos;

namespace SistemaGestionEducativa.Servicios;

public class GestionEstudiantes
{
    private List<Estudiante> _estudiantes = new();
    private int _siguienteId = 1;
    private readonly string _archivo = "estudiantes.txt";
    
    public void Cargar()
    {
        if (!File.Exists(_archivo)) return;
        
        _estudiantes = File.ReadAllLines(_archivo)
            .Where(l => !string.IsNullOrWhiteSpace(l))
            .Select(Estudiante.FromFileString)
            .ToList();
        
        _siguienteId = _estudiantes.Count > 0 ? _estudiantes.Max(e => e.Id) + 1 : 1;
    }
    
    public void Guardar()
    {
        File.WriteAllLines(_archivo, _estudiantes.Select(e => e.ToFileString()));
    }
    
    public Estudiante Agregar(string nombre, string email, int edad)
    {
        var estudiante = new Estudiante
        {
            Id = _siguienteId++,
            Nombre = nombre,
            Email = email,
            Edad = edad
        };
        _estudiantes.Add(estudiante);
        Guardar();
        return estudiante;
    }
    
    public List<Estudiante> Listar() => _estudiantes.ToList();
    
    public Estudiante? Buscar(int id) => _estudiantes.FirstOrDefault(e => e.Id == id);
    
    public Estudiante? Buscar(string nombre)
        => _estudiantes.FirstOrDefault(e =>
            e.Nombre.Contains(nombre, StringComparison.OrdinalIgnoreCase));
    
    public bool Eliminar(int id)
    {
        var estudiante = Buscar(id);
        if (estudiante is null) return false;
        _estudiantes.Remove(estudiante);
        Guardar();
        return true;
    }
    
    public int TotalEstudiantes => _estudiantes.Count;
    
    public (int menores, int adultos, int mayores) EstadisticasEdad()
    {
        int menores = _estudiantes.Count(e => e.Edad < 18);
        int adultos = _estudiantes.Count(e => e.Edad >= 18 && e.Edad < 65);
        int mayores = _estudiantes.Count(e => e.Edad >= 65);
        return (menores, adultos, mayores);
    }
}
```

**Archivo: Servicios/GestionInscripciones.cs**

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using SistemaGestionEducativa.Modelos;

namespace SistemaGestionEducativa.Servicios;

public class GestionInscripciones
{
    private List<Inscripcion> _inscripciones = new();
    private int _siguienteId = 1;
    private readonly string _archivo = "inscripciones.txt";
    
    private readonly GestionEstudiantes _gestionEstudiantes;
    private readonly GestionCursos _gestionCursos;
    
    public GestionInscripciones(GestionEstudiantes gestionEstudiantes, GestionCursos gestionCursos)
    {
        _gestionEstudiantes = gestionEstudiantes;
        _gestionCursos = gestionCursos;
    }
    
    public void Cargar()
    {
        if (!File.Exists(_archivo)) return;
        
        _inscripciones = File.ReadAllLines(_archivo)
            .Where(l => !string.IsNullOrWhiteSpace(l))
            .Select(Inscripcion.FromFileString)
            .ToList();
        
        _siguienteId = _inscripciones.Count > 0 ? _inscripciones.Max(i => i.Id) + 1 : 1;
    }
    
    public void Guardar()
    {
        File.WriteAllLines(_archivo, _inscripciones.Select(i => i.ToFileString()));
    }
    
    public (bool exito, string mensaje) Inscribir(int estudianteId, int cursoId)
    {
        var estudiante = _gestionEstudiantes.Buscar(estudianteId);
        if (estudiante is null)
            return (false, "Estudiante no encontrado");
        
        var curso = _gestionCursos.Buscar(cursoId);
        if (curso is null)
            return (false, "Curso no encontrado");
        
        var inscripcionExistente = _inscripciones
            .FirstOrDefault(i => i.EstudianteId == estudianteId && i.CursoId == cursoId);
        if (inscripcionExistente is not null)
            return (false, "El estudiante ya está inscrito en este curso");
        
        int inscritosEnCurso = _inscripciones.Count(i => i.CursoId == cursoId);
        if (inscritosEnCurso >= curso.CupoMaximo)
            return (false, $"El curso '{curso.Nombre}' ha alcanzado su cupo máximo ({curso.CupoMaximo})");
        
        var inscripcion = new Inscripcion
        {
            Id = _siguienteId++,
            EstudianteId = estudianteId,
            CursoId = cursoId
        };
        _inscripciones.Add(inscripcion);
        Guardar();
        return (true, $"Estudiante '{estudiante.Nombre}' inscrito en '{curso.Nombre}'");
    }
    
    public bool Calificar(int inscripcionId, double calificacion)
    {
        var inscripcion = _inscripciones.FirstOrDefault(i => i.Id == inscripcionId);
        if (inscripcion is null) return false;
        
        if (calificacion < 0 || calificacion > 100)
            return false;
        
        inscripcion = inscripcion with { Calificacion = calificacion };
        var idx = _inscripciones.IndexOf(inscripcion);
        _inscripciones[idx] = inscripcion;
        Guardar();
        return true;
    }
    
    public List<Inscripcion> Listar() => _inscripciones.ToList();
    
    public List<Inscripcion> ListarPorEstudiante(int estudianteId)
        => _inscripciones.Where(i => i.EstudianteId == estudianteId).ToList();
    
    public List<Inscripcion> ListarPorCurso(int cursoId)
        => _inscripciones.Where(i => i.CursoId == cursoId).ToList();
    
    public Inscripcion? Buscar(int id) => _inscripciones.FirstOrDefault(i => i.Id == id);
    
    public (double promedio, int aprobados, int reprobados, int total) ReporteGeneral()
    {
        var conNota = _inscripciones.Where(i => i.Calificacion.HasValue).ToList();
        int total = conNota.Count;
        if (total == 0) return (0, 0, 0, 0);
        
        double promedio = conNota.Average(i => i.Calificacion!.Value);
        int aprobados = conNota.Count(i => i.Calificacion >= 70);
        int reprobados = total - aprobados;
        
        return (promedio, aprobados, reprobados, total);
    }
    
    public List<(string Estudiante, string Curso, double? Nota)> ReporteDetallado()
    {
        return _inscripciones.Select(i =>
        {
            var est = _gestionEstudiantes.Buscar(i.EstudianteId);
            var cur = _gestionCursos.Buscar(i.CursoId);
            return (
                Estudiante: est?.Nombre ?? "Desconocido",
                Curso: cur?.Nombre ?? "Desconocido",
                Nota: i.Calificacion
            );
        }).ToList();
    }
}
```

```csharp
using System;

namespace SistemaGestionEducativa.Utilidades;

public static class ConsolaHelper
{
    public static void EscribirTitulo(string titulo)
    {
        Console.ForegroundColor = ConsoleColor.Cyan;
        Console.WriteLine("╔══════════════════════════════════════╗");
        Console.WriteLine($"║   {titulo.ToUpper(),-33} ║");
        Console.WriteLine("╚══════════════════════════════════════╝");
        Console.ResetColor();
    }
    
    public static void EscribirSubtitulo(string subtitulo)
    {
        Console.ForegroundColor = ConsoleColor.Yellow;
        Console.WriteLine($"\n  ► {subtitulo}");
        Console.ResetColor();
    }
    
    public static void EscribirExito(string mensaje)
    {
        Console.ForegroundColor = ConsoleColor.Green;
        Console.WriteLine($"  ✅ {mensaje}");
        Console.ResetColor();
    }
    
    public static void EscribirError(string mensaje)
    {
        Console.ForegroundColor = ConsoleColor.Red;
        Console.WriteLine($"  ❌ {mensaje}");
        Console.ResetColor();
    }
    
    public static void EscribirInfo(string mensaje)
    {
        Console.ForegroundColor = ConsoleColor.Blue;
        Console.WriteLine($"  ℹ️  {mensaje}");
        Console.ResetColor();
    }
    
    public static void MostrarMenu(string titulo, string[] opciones)
    {
        EscribirTitulo(titulo);
        Console.WriteLine();
        for (int i = 0; i < opciones.Length; i++)
        {
            Console.WriteLine($"  {i + 1}. {opciones[i]}");
        }
        Console.WriteLine($"  0. Volver/Salir");
        Console.WriteLine();
    }
    
    public static string? LeerOpcion()
    {
        Console.Write("  Opción: ");
        return Console.ReadLine()?.Trim();
    }
    
    public static string LeerString(string mensaje)
    {
        string? entrada;
        do
        {
            Console.Write($"  {mensaje}: ");
            entrada = Console.ReadLine()?.Trim();
            if (string.IsNullOrWhiteSpace(entrada))
            {
                EscribirError("La entrada no puede estar vacía.");
            }
        } while (string.IsNullOrWhiteSpace(entrada));
        return entrada;
    }
    
    public static int LeerEntero(string mensaje, int min = int.MinValue, int max = int.MaxValue)
    {
        int valor;
        do
        {
            Console.Write($"  {mensaje} ({min}-{max}): ");
            if (!int.TryParse(Console.ReadLine(), out valor))
            {
                EscribirError("Debes ingresar un número entero.");
                continue;
            }
            if (valor < min || valor > max)
            {
                EscribirError($"El valor debe estar entre {min} y {max}.");
                continue;
            }
            break;
        } while (true);
        return valor;
    }
    
    public static double LeerDouble(string mensaje, double min = double.MinValue, double max = double.MaxValue)
    {
        double valor;
        do
        {
            Console.Write($"  {mensaje} ({min}-{max}): ");
            if (!double.TryParse(Console.ReadLine(), out valor))
            {
                EscribirError("Debes ingresar un número válido.");
                continue;
            }
            if (valor < min || valor > max)
            {
                EscribirError($"El valor debe estar entre {min} y {max}.");
                continue;
            }
            break;
        } while (true);
        return valor;
    }
    
    public static void Pausa()
    {
        Console.WriteLine("\n  Presiona cualquier tecla para continuar...");
        Console.ReadKey(true);
    }
    
    public static void MostrarTabla(string titulo, string[] encabezados, string[][] filas)
    {
        EscribirSubtitulo(titulo);
        
        // Calcular anchos de columna
        int[] anchos = new int[encabezados.Length];
        for (int i = 0; i < encabezados.Length; i++)
        {
            anchos[i] = encabezados[i].Length;
            foreach (var fila in filas)
            {
                if (i < fila.Length && fila[i]?.Length > anchos[i])
                    anchos[i] = fila[i].Length;
            }
            anchos[i] = Math.Min(anchos[i] + 2, 40);
        }
        
        // Línea superior
        Console.Write("  ┌");
        for (int i = 0; i < encabezados.Length; i++)
        {
            Console.Write(new string('─', anchos[i]));
            Console.Write(i < encabezados.Length - 1 ? "┬" : "┐");
        }
        Console.WriteLine();
        
        // Encabezados
        Console.Write("  │");
        for (int i = 0; i < encabezados.Length; i++)
        {
            Console.Write($" {encabezados[i].PadRight(anchos[i] - 1)}│");
        }
        Console.WriteLine();
        
        // Línea separadora
        Console.Write("  ├");
        for (int i = 0; i < encabezados.Length; i++)
        {
            Console.Write(new string('─', anchos[i]));
            Console.Write(i < encabezados.Length - 1 ? "┼" : "┤");
        }
        Console.WriteLine();
        
        // Filas
        foreach (var fila in filas)
        {
            Console.Write("  │");
            for (int i = 0; i < encabezados.Length; i++)
            {
                string valor = i < fila.Length ? fila[i] ?? "" : "";
                Console.Write($" {valor.PadRight(anchos[i] - 1)}│");
            }
            Console.WriteLine();
        }
        
        // Línea inferior
        Console.Write("  └");
        for (int i = 0; i < encabezados.Length; i++)
        {
            Console.Write(new string('─', anchos[i]));
            Console.Write(i < encabezados.Length - 1 ? "┴" : "┘");
        }
        Console.WriteLine();
    }
}
```

```csharp
using System;

void AsignarCalificacion()
{
    Console.Clear();
    EscribirTitulo("ASIGNAR CALIFICACIÓN");
    
    var inscripciones = gestionInscripciones.Listar()
        .Where(i => !i.Calificacion.HasValue)
        .ToList();
    
    if (inscripciones.Count == 0)
    {
        EscribirInfo("No hay inscripciones pendientes de calificar.");
        Pausa();
        return;
    }
    
    var filas = inscripciones.Select(i => {
        var est = gestionEstudiantes.Buscar(i.EstudianteId);
        var cur = gestionCursos.Buscar(i.CursoId);
        return new[] {
            i.Id.ToString(),
            est?.Nombre ?? "?",
            cur?.Nombre ?? "?",
            "Pendiente"
        };
    }).ToArray();
    
    MostrarTabla("INSCRIPCIONES PENDIENTES",
        new[] { "ID Insc.", "Estudiante", "Curso", "Estado" },
        filas);
    
    int idInscripcion = LeerEntero("ID de la inscripción a calificar", 1);
    double calificacion = LeerDouble("Calificación (0-100)", 0, 100);
    
    if (gestionInscripciones.Calificar(idInscripcion, calificacion))
        EscribirExito("Calificación asignada exitosamente.");

        EscribirError("No se pudo asignar la calificación. Verifica el ID.");
    
    Pausa();
}

{

    if (estudiante is null)
    {
        EscribirError("Estudiante no encontrado.");
        Pausa();
        return;
    }
    
    var inscripciones = gestionInscripciones.ListarPorEstudiante(estudianteId);
    
    if (inscripciones.Count == 0)
    {
        EscribirInfo($"El estudiante '{estudiante.Nombre}' no tiene inscripciones.");
        Pausa();
        return;
    }
    
    var filas = inscripciones.Select(i => {
        var cur = gestionCursos.Buscar(i.CursoId);
        return new[] {
            cur?.Nombre ?? "?",
            i.Calificacion.HasValue ? i.Calificacion.Value.ToString("F1") : "Pendiente",
            i.Calificacion.HasValue ? (i.Calificacion >= 70 ? "✅" : "❌") : "—"

    }).ToArray();
    
    MostrarTabla($"CALIFICACIONES DE {estudiante.Nombre.ToUpper()}",
        new[] { "Curso", "Nota", "Estado" },
        filas);
    
    double promedio = inscripciones.Where(i => i.Calificacion.HasValue)
        .DefaultIfEmpty()
        .Average(i => i?.Calificacion ?? 0);
    
    int calificadas = inscripciones.Count(i => i.Calificacion.HasValue);
    Console.WriteLine($"\n  Promedio: {promedio:F1}  (basado en {calificadas} calificaciones)");

{
    Console.Clear();
    EscribirTitulo("ESTUDIANTES POR RANGO DE EDAD");
    
    var estudiantes = gestionEstudiantes.Listar();
    if (estudiantes.Count == 0)
    {
        EscribirInfo("No hay estudiantes registrados.");
        Pausa();
        return;
    }
    
    var rangos = estudiantes.GroupBy(e => e.Edad switch
    {
        < 18 => "Menores (< 18)",
        < 25 => "Jóvenes (18-24)",
        < 40 => "Adultos (25-39)",
        < 60 => "Adultos maduros (40-59)",
        _ => "Adultos mayores (60+)"
    }).OrderBy(g => g.Key);
    
    Console.WriteLine();
    foreach (var rango in rangos)
    {
        double porcentaje = (double)rango.Count() / estudiantes.Count * 100;
        int barras = (int)(porcentaje / 5);
        Console.WriteLine($"  {rango.Key,-25} {rango.Count(),3}  {new string('█', barras)} {porcentaje:F0}%");
    }
    
    Console.WriteLine($"\n  Total: {estudiantes.Count} estudiantes");
    Pausa();
}
```

```csharp
void ExportarCSV(string archivo, string[] encabezados, string[][] filas)
{
```

- La arquitectura sigue el principio de separación de responsabilidades:
  Modelos (datos), Servicios (lógica de negocio), Utilidades (helpers).
- Los modelos usan records de C# 9+ con propiedades inmutables (init) para
  ID y fechas.
- Los servicios gestionan la lógica de negocio con persistencia en archivos
  de texto (formato delimitado por pipes).
- La clase ConsolaHelper proporciona una interfaz de usuario profesional con
  tablas, colores y validación.
- El menú principal navega a submódulos: Estudiantes, Cursos, Inscripciones,
  Calificaciones y Reportes.
- La validación de entrada usa bucles do-while con TryParse para garantizar
  datos correctos.
- Los reportes incluyen resumen general, distribución por edad, top cursos y
  rendimiento académico.
- Los datos se guardan automáticamente al realizar cambios y se cargan al
  iniciar el programa.
- El proyecto es extensible: se puede agregar exportación CSV, base de datos
  SQLite, interfaz web, etc.
- La autoevaluación ayuda a verificar que todos los conceptos del módulo se
  han aplicado correctamente.

PREGUNTAS FRECUENTES
===============================================================================
❓ **¿Dónde se guardan los datos del programa?**
En archivos de texto en el mismo directorio del proyecto: `estudiantes.txt`,
`cursos.txt`, `inscripciones.txt`. Si los archivos no existen, se crean
automáticamente.

❓ **¿Qué pasa si modifico manualmente los archivos de datos?**
El programa lee los archivos al iniciar. Si modificas el formato incorrectamente,
puede fallar la carga. Siempre haz una copia de seguridad antes de editar
manualmente.

❓ **¿Puedo compartir los datos entre diferentes ejecuciones?**
Sí. Los datos persisten en los archivos de texto. Puedes cerrar y abrir el
programa y los datos se mantendrán.

❓ **¿El proyecto funciona en Linux, Mac y Windows?**
Sí. .NET 8 es multiplataforma. Solo asegúrate de tener las fuentes Unicode
instaladas para los caracteres especiales (╔, ╗, etc.).

❓ **¿Cómo agrego más funcionalidades al menú?**
Agrega nuevas opciones al array del menú principal y añade un case en el
switch correspondiente. Luego implementa el método con la lógica nueva.

❓ **¿Qué significan los archivos con formato | (pipe)?**
Es un formato de texto delimitado similar a CSV pero usando el carácter | en
lugar de coma. Cada línea es un registro y cada campo está separado por |.

❓ **¿Puedo cambiar el sistema de archivos por una base de datos?**
Sí. La arquitectura está diseñada para ser modular. Crea una clase
`RepositorioSQLite` que implemente los mismos métodos que los servicios
actuales y reemplaza la persistencia.

❓ **¿Por qué usar records en lugar de clases?**
Los records proporcionan: igualdad por valor (comparación automática),
inmutabilidad con init, toString automático y deconstrucción. Son ideales
para modelos de datos.

❓ **¿Cómo maneja el programa los errores de archivo?**
El programa usa File.Exists antes de leer y captura excepciones básicas. En
una versión de producción, deberías agregar try-catch más robustos.

❓ **¿Puedo usar este proyecto como base para mi portafolio?**
¡Absolutamente! Es un proyecto completo que demuestra todos los fundamentos
de C#. Agrégale las mejoras sugeridas y será un excelente portafolio.

🏗️ **Diseña una Evolución del Sistema (SIGE v2.0)**

Como arquitecto de software, tu desafío es rediseñar el Sistema de Gestión
Educativa para una universidad real con miles de estudiantes. La versión 2.0
debe:

1. **Arquitectura en capas**: UI → Servicios → Repositorios → Base de datos.
2. **Persistencia con SQL Server/PostgreSQL**: Usar Entity Framework Core.
3. **API RESTful**: Exponer funcionalidades como endpoints REST.
4. **Autenticación y autorización**: Roles (admin, profesor, estudiante).
5. **Notificaciones**: Email automático al inscribir, calificar, etc.
6. **Dashboard en tiempo real**: Estadísticas actualizadas al instante.
7. **Módulo de pagos**: Gestionar matrículas y cuotas.
8. **Historial académico completo**: Todas las calificaciones por semestre.
9. **Reportes avanzados**: Gráficos, PDF, Excel exportable.
10. **Escalabilidad**: Soportar 10,000+ estudiantes concurrentes.

Consideraciones arquitectónicas:
- ¿Microservicios o monolito modular?
- ¿CQRS (Command Query Responsibility Segregation)?
- ¿Event Sourcing para el historial académico?
- ¿Cache distribuida (Redis) para consultas frecuentes?
- ¿Almacenamiento de archivos (PDF, imágenes) en la nube?
- ¿Frontend (Blazor, React, Angular) separado del backend?
- ¿Contenedores Docker para despliegue?
- ¿CI/CD con GitHub Actions o Azure DevOps?

Diseña la arquitectura completa: diagrama de componentes, flujo de datos,
modelo de datos relacional, interfaces de API REST y estrategia de despliegue.
Proporciona ejemplos de código para el patrón Repository y Unit of Work.
