```csharp
// ============ Models/Empleado.cs ============
using System;
using SistemaGestionEmpleados.Interfaces;

namespace SistemaGestionEmpleados.Models
{
    public abstract class Empleado : ICalculable, IReportable, IAlmacenable, IDisposable
    {
        private static int _contadorId = 1000;
        private string _nombre;
        private string _apellido;
        private string _email;
        private decimal _salarioBase;
        private bool _activo;
        private bool _disposed;

        public int Id { get; }
        public DateTime FechaCreacion { get; }
        public DateTime? FechaDesactivacion { get; private set; }

        public string Nombre
        {
            get => _nombre;
            set
            {
                if (string.IsNullOrWhiteSpace(value))
                    throw new ArgumentException("El nombre no puede estar vacio.");
                if (value.Length < 2)
                    throw new ArgumentException("El nombre debe tener al menos 2 caracteres.");
                _nombre = value;
            }
        }

        public string Apellido
        {
            get => _apellido;
            set
            {
                if (string.IsNullOrWhiteSpace(value))
                    throw new ArgumentException("El apellido no puede estar vacio.");
                _apellido = value;
            }
        }

        public string NombreCompleto => $"{_nombre} {_apellido}";

        public string Email
        {
            get => _email;
            set
            {
                if (!string.IsNullOrEmpty(value) && !value.Contains("@"))
                    throw new ArgumentException("Email invalido.");
                _email = value;
            }
        }

        public decimal SalarioBase
        {
            get => _salarioBase;
            set
            {
                if (value < 0)
                    throw new ArgumentException("El salario base no puede ser negativo.");
                _salarioBase = value;
            }
        }

        public int AntiguedadAnios => DateTime.Now.Year - FechaCreacion.Year;

        public bool EstaActivo
        {
            get => _activo;
            private set => _activo = value;
        }

        public string Estado => _activo ? "Activo" : "Inactivo";

        // Constructor protegido (solo clases derivadas)
        protected Empleado(string nombre, string apellido, string email, decimal salarioBase)
        {
            Id = ++_contadorId;
            FechaCreacion = DateTime.Now;
            _activo = true;

            Nombre = nombre;
            Apellido = apellido;
            Email = email;
            SalarioBase = salarioBase;

            Console.WriteLine($"[CREAR] Empleado {Id}: {NombreCompleto}");
        }

        // Metodos abstractos
        public abstract decimal CalcularPagoMensual();
        public abstract string ObtenerDescripcion();

        // Metodos virtuales
        public virtual decimal CalcularImpuesto()
        {
            return CalcularPagoMensual() * 0.16m;
        }

        public virtual decimal CalcularBono()
        {
            if (AntiguedadAnios < 1) return 0;
            return SalarioBase * 0.05m * AntiguedadAnios;
        }

        public virtual string GenerarReporte()
        {
            return $"=== REPORTE EMPLEADO ===\n" +
                   $"ID: {Id}\n" +
                   $"Nombre: {NombreCompleto}\n" +
                   $"Email: {_email ?? "N/A"}\n" +
                   $"Salario Base: ${SalarioBase:F2}\n" +
                   $"Pago Mensual: ${CalcularPagoMensual():F2}\n" +
                   $"Bono: ${CalcularBono():F2}\n" +
                   $"Impuesto: ${CalcularImpuesto():F2}\n" +
                   $"Antiguedad: {AntiguedadAnios} anios\n" +
                   $"Estado: {Estado}";
        }

        public string ObtenerResumen()
        {
            return $"[{Id}] {NombreCompleto} - ${CalcularPagoMensual():F2}/mes - {Estado}";
        }

        public void Desactivar()
        {
            if (!_activo) return;
            _activo = false;
            FechaDesactivacion = DateTime.Now;
            Console.WriteLine($"[DESACTIVAR] Empleado {Id}: {NombreCompleto}");
        }

        public void Activar()
        {
            if (_activo) return;
            _activo = true;
            FechaDesactivacion = null;
            Console.WriteLine($"[ACTIVAR] Empleado {Id}: {NombreCompleto}");
        }

        // Implementacion de IDisposable
        public void Dispose()
        {
            Dispose(true);
            GC.SuppressFinalize(this);
        }

        protected virtual void Dispose(bool disposing)
        {
            if (_disposed) return;
            if (disposing)
            {
                Console.WriteLine($"[DISPOSE] Limpiando recursos de empleado {Id}");
            }
            _disposed = true;
        }

        ~Empleado()
        {
            Console.WriteLine($"[FINALIZADOR] Empleado {Id}: {NombreCompleto}");
            Dispose(false);
        }

        public override string ToString() => ObtenerResumen();
    }
}
```

```csharp
using System;

namespace SistemaGestionEmpleados.Models
{
    public sealed class EmpleadoTiempoCompleto : Empleado
    {
        public decimal BonoAnual { get; set; }
        public string Puesto { get; set; }

        public EmpleadoTiempoCompleto(string nombre, string apellido, string email,
            decimal salarioBase, string puesto, decimal bonoAnual)
            : base(nombre, apellido, email, salarioBase)
        {
            Puesto = puesto;
            BonoAnual = bonoAnual >= 0 ? bonoAnual :
                throw new ArgumentException("El bono anual no puede ser negativo.");
        }

        public override decimal CalcularPagoMensual()
        {
            return (SalarioBase + BonoAnual) / 12;
        }

        public override decimal CalcularBono()
        {
            return base.CalcularBono() + (BonoAnual * 0.1m);
        }

        public override string ObtenerDescripcion()
        {
            return $"Empleado Tiempo Completo - {Puesto}";
        }

        public override string GenerarReporte()
        {
            return base.GenerarReporte() +
                   $"\nTipo: Tiempo Completo\n" +
                   $"Puesto: {Puesto}\n" +
                   $"Bono Anual: ${BonoAnual:F2}";
        }
    }
}
```

```csharp
using System;

namespace SistemaGestionEmpleados.Models
{
    public sealed class EmpleadoPorHoras : Empleado
    {
        private decimal _tarifaPorHora;
        private int _horasSemanales;

        public decimal TarifaPorHora
        {
            get => _tarifaPorHora;
            set
            {
                if (value <= 0)
                    throw new ArgumentException("La tarifa debe ser positiva.");
                _tarifaPorHora = value;
            }
        }

        public int HorasSemanales
        {
            get => _horasSemanales;
            set
            {
                if (value < 1 || value > 80)
                    throw new ArgumentException("Las horas semanales deben estar entre 1 y 80.");
                _horasSemanales = value;
            }
        }

        public EmpleadoPorHoras(string nombre, string apellido, string email,
            decimal tarifaPorHora, int horasSemanales)
            : base(nombre, apellido, email, 0)
        {
            TarifaPorHora = tarifaPorHora;
            HorasSemanales = horasSemanales;
        }

        public override decimal CalcularPagoMensual()
        {
            return TarifaPorHora * HorasSemanales * 4; // 4 semanas
        }

        public override decimal CalcularBono()
        {
            return 0; // Los empleados por horas no reciben bono
        }

        public override decimal CalcularImpuesto()
        {
            return CalcularPagoMensual() * 0.10m; // Impuesto reducido
        }

        public override string ObtenerDescripcion()
        {
            return $"Empleado por Horas - ${TarifaPorHora:F2}/h, {HorasSemanales}h/sem";
        }

        public override string GenerarReporte()
        {
            return base.GenerarReporte() +
                   $"\nTipo: Por Horas\n" +
                   $"Tarifa: ${TarifaPorHora:F2}/h\n" +
                   $"Horas/semana: {HorasSemanales}";
        }
    }
}
```

```csharp
using System;

namespace SistemaGestionEmpleados.Models
{
    public sealed class EmpleadoPorComision : Empleado
    {
        private decimal _ventasRealizadas;
        private decimal _porcentajeComision;

        public decimal VentasRealizadas
        {
            get => _ventasRealizadas;
            set
            {
                if (value < 0)
                    throw new ArgumentException("Las ventas no pueden ser negativas.");
                _ventasRealizadas = value;
            }
        }

        public decimal PorcentajeComision
        {
            get => _porcentajeComision;
            set
            {
                if (value < 0 || value > 0.5m)
                    throw new ArgumentException("La comision debe estar entre 0% y 50%.");
                _porcentajeComision = value;
            }
        }

        public EmpleadoPorComision(string nombre, string apellido, string email,
            decimal salarioBase, decimal ventasRealizadas, decimal porcentajeComision)
            : base(nombre, apellido, email, salarioBase)
        {
            VentasRealizadas = ventasRealizadas;
            PorcentajeComision = porcentajeComision;
        }

        public override decimal CalcularPagoMensual()
        {
            return SalarioBase + (VentasRealizadas * PorcentajeComision);
        }

        public override decimal CalcularBono()
        {
            if (VentasRealizadas > 100000)
                return base.CalcularBono() * 2;
            return base.CalcularBono();
        }

        public override string ObtenerDescripcion()
        {
            return $"Empleado por Comision - {PorcentajeComision:P} de ventas";
        }

        public override string GenerarReporte()
        {
            return base.GenerarReporte() +
                   $"\nTipo: Por Comision\n" +
                   $"Ventas: ${VentasRealizadas:F2}\n" +
                   $"Comision: {PorcentajeComision:P}";
        }
    }
```

```csharp
// ============ Models/Departamento.cs ============
using System;
using System.Collections.Generic;
using System.Linq;

namespace SistemaGestionEmpleados.Models
{
    public sealed class Departamento
    {
        private readonly List<Empleado> _empleados = new List<Empleado>();

        public string Nombre { get; }
        public string Codigo { get; }
        public Gerente Director { get; set; }

        public int TotalEmpleados => _empleados.Count;
        public int EmpleadosActivos => _empleados.Count(e => e.EstaActivo);

        public Departamento(string nombre, string codigo)
        {
            Nombre = nombre ?? throw new ArgumentNullException(nameof(nombre));
            Codigo = codigo ?? throw new ArgumentNullException(nameof(codigo));
        }

        public void AgregarEmpleado(Empleado empleado)
        {
            if (empleado == null)
                throw new ArgumentNullException(nameof(empleado));
            _empleados.Add(empleado);
            Console.WriteLine($"[DEPTO] {empleado.NombreCompleto} agregado a {Nombre}");
        }

        public bool RemoverEmpleado(int id)
        {
            var emp = _empleados.FirstOrDefault(e => e.Id == id);
            if (emp == null) return false;
            _empleados.Remove(emp);
            emp.Desactivar();
            return true;
        }

        public Empleado BuscarEmpleado(int id) =>
            _empleados.FirstOrDefault(e => e.Id == id);

        public Empleado BuscarEmpleado(string nombreCompleto) =>
            _empleados.FirstOrDefault(e =>
                e.NombreCompleto.Contains(nombreCompleto, StringComparison.OrdinalIgnoreCase));

        public List<Empleado> ObtenerEmpleados() =>
            _empleados.ToList(); // Copia defensiva

        public decimal CalcularNominaTotal()
        {
            return _empleados.Where(e => e.EstaActivo)
                             .Sum(e => e.CalcularPagoMensual());
        }

        public void MostrarEmpleados()
        {
            Console.WriteLine($"\n=== DEPARTAMENTO: {Nombre} ({Codigo}) ===");
            if (Director != null)
                Console.WriteLine($"Director: {Director.NombreCompleto}");
            Console.WriteLine($"Total empleados: {TotalEmpleados} (Activos: {EmpleadosActivos})");
            Console.WriteLine($"Nomina mensual: ${CalcularNominaTotal():F2}");

            if (_empleados.Count == 0)
            {
                Console.WriteLine("No hay empleados en este departamento.");
                return;
            }

            foreach (var emp in _empleados)
            {
                Console.WriteLine($"  {emp.ObtenerResumen()}");
            }
        }
    }
}
```

```csharp
// ============ Models/Empresa.cs ============
using System;
using System.Collections.Generic;
using System.Linq;

namespace SistemaGestionEmpleados.Models
{
    public sealed class Empresa
    {
        private readonly List<Departamento> _departamentos = new List<Departamento>();
        private readonly List<Empleado> _todosEmpleados = new List<Empleado>();

        public string Nombre { get; }
        public string RUC { get; }
        public int TotalDepartamentos => _departamentos.Count;
        public int TotalEmpleados => _todosEmpleados.Count;
        public int EmpleadosActivos => _todosEmpleados.Count(e => e.EstaActivo);

        public Empresa(string nombre, string ruc)
        {
            Nombre = nombre ?? throw new ArgumentNullException(nameof(nombre));
            RUC = ruc ?? throw new ArgumentNullException(nameof(ruc));
        }

        public void AgregarDepartamento(Departamento depto)
        {
            if (depto == null)
                throw new ArgumentNullException(nameof(depto));
            _departamentos.Add(depto);
            Console.WriteLine($"[EMPRESA] Departamento '{depto.Nombre}' agregado.");
        }

        public void AgregarEmpleado(Empleado empleado, string codigoDepartamento)
        {
            if (empleado == null)
                throw new ArgumentNullException(nameof(empleado));

            var depto = _departamentos.FirstOrDefault(d => d.Codigo == codigoDepartamento);
            if (depto == null)
                throw new ArgumentException($"Departamento {codigoDepartamento} no encontrado.");

            depto.AgregarEmpleado(empleado);
            _todosEmpleados.Add(empleado);
        }

        public void MostrarEstructura()
        {
            Console.WriteLine($"\n{'='.PadRight(60, '=')}");
            Console.WriteLine($"EMPRESA: {Nombre} (RUC: {RUC})");
            Console.WriteLine($"Total Departamentos: {TotalDepartamentos}");
            Console.WriteLine($"Total Empleados: {TotalEmpleados} (Activos: {EmpleadosActivos})");
            Console.WriteLine($"Nomina Global: ${CalcularNominaGlobal():F2}");
            Console.WriteLine($"{'='.PadRight(60, '=')}");

            foreach (var depto in _departamentos)
            {
                depto.MostrarEmpleados();
            }
        }

        public decimal CalcularNominaGlobal()
        {
            return _departamentos.Sum(d => d.CalcularNominaTotal());
        }

        public Departamento BuscarDepartamento(string codigo) =>
            _departamentos.FirstOrDefault(d => d.Codigo == codigo);

        public Empleado BuscarEmpleado(int id) =>
            _todosEmpleados.FirstOrDefault(e => e.Id == id);

        public List<Empleado> BuscarPorTipo<T>() where T : Empleado =>
            _todosEmpleados.Where(e => e is T && e.EstaActivo).ToList();

        public void GenerarReporteCompleto()
        {
            Console.WriteLine($"\n{'='.PadRight(70, '=')}");
            Console.WriteLine($"REPORTE COMPLETO DE NOMINA - {Nombre}");
            Console.WriteLine($"Fecha: {DateTime.Now:dd/MM/yyyy HH:mm}");
            Console.WriteLine($"{'='.PadRight(70, '=')}");

            foreach (var emp in _todosEmpleados.Where(e => e.EstaActivo))
            {
                Console.WriteLine();
                Console.WriteLine(emp.GenerarReporte());
            }

            Console.WriteLine($"\n{'='.PadRight(70, '=')}");
            Console.WriteLine($"TOTAL NOMINA MENSUAL: ${CalcularNominaGlobal():F2}");
            Console.WriteLine($"TOTAL EMPLEADOS: {EmpleadosActivos}");
            Console.WriteLine($"{'='.PadRight(70, '=')}");
        }
    }
}
```

```csharp
using System;

        public static void MostrarEstadisticas(Empresa empresa)

            Console.WriteLine($"Total empleados: {empresa.TotalEmpleados}");

            Console.WriteLine($"Departamentos: {empresa.TotalDepartamentos}");
            Console.WriteLine($"Nomina mensual: ${empresa.CalcularNominaGlobal():F2}");
            Console.WriteLine($"Nomina anual estimada: ${empresa.CalcularNominaGlobal() * 12:F2}");

            var tc = empresa.BuscarPorTipo<EmpleadoTiempoCompleto>();
            var ph = empresa.BuscarPorTipo<EmpleadoPorHoras>();
            var com = empresa.BuscarPorTipo<EmpleadoPorComision>();
            Console.WriteLine($"\nPor tipo:");
            Console.WriteLine($"  Tiempo Completo: {tc.Count}");
            Console.WriteLine($"  Por Horas: {ph.Count}");
            Console.WriteLine($"  Por Comision: {com.Count}");
        }

        public static void SimularOperaciones(Empresa empresa)
        {
            Console.WriteLine("\n=== SIMULACION DE OPERACIONES ===");

            var activos = empresa.BuscarPorTipo<Empleado>();

            if (activos.Count == 0)
            {
                Console.WriteLine("No hay empleados activos para simular.");
                return;
            }

            foreach (var emp in activos)
            {
                Console.WriteLine($"\nProcesando: {emp.ObtenerResumen()}");
                Console.WriteLine($"  Pago mensual: ${emp.CalcularPagoMensual():F2}");
                Console.WriteLine($"  Bono: ${emp.CalcularBono():F2}");
                Console.WriteLine($"  Impuesto: ${emp.CalcularImpuesto():F2}");
                Console.WriteLine($"  Neto: ${emp.CalcularPagoMensual() - emp.CalcularImpuesto():F2}");
```

```csharp
        static void MostrarEmpleados(Empresa empresa, bool soloActivos)
        {
            string estado = soloActivos ? "ACTIVOS" : "INACTIVOS";
            Console.WriteLine($"\n=== EMPLEADOS {estado} ===");

            var empleados = empresa.BuscarPorTipo<Empleado>();
            if (!soloActivos)
            {
                empleados = empresa.BuscarPorTipo<Empleado>();
                // BuscarPorTipo solo devuelve activos, asi que
                // para inactivos filtramos de otra forma:
                empleados.Clear();
            }

            if (empleados.Count == 0)
            {
                Console.WriteLine($"No hay empleados {estado.ToLower()}.");
                return;
            }

            foreach (var emp in empleados.OrderBy(e => e.Id))
            {
                Console.WriteLine($"  {emp.ObtenerResumen()} - {emp.ObtenerDescripcion()}");
            }
            Console.WriteLine($"Total: {empleados.Count}");
```

================================================================================
  PREGUNTAS FRECUENTES (FAQ)
================================================================================

  ───
  ❓ ¿Puedo agregar más tipos de empleados?
  ───
  Sí. Crea una nueva clase que herede de Empleado e implemente los
  metodos abstractos. El sistema aceptara el nuevo tipo sin cambios
  (polimorfismo + Open/Closed).

  ───
  ❓ ¿Cómo puedo persistir los datos?
  ───
  Implementa IAlmacenable con metodos Guardar() y Cargar(). Usa
  System.Text.Json o XmlSerializer.

  ───
  ❓ ¿Por qué algunas clases son sealed?
  ───
  Para indicar que no deben heredarse mas alla de ese nivel. Es una
  decision de diseno que evita jerarquias demasiado profundas.

  ───
  ❓ ¿El finalizador se ejecuta realmente?
  ───
  Si no se llama a Dispose(), el GC ejecutara el finalizador en algun
  momento futuro. En este proyecto, el finalizador es demostrativo.

  ───
  ❓ ¿Puedo convertir esto a una API web?
  ───
  Si. Los modelos son reutilizables. Agrega controllers ASP.NET Core
  y expone los endpoints.
