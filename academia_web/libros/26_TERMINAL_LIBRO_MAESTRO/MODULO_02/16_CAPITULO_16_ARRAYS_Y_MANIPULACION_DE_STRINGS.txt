1. ARRAYS

array=(elem1 elem2 elem3). ${array[0]} primer elemento. ${array[@]} todos.

${#array[@]} longitud. array+=(elem4) agregar. unset array[1] eliminar.



2. ARRAYS ASOCIATIVOS (Bash 4+)

declare -A dict. dict[clave]=valor. ${!dict[@]} claves. ${dict[@]} valores.

Útil para configuraciones y mapeos.



3. MANIPULACIÓN DE STRINGS

${#str} longitud. ${str:pos:len} substring. ${str#patron} eliminar prefijo.

${str%patron} eliminar sufijo. ${str/old/new} reemplazar primero.

${str//old/new} reemplazar todos.



4. EXPRESIONES REGULARES

[[ $str =~ ^[0-9]+$ ]] valida número. =~ operador regex. BASH_REMATCH

contiene grupos capturados ($ {BASH_REMATCH[1]}).

EJEMPLO DE CÓDIGO
----------------------------------------

#!/bin/bash
frutas=(manzana pera uva fresa)
echo "Primera: ${frutas[0]}"
echo "Todas: ${frutas[@]}"
echo "Total: ${#frutas[@]}"

fecha="2024-01-15"
echo "Anio: ${fecha:0:4}"
echo "Mes: ${fecha:5:2}"
echo "Dia: ${fecha:8:2}"

[[ "$1" =~ ^[0-9]+$ ]] && echo "Es numero" || echo "No es numero"
