API Resources: Resource, ResourceCollection.

Conditional attributes. Pagination.

Fractal anterior, preferir resources.

EJEMPLO DE CÓDIGO
----------------------------------------

class UserResource extends JsonResource {
  public function toArray(Request $request): array {
    return [
      "id" => $this->id,
      "name" => $this->name,
      "email" => $this->email,
      "posts_count" => $this->whenCounted("posts"),
      "last_login" => $this->last_login?->diffForHumans(),
    ];
  }
}

Route::get("/users", fn() => UserResource::collection(User::paginate()));
