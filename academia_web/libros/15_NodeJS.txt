¿Qué es Node.js?
----------------
Node.js es un entorno de ejecución de JavaScript basado en el motor V8 de Google
Chrome. Permite ejecutar JavaScript en el servidor, fuera del navegador. Fue
creado por Ryan Dahl en 2009. Es asíncrono, orientado a eventos y muy eficiente
para aplicaciones I/O intensivas (entrada/salida de datos).

Características principales:
- Asíncrono y no bloqueante (event loop).
- Módulos (CommonJS y ES Modules).
- NPM (el gestor de paquetes más grande del mundo).
- Ideal para APIs, microservicios, websockets, y aplicaciones en tiempo real.
- Gran ecosistema de bibliotecas.

Filosofía: "Pequeño y modular" - Node.js promueve construir aplicaciones con
módulos pequeños y especializados.

Versiones: Node.js 20 LTS (Long-Term Support) y Node.js 22 (2024) son las
versiones actuales.

CAPITULO 4: SISTEMA DE ARCHIVOS (FS)
================================================================================

El módulo fs permite interactuar con el sistema de archivos.

const fs = require('fs');

Leer archivo (síncrono - bloquea):
---------------------------------
try {
    const data = fs.readFileSync('archivo.txt', 'utf8');
    console.log(data);
} catch (err) {
    console.error('Error:', err);
}

Leer archivo (asíncrono con callback):
--------------------------------------
fs.readFile('archivo.txt', 'utf8', (err, data) => {
    if (err) {
        console.error('Error:', err);
        return;
    }
    console.log(data);
});

Leer archivo (con promesas - fs/promises - recomendado):
--------------------------------------------------------
const fsPromises = require('fs/promises');

async function leerArchivo() {
    try {
        const data = await fsPromises.readFile('archivo.txt', 'utf8');
        console.log(data);
    } catch (err) {
        console.error('Error:', err);
    }
}
leerArchivo();

Escribir archivo:
-----------------
// Síncrono:
fs.writeFileSync('salida.txt', 'Contenido', 'utf8');

// Asíncrono con callback:
fs.writeFile('salida.txt', 'Contenido', 'utf8', (err) => {
    if (err) console.error(err);
    else console.log('Archivo guardado');
});

// Con promesas:
await fsPromises.writeFile('salida.txt', 'Contenido');

Añadir al final (append):
-------------------------
await fsPromises.appendFile('log.txt', 'Nueva línea\n');

Verificar existencia y estadísticas:
------------------------------------
const existe = await fsPromises.exists('archivo.txt');
const stats = await fsPromises.stat('archivo.txt');
console.log(stats.isFile());    // true
console.log(stats.size);        // tamaño en bytes
console.log(stats.mtime);       // fecha de modificación

Operaciones con directorios:
----------------------------
await fsPromises.mkdir('carpeta', { recursive: true });
const archivos = await fsPromises.readdir('.');
await fsPromises.rmdir('carpeta');
await fsPromises.unlink('archivo.txt');  // eliminar archivo

CAPITULO 5: EVENTOS Y EVENTEMITTER
================================================================================

Node.js se basa en eventos. El módulo events proporciona EventEmitter.

Crear emisor y escuchar:
------------------------
const EventEmitter = require('events');
const miEmisor = new EventEmitter();

// Registrar evento
miEmisor.on('saludo', (nombre) => {
    console.log(`¡Hola ${nombre}!`);
});

// Emitir evento
miEmisor.emit('saludo', 'Esteban');  // "¡Hola Esteban!"

Evento con una sola escucha (once):
-----------------------------------
miEmisor.once('unico', () => {
    console.log('Solo una vez');
});
miEmisor.emit('unico');  // funciona
miEmisor.emit('unico');  // no pasa nada

Clase que extiende EventEmitter:
--------------------------------
class Reloj extends EventEmitter {
    iniciar() {
        this.timer = setInterval(() => {
            this.emit('tick', new Date());
        }, 1000);
    }
    detener() {
        clearInterval(this.timer);
        this.emit('stop');
    }
}

const reloj = new Reloj();
reloj.on('tick', (hora) => console.log(`Tick: ${hora.toLocaleTimeString()}`));
reloj.on('stop', () => console.log('Reloj detenido'));
reloj.iniciar();
// Para detener después de 5 segundos:
setTimeout(() => reloj.detener(), 5000);

Eliminar listeners:
-------------------
miEmisor.removeListener('saludo', miHandler);
miEmisor.removeAllListeners();

const server = http.createServer((req, res) => {
    const url = req.url;
    const method = req.method;

    if (url === '/' && method === 'GET') {
        res.writeHead(200, { 'Content-Type': 'text/html' });
        res.end('<h1>Inicio</h1>');
    } else if (url === '/about' && method === 'GET') {
        res.writeHead(200, { 'Content-Type': 'text/html' });
        res.end('<h1>Acerca</h1>');
    } else {
        res.writeHead(404, { 'Content-Type': 'text/plain' });
        res.end('404 - No encontrado');
    }
});

Servidor con JSON:
------------------
const server = http.createServer((req, res) => {
    const data = { nombre: "Esteban", edad: 30 };
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(data));
});

Leer datos del body (POST):
---------------------------
const server = http.createServer((req, res) => {
    if (req.method === 'POST' && req.url === '/datos') {
        let body = '';
        req.on('data', chunk => {
            body += chunk.toString();
        });
        req.on('end', () => {
            try {
                const data = JSON.parse(body);
                res.writeHead(200, { 'Content-Type': 'application/json' });
                res.end(JSON.stringify({ recibido: data }));
            } catch (error) {
                res.writeHead(400);
                res.end('JSON inválido');
            }
        });
    }
});

CAPITULO 8: MIDDLEWARES Y ENRUTAMIENTO
================================================================================

Middleware es una función que se ejecuta antes de las rutas.

Middleware básico:
------------------
app.use((req, res, next) => {
    console.log(`${req.method} ${req.url} - ${new Date().toISOString()}`);
    next();  // Pasa al siguiente middleware o ruta
});

Middleware para autenticación:
------------------------------
app.use('/admin', (req, res, next) => {
    const token = req.headers.authorization;
    if (!token || token !== 'Bearer secreto') {
        return res.status(401).json({ error: 'No autorizado' });
    }
    next();
});

Middleware de error (4 parámetros):
-----------------------------------
app.use((err, req, res, next) => {
    console.error(err.stack);
    res.status(500).json({ error: 'Algo salió mal' });
});

Enrutador (Router - para organizar rutas):
------------------------------------------
// routes/usuarios.js
const express = require('express');
const router = express.Router();

router.get('/', (req, res) => {
    res.json({ usuarios: [] });
});

router.post('/', (req, res) => {

});

router.get('/:id', (req, res) => {
    const id = req.params.id;
    res.json({ id, nombre: 'Usuario ' + id });

module.exports = router;

// app.js
const usuariosRouter = require('./routes/usuarios');

Middleware para validación:
---------------------------
function validarUsuario(req, res, next) {
    const { nombre, email } = req.body;
    if (!nombre || !email) {
        return res.status(400).json({ error: 'Nombre y email son requeridos' });
    }
    if (email.indexOf('@') === -1) {
        return res.status(400).json({ error: 'Email inválido' });
    }
    next();
}

app.post('/api/usuarios', validarUsuario, (req, res) => {
    // Procesar usuario
});

JWT (JSON Web Token) para autenticación.

Instalación:

bcrypt para hashear contraseñas:
--------------------------------
const bcrypt = require('bcrypt');
const saltRounds = 10;

JWT (generar y verificar):
--------------------------
const jwt = require('jsonwebtoken');
const secret = process.env.JWT_SECRET || 'mi_secreto';

// Generar token:
const token = jwt.sign({ userId: usuario.id, email: usuario.email }, secret, { expiresIn: '7d' });

// Verificar token:
try {
    const decoded = jwt.verify(token, secret);
    console.log(decoded.userId);
} catch (error) {
    console.error('Token inválido');
}

Middleware de autenticación:
----------------------------
function verificarToken(req, res, next) {
    const authHeader = req.headers.authorization;
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
        return res.status(401).json({ error: 'Token no proporcionado' });
    }
    const token = authHeader.split(' ')[1];
    try {
        const decoded = jwt.verify(token, process.env.JWT_SECRET);
        req.userId = decoded.userId;
        next();
    } catch (error) {
        return res.status(403).json({ error: 'Token inválido o expirado' });
    }
}

Ruta de login:
--------------
app.post('/api/auth/login', async (req, res) => {
    const { email, password } = req.body;
    const usuario = await Usuario.findOne({ email });
    if (!usuario) {
        return res.status(400).json({ error: 'Credenciales inválidas' });
    }
    const match = await bcrypt.compare(password, usuario.password);
    if (!match) {
        return res.status(400).json({ error: 'Credenciales inválidas' });
    }
    const token = jwt.sign({ userId: usuario._id, email: usuario.email }, process.env.JWT_SECRET);
    res.json({ token, usuario: { id: usuario._id, nombre: usuario.nombre, email: usuario.email } });
});

Ruta protegida:
---------------
app.get('/api/perfil', verificarToken, async (req, res) => {
    const usuario = await Usuario.findById(req.userId).select('-password');
    res.json(usuario);

CAPITULO 11: WEBSOCKETS (SOCKET.IO)
================================================================================

WebSockets permiten comunicación bidireccional en tiempo real.

Instalación:
------------
npm install socket.io socket.io-client

Servidor con Express + Socket.io:
---------------------------------
const express = require('express');
const http = require('http');
const { Server } = require('socket.io');

const app = express();
const server = http.createServer(app);
const io = new Server(server);

io.on('connection', (socket) => {
    console.log('Nuevo cliente conectado:', socket.id);

    // Evento personalizado
    socket.on('mensaje', (data) => {
        console.log('Mensaje recibido:', data);
        // Enviar a todos los clientes (broadcast)
        io.emit('mensaje', { usuario: data.usuario, texto: data.texto });
    });

    // Sala privada (room)
    socket.on('unirse_sala', (sala) => {
        socket.join(sala);
        io.to(sala).emit('mensaje_sala', `Usuario ${socket.id} se unió a ${sala}`);
    });

    socket.on('disconnect', () => {
        console.log('Cliente desconectado:', socket.id);
    });
});

server.listen(3000, () => {
    console.log('Servidor en http://localhost:3000');
});

Cliente (frontend):
-------------------
import io from 'socket.io-client';

const socket = io('http://localhost:3000');

socket.on('connect', () => {
    console.log('Conectado al servidor');
});

socket.on('mensaje', (data) => {
    console.log(`${data.usuario}: ${data.texto}`);
});

// Para desconectar:
socket.disconnect();

CAPITULO 12: PROCESAMIENTO ASÍNCRONO (PROMESAS, ASYNC/AWAIT)
================================================================================

Callbacks (tradicional - menos recomendado hoy):
------------------------------------------------
fs.readFile('archivo.txt', 'utf8', (err, data) => {
    if (err) {
        console.error(err);
        return;
    }
    console.log(data);
});

Promesas:
---------
function leerArchivoPromesa(ruta) {
    return new Promise((resolve, reject) => {
        fs.readFile(ruta, 'utf8', (err, data) => {
            if (err) reject(err);
            else resolve(data);
        });
    });
}

leerArchivoPromesa('archivo.txt')
    .then(data => console.log(data))
    .catch(err => console.error(err));

async/await (recomendado - más legible):
----------------------------------------
async function main() {
    try {
        const data = await leerArchivoPromesa('archivo.txt');
        console.log(data);
    } catch (error) {
        console.error(error);
    }
}
main();

// Múltiples promesas en paralelo:
async function procesarTodo() {
    try {
        const [data1, data2] = await Promise.all([
            leerArchivoPromesa('archivo1.txt'),
            leerArchivoPromesa('archivo2.txt')
        ]);
        console.log(data1, data2);
    } catch (error) {
        console.error(error);
    }
}

fs/promises (módulo nativo con promesas):
-----------------------------------------
const fs = require('fs/promises');

async function leer() {
    const data = await fs.readFile('archivo.txt', 'utf8');
    console.log(data);
}

-------------------------------
// En callbacks: manejar error en cada callback
// En promesas: .catch() al final
// En async/await: try-catch

-------------------------

const postSchema = new mongoose.Schema({
    titulo: { type: String, required: true, trim: true },

    autor: { type: mongoose.Schema.Types.ObjectId, ref: 'Usuario', required: true },

    fechaCreacion: { type: Date, default: Date.now },

    publicado: { type: Boolean, default: true }
});

postSchema.pre('save', function(next) {

    next();
});

const Post = mongoose.model('Post', postSchema);

const Post = require('../models/Post');

exports.listarPosts = async (req, res) => {
    try {
        const { page = 1, limit = 10, categoria, busqueda } = req.query;
        const skip = (page - 1) * limit;

        // Construir filtro
        let filtro = {};
        if (categoria) filtro.categoria = categoria;
        if (busqueda) {
            filtro.$or = [
                { titulo: { $regex: busqueda, $options: 'i' } },
                { contenido: { $regex: busqueda, $options: 'i' } }
            ];
        }

        const posts = await Post.find(filtro)
            .populate('autor', 'nombre email')
            .sort({ fechaCreacion: -1 })
            .skip(skip)
            .limit(parseInt(limit));

        const total = await Post.countDocuments(filtro);

        res.json({
            posts,
            pagina: parseInt(page),
            totalPaginas: Math.ceil(total / limit),
            total
        });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.crearPost = async (req, res) => {
    try {
        const { titulo, contenido, categoria } = req.body;
        const post = new Post({
            titulo,
            contenido,
            categoria,
            autor: req.userId
        });
        await post.save();
        res.status(201).json(post);
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.actualizarPost = async (req, res) => {
    try {
        const { id } = req.params;
        const { titulo, contenido, categoria, publicado } = req.body;

        const post = await Post.findById(id);
        if (!post) {
            return res.status(404).json({ error: 'Post no encontrado' });
        }

        // Verificar que el usuario es el autor
        if (post.autor.toString() !== req.userId) {
            return res.status(403).json({ error: 'No autorizado para modificar este post' });
        }

        post.titulo = titulo || post.titulo;
        post.contenido = contenido || post.contenido;
        post.categoria = categoria || post.categoria;
        post.publicado = publicado !== undefined ? publicado : post.publicado;
        post.actualizado = new Date();

        await post.save();
        res.json(post);
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};

exports.eliminarPost = async (req, res) => {
    try {
        const { id } = req.params;
        const post = await Post.findById(id);
        if (!post) {
            return res.status(404).json({ error: 'Post no encontrado' });
        }

        if (post.autor.toString() !== req.userId) {
            return res.status(403).json({ error: 'No autorizado para eliminar este post' });
        }

        await Post.findByIdAndDelete(id);
        res.status(204).send();
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
};
