"Las transiciones animan de A a B. Los keyframes animan de A a B
a C a D... Controlas cada paso del camino."
                         — Profesor Búho AI

===============================================================================
OBJETIVOS
===============================================================================

✓ Crear animaciones con @keyframes.
✓ Usar from y to.
✓ Usar porcentajes para múltiples pasos.
✓ Aplicar animaciones con animation-name.

===============================================================================
SINTAXIS
===============================================================================

@keyframes nombre {
    from { opacity: 0; transform: translateX(-20px); }
    to   { opacity: 1; transform: translateX(0); }
}

@keyframes rebote {
    0%   { transform: translateY(0); }
    50%  { transform: translateY(-50px); }
    80%  { transform: translateY(-10px); }
    100% { transform: translateY(0); }
}

.elemento { animation: rebote 1s infinite; }

===============================================================================
CÓDIGO PRÁCTICO
===============================================================================

<style>
* { margin: 0; padding: 0; box-sizing: border-box; }
body { font-family: Arial; background: #f0f2f5; padding: 40px; }
h1 { margin-bottom: 20px; }

.demo { display: flex; gap: 40px; flex-wrap: wrap; margin-bottom: 30px; align-items: flex-end; }

.box { width: 80px; height: 80px; background: #3498db; border-radius: 6px; }

@keyframes fadeIn {
    from { opacity: 0; transform: translateY(20px); }
    to   { opacity: 1; transform: translateY(0); }
}
.fade-in { animation: fadeIn 0.6s ease; }

@keyframes bounce {
    0%   { transform: translateY(0); }
    30%  { transform: translateY(-60px); }
    50%  { transform: translateY(0); }
    70%  { transform: translateY(-30px); }
    90%  { transform: translateY(0); }
    100% { transform: translateY(0); }
}
.bounce { animation: bounce 1.5s infinite; background: #e74c3c; }

@keyframes colors {
    0%   { background: #3498db; }
    25%  { background: #e74c3c; transform: rotate(0deg); }
    50%  { background: #2ecc71; transform: rotate(180deg); }
    75%  { background: #f39c12; transform: rotate(270deg); }
    100% { background: #3498db; transform: rotate(360deg); }
}
.colors { animation: colors 4s infinite; border-radius: 50%; }

@keyframes pulse {
    0%   { transform: scale(1); box-shadow: 0 0 0 0 rgba(108,99,255,0.7); }
    50%  { transform: scale(1.05); box-shadow: 0 0 0 15px rgba(108,99,255,0); }
    100% { transform: scale(1); box-shadow: 0 0 0 0 rgba(108,99,255,0); }
}
.pulse { animation: pulse 2s infinite; background: #6c63ff; }

@keyframes spin {
    from { transform: rotate(0deg); }
    to   { transform: rotate(360deg); }
}
.spinner {
    width: 40px; height: 40px; border: 4px solid #ddd;
    border-top-color: #3498db; border-radius: 50%;
    animation: spin 1s linear infinite;
}
</style>
