#include "cuenta_corriente.hpp"

CuentaCorriente::CuentaCorriente(const std::string& numero, Cliente* titular,
                                 double saldoInicial, double descubiertoMaximo)
    : Cuenta(numero, titular, saldoInicial, "Corriente"),
      descubiertoMaximo_(descubiertoMaximo) {}

bool CuentaCorriente::retirar(double cantidad) {
    if (cantidad <= 0) {
        std::cout << "Error: Cantidad debe ser positiva\n";
        return false;
    }
    double saldoDisponible = saldo_ + descubiertoMaximo_;
    if (cantidad > saldoDisponible) {
        std::cout << "Error: Excede el limite de descubierto\n";
        return false;
    }
    saldo_ -= cantidad;
    return true;
}

void CuentaCorriente::calcularIntereses() {
    // No genera intereses, solo cobra comision si esta en negativo
    if (saldo_ < 0) {
        saldo_ -= COMISION_MENSUAL;
        std::cout << "Comision de mantenimiento aplicada: $"
                  << COMISION_MENSUAL << std::endl;
    }
}

double CuentaCorriente::getDescubiertoMaximo() const {
    return descubiertoMaximo_;
}

void CuentaCorriente::setDescubiertoMaximo(double max) {
    descubiertoMaximo_ = max;
}

void CuentaCorriente::mostrar() const {
    Cuenta::mostrar();
    std::cout << "Descubierto Maximo: $" << descubiertoMaximo_ << std::endl;
}

#include "cuenta_ahorros.hpp"

CuentaAhorros::CuentaAhorros(const std::string& numero, Cliente* titular,
                             double saldoInicial, double interesAnual)
    : Cuenta(numero, titular, saldoInicial, "Ahorros"),
      interesAnual_(interesAnual),
      limiteRetirosMensuales_(MAX_RETIROS_MENSUALES),
      retirosEsteMes_(0) {}

bool CuentaAhorros::retirar(double cantidad) {
    if (cantidad <= 0) {
        std::cout << "Error: Cantidad debe ser positiva\n";
        return false;
    }
    if (retirosEsteMes_ >= limiteRetirosMensuales_) {
        std::cout << "Error: Ha excedido el limite mensual de retiros\n";
        return false;
    }
    if (cantidad > saldo_) {
        std::cout << "Error: Saldo insuficiente\n";
        return false;
    }
    saldo_ -= cantidad;
    ++retirosEsteMes_;
    return true;
}

void CuentaAhorros::calcularIntereses() {
    double interesMensual = (interesAnual_ / 12.0 / 100.0) * saldo_;
    saldo_ += interesMensual;
    std::cout << "Interes aplicado: $" << interesMensual << std::endl;
}

double CuentaAhorros::getInteresAnual() const { return interesAnual_; }

void CuentaAhorros::aplicarInteresMensual() {
    calcularIntereses();
    retirosEsteMes_ = 0;
}

void CuentaAhorros::mostrar() const {
    Cuenta::mostrar();
    std::cout << "Interes Anual: " << interesAnual_ << "%"
              << " | Retiros este mes: " << retirosEsteMes_
              << "/" << limiteRetirosMensuales_ << std::endl;
}

#include "cuenta_credito.hpp"

CuentaCredito::CuentaCredito(const std::string& numero, Cliente* titular,
                             double limiteCredito, double tasaInteres)
    : Cuenta(numero, titular, 0.0, "Credito"),
      limiteCredito_(limiteCredito),
      tasaInteres_(tasaInteres),
      saldoPendiente_(0.0) {}

bool CuentaCredito::retirar(double cantidad) {
    if (cantidad <= 0) {
        std::cout << "Error: Cantidad debe ser positiva\n";
        return false;
    }
    if (saldoPendiente_ + cantidad > limiteCredito_) {
        std::cout << "Error: Excede el limite de credito\n";
        return false;
    }
    saldo_ += cantidad;
    saldoPendiente_ += cantidad;
    return true;
}

void CuentaCredito::calcularIntereses() {
    double interes = saldoPendiente_ * (tasaInteres_ / 100.0 / 12.0);
    saldoPendiente_ += interes;
    saldo_ += interes;
    std::cout << "Interes de credito: $" << interes << std::endl;
}

bool CuentaCredito::pagarTarjeta(double cantidad) {
    if (cantidad <= 0) {
        std::cout << "Error: Cantidad debe ser positiva\n";
        return false;
    }
    if (cantidad > saldoPendiente_) {
        cantidad = saldoPendiente_;
    }
    saldoPendiente_ -= cantidad;
    saldo_ -= cantidad;
    return true;
}

double CuentaCredito::getLimiteCredito() const { return limiteCredito_; }
double CuentaCredito::getSaldoPendiente() const { return saldoPendiente_; }
double CuentaCredito::getCreditoDisponible() const {
    return limiteCredito_ - saldoPendiente_;
}

void CuentaCredito::mostrar() const {
    std::cout << "Cuenta: " << numeroCuenta_

    switch (t) {
        case DEPOSITO:      return "DEPOSITO";
        case RETIRO:        return "RETIRO";
        case TRANSFERENCIA: return "TRANSFERENCIA";
        case PAGO:          return "PAGO";
        default:            return "DESCONOCIDO";
    }
}

void Transaccion::mostrar() const {
    std::cout << "[" << id_ << "] "
              << fecha_ << " | "
              << tipoToString(tipo_) << " | $"
              << monto_ << " | "
              << cuentaOrigen_;
    if (!cuentaDestino_.empty()) {
        std::cout << " -> " << cuentaDestino_;
    }
    if (!descripcion_.empty()) {
        std::cout << " | " << descripcion_;
    }
    std::cout << std::endl;
}

        if (c->isActivo()) ++activos;
    }
    std::cout << activos << std::endl;

    std::cout << "\nCUENTAS: " << cuentas_.size() << std::endl;
    double totalDepositos = 0;
    double totalDeuda = 0;
    for (auto c : cuentas_) {
        if (c->getTipo() == "Credito") {
            CuentaCredito* cc = dynamic_cast<CuentaCredito*>(c);
            if (cc) totalDeuda += cc->getSaldoPendiente();
        } else {
            totalDepositos += c->getSaldo();
        }
    }
    std::cout << "Total en depositos: $" << totalDepositos << std::endl;
    std::cout << "Total en creditos: $" << totalDeuda << std::endl;
    std::cout << "Balance neto: $" << (totalDepositos - totalDeuda) << std::endl;

    std::cout << "\nTRANSACCIONES: " << transacciones_.size() << std::endl;
    std::cout << "========================================\n";
}
