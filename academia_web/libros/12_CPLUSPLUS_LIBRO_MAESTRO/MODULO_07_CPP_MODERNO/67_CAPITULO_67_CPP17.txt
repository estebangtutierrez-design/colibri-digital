--- 1. IF CONSTEXPR ---

if constexpr evalúa condiciones en tiempo de compilación.

```cpp
#include <iostream>
#include <type_traits>
#include <string>
#include <vector>

template<typename T>
auto procesar(T&& valor) {
    if constexpr (std::is_integral_v<std::decay_t<T>>) {
        std::cout << "Entero: " << valor << "\n";
        return valor * 2;
    } else if constexpr (std::is_floating_point_v<std::decay_t<T>>) {
        std::cout << "Flotante: " << valor << "\n";
        return valor * 1.5;
    } else if constexpr (std::is_same_v<std::decay_t<T>, std::string>) {
        std::cout << "String: " << valor << "\n";
        return valor + " procesado";
    } else {
        static_assert(std::is_same_v<std::decay_t<T>, void>,
                      "Tipo no soportado");
    }
}

// if constexpr con sizeof...
template<typename... Args>
void procesarPack(Args... args) {
    if constexpr (sizeof...(Args) == 0) {
        std::cout << "Pack vacio\n";
    } else if constexpr (sizeof...(Args) == 1) {
        std::cout << "Un elemento\n";
    } else {
        std::cout << sizeof...(Args) << " elementos\n";
    }
}

// if constexpr en clases CRTP
template<typename Derived>
class Base {
public:
    void interfaz() {
        if constexpr (requires { static_cast<Derived*>(this)->implementar(); }) {
            static_cast<Derived*>(this)->implementar();
        } else {
            std::cout << "Implementacion por defecto\n";
        }
    }
};

class ConImpl : public Base<ConImpl> {
public:
    void implementar() { std::cout << "Implementacion personalizada\n"; }
};

class SinImpl : public Base<SinImpl> {};

int main() {
    std::cout << "=== IF CONSTEXPR ===\n";

    std::cout << procesar(42) << "\n";
    std::cout << procesar(3.14) << "\n";
    std::cout << procesar(std::string("Hola")) << "\n";

    procesarPack();
    procesarPack(1);
    procesarPack(1, 2, 3);

    ConImpl obj1;
    obj1.interfaz();

    SinImpl obj2;
    obj2.interfaz();

    // La rama falsa de if constexpr NO se instancia
    // Por eso esto compila aunque el tipo no tenga el método:
    // if constexpr (condicion) { ... } else { ... }

    return 0;
}
```

--- 2. STRUCTURED BINDINGS ---

Desempaquetado de tuplas, arrays y structs en variables individuales.

```cpp
#include <iostream>
#include <tuple>
#include <map>
#include <string>
#include <array>
#include <set>

struct Persona {
    std::string nombre;
    int edad;
    double altura;
};

int main() {
    std::cout << "=== STRUCTURED BINDINGS ===\n";

    // Con arrays
    int arr[] = {10, 20, 30};
    auto [a, b, c] = arr;
    std::cout << a << ", " << b << ", " << c << "\n";

    // Con tuplas
    std::tuple<std::string, int, double> t("Ana", 25, 1.75);
    auto [nombre, edad, altura] = t;
    std::cout << nombre << ", " << edad << " anios, " << altura << "m\n";

    // Con structs
    Persona p{"Carlos", 30, 1.80};
    auto [nom, ed, alt] = p;
    std::cout << nom << ", " << ed << " anios, " << alt << "m\n";

    // Con pares (útil para mapas)
    std::map<std::string, int> edades = {
        {"Ana", 25}, {"Luis", 30}, {"Pedro", 35}
    };

    for (const auto& [nombre, edad] : edades) {
        std::cout << nombre << ": " << edad << "\n";
    }

    // Modificar valores
    std::map<int, std::string> items;
    items[1] = "Original";

    for (auto& [id, valor] : items) {
        valor = "Modificado";
    }
    std::cout << "items[1]: " << items[1] << "\n";

    // Con bindings anidados
    std::map<std::string, std::tuple<int, double>> datos = {
        {"obj1", {10, 3.14}}
    };
    for (const auto& [key, [val1, val2]] : datos) {
        std::cout << key << ": " << val1 << ", " << val2 << "\n";
    }

    // Con std::set (insert devuelve par)
    std::set<int> conjunto;
    auto [iter, inserted] = conjunto.insert(42);
    std::cout << "Insertado: " << inserted << "\n";

    return 0;
}
```

--- 3. STD::OPTIONAL ---

Representa un valor que puede estar presente o no.

```cpp
#include <iostream>
#include <optional>
#include <string>
#include <vector>
#include <map>
#include <cmath>

// Función que puede fallar
std::optional<double> raizCuadrada(double x) {
    if (x >= 0) return std::sqrt(x);
    return std::nullopt;
}

// Buscar en contenedor
template<typename Container, typename Key>
std::optional<typename Container::mapped_type>
buscar(const Container& c, const Key& k) {
    auto it = c.find(k);
    if (it != c.end()) return it->second;
    return std::nullopt;
}

// Parseo de enteros
std::optional<int> parseInt(const std::string& s) {
    try {
        return std::stoi(s);
    } catch (...) {
        return std::nullopt;
    }
}

struct Config {
    std::optional<int> timeout;
    std::optional<std::string> host;
    std::optional<int> puerto;
};

int main() {
    std::cout << "=== STD::OPTIONAL ===\n";

    // Uso básico
    auto r1 = raizCuadrada(16.0);
    auto r2 = raizCuadrada(-4.0);

    if (r1) {
        std::cout << "Raiz(16): " << *r1 << "\n";
    }

    if (r2.has_value()) {
        std::cout << "Raiz(-4): " << r2.value() << "\n";
    } else {
        std::cout << "Raiz(-4): no es un numero real\n";
    }

    // value_or
    std::cout << "Raiz(16): " << r1.value_or(0.0) << "\n";
    std::cout << "Raiz(-4): " << r2.value_or(0.0) << "\n";

    // Búsqueda en mapa
    std::map<std::string, int> puertos = {
        {"http", 80}, {"https", 443}, {"ssh", 22}
    };

    auto p1 = buscar(puertos, std::string("http"));
    auto p2 = buscar(puertos, std::string("ftp"));

    std::cout << "http: " << (p1 ? std::to_string(*p1) : "no encontrado") << "\n";
    std::cout << "ftp: " << (p2 ? std::to_string(*p2) : "no encontrado") << "\n";

    // Parseo de enteros
    auto n1 = parseInt("42");
    auto n2 = parseInt("no es numero");

    std::cout << "parseInt('42'): " << n1.value_or(-1) << "\n";
    std::cout << "parseInt('no es numero'): " << n2.value_or(-1) << "\n";

    // Optional como miembro de struct
    Config cfg;
    cfg.timeout = 30;
    cfg.host = "localhost";

    if (cfg.timeout) std::cout << "Timeout: " << *cfg.timeout << "s\n";
    if (cfg.host) std::cout << "Host: " << *cfg.host << "\n";
    if (!cfg.puerto) std::cout << "Puerto no configurado\n";

    // Uso en contenedores
    std::vector<std::optional<int>> valores = {1, std::nullopt, 3, std::nullopt, 5};
    for (const auto& v : valores) {
        std::cout << v.value_or(0) << " ";
    }
    std::cout << "\n";

    return 0;
}
```

--- 4. STD::VARIANT ---

Unión type-safe que puede contener uno de varios tipos.

```cpp
#include <iostream>
#include <variant>
#include <string>
#include <vector>
#include <type_traits>

// Tipos para la variante
using Valor = std::variant<int, double, std::string, bool>;

// Visitor con overloaded
template<typename... Ts>
struct Overloaded : Ts... { using Ts::operator()...; };

template<typename... Ts>
Overloaded(Ts...) -> Overloaded<Ts...>;

// Función que imprime cualquier variante
void imprimirValor(const Valor& v) {
    std::visit(Overloaded{
        [](int i) { std::cout << "Entero: " << i; },
        [](double d) { std::cout << "Double: " << d; },
        [](const std::string& s) { std::cout << "String: " << s; },
        [](bool b) { std::cout << "Bool: " << (b ? "true" : "false"); }
    }, v);
    std::cout << "\n";
}

// Expresión aritmética con variant
using Expr = std::variant<int, double>;

Expr sumar(const Expr& a, const Expr& b) {
    return std::visit(Overloaded{
        [](int i1, int i2) -> Expr { return i1 + i2; },
        [](int i, double d) -> Expr { return i + d; },
        [](double d, int i) -> Expr { return d + i; },
        [](double d1, double d2) -> Expr { return d1 + d2; }
    }, a, b);
}

int main() {
    std::cout << "=== STD::VARIANT ===\n";

    // Creación y asignación
    Valor v1 = 42;
    Valor v2 = 3.14;
    Valor v3 = std::string("Hola");
    Valor v4 = true;

    imprimirValor(v1);
    imprimirValor(v2);
    imprimirValor(v3);
    imprimirValor(v4);

    // Alternar entre tipos
    v1 = 3.14;
    imprimirValor(v1);

    // Verificar tipo actual
    if (std::holds_alternative<int>(v2)) {
        std::cout << "v2 es int\n";
    }
    if (std::holds_alternative<double>(v2)) {
        std::cout << "v2 es double: " << std::get<double>(v2) << "\n";
    }

    // get_if
    if (auto* d = std::get_if<double>(&v2)) {
        std::cout << "get_if double: " << *d << "\n";
    }

    // std::visit
    std::cout << "\nVisit con lambda generica:\n";
    std::visit([](const auto& val) {
        std::cout << "Valor: " << val << "\n";
    }, v2);

    // Múltiples variantes
    auto resultado = sumar(42, 3.5);
    std::visit([](const auto& v) {
        std::cout << "Suma: " << v << "\n";
    }, resultado);

    // Vector de variants
    std::vector<Valor> valores = {10, 2.5, "test", false, 42};
    for (const auto& v : valores) {
        imprimirValor(v);
    }

    // Variant con tipos personalizados
    struct Error { std::string mensaje; };
    struct Exito { int resultado; };
    using Resultado = std::variant<Exito, Error>;

    Resultado res = Exito{42};
    if (std::holds_alternative<Exito>(res)) {
        std::cout << "Exito: " << std::get<Exito>(res).resultado << "\n";
    }

    res = Error{"Algo salio mal"};
    if (std::holds_alternative<Error>(res)) {
        std::cout << "Error: " << std::get<Error>(res).mensaje << "\n";
    }

    return 0;
}
```

--- 5. STD::ANY ---

Contenedor type-safe para cualquier tipo (similar a void*).

```cpp
#include <iostream>
#include <any>
#include <string>
#include <vector>
#include <type_traits>

int main() {
    std::cout << "=== STD::ANY ===\n";

    // Almacenar cualquier tipo
    std::any valor;

    valor = 42;
    std::cout << "int: " << std::any_cast<int>(valor) << "\n";

    valor = 3.14;
    std::cout << "double: " << std::any_cast<double>(valor) << "\n";

    valor = std::string("Hola Mundo");
    std::cout << "string: " << std::any_cast<std::string>(valor) << "\n";

    // Verificar tipo
    if (valor.type() == typeid(std::string)) {
        std::cout << "Es un string\n";
    }

    // any_cast seguro (retorna puntero)
    if (auto* s = std::any_cast<std::string>(&valor)) {
        std::cout << "Contenido: " << *s << "\n";
    }

    if (auto* d = std::any_cast<double>(&valor)) {
        std::cout << "Es double: " << *d << "\n";
    } else {
        std::cout << "No es double\n";
    }

    // any_cast lanza excepción si el tipo no coincide
    try {
        int i = std::any_cast<int>(valor);  // Error: es string
    } catch (const std::bad_any_cast& e) {
        std::cout << "Error: " << e.what() << "\n";
    }

    // Vector de any (pérdida de type safety)
    std::vector<std::any> varios = {42, 3.14, std::string("Hola"), true};

    for (const auto& a : varios) {
        if (auto* i = std::any_cast<int>(&a)) {
            std::cout << "int: " << *i << "\n";
        } else if (auto* d = std::any_cast<double>(&a)) {
            std::cout << "double: " << *d << "\n";
        } else if (auto* s = std::any_cast<std::string>(&a)) {
            std::cout << "string: " << *s << "\n";
        } else if (auto* b = std::any_cast<bool>(&a)) {
            std::cout << "bool: " << *b << "\n";
        }
    }

    // any con tipos personalizados
    struct Punto { int x, y; };
    valor = Punto{10, 20};
    auto p = std::any_cast<Punto>(valor);
    std::cout << "Punto: (" << p.x << ", " << p.y << ")\n";

    // any vacío
    std::any vacio;
    std::cout << "Tiene valor: " << vacio.has_value() << "\n";
    vacio.reset();
    std::cout << "Despues de reset: " << vacio.has_value() << "\n";

    return 0;
}
```

--- 6. STD::FILESYSTEM ---

La biblioteca de sistema de archivos.

```cpp
#include <iostream>
#include <filesystem>
#include <string>
#include <vector>
#include <fstream>

namespace fs = std::filesystem;

int main() {
    std::cout << "=== STD::FILESYSTEM ===\n";

    // Directorio actual
    fs::path current = fs::current_path();
    std::cout << "Directorio actual: " << current << "\n";

    // Componer rutas
    fs::path archivo = current / "test.txt";
    std::cout << "Ruta completa: " << archivo << "\n";
    std::cout << "  Nombre: " << archivo.filename() << "\n";
    std::cout << "  Extension: " << archivo.extension() << "\n";
    std::cout << "  Stem: " << archivo.stem() << "\n";
    std::cout << "  Parent: " << archivo.parent_path() << "\n";

    // Crear directorio
    fs::path dir = current / "test_dir";
    if (!fs::exists(dir)) {
        fs::create_directory(dir);
        std::cout << "Directorio creado: " << dir << "\n";
    }

    // Crear archivo
    std::ofstream(archivo) << "Contenido de prueba\n";
    std::cout << "Archivo creado\n";

    // Información del archivo
    if (fs::exists(archivo)) {
        std::cout << "  Tamano: " << fs::file_size(archivo) << " bytes\n";
        std::cout << "  Es archivo: " << fs::is_regular_file(archivo) << "\n";
        std::cout << "  Ultima modificacion: " << fs::last_write_time(archivo) << "\n";
    }

    // Iterar directorio
    std::cout << "\nContenido de " << current << ":\n";
    for (const auto& entry : fs::directory_iterator(current)) {
        fs::path p = entry.path();
        std::cout << "  " << (fs::is_directory(p) ? "[DIR]" : "[FILE]")
                  << " " << p.filename() << "\n";
        if (entry.file_size() > 0) {
            std::cout << "    (" << entry.file_size() << " bytes)\n";
        }
    }

    // Recursivo
    std::cout << "\nRecursivo (limitado):\n";
    int count = 0;
    for (const auto& entry : fs::recursive_directory_iterator(
             current, fs::directory_options::skip_permission_denied)) {
        if (++count > 10) break;
        std::cout << "  " << entry.path() << "\n";
    }

    // Copiar y renombrar
    fs::path copia = current / "copia_test.txt";
    fs::copy_file(archivo, copia, fs::copy_options::overwrite_existing);
    std::cout << "\nArchivo copiado\n";

    // Mover/renombrar
    fs::path movido = current / "movido_test.txt";
    fs::rename(copia, movido);
    std::cout << "Archivo movido/renombrado\n";

    // Espacio en disco
    fs::space_info si = fs::space(current);
    std::cout << "\nEspacio en disco:\n";
    std::cout << "  Capacidad: " << si.capacity / (1024*1024*1024) << " GB\n";
    std::cout << "  Libre: " << si.free / (1024*1024*1024) << " GB\n";

    // Limpiar
    fs::remove(archivo);
    fs::remove(movido);
    fs::remove(dir);
    std::cout << "\nArchivos temporales eliminados\n";

    return 0;
}
```

--- 7. PARALLEL ALGORITHMS ---

C++17 añade políticas de ejecución paralela para algoritmos STL.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <execution>
#include <chrono>
#include <numeric>
#include <random>

int main() {
    std::cout << "=== PARALLEL ALGORITHMS ===\n";

    // Preparar datos
    std::vector<long long> datos(10'000'000);
    std::iota(datos.begin(), datos.end(), 1);

    // Políticas de ejecución:
    // std::execution::seq    - secuencial
    // std::execution::par    - paralelo
    // std::execution::par_unseq - paralelo + vectorizado
    // std::execution::unseq  - vectorizado (C++20)

    auto medir = [&](const std::string& nombre, auto policy) {
        std::vector<long long> copia = datos;
        auto inicio = std::chrono::high_resolution_clock::now();

        // Transformación
        std::transform(policy, copia.begin(), copia.end(), copia.begin(),
                       [](long long x) { return x * x; });

        auto fin = std::chrono::high_resolution_clock::now();
        auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
            fin - inicio);
        std::cout << nombre << ": " << ms.count() << " ms\n";
    };

    std::cout << "\nComparacion de politicas:\n";
    // medir("seq      ", std::execution::seq);
    // medir("par      ", std::execution::par);
    // medir("par_unseq", std::execution::par_unseq);

    // Reducción paralela
    std::vector<long long> nums(10'000'000);
    std::iota(nums.begin(), nums.end(), 1);

    auto total = std::reduce(std::execution::par,
                             nums.begin(), nums.end(), 0LL);
    std::cout << "\nSuma con reduce paralelo: " << total << "\n";

    // Count con paralelo
    std::mt19937 gen(42);
    std::uniform_int_distribution<> dist(0, 100);
    std::vector<int> aleatorios(1'000'000);
    for (auto& n : aleatorios) n = dist(gen);

    auto count = std::count_if(std::execution::par,
                                aleatorios.begin(), aleatorios.end(),
                                [](int n) { return n > 50; });
    std::cout << "Elementos > 50: " << count << "\n";

    return 0;
}
```

--- 8. STRING_VIEW ---

std::string_view es una vista no-propietaria de cadenas.

```cpp
#include <iostream>
#include <string_view>
#include <string>
#include <vector>
#include <algorithm>

// Función que acepta cualquier tipo de cadena (sin copias)
void procesarSV(std::string_view sv) {
    std::cout << "String_view: " << sv
              << " (size=" << sv.size() << ")\n";
}

// Extraer extensión de archivo
std::string_view obtenerExtension(std::string_view archivo) {
    auto pos = archivo.rfind('.');
    if (pos != std::string_view::npos) {
        return archivo.substr(pos);
    }
    return {};
}

int main() {
    std::cout << "=== STRING_VIEW ===\n";

    // Creación desde diferentes fuentes
    std::string_view sv1 = "Literal de string";
    std::string s = "std::string";
    std::string_view sv2 = s;
    std::string_view sv3 = sv2.substr(0, 3);

    procesarSV(sv1);
    procesarSV(sv2);
    procesarSV(sv3);
    procesarSV("Temporal");

    // Extracción de extensiones
    std::cout << "\nExtensiones:\n";
    std::cout << "archivo.txt: " << obtenerExtension("archivo.txt") << "\n";
    std::cout << "imagen.jpg: " << obtenerExtension("imagen.jpg") << "\n";
    std::cout << "sin_extension: " << obtenerExtension("sin_extension") << "\n";

    // Búsqueda
    std::string_view texto = "Hola Mundo desde C++17";
    auto pos = texto.find("Mundo");
    if (pos != std::string_view::npos) {
        std::cout << "'Mundo' encontrado en posicion " << pos << "\n";
    }

    // Comparación
    if (sv1 == "Literal de string") {
        std::cout << "sv1 coincide con literal\n";
    }

    // Iteración
    std::cout << "\nCaracteres: ";
    for (char c : sv1) {
        std::cout << c << " ";
    }
    std::cout << "\n";

    // Remove_prefix y remove_suffix
    std::string_view path = "/home/user/documentos/archivo.txt";
    path.remove_prefix(1);  // quita '/'
    auto dir = path.substr(0, path.find('/'));
    std::cout << "Directorio: " << dir << "\n";

    // Uso con contenedores (vista de vector de chars)
    std::vector<char> chars = {'H', 'o', 'l', 'a'};
    std::string_view desdeChars(chars.data(), chars.size());
    std::cout << "Desde vector: " << desdeChars << "\n";

    return 0;
}
```

--- 9. NESTED NAMESPACES Y OTRAS CARACTERÍSTICAS ---

```cpp
#include <iostream>
#include <string>
#include <map>
#include <optional>

// Nested namespaces (C++17)
namespace A::B::C {
    int valor = 42;
}

// Inline variables (C++17)
struct MiClase {
    static inline int contador = 0;
    static inline const double PI = 3.14159;
};

// if con inicializador (C++17)
void demoIfInicializador() {
    std::map<std::string, int> mapa;
    mapa["clave"] = 42;

    if (auto it = mapa.find("clave"); it != mapa.end()) {
        std::cout << "Encontrado: " << it->second << "\n";
    } else {
        std::cout << "No encontrado\n";
    }
}

// switch con inicializador (C++17)
void demoSwitchInicializador() {
    std::optional<int> opt = 42;

    switch (auto v = opt; v.has_value()) {
        case true:
            std::cout << "Valor: " << v.value() << "\n";
            break;
        case false:
            std::cout << "Sin valor\n";
            break;
    }
}

// constexpr lambda (C++17)
constexpr auto cuadrado = [](int x) { return x * x; };

// __has_include (C++17)
#if __has_include(<optional>)
    #define TIENE_OPTIONAL true
#else
    #define TIENE_OPTIONAL false
#endif

// std::clamp (C++17)
#include <algorithm>

template<typename T>
T clamp(T valor, T min, T max) {
    return std::clamp(valor, min, max);
}

int main() {
    std::cout << "=== NESTED NAMESPACES Y OTRAS ===\n";

    std::cout << "A::B::C::valor = " << A::B::C::valor << "\n";

    MiClase a, b;
    a.contador = 5;
    std::cout << "MiClase::contador = " << MiClase::contador << "\n";  // 5
    std::cout << "MiClase::PI = " << MiClase::PI << "\n";

    demoIfInicializador();
    demoSwitchInicializador();

    // constexpr lambda
    constexpr int resultado = cuadrado(5);
    static_assert(resultado == 25);

    std::cout << "constexpr lambda: " << cuadrado(10) << "\n";

    // std::clamp
    std::cout << "clamp(5, 0, 10): " << clamp(5, 0, 10) << "\n";
    std::cout << "clamp(-5, 0, 10): " << clamp(-5, 0, 10) << "\n";
    std::cout << "clamp(15, 0, 10): " << clamp(15, 0, 10) << "\n";

    // __has_include
    std::cout << "Tiene <optional>: " << TIENE_OPTIONAL << "\n";

    std::byte b2{0xFF};
```

```cpp
#include <iostream>
#include <filesystem>
#include <optional>
#include <variant>
#include <string>
#include <vector>
#include <fstream>
#include <sstream>

namespace fs = std::filesystem;

// Representación de un archivo de configuración
using ValorConfig = std::variant<int, double, std::string, bool>;

struct Configuracion {
    std::map<std::string, ValorConfig> valores;

    template<typename T>
    std::optional<T> obtener(const std::string& clave) const {
        auto it = valores.find(clave);
        if (it == valores.end()) return std::nullopt;
        if (auto* v = std::get_if<T>(&it->second)) {
            return *v;
        }
        return std::nullopt;
    }

    void mostrar() const {
        for (const auto& [clave, valor] : valores) {
            std::cout << clave << " = ";
            std::visit([](const auto& v) {
                std::cout << v;
            }, valor);
            std::cout << "\n";
        }
    }
};

// Parseador simple de archivos .conf
std::optional<Configuracion> cargarConfig(const fs::path& ruta) {
    if (!fs::exists(ruta)) {
        return std::nullopt;
    }

    std::ifstream archivo(ruta);
    if (!archivo.is_open()) {
        return std::nullopt;
    }

    Configuracion config;
    std::string linea;

    while (std::getline(archivo, linea)) {
        // Ignorar comentarios y líneas vacías
        if (linea.empty() || linea[0] == '#') continue;

        auto sep = linea.find('=');
        if (sep == std::string::npos) continue;

        std::string clave = linea.substr(0, sep);
        std::string valor_str = linea.substr(sep + 1);

        // Trim whitespace
        auto trim = [](std::string_view sv) {
            sv.remove_prefix(std::min(sv.find_first_not_of(" \t"),
                                       sv.size()));
            auto end = sv.find_last_not_of(" \t");
            if (end != std::string_view::npos) sv = sv.substr(0, end + 1);
            return sv;
        };

        clave = std::string(trim(clave));
        valor_str = std::string(trim(valor_str));

        // Intentar parsear como int, double, bool, string
        ValorConfig valor;

        if (valor_str == "true" || valor_str == "false") {
            valor = (valor_str == "true");
        } else if (valor_str.find('.') != std::string::npos) {
            try { valor = std::stod(valor_str); }
            catch (...) { valor = valor_str; }
        } else {
            try { valor = std::stoi(valor_str); }
            catch (...) { valor = valor_str; }
        }

        config.valores[clave] = valor;
    }

    return config;
}

int main() {
    std::cout << "=== EJEMPLO INTEGRADOR C++17 ===\n";

    // Crear archivo de configuración
    fs::path config_path = fs::current_path() / "app.conf";

    {
        std::ofstream conf(config_path);
        conf << "# Configuracion de aplicacion\n";
        conf << "host = localhost\n";
        conf << "puerto = 8080\n";
        conf << "timeout = 30.5\n";
        conf << "debug = true\n";
        conf << "nombre = Servidor Principal\n";
    }

    // Cargar y mostrar configuración
    auto config = cargarConfig(config_path);

    if (config) {
        std::cout << "\nConfiguracion cargada:\n";
        config->mostrar();

        // Acceder a valores específicos
        if (auto puerto = config->obtener<int>("puerto")) {
            std::cout << "\nPuerto: " << *puerto << "\n";
        }

        if (auto debug = config->obtener<bool>("debug")) {
            std::cout << "Debug: " << std::boolalpha << *debug << "\n";
        }

        if (auto timeout = config->obtener<double>("timeout")) {
            std::cout << "Timeout: " << *timeout << "s\n";
        }
    } else {
        std::cout << "Error al cargar configuracion\n";
    }

    // Limpiar
    fs::remove(config_path);

    // Demostración de structured bindings con filesystem
    std::cout << "\nRecorriendo directorio actual:\n";
    for (const auto& entry : fs::directory_iterator(".")) {
        const auto& path = entry.path();
        auto [tam, es_dir] = [&]() -> std::pair<uintmax_t, bool> {
            if (fs::is_regular_file(path))
                return {fs::file_size(path), false};
            return {0, true};
        }();

        std::cout << (es_dir ? "[DIR]" : "[FILE]")
                  << " " << path.filename();
        if (!es_dir) std::cout << " (" << tam << " bytes)";
        std::cout << "\n";
    }

    // if constexpr + variant + optional
    auto procesarValor = [](const ValorConfig& v) -> std::optional<std::string> {
        if constexpr (true) {
            return std::visit([](const auto& val) -> std::string {
                std::ostringstream oss;
                oss << val;
                return oss.str();
            }, v);
        }
    };

    ValorConfig v = 42;
    auto str = procesarValor(v);
    if (str) std::cout << "\nValor como string: " << *str << "\n";

    return 0;
}
```

C++17 introduce características transformadoras:

• if constexpr: compilación condicional que elimina ramas muertas.
• Structured bindings: descomposición de tuplas/structs/arrays.
• std::optional: valores que pueden estar ausentes.
• std::variant: unión type-safe con visita de tipos.
• std::any: contenedor para cualquier tipo (similar a void*).
• std::filesystem: manipulación portátil del sistema de archivos.
• Parallel algorithms: políticas seq/par/par_unseq para algoritmos STL.
• std::string_view: vista no-propietaria de cadenas.
• Nested namespaces: A::B::C en lugar de namespace A { namespace B { ... } }
• if/switch con inicializador: if (auto x = expr; condicion).

================================================================================
  PREGUNTAS FRECUENTES (FAQ)
================================================================================

P: ¿if constexpr es como #if?
R: No. if constexpr es parte del lenguaje, evalúa en tiempo de
compilación, y el código de la rama falsa no se instancia.

P: ¿Cuándo usar std::optional vs std::variant?
R: optional para "valor o nada". variant para "uno de varios tipos
posibles".

P: ¿std::any es mejor que void*?
R: Sí. any es type-safe, sabe el tipo almacenado, y lanza excepciones
en conversiones incorrectas.

P: ¿filesystem reemplaza a boost::filesystem?
R: Sí. std::filesystem está basado en boost::filesystem y es
prácticamente compatible.

P: ¿Los algoritmos paralelos siempre son más rápidos?
R: No. Para datos pequeños, la sobrecarga de paralelismo puede ser
mayor que el beneficio. Para datos grandes (millones de elementos),
sí suelen ser más rápidos.
