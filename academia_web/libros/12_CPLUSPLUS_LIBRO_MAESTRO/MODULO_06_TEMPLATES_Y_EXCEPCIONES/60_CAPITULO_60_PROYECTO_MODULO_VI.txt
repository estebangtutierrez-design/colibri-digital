ESPECIFICACIÓN DEL PROYECTO
---------------------------

Nombre: SafeContainer<T>

Requisitos funcionales:
  - Contenedor genérico que almacene elementos de tipo T.
  - Inserción, eliminación, acceso por índice, búsqueda.
  - Redimensionamiento automático con strong guarantee.
  - Serialización a archivo binario.
  - Deserialización desde archivo binario.
  - Iteradores forward básicos.
  - Constructor de copia, movimiento, asignación.
  - Operaciones: push_back, pop_back, insert, erase, clear, size, empty.

Requisitos de seguridad:
  - Strong guarantee para push_back, insert, erase.
  - Basic guarantee para todas las operaciones.
  - Nothrow guarantee para size, empty, begin, end.
  - RAII para el buffer interno.
  - Las excepciones no deben causar fugas de memoria.

Requisitos de diseño:
  - Uso de smart pointers para el buffer interno.
  - Type traits para validar tipos.
  - Jerarquía de excepciones personalizada.
  - Namespace propio.
  - Archivos separados: .hpp, .cpp, test.cpp.

#include <iostream>
#include <stdexcept>
#include <string>
#include <type_traits>

namespace safe_container {

// Excepción base del contenedor
class ContainerException : public std::runtime_error {
    using std::runtime_error::runtime_error;
};

class OutOfRangeException : public ContainerException {
    size_t index;
    size_t size;
public:
    OutOfRangeException(size_t idx, size_t sz)
        : ContainerException("Indice " + std::to_string(idx)
              + " fuera de rango [0, " + std::to_string(sz) + ")"),
          index(idx), size(sz) {}
    size_t getIndex() const { return index; }
    size_t getSize() const { return size; }
};

class InvalidTypeException : public ContainerException {
    std::string typeName;
public:
    explicit InvalidTypeException(const std::string& t)
        : ContainerException("Tipo invalido: " + t), typeName(t) {}
    std::string getTypeName() const { return typeName; }
};

class SerializationException : public ContainerException {
    std::string fileName;
public:
    SerializationException(const std::string& msg, const std::string& file)
        : ContainerException(msg + ": " + file), fileName(file) {}
    std::string getFileName() const { return fileName; }
};

class EmptyContainerException : public ContainerException {
public:
    EmptyContainerException()
        : ContainerException("Operacion en contenedor vacio") {}
};

} // namespace safe_container

// FASE 2: Buffer RAII con strong guarantee
//==============================================================================
namespace safe_container {

template <typename T>
class BufferRAII {
    T* datos;
    size_t capacidad;

    void liberar() noexcept {
        delete[] datos;
    }

public:
    BufferRAII() : datos(nullptr), capacidad(0) {}

    explicit BufferRAII(size_t cap)
        : datos(new T[cap]()), capacidad(cap) {}

    ~BufferRAII() noexcept {
        liberar();
    }

    BufferRAII(const BufferRAII& otro)
        : datos(new T[otro.capacidad]),
          capacidad(otro.capacidad) {
        for (size_t i = 0; i < capacidad; ++i) {
            datos[i] = otro.datos[i];
        }
    }

    BufferRAII& operator=(const BufferRAII& otro) {
        if (this != &otro) {
            BufferRAII tmp(otro);
            swap(tmp);
        }
        return *this;
    }

    BufferRAII(BufferRAII&& otro) noexcept
        : datos(otro.datos), capacidad(otro.capacidad) {
        otro.datos = nullptr;
        otro.capacidad = 0;
    }

    BufferRAII& operator=(BufferRAII&& otro) noexcept {
        if (this != &otro) {
            liberar();
            datos = otro.datos;
            capacidad = otro.capacidad;
            otro.datos = nullptr;
            otro.capacidad = 0;
        }
        return *this;
    }

    void swap(BufferRAII& otro) noexcept {
        std::swap(datos, otro.datos);
        std::swap(capacidad, otro.capacidad);
    }

    T& operator[](size_t i) { return datos[i]; }
    const T& operator[](size_t i) const { return datos[i]; }

    T* get() { return datos; }
    const T* get() const { return datos; }

    size_t size() const noexcept { return capacidad; }

    // Strong guarantee: si falla, el buffer no cambia
    void redimensionar(size_t nuevaCap) {
        T* nuevos = new T[nuevaCap]();
        size_t copiar = std::min(capacidad, nuevaCap);
        for (size_t i = 0; i < copiar; ++i) {
            nuevos[i] = datos[i];
        }
        delete[] datos;
        datos = nuevos;
        capacidad = nuevaCap;
    }
};

} // namespace safe_container

// FASE 3: Implementación del contenedor SafeContainer
//==============================================================================
namespace safe_container {

template <typename T>
class SafeContainer {
    static_assert(std::is_nothrow_destructible_v<T>,
                  "T debe tener destructor noexcept");

    BufferRAII<T> buffer;
    size_t longitud;

    static constexpr size_t CAPACIDAD_INICIAL = 8;

    struct NothrowCopy {
        static constexpr bool value =
            std::is_nothrow_copy_constructible_v<T>;
    };

    struct NothrowMove {
        static constexpr bool value =
            std::is_nothrow_move_constructible_v<T>;
    };

public:
    SafeContainer() : buffer(CAPACIDAD_INICIAL), longitud(0) {}

    explicit SafeContainer(size_t n, const T& valor = T{})
        : buffer(n), longitud(n) {
        for (size_t i = 0; i < n; ++i) {
            buffer[i] = valor;
        }
    }

    // Strong guarantee for copy constructor
    SafeContainer(const SafeContainer& otro)
        : buffer(otro.buffer), longitud(otro.longitud) {}

    // Nothrow move constructor
    SafeContainer(SafeContainer&& otro) noexcept
        : buffer(std::move(otro.buffer)),
          longitud(otro.longitud) {
        otro.longitud = 0;
    }

    SafeContainer& operator=(const SafeContainer& otro) {
        if (this != &otro) {
            SafeContainer tmp(otro);
            swap(tmp);
        }
        return *this;
    }

    SafeContainer& operator=(SafeContainer&& otro) noexcept {
        if (this != &otro) {
            buffer = std::move(otro.buffer);
            longitud = otro.longitud;
            otro.longitud = 0;
        }
        return *this;
    }

    ~SafeContainer() noexcept = default;

    // Strong guarantee for push_back
    void push_back(const T& valor) {
        if (longitud >= buffer.size()) {
            BufferRAII<T> nuevo(buffer.size() * 2);
            for (size_t i = 0; i < longitud; ++i) {
                nuevo[i] = buffer[i];
            }
            nuevo[longitud] = valor;
            buffer.swap(nuevo);
            ++longitud;
        } else {
            buffer[longitud] = valor;
            ++longitud;
        }
    }

    // Strong guarantee for push_back move
    void push_back(T&& valor) {
        if (longitud >= buffer.size()) {
            BufferRAII<T> nuevo(buffer.size() * 2);
            for (size_t i = 0; i < longitud; ++i) {
                nuevo[i] = std::move_if_noexcept(buffer[i]);
            }
            nuevo[longitud] = std::move(valor);
            buffer.swap(nuevo);
            ++longitud;
        } else {
            buffer[longitud] = std::move(valor);
            ++longitud;
        }
    }

    // Strong guarantee for insert
    void insert(size_t pos, const T& valor) {
        if (pos > longitud) {
            throw OutOfRangeException(pos, longitud);
        }
        if (longitud >= buffer.size()) {
            BufferRAII<T> nuevo(buffer.size() * 2);
            for (size_t i = 0; i < pos; ++i) {
                nuevo[i] = buffer[i];
            }
            nuevo[pos] = valor;
            for (size_t i = pos; i < longitud; ++i) {
                nuevo[i + 1] = buffer[i];
            }
            buffer.swap(nuevo);
        } else {
            for (size_t i = longitud; i > pos; --i) {
                buffer[i] = buffer[i - 1];
            }
            buffer[pos] = valor;
        }
        ++longitud;
    }

    // Nothrow guarantee for pop_back
    void pop_back() noexcept {
        if (longitud > 0) {
            --longitud;
            buffer[longitud] = T{};
        }
    }

    // Strong guarantee for erase
    T erase(size_t pos) {
        if (pos >= longitud) {
            throw OutOfRangeException(pos, longitud);
        }
        T valor = buffer[pos];
        for (size_t i = pos + 1; i < longitud; ++i) {
            buffer[i - 1] = buffer[i];
        }
        --longitud;
        return valor;
    }

    void clear() noexcept {
        for (size_t i = 0; i < longitud; ++i) {
            buffer[i] = T{};
        }
        longitud = 0;
    }

    // Nothrow accessors
    T& operator[](size_t i) noexcept { return buffer[i]; }
    const T& operator[](size_t i) const noexcept { return buffer[i]; }

    T& at(size_t i) {
        if (i >= longitud) throw OutOfRangeException(i, longitud);
        return buffer[i];
    }

    const T& at(size_t i) const {
        if (i >= longitud) throw OutOfRangeException(i, longitud);
        return buffer[i];
    }

    T& front() {
        if (longitud == 0) throw EmptyContainerException();
        return buffer[0];
    }

    T& back() {
        if (longitud == 0) throw EmptyContainerException();
        return buffer[longitud - 1];
    }

    size_t size() const noexcept { return longitud; }
    size_t capacity() const noexcept { return buffer.size(); }
    bool empty() const noexcept { return longitud == 0; }

    void swap(SafeContainer& otro) noexcept {
        buffer.swap(otro.buffer);
        std::swap(longitud, otro.longitud);
    }

    // Iterators (nothrow)
    T* begin() noexcept { return buffer.get(); }
    T* end() noexcept { return buffer.get() + longitud; }
    const T* begin() const noexcept { return buffer.get(); }
    const T* end() const noexcept { return buffer.get() + longitud; }
};

} // namespace safe_container

// FASE 4: Serialización segura
//==============================================================================
namespace safe_container {

template <typename T>
class SafeContainerSerializer {
    static_assert(std::is_trivially_copyable_v<T>,
                  "SafeContainerSerializer requiere tipos trivialmente copiables");

public:
    static void guardar(const SafeContainer<T>& cont,
                        const std::string& archivo) {
        std::string tmp = archivo + ".sct.tmp";

        try {
            std::ofstream out(tmp, std::ios::binary);
            out.exceptions(std::ios::failbit | std::ios::badbit);

            size_t size = cont.size();
            out.write(reinterpret_cast<const char*>(&size), sizeof(size));

            if (size > 0) {
                out.write(reinterpret_cast<const char*>(cont.begin()),
                          size * sizeof(T));
            }

            out.close();

            std::error_code ec;
            if (!std::filesystem::rename(tmp, archivo, ec)) {
                throw SerializationException(
                    "No se pudo renombrar a", archivo);
            }
        }
        catch (const SerializationException&) {
            throw;
        }
        catch (const std::exception& e) {
            std::error_code ec;
            std::filesystem::remove(tmp, ec);
            throw SerializationException(
                std::string("Error de serializacion: ") + e.what(),
                archivo);
        }
    }

    static SafeContainer<T> cargar(const std::string& archivo) {
        std::ifstream in(archivo, std::ios::binary);
        if (!in) {
            throw SerializationException(
                "No se pudo abrir", archivo);
        }

        size_t size;
        in.read(reinterpret_cast<char*>(&size), sizeof(size));
        if (!in) {
            throw SerializationException(
                "Error al leer tamano de", archivo);
        }

        SafeContainer<T> cont(size);

        if (size > 0) {
            in.read(reinterpret_cast<char*>(cont.begin()),
                    size * sizeof(T));
            if (!in) {
                throw SerializationException(
                    "Error al leer datos de", archivo);
            }
        }

        return cont;
    }
};

// Serialización con verificación de integridad (checksum simple)
template <typename T>
class SafeContainerSerializerV2 {
public:
    static void guardar(const SafeContainer<T>& cont,
                        const std::string& archivo) {
        std::string tmp = archivo + ".sct2.tmp";

        try {
            std::ofstream out(tmp, std::ios::binary);
            out.exceptions(std::ios::failbit | std::ios::badbit);

            const char magia[4] = {'S', 'C', 'T', '2'};
            out.write(magia, 4);

            size_t size = cont.size();
            out.write(reinterpret_cast<const char*>(&size), sizeof(size));

            if (size > 0) {
                out.write(reinterpret_cast<const char*>(cont.begin()),
                          size * sizeof(T));
            }

            uint32_t checksum = 0;
            for (size_t i = 0; i < size; ++i) {
                const char* bytes = reinterpret_cast<const char*>(&cont[i]);
                for (size_t j = 0; j < sizeof(T); ++j) {
                    checksum += static_cast<uint8_t>(bytes[j]);
                }
            }
            out.write(reinterpret_cast<const char*>(&checksum),
                      sizeof(checksum));

            out.close();

            std::error_code ec;
            if (!std::filesystem::rename(tmp, archivo, ec)) {
                throw SerializationException(
                    "No se pudo renombrar a", archivo);
            }
        }
        catch (...) {
            std::error_code ec;
            std::filesystem::remove(tmp, ec);
            throw;
        }
    }

    static SafeContainer<T> cargar(const std::string& archivo) {
        std::ifstream in(archivo, std::ios::binary);
        if (!in) {
            throw SerializationException(
                "No se pudo abrir", archivo);
        }

        char magia[4];
        in.read(magia, 4);
        if (std::memcmp(magia, "SCT2", 4) != 0) {
            throw SerializationException(
                "Formato invalido (magia incorrecta)", archivo);
        }

        size_t size;
        in.read(reinterpret_cast<char*>(&size), sizeof(size));

        SafeContainer<T> cont(size);
        T* data = cont.begin();

        if (size > 0) {
            in.read(reinterpret_cast<char*>(data),
                    size * sizeof(T));
        }

        uint32_t checksumLeido;
        in.read(reinterpret_cast<char*>(&checksumLeido),
                sizeof(checksumLeido));

        uint32_t checksumCalc = 0;
        for (size_t i = 0; i < size; ++i) {
            const char* bytes = reinterpret_cast<const char*>(&data[i]);
            for (size_t j = 0; j < sizeof(T); ++j) {
                checksumCalc += static_cast<uint8_t>(bytes[j]);
            }
        }

        if (checksumCalc != checksumLeido) {
            throw SerializationException(
                "Checksum invalido - datos corruptos", archivo);
        }

        return cont;
    }
};

} // namespace safe_container

// FASE 5: Type traits personalizados para SafeContainer
//==============================================================================
namespace safe_container {

// Trait: verificar si T es válido para SafeContainer
template <typename T>
struct is_valid_container_type {
    static constexpr bool value =
        std::is_nothrow_destructible_v<T> &&
        (std::is_copy_constructible_v<T> || std::is_move_constructible_v<T>);
};

// Trait: verificar si T es serializable
template <typename T>
struct is_serializable {
    static constexpr bool value =
        std::is_trivially_copyable_v<T>;
};

// Trait: verificar si T es comparable
template <typename T, typename = void>
struct is_comparable : std::false_type {};

template <typename T>
struct is_comparable<T, std::void_t<
    decltype(std::declval<T>() == std::declval<T>()),
    decltype(std::declval<T>() < std::declval<T>())
>> : std::true_type {};

// Ejemplo de uso: SafeContainer con búsqueda (solo si T es comparable)
template <typename T>
class SafeContainerConBusqueda : public SafeContainer<T> {
    static_assert(is_comparable<T>::value,
                  "SafeContainerConBusqueda requiere tipos comparables");

    using Base = SafeContainer<T>;

public:
    using Base::Base;

    int buscar(const T& valor) const noexcept {
        for (size_t i = 0; i < this->size(); ++i) {
            if ((*this)[i] == valor) {
                return static_cast<int>(i);
            }
        }
        return -1;
    }

    bool contiene(const T& valor) const noexcept {
        return buscar(valor) != -1;
    }
};

// SafeContainer con ordenación (solo si T es ordenable)
template <typename T>
class SafeContainerOrdenable : public SafeContainer<T> {
    static_assert(is_comparable<T>::value,
                  "SafeContainerOrdenable requiere tipos comparables");

    using Base = SafeContainer<T>;

public:
    using Base::Base;

    void ordenar() {
        // Bubble sort con strong guarantee
        auto copia = *this;
        size_t n = copia.size();
        for (size_t i = 0; i < n - 1; ++i) {
            for (size_t j = 0; j < n - i - 1; ++j) {
                if (copia[j] > copia[j + 1]) {
                    T tmp = std::move(copia[j]);
                    copia[j] = std::move(copia[j + 1]);
                    copia[j + 1] = std::move(tmp);
                }
            }
        }
        Base::swap(copia);
    }
};

} // namespace safe_container

namespace safe_container {
namespace tests {

    std::cout << "  test_construccion... ";

    assert(c[0] == 10);

void test_push_back_strong_guarantee() {
    std::cout << "  test_push_back_strong_guarantee... ";
    SafeContainer<int> c;
    c.push_back(1);
    c.push_back(2);
    c.push_back(3);
    // Si push_back lanza, el contenedor debe permanecer intacto
    // (difícil de probar con int, pero verificamos estado)
    size_t old_size = c.size();
    int old_val = c[0];
    // No debería lanzar
    c.push_back(4);
    assert(c.size() == old_size + 1);
    assert(c[0] == old_val);
    std::cout << "OK\n";
}

FAQ - PREGUNTAS FRECUENTES
---------------------------

  P: ¿Por qué usar un buffer interno con new[] en lugar de vector?
  R: Para demostrar el manejo manual con RAII y strong guarantee.
  En producción usarías std::vector directamente.

  P: ¿Qué es `std::move_if_noexcept`?
  R: Es una función que mueve si el tipo tiene noexcept move,
  y copia si no. Útil para strong guarantee en contenedores.

  P: ¿Por qué static_assert para tipo válido?
  R: Para dar mensajes de error claros en compilación, en lugar
  de crípticos errores de template.

  P: ¿Cómo sé si mi contenedor es realmente exception-safe?
  R: Prueba insertando en medio de operaciones que lanzan, verifica
  que el estado sea consistente, y usa herramientas como sanitizers.

  P: ¿Debo usar siempre strong guarantee?
  R: Solo cuando es posible y tiene sentido. A veces strong guarantee
  es demasiado costosa y basic guarantee es suficiente.
