namespace dominio {

// Tipos soportados
enum class TipoColumna {
    Entero,
    Texto,
    Decimal
};

// Valor de celda
using ValorCelda = std::variant<int, std::string, double>;

// Esquema de columna
struct Columna {
    std::string nombre;
    TipoColumna tipo;
};

// Esquema de tabla
struct EsquemaTabla {
    std::string nombre;
    std::vector<Columna> columnas;
};

// Fila de datos
class Fila {
public:
    explicit Fila(std::vector<ValorCelda> valores)
        : valores_(std::move(valores)) {}

    const ValorCelda& get_columna(size_t idx) const {
        return valores_.at(idx);
    }

    size_t size() const { return valores_.size(); }

private:
    std::vector<ValorCelda> valores_;
};

// Resultado de consulta
class ResultadoConsulta {
public:
    ResultadoConsulta(
        std::vector<std::string> nombres_columnas,
        std::vector<Fila> filas)
        : nombres_(std::move(nombres_columnas))
        , filas_(std::move(filas)) {}

    const std::vector<std::string>& columnas() const { return nombres_; }
    const std::vector<Fila>& filas() const { return filas_; }
    size_t filas_afectadas() const { return filas_.size(); }

private:
    std::vector<std::string> nombres_;
    std::vector<Fila> filas_;
};

// Índice para búsqueda rápida
template<typename T>
class Indice {
public:
    void insertar(const T& clave, size_t posicion) {
        mapa_[clave].push_back(posicion);
    }

    std::vector<size_t> buscar(const T& clave) const {
        auto it = mapa_.find(clave);
        if (it != mapa_.end()) {
            return it->second;
        }
        return {};
    }

    void clear() { mapa_.clear(); }

private:
    std::map<T, std::vector<size_t>> mapa_;
};

} // namespace dominio

namespace aplicacion {

// Puerto de salida: almacenamiento persistente
class Almacenamiento {
public:
    virtual ~Almacenamiento() = default;
    virtual void guardar_tabla(const std::string& nombre,
        const std::vector<dominio::Fila>& filas) = 0;
    virtual std::optional<std::vector<dominio::Fila>>
        cargar_tabla(const std::string& nombre) = 0;
    virtual bool tabla_existe(const std::string& nombre) = 0;
};

// Puerto de salida: interfaz de usuario
class InterfazUsuario {
public:
    virtual ~InterfazUsuario() = default;
    virtual void mostrar_resultado(
        const dominio::ResultadoConsulta& resultado) = 0;
    virtual void mostrar_error(const std::string& mensaje) = 0;
    virtual void mostrar_mensaje(const std::string& mensaje) = 0;
};

// Servicio principal de base de datos
class ServicioBaseDatos {
public:
    ServicioBaseDatos(
        std::unique_ptr<Almacenamiento> almacenamiento,
        std::unique_ptr<InterfazUsuario> interfaz)
        : almacen_(std::move(almacenamiento))
        , interfaz_(std::move(interfaz))
    {}

    void crear_tabla(const std::string& nombre,
                     const std::vector<dominio::Columna>& columnas) {
        if (almacen_->tabla_existe(nombre)) {
            interfaz_->mostrar_error(
                "La tabla '" + nombre + "' ya existe");
            return;
        }

        esquemas_[nombre] = dominio::EsquemaTabla{nombre, columnas};
        almacen_->guardar_tabla(nombre, {});

        interfaz_->mostrar_mensaje(
            "Tabla '" + nombre + "' creada correctamente");
    }

    void insertar(const std::string& nombre_tabla,
                  std::vector<dominio::ValorCelda> valores) {
        auto it = esquemas_.find(nombre_tabla);
        if (it == esquemas_.end()) {
            interfaz_->mostrar_error(
                "Tabla no encontrada: " + nombre_tabla);
            return;
        }

        if (valores.size() != it->second.columnas.size()) {
            interfaz_->mostrar_error(
                "Número incorrecto de columnas");
            return;
        }

        auto filas = almacen_->cargar_tabla(nombre_tabla)
                         .value_or(std::vector<dominio::Fila>{});
        filas.emplace_back(std::move(valores));
        almacen_->guardar_tabla(nombre_tabla, filas);

        interfaz_->mostrar_mensaje("1 fila insertada");
    }

    dominio::ResultadoConsulta seleccionar(
        const std::string& nombre_tabla) {
        auto it = esquemas_.find(nombre_tabla);
        if (it == esquemas_.end()) {
            throw std::runtime_error(
                "Tabla no encontrada: " + nombre_tabla);
        }

        std::vector<std::string> nombres;
        for (const auto& col : it->second.columnas) {
            nombres.push_back(col.nombre);
        }

        auto filas = almacen_->cargar_tabla(nombre_tabla)
                         .value_or(std::vector<dominio::Fila>{});

        return {nombres, std::move(filas)};
    }

private:
    std::unique_ptr<Almacenamiento> almacen_;
    std::unique_ptr<InterfazUsuario> interfaz_;
    std::map<std::string, dominio::EsquemaTabla> esquemas_;
};

} // namespace aplicacion

namespace infraestructura {

// Almacenamiento en archivo binario
class AlmacenamientoArchivo : public aplicacion::Almacenamiento {
public:
    explicit AlmacenamientoArchivo(const std::string& directorio)
        : directorio_(directorio) {}

    void guardar_tabla(const std::string& nombre,
        const std::vector<dominio::Fila>& filas) override {
        std::string ruta = directorio_ + "/" + nombre + ".db";
        std::ofstream archivo(ruta, std::ios::binary);

        size_t num_filas = filas.size();
        archivo.write(reinterpret_cast<const char*>(&num_filas),
                      sizeof(num_filas));

        for (const auto& fila : filas) {
            size_t num_cols = fila.size();
            archivo.write(reinterpret_cast<const char*>(&num_cols),
                          sizeof(num_cols));
            // Serializar cada celda...
        }
    }

    std::optional<std::vector<dominio::Fila>>
    cargar_tabla(const std::string& nombre) override {
        std::string ruta = directorio_ + "/" + nombre + ".db";
        std::ifstream archivo(ruta, std::ios::binary);
        if (!archivo) {
            return std::nullopt;
        }
        // Deserializar...
        return std::vector<dominio::Fila>{};
    }

    bool tabla_existe(const std::string& nombre) override {
        std::string ruta = directorio_ + "/" + nombre + ".db";
        std::ifstream archivo(ruta);
        return archivo.good();
    }

private:
    std::string directorio_;
};

// Interfaz de línea de comandos
class InterfazCLI : public aplicacion::InterfazUsuario {
public:
    void mostrar_resultado(
        const dominio::ResultadoConsulta& resultado) override {
        // Mostrar nombres de columnas
        for (const auto& col : resultado.columnas()) {
            std::cout << col << "\t";
        }
        std::cout << "\n" << std::string(40, '-') << "\n";

        // Mostrar filas
        for (const auto& fila : resultado.filas()) {
            for (size_t i = 0; i < fila.size(); ++i) {
                const auto& val = fila.get_columna(i);
                if (std::holds_alternative<int>(val)) {
                    std::cout << std::get<int>(val);
                } else if (std::holds_alternative<double>(val)) {
                    std::cout << std::get<double>(val);
                } else {
                    std::cout << std::get<std::string>(val);
                }
                std::cout << "\t";
            }
            std::cout << "\n";
        }

        std::cout << resultado.filas_afectadas()
                  << " filas encontradas\n";
    }

    void mostrar_error(const std::string& mensaje) override {
        std::cerr << "[ERROR] " << mensaje << "\n";
    }

    void mostrar_mensaje(const std::string& mensaje) override {
        std::cout << mensaje << "\n";
    }
};

} // namespace infraestructura

int main_proyecto_final() {
    // Configurar dependencias
    auto almacenamiento = std::make_unique<
        infraestructura::AlmacenamientoArchivo>("./data");
    auto interfaz = std::make_unique<
        infraestructura::InterfazCLI>();

    auto db = aplicacion::ServicioBaseDatos(
        std::move(almacenamiento),
        std::move(interfaz));

    // Crear tabla
    db.crear_tabla("usuarios", {
        {"id", dominio::TipoColumna::Entero},
        {"nombre", dominio::TipoColumna::Texto},
        {"email", dominio::TipoColumna::Texto}
    });

    // Insertar datos
    db.insertar("usuarios", {
        dominio::ValorCelda(1),
        dominio::ValorCelda(std::string("Ana García")),
        dominio::ValorCelda(std::string("ana@email.com"))
    });

    db.insertar("usuarios", {
        dominio::ValorCelda(2),
        dominio::ValorCelda(std::string("Carlos López")),
        dominio::ValorCelda(std::string("carlos@email.com"))
    });

    // Consultar
    auto resultado = db.seleccionar("usuarios");
    // Mostrar resultado sería responsabilidad de la interfaz

    return 0;
}
