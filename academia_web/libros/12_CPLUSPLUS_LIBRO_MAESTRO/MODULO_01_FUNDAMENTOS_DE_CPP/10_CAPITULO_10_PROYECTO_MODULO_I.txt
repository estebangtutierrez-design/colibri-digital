===============================================================================
PREGUNTAS FRECUENTES
===============================================================================

P: Por que usar memoria dinamica en lugar de un arreglo fijo?
R: Para que el programa pueda manejar cualquier cantidad de
registros sin limites predefinidos. El arreglo crece
dinamicamente segun se necesite.

P: Que hace exactamente el operador new?
R: Reserva memoria en el heap (monticulo) y llama al constructor
del objeto. Devuelve un puntero a la memoria reservada.

P: Por que necesito delete[] y no delete para arreglos?
R: delete[] llama a los destructores de TODOS los elementos
del arreglo. delete solo llama al destructor del primer
elemento, causando memory leak para el resto.

P: Como se evita la fragmentacion del arreglo al eliminar?
R: Al eliminar un elemento, se desplazan los elementos
siguientes hacia la izquierda para llenar el hueco.
Esto mantiene el arreglo compacto.

P: Es necesario ordenar por fecha?
R: Para calcular ciertas estadisticas (como mediana) se requiere
orden. Para el calculo de maximos/minimos no es necesario,
pero mejora la presentacion.

// tests/test_temperatura.cpp
#include "../src/temperatura.h"
#include <iostream>
#include <cassert>
#include <cmath>

bool aproximadamente(double a, double b, double eps = 1e-9) {
    return std::fabs(a - b) < eps;
}

void testConversiones() {
    std::cout << "Probando conversiones..." << std::endl;

    // Celsius -> Fahrenheit
    assert(aproximadamente(celsiusAFahrenheit(0), 32.0));
    assert(aproximadamente(celsiusAFahrenheit(100), 212.0));
    assert(aproximadamente(celsiusAFahrenheit(-40), -40.0));
    std::cout << "  Celsius->Fahrenheit: OK" << std::endl;

    // Fahrenheit -> Celsius
    assert(aproximadamente(fahrenheitACelsius(32), 0.0));
    assert(aproximadamente(fahrenheitACelsius(212), 100.0));
    assert(aproximadamente(fahrenheitACelsius(-40), -40.0));
    std::cout << "  Fahrenheit->Celsius: OK" << std::endl;

    // Celsius -> Kelvin
    assert(aproximadamente(celsiusAKelvin(0), 273.15));
    assert(aproximadamente(celsiusAKelvin(100), 373.15));
    std::cout << "  Celsius->Kelvin: OK" << std::endl;

    // Kelvin -> Celsius
    assert(aproximadamente(kelvinACelsius(273.15), 0.0));
    assert(aproximadamente(kelvinACelsius(373.15), 100.0));
    std::cout << "  Kelvin->Celsius: OK" << std::endl;

    // Redondeo
    assert(aproximadamente(celsiusAFahrenheit(37), 98.6, 0.1));
    std::cout << "  Conversion cuerpo humano: OK" << std::endl;

    std::cout << "  Todas las conversiones: PASARON!" << std::endl;
}

void testEstadisticas() {
    std::cout << "Probando estadisticas..." << std::endl;

    SistemaTemperatura sis;
    inicializarSistema(sis, 5);

    // Agregar datos conocidos
    agregarRegistro(sis, {1, 1, 2024, 30.0, 20.0});  // Max:30 Min:20
    agregarRegistro(sis, {2, 1, 2024, 32.0, 22.0});  // Max:32 Min:22
    agregarRegistro(sis, {3, 1, 2024, 28.0, 18.0});  // Max:28 Min:18

    // Promedios
    assert(aproximadamente(calcularPromedioMax(sis), 30.0));
    assert(aproximadamente(calcularPromedioMin(sis), 20.0));
    std::cout << "  Promedios: OK" << std::endl;

    // Max y min global
    assert(aproximadamente(calcularMaxGlobal(sis), 32.0));
    assert(aproximadamente(calcularMinGlobal(sis), 18.0));
    std::cout << "  Max/Min global: OK" << std::endl;

    // Sistema vacio
    SistemaTemperatura vacio;
    inicializarSistema(vacio, 1);
    assert(aproximadamente(calcularPromedioMax(vacio), 0.0));
    assert(aproximadamente(calcularMaxGlobal(vacio), 0.0));
    std::cout << "  Sistema vacio: OK" << std::endl;

    liberarSistema(vacio);
    liberarSistema(sis);

    std::cout << "  Todas las estadisticas: PASARON!" << std::endl;
}

void testSistemaGestion() {
    std::cout << "Probando sistema de gestion..." << std::endl;

    // Inicializar
    SistemaTemperatura sis;
    inicializarSistema(sis, 2);
    assert(sis.cantidad == 0);
    assert(sis.capacidad == 2);
    std::cout << "  Inicializacion: OK" << std::endl;

    // Agregar hasta exceder capacidad
    agregarRegistro(sis, {1, 1, 2024, 25.0, 15.0});
    agregarRegistro(sis, {2, 1, 2024, 26.0, 16.0});
    agregarRegistro(sis, {3, 1, 2024, 27.0, 17.0});  // Redimensiona
    assert(sis.cantidad == 3);
    assert(sis.capacidad == 4);  // Se duplico de 2 a 4
    std::cout << "  Redimension automatica: OK" << std::endl;

    // Ordenar por fecha
    SistemaTemperatura desordenado;
    inicializarSistema(desordenado, 5);
    agregarRegistro(desordenado, {3, 1, 2024, 28.0, 18.0});
    agregarRegistro(desordenado, {1, 1, 2024, 30.0, 20.0});
    agregarRegistro(desordenado, {2, 1, 2024, 32.0, 22.0});
    ordenarPorFecha(desordenado);
    assert(desordenado.registros[0].dia == 1);
    assert(desordenado.registros[1].dia == 2);
    assert(desordenado.registros[2].dia == 3);
    std::cout << "  Orden por fecha: OK" << std::endl;

    liberarSistema(desordenado);
    liberarSistema(sis);

    std::cout << "  Sistema de gestion: PASARON!" << std::endl;
}

16. POSIBLES ERRORES Y SOLUCIONES
===============================================================================

| ERROR                           | CAUSA                          | SOLUCION                          |
|----------------------------------|--------------------------------|------------------------------------|
| Segmentation fault              | Puntero nulo o invalido        | Verificar nullptr antes de usar    |
| double free or corruption       | Dos delete al mismo ptr        | nullptr despues de delete          |
| Undefined reference             | Falta enlazar .cpp             | Compilar todos los .cpp            |
| No such file or directory       | Archivo no existe              | Verificar ruta del archivo         |
| Permission denied               | Sin permisos de escritura      | Cambiar permisos del directorio    |
| Memory leak                     | Olvidar delete                 | Usar smart pointers                |
| Archivo corrupto                | Formato CSV incorrecto         | Validar formato al cargar          |
| Bucle infinito                  | Condicion de salida incorrecta | Revisar logica del bucle           |
