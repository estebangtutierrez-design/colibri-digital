def login():
    data = request.get_json()
    username = data.get("username")
    password = data.get("password")
    
    if username == "admin" and password == "secret":
        session["user"] = username
        return jsonify({"status": "ok"})
    return jsonify({"error": "Invalid credentials"}), 401
