Coverage: pytest-cov. Factory boy para test data.

Tox: multi-version testing.

EJEMPLO DE CÓDIGO
----------------------------------------

import pytest
from unittest.mock import patch, MagicMock

@pytest.fixture
def mock_mail():
    with patch("app.mail.send") as mock:
        yield mock

def test_registration_sends_email(client, mock_mail):
    response = client.post("/register", json={
        "name": "Alice", "email": "alice@test.com"
    })
    assert response.status_code == 201
    mock_mail.assert_called_once()
