Broadcast. Emit desde fuera del handler.

EJEMPLO DE CÓDIGO
----------------------------------------

from flask_socketio import SocketIO, emit, join_room

socketio = SocketIO(app, cors_allowed_origins="*")

@socketio.on("message")
def handle_message(data):
    emit("response", {"text": f"Echo: {data}"}, broadcast=True)

@socketio.on("join")
def on_join(data):
    join_room(data["room"])
    emit("status", f"{data["user"]} joined", room=data["room"])

if __name__ == "__main__":
    socketio.run(app, debug=True)
