Querido estudiante…

¿Qué pasa si quieres controlar CÓMO se lee o modifica una propiedad?

Ejemplo: cuando asignas una edad, quieres validar que sea positiva.

Ahí entran los GETTERS y SETTERS.

- Getter: método que se lee como propiedad.
- Setter: método que se asigna como propiedad.

===============================================================================
1. GETTER
===============================================================================

```javascript
class Persona {
    constructor(nombre, apellido) {
        this.nombre = nombre;
        this.apellido = apellido;
    }

    // Getter: se accede sin ()
    get nombreCompleto() {
        return this.nombre + " " + this.apellido;
    }

    get iniciales() {
        return this.nombre[0] + "." + this.apellido[0] + ".";
    }
}

let p = new Persona("Ana", "García");
console.log(p.nombreCompleto);  // Ana García (sin paréntesis)
console.log(p.iniciales);       // A.G.
```

===============================================================================
2. SETTER CON VALIDACIÓN
===============================================================================

```javascript
class Estudiante {
    constructor(nombre) {
        this._nombre = nombre;
        this._edad = 0;
    }

    // Getter
    get edad() {
        return this._edad;
    }

    // Setter con validación
    set edad(valor) {
        if (typeof valor !== "number") {
            console.log("❌ La edad debe ser un número");
            return;
        }
        if (valor < 0 || valor > 120) {
            console.log("❌ Edad inválida (0-120)");
            return;
        }
        this._edad = valor;
    }
}

let est = new Estudiante("Luis");
est.edad = 25;     // Usa el setter
console.log(est.edad);  // 25 (usa el getter)

est.edad = -5;     // ❌ Edad inválida
est.edad = "treinta";  // ❌ La edad debe ser un número
```

3. PROPIEDADES CALCULADAS
===============================================================================

```javascript
class Circulo {
    constructor(radio) {
        this.radio = radio;
    }

    get area() {
        return Math.PI * this.radio * this.radio;
    }

    get circunferencia() {
        return 2 * Math.PI * this.radio;
    }

    set diametro(valor) {
        this.radio = valor / 2;
    }

    get diametro() {
        return this.radio * 2;
    }
}

let c = new Circulo(5);
console.log("Radio: " + c.radio);               // 5
console.log("Diámetro: " + c.diametro);          // 10
console.log("Área: " + c.area.toFixed(2));       // 78.54
console.log("Circunferencia: " + c.circunferencia.toFixed(2));  // 31.42

c.diametro = 20;  // Setter → cambia radio a 10
console.log("Nuevo radio: " + c.radio);          // 10
```

4. SETTER CON TRANSFORMACIÓN
===============================================================================

```javascript
class Usuario {
    constructor(email) {
        this._email = "";
        this.email = email;  // Usa el setter
    }

    get email() {
        return this._email;
    }

    set email(valor) {
        // Guardar siempre en minúsculas
        this._email = valor.toLowerCase().trim();
    }

    get dominio() {
        return this._email.split("@")[1] || "dominio inválido";
    }
}

let u = new Usuario("  Ana@Gmail.COM  ");
console.log(u.email);     // ana@gmail.com
console.log(u.dominio);   // gmail.com
```

MINI PROYECTO: CUENTA BANCARIA CON SETTERS

```javascript
class Cuenta {
    #saldo = 0;

    constructor(titular) {
        this.titular = titular;
    }

    get saldo() {
        return "$" + this.#saldo.toFixed(2);
    }

    set saldo(valor) {
        console.log("⚠️ No puedes asignar saldo directamente. Usa depositar/retirar");
    }

    depositar(monto) {
        if (monto <= 0) return console.log("❌ Monto inválido");
        this.#saldo += monto;
        console.log("✅ Depósito: $" + monto + ". Saldo: " + this.saldo);
    }

    retirar(monto) {
        if (monto <= 0) return console.log("❌ Monto inválido");
        if (monto > this.#saldo) return console.log("❌ Saldo insuficiente");
        this.#saldo -= monto;
        console.log("✅ Retiro: $" + monto + ". Saldo: " + this.saldo);
    }

    get enDescubierto() {
        return this.#saldo < 0;
    }
}

let cuenta = new Cuenta("Ana");
cuenta.depositar(1000);      // ✅ Depósito: $1000
console.log(cuenta.saldo);   // $1000.00
cuenta.saldo = 500000;       // ⚠️ No puedes asignar saldo directamente
console.log(cuenta.saldo);   // $1000.00
```
