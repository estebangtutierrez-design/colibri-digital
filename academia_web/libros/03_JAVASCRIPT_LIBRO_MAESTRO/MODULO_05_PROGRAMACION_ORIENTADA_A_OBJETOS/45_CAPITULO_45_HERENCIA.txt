1. CLASE PADRE E HIJA
===============================================================================

```javascript
// Clase padre
class Animal {
    constructor(nombre) {
        this.nombre = nombre;
        this.vivo = true;
    }

    respirar() {
        console.log(this.nombre + " está respirando");
    }

    comer() {
        console.log(this.nombre + " está comiendo");
    }
}

// Clase hija
class Perro extends Animal {
    constructor(nombre, raza) {
        super(nombre);  // Llama al constructor de Animal
        this.raza = raza;
    }

    ladrar() {
        console.log(this.nombre + " dice: ¡Guau!");
    }
}

let firulais = new Perro("Firulais", "Labrador");
firulais.respirar();  // Heredado de Animal
firulais.comer();      // Heredado de Animal
firulais.ladrar();     // Propio de Perro
```

2. SOBRESCRIBIR MÉTODOS
===============================================================================

```javascript
class Animal2 {
    constructor(nombre) {
        this.nombre = nombre;
    }

    hacerSonido() {
        console.log(this.nombre + " hace un sonido genérico");
    }
}

class Gato extends Animal2 {
    hacerSonido() {
        console.log(this.nombre + " dice: ¡Miau!");
    }
}

class Vaca extends Animal2 {
    hacerSonido() {
        console.log(this.nombre + " dice: ¡Muuu!");
    }
}

let animales = [
    new Gato("Whiskers"),
    new Vaca("Lola"),
    new Animal2("Genérico")
];

animales.forEach(a => a.hacerSonido());
// Whiskers dice: ¡Miau!
// Lola dice: ¡Muuu!
// Genérico hace un sonido genérico
```

3. SUPER PARA LLAMAR AL PADRE
===============================================================================

```javascript
class Vehiculo {
    constructor(marca, año) {
        this.marca = marca;
        this.año = año;
        this.encendido = false;
    }

    arrancar() {
        this.encendido = true;
        console.log(this.marca + " arrancó");
    }

    info() {
        return this.marca + " (" + this.año + ")";
    }
}

class Coche extends Vehiculo {
    constructor(marca, año, puertas) {
        super(marca, año);
        this.puertas = puertas;
    }

    arrancar() {
        super.arrancar();  // Llama al método del padre
        console.log("✅ " + this.puertas + " puertas bloqueadas");
    }

    info() {
        return super.info() + " - " + this.puertas + " puertas";
    }
}

let coche = new Coche("Toyota", 2024, 5);
coche.arrancar();
// Toyota arrancó
// ✅ 5 puertas bloqueadas
console.log(coche.info());  // Toyota (2024) - 5 puertas
```

4. CADENA DE HERENCIA
===============================================================================

```javascript
class Persona {
    constructor(nombre) {
        this.nombre = nombre;
    }
    saludar() {
        return "Hola, soy " + this.nombre;
    }
}

class Empleado extends Persona {
    constructor(nombre, empresa) {
        super(nombre);
        this.empresa = empresa;
    }
    trabajar() {
        return this.nombre + " trabaja en " + this.empresa;
    }
}

class Desarrollador extends Empleado {
    constructor(nombre, empresa, lenguaje) {
        super(nombre, empresa);
        this.lenguaje = lenguaje;
    }
    programar() {
        return this.nombre + " programa en " + this.lenguaje;
    }
}

let dev = new Desarrollador("Ana", "Google", "JavaScript");
console.log(dev.saludar());    // Hola, soy Ana
console.log(dev.trabajar());   // Ana trabaja en Google
console.log(dev.programar());  // Ana programa en JavaScript
```

MINI PROYECTO: SISTEMA DE EMPLEADOS
===============================================================================

```javascript
class Empleado2 {
    constructor(nombre, salarioBase) {
        this.nombre = nombre;
        this.salarioBase = salarioBase;
    }

    calcularSalario() {
        return this.salarioBase;
    }

    descripcion() {
        return this.nombre + ": $" + this.calcularSalario();
    }
}

class Gerente extends Empleado2 {
    calcularSalario() {
        return this.salarioBase + 500;  // Bono de gerente
    }
}

class Vendedor extends Empleado2 {
    constructor(nombre, salarioBase, comisiones) {
        super(nombre, salarioBase);
        this.comisiones = comisiones;
    }
    calcularSalario() {
        return this.salarioBase + this.comisiones;
    }
}

class Pasante extends Empleado2 {
    calcularSalario() {
        return this.salarioBase * 0.5;  // Medio sueldo
    }
}

let empleados = [
    new Gerente("Ana", 3000),
    new Vendedor("Luis", 1500, 800),
    new Pasante("Sofía", 2000)
];

empleados.forEach(e => console.log(e.descripcion()));
// Ana: $3500
// Luis: $2300
// Sofía: $1000
```

RESUMEN
===============================================================================

✔ extends: hereda de otra clase.
✔ super(): llama al constructor del padre.
✔ super.metodo(): llama a un método del padre.
✔ Sobrescribir: redefinir método en la hija.
✔ Jerarquía: clase padre → hija → nieta.
