1. ABSTRACCIÓN PRÁCTICA
===============================================================================

```javascript
class Cafetera {
    #agua = 0;
    #cafe = 0;
    #encendida = false;

    // Interfaz simple
    encender() {
        this.#encendida = true;
        console.log("☕ Cafetera encendida");
    }

    hacerCafe() {
        if (!this.#encendida) return console.log("❌ Enciende la cafetera");
        if (this.#agua < 200) return console.log("❌ Sin agua");
        if (this.#cafe < 10) return console.log("❌ Sin café");

        this.#agua -= 200;
        this.#cafe -= 10;
        console.log("☕ ¡Café listo!");
    }

    llenarAgua() {
        this.#agua = 1000;
        console.log("💧 Agua llenada");
    }

    llenarCafe() {
        this.#cafe = 100;
        console.log("🫘 Café llenado");
    }
}

// El usuario solo ve: encender, hacerCafe, llenarAgua, llenarCafe
let cafetera = new Cafetera();
cafetera.llenarAgua();
cafetera.llenarCafe();
cafetera.encender();
cafetera.hacerCafe();
// El usuario no necesita saber cómo se gestiona el agua o el café internamente
```

2. CLASE ABSTRACTA (SIMULADA)
===============================================================================

JavaScript no tiene clases abstractas nativas.
Pero puedes simularlas lanzando errores.

```javascript
class Animal {
    constructor(nombre) {
        if (new.target === Animal) {
            throw new Error("No puedes instanciar Animal directamente");
        }
        this.nombre = nombre;
    }

    hacerSonido() {
        throw new Error("Método abstracto: implementar en subclase");
    }

    moverse() {
        throw new Error("Método abstracto: implementar en subclase");
    }

    dormir() {
        console.log(this.nombre + " está durmiendo zzz");
    }
}

class Perro extends Animal {
    hacerSonido() { console.log(this.nombre + " dice: ¡Guau!"); }
    moverse() { console.log(this.nombre + " corre en 4 patas"); }
}

class Pajaro extends Animal {
    hacerSonido() { console.log(this.nombre + " dice: ¡Pío!"); }
    moverse() { console.log(this.nombre + " vuela"); }
}

// let a = new Animal("Genérico");  // ❌ Error
let perro = new Perro("Firulais");
perro.hacerSonido();  // Firulais dice: ¡Guau!
perro.moverse();      // Firulais corre en 4 patas
perro.dormir();       // Firulais está durmiendo (heredado)
```

3. INTERFAZ SIMPLE, COMPLEJIDAD OCULTA
===============================================================================

```javascript
class ReproductorMusica {
    #estado = "detenido";
    #cancionActual = null;
    #volumen = 50;
    #playlist = [];

    // Interfaz simple
    play() {
        if (this.#playlist.length === 0) return console.log("❌ Playlist vacía");
        this.#estado = "reproduciendo";
        if (!this.#cancionActual) {
            this.#cancionActual = this.#playlist[0];
        }
        console.log("▶️ Reproduciendo: " + this.#cancionActual);
    }

    stop() {
        this.#estado = "detenido";
        console.log("⏹️ Detenido");
    }

    next() {
        let idx = this.#playlist.indexOf(this.#cancionActual);
        let nextIdx = (idx + 1) % this.#playlist.length;
        this.#cancionActual = this.#playlist[nextIdx];
        console.log("⏭️ Siguiente: " + this.#cancionActual);
        if (this.#estado === "reproduciendo") this.play();
    }

    agregarCancion(cancion) {
        this.#playlist.push(cancion);
        console.log("🎵 Agregada: " + cancion);
    }

    subirVolumen() {
        this.#volumen = Math.min(100, this.#volumen + 10);
        console.log("🔊 Volumen: " + this.#volumen);
    }

    bajarVolumen() {
        this.#volumen = Math.max(0, this.#volumen - 10);
        console.log("🔉 Volumen: " + this.#volumen);
    }
}

let reproductor = new ReproductorMusica();
reproductor.agregarCancion("Bohemian Rhapsody");
reproductor.agregarCancion("Imagine");
reproductor.agregarCancion("Billie Jean");
reproductor.play();
reproductor.next();
reproductor.subirVolumen();
reproductor.stop();
// Usuario no necesita saber cómo se gestiona la playlist internamente
```

4. BUENA VS MALA ABSTRACCIÓN
===============================================================================

```javascript
// 🚫 MALA abstracción: expone demasiado
class UsuarioMal {
    constructor(nombre, password) {
        this.nombre = nombre;
        this._passwordHash = this.#hash(password);
        this._intentosLogin = 0;
        this._bloqueado = false;
        this._ultimoAcceso = null;
    }

    #hash(texto) {
        let hash = 0;
        for (let i = 0; i < texto.length; i++) {
            hash = ((hash << 5) - hash) + texto.charCodeAt(i);
        }
        return hash;
    }

    login(password) { /* ... */ }
    getNombre() { return this.nombre; }
}

// ✅ BUENA abstracción: solo expone lo necesario
class UsuarioBien {
    #nombre;
    #passwordHash;
    #intentosLogin = 0;
    #bloqueado = false;

    constructor(nombre, password) {
        this.#nombre = nombre;
        this.#passwordHash = this.#hash(password);
    }

    #hash(texto) { /* ... */ }

    login(password) {
        if (this.#bloqueado) return "Usuario bloqueado";
        if (this.#hash(password) === this.#passwordHash) {
            this.#intentosLogin = 0;
            return "✅ Login exitoso";
        }
        this.#intentosLogin++;
        if (this.#intentosLogin >= 3) this.#bloqueado = true;
        return "❌ Contraseña incorrecta";
    }

    getNombre() { return this.#nombre; }
}
```

```javascript
class Microondas {
    #encendido = false;
    #tiempo = 0;
    #cocinando = false;

    abrirPuerta() {
        this.#cocinando = false;
        console.log("🚪 Puerta abierta");
    }

    cerrarPuerta() {
        console.log("🚪 Puerta cerrada");
    }

    setTiempo(segundos) {
        if (segundos < 0 || segundos > 600) {
            return console.log("❌ Tiempo: 0-600 segundos");
        }
        this.#tiempo = segundos;
        console.log("⏱️ Tiempo: " + segundos + "s");
    }

    iniciar() {
        if (this.#tiempo <= 0) return console.log("❌ Configura el tiempo");
        this.#cocinando = true;
        console.log("🔥 Calentando... (" + this.#tiempo + "s)");
        setTimeout(() => {
            this.#cocinando = false;
            console.log("🔔 ¡Comida lista!");
        }, this.#tiempo * 100);
    }

    get estado() {
        if (this.#cocinando) return "🔥 Cocinando";
        return "💤 En reposo";
    }
}

let micro = new Microondas();
micro.setTiempo(30);
micro.iniciar();
console.log(micro.estado);
```
