Querido estudiante…

Cuando usas new Coche(), JavaScript busca el CONSTRUCTOR.

El constructor es un método especial que:
1. Se ejecuta automáticamente.
2. Inicializa las propiedades del objeto.
3. Puede recibir parámetros.

Es el "nacimiento" de cada objeto.

===============================================================================
1. SINTAXIS BÁSICA
===============================================================================

```javascript
class Usuario {
    constructor(nombre, email) {
        this.nombre = nombre;
        this.email = email;
        this.activo = true;  // Valor por defecto
    }
}

let ana = new Usuario("Ana García", "ana@email.com");
console.log(ana.nombre);   // Ana García
console.log(ana.email);    // ana@email.com
console.log(ana.activo);   // true
```

===============================================================================
2. CONSTRUCTOR CON PARÁMETROS POR DEFECTO
===============================================================================

```javascript
class Producto {
    constructor(nombre, precio = 0, stock = 0) {
        this.nombre = nombre;
        this.precio = precio;
        this.stock = stock;
        this.id = Date.now();  // ID único basado en timestamp
    }
}

let laptop = new Producto("Laptop", 1200, 10);
let generico = new Producto("Genérico");

console.log(laptop.precio);   // 1200
console.log(laptop.id);       // 123456789...
console.log(generico.precio); // 0
console.log(generico.stock);  // 0
```

3. VALIDACIÓN EN EL CONSTRUCTOR
===============================================================================

```javascript
class CuentaBancaria {
    constructor(titular, saldoInicial = 0) {
        if (!titular || typeof titular !== "string") {
            throw new Error("El titular es obligatorio");
        }
        if (saldoInicial < 0) {
            throw new Error("El saldo inicial no puede ser negativo");
        }
        this.titular = titular;
        this.saldo = saldoInicial;
        this.numeroCuenta = "ES" + Math.floor(Math.random() * 1000000000);
    }
}

let cuenta = new CuentaBancaria("Ana Pérez", 500);
console.log(cuenta.titular);      // Ana Pérez
console.log(cuenta.numeroCuenta); // ES123456789

// let mala = new CuentaBancaria("", -100);  // ❌ Error
```

MINI PROYECTO: FÁBRICA DE USUARIOS
===============================================================================

```javascript
class Libro {
    constructor(titulo, autor, año, genero) {
        this.titulo = titulo;
        this.autor = autor;
        this.año = año;
        this.genero = genero;
        this.prestado = false;
        this.fechaPrestamo = null;
    }

    prestar() {
        if (this.prestado) return console.log("❌ Ya está prestado");
        this.prestado = true;
        this.fechaPrestamo = new Date();
        console.log("📖 Prestado: " + this.titulo);
    }

    devolver() {
        if (!this.prestado) return console.log("❌ No estaba prestado");
        this.prestado = false;
        this.fechaPrestamo = null;
        console.log("📚 Devuelto: " + this.titulo);
    }

    info() {
        let estado = this.prestado ? "Prestado" : "Disponible";
        return this.titulo + " - " + this.autor + " (" + this.año + ") [" + estado + "]";
    }
}

let libros = [
    new Libro("Cien años de soledad", "G. García Márquez", 1967, "Novela"),
    new Libro("1984", "George Orwell", 1949, "Ciencia Ficción"),
    new Libro("El Quijote", "Cervantes", 1605, "Clásico")
];

libros[0].prestar();
libros.forEach(l => console.log(l.info()));
libros[0].devolver();
libros.forEach(l => console.log(l.info()));
```
