BIENVENIDA DEL PROFESOR BÚHO
===============================================================================

Querido estudiante…

Hasta ahora, cada objeto tiene sus propias propiedades y métodos.

Pero ¿y si quieres que algo sea COMPARTIDO por todas las instancias?
¿O un método que no necesite un objeto para funcionar?

Ahí entran los miembros ESTÁTICOS (static).

- Propiedad estática: pertenece a la clase.
- Método estático: se llama desde la clase, no desde la instancia.

===============================================================================
1. PROPIEDADES ESTÁTICAS
===============================================================================

```javascript
class Usuario {
    static contador = 0;

    constructor(nombre) {
        this.nombre = nombre;
        Usuario.contador++;  // Incrementa en cada new
        this.id = Usuario.contador;
    }
}

let u1 = new Usuario("Ana");
let u2 = new Usuario("Luis");
let u3 = new Usuario("Sofía");

console.log(Usuario.contador);  // 3
console.log(u1.id);             // 1
console.log(u2.id);             // 2
console.log(u3.id);             // 3

// Las instancias NO tienen acceso a la estática
console.log(u1.contador);  // undefined
```

===============================================================================
2. MÉTODOS ESTÁTICOS
===============================================================================

```javascript
class CalculadoraMatematica {
    static sumar(a, b) { return a + b; }
    static restar(a, b) { return a - b; }
    static multiplicar(a, b) { return a * b; }
    static dividir(a, b) {
        if (b === 0) return "Error";
        return a / b;
    }
}

// Se llaman desde la clase, sin instanciar
console.log(CalculadoraMatematica.sumar(10, 5));         // 15
console.log(CalculadoraMatematica.multiplicar(4, 3));    // 12
console.log(CalculadoraMatematica.dividir(10, 0));       // Error
```

3. EJEMPLOS ÚTILES DE MÉTODOS ESTÁTICOS
===============================================================================

```javascript
class Validador {
    static email(email) {
        return email.includes("@") && email.includes(".");
    }

    static telefono(tel) {
        return tel.length >= 10 && /^\d+$/.test(tel);
    }

    static edad(valor) {
        return typeof valor === "number" && valor >= 0 && valor <= 120;
    }

    static noVacio(valor) {
        return typeof valor === "string" && valor.trim().length > 0;
    }
}

console.log(Validador.email("ana@email.com"));      // true
console.log(Validador.email("invalido"));            // false
console.log(Validador.telefono("5551234"));          // false (muy corto)
console.log(Validador.edad(25));                     // true
console.log(Validador.edad(150));                    // false
```

4. CONTADOR DE INSTANCIAS AVANZADO
===============================================================================

```javascript
class Producto {
    static #totalProductos = 0;
    static #ingresosTotales = 0;

    constructor(nombre, precio) {
        this.nombre = nombre;
        this.precio = precio;
        this.id = ++Producto.#totalProductos;
    }

    vender(cantidad = 1) {
        let ingreso = this.precio * cantidad;
        Producto.#ingresosTotales += ingreso;
        return "💰 Vendido: " + this.nombre + " x" + cantidad +
               " ($" + ingreso + ")";
    }

    static getTotalProductos() {
        return Producto.#totalProductos;
    }

    static getIngresosTotales() {
        return "$" + Producto.#ingresosTotales;
    }

    static reporte() {
        console.log("=== REPORTE ESTÁTICO ===");
        console.log("Productos creados: " + Producto.#totalProductos);
        console.log("Ingresos totales: " + Producto.getIngresosTotales());
    }
}

let p1 = new Producto("Laptop", 1200);
let p2 = new Producto("Mouse", 25);
let p3 = new Producto("Teclado", 80);

console.log(p1.vender(2));
console.log(p2.vender(5));
console.log(p3.vender(1));

Producto.reporte();
// Productos creados: 3
// Ingresos totales: $2605
```

MINI PROYECTO: GESTOR DE CONFIGURACIÓN
===============================================================================

```javascript
class Config {
    static #valores = {
        appName: "MiApp",
        version: "1.0.0",
        idioma: "es",
        maxUsuarios: 100,
        debug: false
    };

    static get(clave) {
        return this.#valores[clave];
    }

    static set(clave, valor) {
        if (!(clave in this.#valores)) {
            console.log("❌ Config: '" + clave + "' no existe");
            return;
        }
        this.#valores[clave] = valor;
        console.log("✅ Config: " + clave + " = " + valor);
    }

    static listar() {
        console.log("=== CONFIGURACIÓN ===");
        for (let clave in this.#valores) {
            console.log(clave + ": " + this.#valores[clave]);
        }
    }

    static reset() {
        this.#valores = {
            appName: "MiApp",
            version: "1.0.0",
            idioma: "es",
            maxUsuarios: 100,
            debug: false
        };
        console.log("🔄 Configuración restablecida");
    }
}

Config.listar();
Config.set("idioma", "en");
Config.set("maxUsuarios", 200);
console.log("Idioma actual: " + Config.get("idioma"));  // en
```
