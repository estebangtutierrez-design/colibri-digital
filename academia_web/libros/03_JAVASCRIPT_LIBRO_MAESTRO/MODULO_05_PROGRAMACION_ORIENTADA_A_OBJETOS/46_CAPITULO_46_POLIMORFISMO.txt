"Polimorfismo: la capacidad de un mismo método
de comportarse de manera diferente según el objeto que lo llama."
                                  — Profesor Búho AI

===============================================================================
BIENVENIDA DEL PROFESOR BÚHO
===============================================================================

Querido estudiante…

Polimorfismo viene del griego: "muchas formas".

En POO significa: un mismo método (mismo nombre)
se comporta diferente según la clase del objeto.

"Polimorfismo" suena complejo, pero en la práctica
es simplemente sobrescribir métodos y que el programa
elija automáticamente cuál ejecutar.

===============================================================================
1. POLIMORFISMO BÁSICO
===============================================================================

```javascript
class Figura {
    area() {
        return 0;
    }
    descripcion() {
        return "Figura genérica";
    }
}

class Circulo extends Figura {
    constructor(radio) {
        super();
        this.radio = radio;
    }
    area() {
        return Math.PI * this.radio * this.radio;
    }
    descripcion() {
        return "Círculo (radio: " + this.radio + ")";
    }
}

class Rectangulo extends Figura {
    constructor(base, altura) {
        super();
        this.base = base;
        this.altura = altura;
    }
    area() {
        return this.base * this.altura;
    }
    descripcion() {
        return "Rectángulo (" + this.base + "x" + this.altura + ")";
    }
}

// Polimorfismo en acción
let figuras = [
    new Circulo(5),
    new Rectangulo(4, 6),
    new Circulo(3)
];

figuras.forEach(f => {
    console.log(f.descripcion() + " → Área: " + f.area().toFixed(2));
});
// Círculo (radio: 5) → Área: 78.54
// Rectángulo (4x6) → Área: 24.00
// Círculo (radio: 3) → Área: 28.27
```

2. SISTEMA DE NOTIFICACIONES
===============================================================================

```javascript
class Notificacion {
    enviar(mensaje) {
        throw new Error("Método abstracto: implementar en subclase");
    }
}

class Email extends Notificacion {
    constructor(direccion) {
        super();
        this.direccion = direccion;
    }
    enviar(mensaje) {
        console.log("📧 Enviando email a " + this.direccion);
        console.log("   Asunto: " + mensaje.asunto);
        console.log("   Cuerpo: " + mensaje.cuerpo);
    }
}

class SMS extends Notificacion {
    constructor(telefono) {
        super();
        this.telefono = telefono;
    }
    enviar(mensaje) {
        console.log("📱 Enviando SMS a " + this.telefono);
        console.log("   Mensaje: " + mensaje.cuerpo.substring(0, 50));
    }
}

class PushNotification extends Notificacion {
    constructor(dispositivo) {
        super();
        this.dispositivo = dispositivo;
    }
    enviar(mensaje) {
        console.log("🔔 Push notification a " + this.dispositivo);
        console.log("   Título: " + mensaje.asunto);
    }
}

let mensaje = {
    asunto: "Oferta especial",
    cuerpo: "50% de descuento en todos los cursos"
};

let canales = [
    new Email("ana@email.com"),
    new SMS("555-1234"),
    new PushNotification("iPhone-Ana")
];

console.log("=== ENVIANDO NOTIFICACIONES ===");
canales.forEach(canal => canal.enviar(mensaje));
// Cada canal se comporta diferente: polimorfismo
```

4. POLIMORFISMO CON DUCK TYPING
===============================================================================

JavaScript usa "duck typing": si camina como pato y suena como pato, es un pato.
No necesitas herencia para polimorfismo.

```javascript
class Pato {
    hacerSonido() { return "¡Cuac!"; }
}

class RobotPato {
    hacerSonido() { return "🤖 ¡Cuac! (modo robot)"; }
}

class PersonaDisfrazada {
    hacerSonido() { return "¡Cuac! (soy Juan)"; }
}

function ejecutarEspectaculo(participante) {
    console.log(participante.hacerSonido());
}

ejecutarEspectaculo(new Pato());             // ¡Cuac!
ejecutarEspectaculo(new RobotPato());        // 🤖 ¡Cuac!
ejecutarEspectaculo(new PersonaDisfrazada());// ¡Cuac! (soy Juan)
```

MINI PROYECTO: PROCESADOR DE PAGOS
===============================================================================

```javascript
class Pago {
    procesar(monto) {
        throw new Error("Implementar en subclase");
    }
}

class TarjetaCredito extends Pago {
    constructor(numero, titular) {
        super();
        this.numero = numero;
        this.titular = titular;
    }
    procesar(monto) {
        console.log("💳 Tarjeta ****" + this.numero.slice(-4));
        console.log("   Titular: " + this.titular);
        console.log("   Cargo: $" + monto.toFixed(2));
        console.log("   ✅ Aprobado");
    }
}

class PayPal extends Pago {
    constructor(email) {
        super();
        this.email = email;
    }
    procesar(monto) {
        console.log("🅿️ PayPal: " + this.email);
        console.log("   Cargo: $" + monto.toFixed(2));
        console.log("   ✅ Pago realizado");
    }
}

class Transferencia extends Pago {
    constructor(banco, cuenta) {
        super();
        this.banco = banco;
        this.cuenta = cuenta;
    }
    procesar(monto) {
        console.log("🏦 Transferencia bancaria");
        console.log("   Banco: " + this.banco + " - Cuenta: " + this.cuenta);
        console.log("   Monto: $" + monto.toFixed(2));
        console.log("   ✅ Transferencia iniciada");
    }
}

function checkout(carrito, metodoPago) {
    let total = carrito.reduce((a, p) => a + p.precio, 0);
    console.log("🛒 Total del carrito: $" + total);
    metodoPago.procesar(total);
}

let carrito = [
    { nombre: "Laptop", precio: 1200 },
    { nombre: "Mouse", precio: 25 }
];

checkout(carrito, new TarjetaCredito("1234567890123456", "Ana García"));
checkout(carrito, new PayPal("ana@email.com"));
checkout(carrito, new Transferencia("Banco Nacional", "ES123456789"));
```

RESUMEN
===============================================================================

✔ Polimorfismo: mismo método, diferente comportamiento.
✔ Se logra sobrescribiendo métodos en subclases.
✔ instanceof: verifica el tipo de objeto.
✔ Duck typing: si tiene el método, funciona.
✔ Útil para: sistemas de pago, notificaciones, figuras.
