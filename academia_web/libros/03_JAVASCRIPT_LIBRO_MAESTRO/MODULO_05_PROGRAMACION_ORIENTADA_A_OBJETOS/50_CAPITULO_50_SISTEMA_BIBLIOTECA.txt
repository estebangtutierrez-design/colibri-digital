1. CLASE PADRE: MATERIAL
===============================================================================

```javascript
class Material {
    #id;
    static #contador = 0;

    constructor(titulo, año, disponible = true) {
        if (new.target === Material) {
            throw new Error("Clase abstracta: no instanciar");
        }
        this.#id = ++Material.#contador;
        this.titulo = titulo;
        this.año = año;
        this._disponible = disponible;
        this._prestadoA = null;
    }

    get id() { return this.#id; }
    get disponible() { return this._disponible; }

    static get totalMateriales() {
        return Material.#contador;
    }

    prestar(usuario) {
        if (!this._disponible) {
            return "❌ '" + this.titulo + "' ya está prestado";
        }
        this._disponible = false;
        this._prestadoA = usuario;
        return "✅ '" + this.titulo + "' prestado a " + usuario;
    }

    devolver() {
        if (this._disponible) {
            return "❌ '" + this.titulo + "' no estaba prestado";
        }
        let usuario = this._prestadoA;
        this._disponible = true;
        this._prestadoA = null;
        return "📚 '" + this.titulo + "' devuelto por " + usuario;
    }

    info() {
        let estado = this._disponible ? "Disponible" : "Prestado a " + this._prestadoA;
        return "[ID:" + this.id + "] " + this.titulo + " (" + this.año + ") - " + estado;
    }

    detalle() {
        throw new Error("Método abstracto: implementar en subclase");
    }
}
```

2. SUBCLASES: LIBRO, REVISTA, DVD
===============================================================================

```javascript
class Libro extends Material {
    constructor(titulo, año, autor, genero, paginas) {
        super(titulo, año);
        this.autor = autor;
        this.genero = genero;
        this.paginas = paginas;
    }

    detalle() {
        return "📖 Libro: " + this.titulo + " - " + this.autor +
               " (" + this.genero + ", " + this.paginas + " pág.)";
    }

    get esLargo() {
        return this.paginas > 300;
    }
}

class Revista extends Material {
    constructor(titulo, año, numero, editorial) {
        super(titulo, año);
        this.numero = numero;
        this.editorial = editorial;
    }

    detalle() {
        return "📰 Revista: " + this.titulo + " #" + this.numero +
               " - " + this.editorial;
    }
}

class DVD extends Material {
    constructor(titulo, año, director, duracion) {
        super(titulo, año);
        this.director = director;
        this.duracion = duracion;
    }

    detalle() {
        return "🎬 DVD: " + this.titulo + " - " + this.director +
               " (" + this.duracion + " min)";
    }
}
```

3. CLASE BIBLIOTECA

```javascript
class Biblioteca {
    #materiales = [];
    #nombre;

    constructor(nombre) {
        this.#nombre = nombre;
    }

    agregar(material) {
        this.#materiales.push(material);
        console.log("➕ Agregado: " + material.detalle());
        return this;
    }

    buscar(termino) {
        return this.#materiales.filter(m =>
            m.titulo.toLowerCase().includes(termino.toLowerCase())
        );
    }

    prestar(id, usuario) {
        let material = this.#materiales.find(m => m.id === id);
        if (!material) return console.log("❌ Material no encontrado");
        console.log(material.prestar(usuario));
        return this;
    }

    devolver(id) {
        let material = this.#materiales.find(m => m.id === id);
        if (!material) return console.log("❌ Material no encontrado");
        console.log(material.devolver());
        return this;
    }

    listarDisponibles() {
        console.log("=== MATERIALES DISPONIBLES ===");
        let disponibles = this.#materiales.filter(m => m.disponible);
        if (disponibles.length === 0) return console.log("📭 No hay");
        disponibles.forEach(m => console.log(m.detalle()));
        return this;
    }

    listarPrestados() {
        console.log("=== MATERIALES PRESTADOS ===");
        let prestados = this.#materiales.filter(m => !m.disponible);
        if (prestados.length === 0) return console.log("📭 No hay");
        prestados.forEach(m => console.log(m.info()));
        return this;
    }

    reporte() {
        console.log("========== BIBLIOTECA: " + this.#nombre + " ==========");
        console.log("Total materiales: " + this.#materiales.length);
        console.log("Disponibles: " + this.#materiales.filter(m => m.disponible).length);
        console.log("Prestados: " + this.#materiales.filter(m => !m.disponible).length);
        console.log("IDs generados: " + Material.totalMateriales);

        if (this.#materiales.length > 0) {
            let masPaginas = this.#materiales
                .filter(m => m instanceof Libro)
                .reduce((max, l) => l.paginas > (max?.paginas || 0) ? l : max, null);
            if (masPaginas) {
                console.log("📖 Libro más extenso: " + masPaginas.titulo +
                    " (" + masPaginas.paginas + " pág.)");
            }
        }
        console.log("===========================================");
        return this;
    }
}
```

4. PROBANDO EL SISTEMA
===============================================================================

```javascript
let biblio = new Biblioteca("Biblioteca Central");

// Agregar materiales
biblio
    .agregar(new Libro("Cien años de soledad", 1967, "G. García Márquez", "Novela", 471))
    .agregar(new Libro("1984", 1949, "George Orwell", "Ciencia Ficción", 328))
    .agregar(new Libro("El Principito", 1943, "Saint-Exupéry", "Infantil", 96))
    .agregar(new Revista("National Geographic", 2024, 275, "NG Society"))
    .agregar(new DVD("Inception", 2010, "Christopher Nolan", 148));

// Préstamos
biblio
    .prestar(1, "Ana García")       // Cien años de soledad
    .prestar(2, "Luis Pérez")       // 1984
    .prestar(1, "Carlos");          // ❌ Ya prestado

// Búsqueda
console.log("=== BÚSQUEDA: 'El' ===");
biblio.buscar("El").forEach(m => console.log("  " + m.detalle()));

// Estado
biblio.listarDisponibles();
biblio.listarPrestados();

// Devolución
biblio
    .devolver(1)
    .listarDisponibles();

// Reporte
biblio.reporte();
```
