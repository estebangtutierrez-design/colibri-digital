BIENVENIDA DEL PROFESOR BÚHO
===============================================================================

Querido estudiante…

Los callbacks tienen el problema del callback hell.

Las PROMESAS llegaron para solucionarlo.

Una Promesa es como un ticket de restaurante:
- Lo recibes al pedir (pendiente).
- Cuando la comida está lista, te llaman (resuelta).
- Si no hay ingredientes, te avisan (rechazada).

===============================================================================
1. CREAR UNA PROMESA
===============================================================================

```javascript
let promesa = new Promise(function(resolve, reject) {
    // Código asíncrono aquí
    let exito = true;

    setTimeout(function() {
        if (exito) {
            resolve("✅ Operación exitosa");
        } else {
            reject("❌ Algo salió mal");
        }
    }, 2000);
});

console.log(promesa);  // Promise { <pending> }

// Consumir la promesa
promesa
    .then(function(resultado) {
        console.log(resultado);  // "✅ Operación exitosa"
    })
    .catch(function(error) {
        console.log(error);     // "❌ Algo salió mal"
    });
```

===============================================================================
2. ESTADOS DE UNA PROMESA
===============================================================================

```javascript
// Pendiente (pending): estado inicial
let pendiente = new Promise(() => {});
console.log(pendiente);  // Promise { <pending> }

// Cumplida (fulfilled): resolve() fue llamado
let cumplida = Promise.resolve("✅ Hecho");
console.log(cumplida);  // Promise { <fulfilled>: "✅ Hecho" }

// Rechazada (rejected): reject() fue llamado
let rechazada = Promise.reject("❌ Error");
console.log(rechazada);  // Promise { <rejected>: "❌ Error" }

// Una promesa solo puede resolverse o rechazarse UNA VEZ
let promesaUnica = new Promise(function(resolve, reject) {
    resolve("Primero");  // Este se ejecuta
    resolve("Segundo");  // Ignorado
    reject("Error");     // Ignorado (ya está resuelta)
});
```

===============================================================================
3. THEN Y CATCH
===============================================================================

```javascript
function obtenerUsuario(id) {
    return new Promise(function(resolve, reject) {
        setTimeout(function() {
            if (id <= 0) {
                reject("ID inválido");
                return;
            }
            resolve({ id: id, nombre: "Ana", email: "ana@email.com" });
        }, 1000);
    });
}

// then recibe el valor de resolve
// catch recibe el valor de reject
obtenerUsuario(1)
    .then(function(usuario) {
        console.log("✅ Usuario:", usuario.nombre);
    })
    .catch(function(error) {
        console.log("❌", error);
    });

// finally: se ejecuta siempre (éxito o error)
obtenerUsuario(1)
    .then(u => console.log(u.nombre))
    .catch(e => console.log(e))
    .finally(() => console.log("🏁 Operación terminada"));
```

===============================================================================
4. CONVERTIR CALLBACK A PROMESA
===============================================================================

```javascript
// Antes (callback)
function leerArchivoCallback(ruta, callback) {
    setTimeout(function() {
        callback(null, "Contenido del archivo");
    }, 1000);
}

// Después (Promise)
function leerArchivoPromise(ruta) {
    return new Promise(function(resolve, reject) {
        setTimeout(function() {
            if (ruta) {
                resolve("Contenido de: " + ruta);
            } else {
                reject("Ruta no especificada");
            }
        }, 1000);
    });
}

// Uso con promesa
leerArchivoPromise("datos.txt")
    .then(function(contenido) {
        console.log("📄", contenido);
    })
    .catch(function(error) {
        console.log("❌", error);
    });
```

===============================================================================
5. EJEMPLO PRÁCTICO
===============================================================================

```javascript
function esperar(ms) {
    return new Promise(function(resolve) {
        setTimeout(resolve, ms);
    });
}

function simularDescarga(archivo) {
    return new Promise(function(resolve) {
        console.log("⏳ Descargando " + archivo + "...");
        setTimeout(function() {
            resolve("📦 " + archivo + " descargado (2MB)");
        }, 2000);
    });
}

// Uso
simularDescarga("foto.jpg")
    .then(function(resultado) {
        console.log(resultado);
        return esperar(1000);  // Esperar 1 segundo
    })
    .then(function() {
        console.log("📸 Procesando imagen...");
        return simularDescarga("video.mp4");
    })
    .then(function(resultado) {
        console.log(resultado);
    })
    .catch(function(error) {
        console.log("❌ Error:", error);
    });
```

MINI PROYECTO: SIMULADOR DE PETICIONES
===============================================================================

```javascript
function peticionAPI(endpoint, delay = 1000) {
    return new Promise(function(resolve, reject) {
        console.log("📡 Llamando a " + endpoint + "...");

        setTimeout(function() {
            let exito = Math.random() > 0.2;  // 80% éxito

            if (exito) {
                resolve({
                    status: 200,
                    datos: { endpoint: endpoint, resultado: "ok" }
                });
            } else {
                reject({
                    status: 500,
                    error: "Error interno del servidor"
                });
            }
        }, delay);
    });
}

peticionAPI("/api/usuarios")
    .then(function(respuesta) {
        console.log("✅ Status:", respuesta.status);
        console.log("   Datos:", respuesta.datos);
    })
    .catch(function(error) {
        console.log("❌ Error", error.status + ":", error.error);
    })
    .finally(function() {
        console.log("🏁 Petición finalizada");
    });
```

===============================================================================
RESUMEN
===============================================================================

✔ Promesa: objeto con valor futuro (pendiente, cumplida, rechazada).
✔ resolve(valor): cumple la promesa.
✔ reject(error): rechaza la promesa.
✔ then(): maneja el éxito.
✔ catch(): maneja el error.
✔ finally(): se ejecuta siempre.
✔ Una promesa solo puede resolverse UNA VEZ.
