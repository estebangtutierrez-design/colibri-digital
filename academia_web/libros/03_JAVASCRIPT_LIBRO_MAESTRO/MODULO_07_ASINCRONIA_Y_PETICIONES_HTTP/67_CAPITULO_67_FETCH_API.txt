Las aplicaciones web necesitan comunicarse con servidores:
- Obtener datos (GET).
- Enviar datos (POST).
- Actualizar (PUT).
- Eliminar (DELETE).

Fetch API es la herramienta moderna para hacerlo.
Devuelve una Promesa, así que puedes usar .then() o async/await.

2. FETCH REAL - ESTRUCTURA
===============================================================================

```javascript
// Estructura de una petición fetch real:

fetch("https://api.ejemplo.com/usuarios")
    .then(function(respuesta) {
        // respuesta.ok: true si status 200-299
        // respuesta.status: 200, 404, 500, etc.
        // respuesta.statusText: "OK", "Not Found"

        if (!respuesta.ok) {
            throw new Error("HTTP " + respuesta.status);
        }
        return respuesta.json();  // También: .text(), .blob()
    })
    .then(function(datos) {
        console.log("✅ Datos:", datos);
    })
    .catch(function(error) {
        console.log("❌ Error:", error.message);
    });

// Otras opciones de respuesta:
// respuesta.json() → objeto JS
// respuesta.text() → string
// respuesta.blob() → archivo binario
// respuesta.formData() → FormData
```

3. FETCH CON ASYNC/AWAIT
===============================================================================

```javascript
async function obtenerUsuario(id) {
    try {
        let respuesta = await fetch(
            "https://jsonplaceholder.typicode.com/users/" + id
        );

        if (!respuesta.ok) {
            throw new Error("Error HTTP: " + respuesta.status);
        }

        let usuario = await respuesta.json();
        console.log("👤", usuario.name);
        console.log("📧", usuario.email);
        console.log("🏙️ Ciudad:", usuario.address?.city);
        return usuario;
    } catch (error) {
        console.log("❌ No se pudo obtener usuario:", error.message);
        return null;
    }
}

// obtenerUsuario(1);
```

4. FETCH POST
===============================================================================

```javascript
async function crearUsuario(nombre, email) {
    try {
        let respuesta = await fetch(
            "https://jsonplaceholder.typicode.com/users",
            {
                method: "POST",
                headers: {
                    "Content-Type": "application/json"
                },
                body: JSON.stringify({
                    name: nombre,
                    email: email
                })
            }
        );

        if (!respuesta.ok) {
            throw new Error("Error al crear: " + respuesta.status);
        }

        let nuevoUsuario = await respuesta.json();
        console.log("✅ Usuario creado con ID:", nuevoUsuario.id);
        return nuevoUsuario;
    } catch (error) {
        console.log("❌ Error:", error.message);
    }
}

// crearUsuario("Luis Pérez", "luis@email.com");
```

5. MÉTODOS HTTP COMUNES
===============================================================================

```javascript
async function apiCliente() {
    const BASE = "https://jsonplaceholder.typicode.com";

    // GET - Obtener
    const get = async (url) => {
        let res = await fetch(BASE + url);
        return res.json();
    };

    // POST - Crear
    const post = async (url, datos) => {
        let res = await fetch(BASE + url, {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(datos)
        });
        return res.json();
    };

    // PUT - Actualizar (completo)
    const put = async (url, datos) => {
        let res = await fetch(BASE + url, {
            method: "PUT",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(datos)
        });
        return res.json();
    };

    // PATCH - Actualizar (parcial)
    const patch = async (url, datos) => {
        let res = await fetch(BASE + url, {
            method: "PATCH",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(datos)
        });
        return res.json();
    };

    // DELETE - Eliminar
    const del = async (url) => {
        let res = await fetch(BASE + url, { method: "DELETE" });
        return res.ok ? "✅ Eliminado" : "❌ Error";
    };

    return { get, post, put, patch, del };
}
```

```javascript
async function explorarAPI() {
    let endpoints = [
        "https://jsonplaceholder.typicode.com/posts/1",
        "https://jsonplaceholder.typicode.com/comments/1",
        "https://jsonplaceholder.typicode.com/albums/1"
    ];

    try {
        let resultados = await Promise.all(
            endpoints.map(url => fetch(url).then(r => r.json()))
        );

        console.log("📊 Resultados de la API:");
        console.log("Post:", resultados[0].title);
        console.log("Comment:", resultados[1].body);
        console.log("Album:", resultados[2].title);
    } catch (error) {
        console.log("❌ Error:", error.message);
    }
}
```
