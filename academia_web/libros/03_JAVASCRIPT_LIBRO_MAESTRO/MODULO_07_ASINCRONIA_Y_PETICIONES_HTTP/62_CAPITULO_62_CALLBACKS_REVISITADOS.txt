Querido estudiante…

Ya viste callbacks en el Módulo III.
Ahora los revisitamos desde la perspectiva asíncrona.

Los callbacks son funciones que se pasan a otra función
para ejecutarse CUANDO una operación asíncrona termina.

Son el mecanismo original de asincronía en JS,
pero tienen problemas cuando se anidan demasiado.

===============================================================================
1. CALLBACK ASÍNCRONO
===============================================================================

```javascript
function pedirDatos(url, callback) {
    // Simular petición asíncrona
    setTimeout(function() {
        let datos = { usuario: "Ana", id: 1 };
        callback(null, datos);  // null = sin error
    }, 1500);
}

function mostrarDatos(error, datos) {
    if (error) {
        console.log("❌ Error:", error);
        return;
    }
    console.log("✅ Datos recibidos:", datos);
}

console.log("📡 Solicitando datos...");
pedirDatos("/api/usuario", mostrarDatos);
console.log("⏳ Esperando respuesta... (no bloquea)");
```

===============================================================================
2. PATRÓN ERROR FIRST
===============================================================================

Los callbacks en Node.js siguen el patrón "error first":
Primer parámetro: error (null si todo bien).
Segundo parámetro: datos.

```javascript
function leerArchivo(ruta, callback) {
    setTimeout(function() {
        let exito = Math.random() > 0.3;  // 70% éxito

        if (exito) {
            callback(null, "Contenido del archivo: Hola mundo");
        } else {
            callback("Error: archivo no encontrado", null);
        }
    }, 1000);
}

leerArchivo("datos.txt", function(err, contenido) {
    if (err) {
        console.log("❌ " + err);
        return;
    }
    console.log("📄 " + contenido);
});
```

3. CALLBACK HELL
===============================================================================

```javascript
// 🚫 Callback Hell: código en forma de pirámide

function obtenerUsuario(id, cb) {
    setTimeout(() => cb(null, { id: id, nombre: "Ana" }), 1000);
}

function obtenerPedidos(usuarioId, cb) {
    setTimeout(() => cb(null, ["Pedido1", "Pedido2"]), 1000);
}

function obtenerDetalle(pedido, cb) {
    setTimeout(() => cb(null, { pedido: pedido, total: 250 }), 1000);
}

// El infierno:
obtenerUsuario(1, function(err, usuario) {
    if (err) return console.log(err);
    console.log("Usuario:", usuario.nombre);

    obtenerPedidos(usuario.id, function(err, pedidos) {
        if (err) return console.log(err);

        obtenerDetalle(pedidos[0], function(err, detalle) {
            if (err) return console.log(err);
            console.log("Detalle del pedido:", detalle);
            // Más anidamiento aquí...
        });
    });
});
```

function handleUsuario(err, usuario) {

    obtenerPedidos(usuario.id, handlePedidos);

function handlePedidos(err, pedidos) {
    if (err) return console.log(err);

===============================================================================
MINI PROYECTO: SIMULADOR DE PROCESO ASÍNCRONO
===============================================================================

```javascript
function paso1(datos, callback) {
    setTimeout(function() {
        let resultado = datos + " → Paso 1 completado";
        console.log("✅ Paso 1");
        callback(null, resultado);
    }, 1000);
}

function paso2(datos, callback) {
    setTimeout(function() {
        let resultado = datos + " → Paso 2 completado";
        console.log("✅ Paso 2");
        callback(null, resultado);
    }, 1000);
}

function paso3(datos, callback) {
    setTimeout(function() {
        let resultado = datos + " → Paso 3 completado";
        console.log("✅ Paso 3");
        callback(null, resultado);
    }, 1000);
}

function ejecutarProceso() {
    console.log("🚀 Iniciando proceso...");

    paso1("Inicio", function(err, r1) {
        if (err) return console.log("❌", err);
        console.log("  →", r1);

        paso2(r1, function(err, r2) {
            if (err) return console.log("❌", err);
            console.log("  →", r2);

            paso3(r2, function(err, r3) {
                if (err) return console.log("❌", err);
                console.log("🏁 Final:", r3);
            });
        });
    });
}

ejecutarProceso();
// 🚀 Iniciando proceso...
// ✅ Paso 1 (1s)
// ✅ Paso 2 (2s)
// ✅ Paso 3 (3s)
// 🏁 Final: Inicio → Paso 1 → Paso 2 → Paso 3
```
