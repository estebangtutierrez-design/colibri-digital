1. LOCALSTORAGE
===============================================================================

Almacenamiento persistente (no se borra al cerrar el navegador).

```javascript
// Guardar
localStorage.setItem("nombre", "Ana");
localStorage.setItem("edad", "25");
localStorage.setItem("usuario", JSON.stringify({ id: 1, rol: "admin" }));

// Leer
console.log(localStorage.getItem("nombre"));  // "Ana"
console.log(localStorage.getItem("no-existe"));  // null

// Parsear JSON
let usuario = JSON.parse(localStorage.getItem("usuario"));
console.log(usuario.rol);  // "admin"

// Eliminar
localStorage.removeItem("edad");

// Vaciar todo
// localStorage.clear();

// Número de items
console.log(localStorage.length);
```

===============================================================================
2. SESSIONSTORAGE
===============================================================================

Igual que localStorage, pero se borra al cerrar la pestaña.

```javascript
// Guardar datos temporales
sessionStorage.setItem("token", "abc123");
sessionStorage.setItem("carrito", JSON.stringify([
    { producto: "Laptop", cantidad: 1 },
    { producto: "Mouse", cantidad: 2 }
]));

// Leer
let token = sessionStorage.getItem("token");
let carrito = JSON.parse(sessionStorage.getItem("carrito"));

// Diferencia clave:
// localStorage → sobrevive a cierres de navegador
// sessionStorage → se borra al cerrar pestaña
```

3. GEOLOCALIZACIÓN
===============================================================================

```javascript
function obtenerUbicacion() {
    if (!navigator.geolocation) {
        console.log("❌ Geolocalización no soportada");
        return;
    }

    navigator.geolocation.getCurrentPosition(
        function(posicion) {
            console.log("📍 Ubicación obtenida:");
            console.log("   Latitud:", posicion.coords.latitude);
            console.log("   Longitud:", posicion.coords.longitude);
            console.log("   Precisión:", posicion.coords.accuracy, "metros");
            console.log("   Altitud:", posicion.coords.altitude);

            // Crear enlace a Google Maps
            let url = "https://www.google.com/maps?q=" +
                posicion.coords.latitude + "," +
                posicion.coords.longitude;
            console.log("🗺️ Maps:", url);
        },
        function(error) {
            let mensajes = {
                1: "Permiso denegado",
                2: "Posición no disponible",
                3: "Tiempo de espera agotado"
            };
            console.log("❌ Error:", mensajes[error.code] || "Desconocido");
        },
        {
            enableHighAccuracy: true,
            timeout: 10000,
            maximumAge: 60000
        }
    );
}
```

4. NOTIFICACIONES
===============================================================================

```javascript
async function enviarNotificacion(titulo, cuerpo) {
    // Solicitar permiso
    if (Notification.permission === "denied") {
        console.log("❌ Notificaciones bloqueadas");
        return;
    }

    if (Notification.permission === "default") {
        let permiso = await Notification.requestPermission();
        if (permiso !== "granted") {
            console.log("❌ Permiso no concedido");
            return;
        }
    }

    // Enviar notificación
    let notif = new Notification(titulo, {
        body: cuerpo,
        icon: "https://via.placeholder.com/64",
        tag: "mensaje-unico"  // Reemplaza notificaciones con mismo tag
    });

    notif.addEventListener("click", function() {
        console.log("🔔 Usuario hizo clic en la notificación");
        window.focus();
        this.close();
    });

    console.log("🔔 Notificación enviada");
}

// enviarNotificacion("🎉 Bienvenido", "Gracias por usar nuestra aplicación");
```

5. OTRAS APIs ÚTILES
===============================================================================

```javascript
// navigator.onLine: estado de la conexión
console.log("¿En línea?", navigator.onLine);

window.addEventListener("online", () => console.log("🌐 Conexión restablecida"));
window.addEventListener("offline", () => console.log("📡 Sin conexión"));

// navigator.userAgent: información del navegador
console.log("Navegador:", navigator.userAgent);

// navigator.clipboard: portapapeles
async function copiarAlPortapapeles(texto) {
    try {
        await navigator.clipboard.writeText(texto);
        console.log("📋 Copiado: " + texto);
    } catch (error) {
        console.log("❌ No se pudo copiar");
    }
}

// navigator.mediaDevices: cámara y micrófono
async function obtenerCamara() {
    try {
        let stream = await navigator.mediaDevices.getUserMedia({ video: true });
        // stream contiene el video de la cámara
        console.log("📷 Cámara accesible");
    } catch (error) {
        console.log("❌ Cámara no disponible:", error.message);
    }
}
```

MINI PROYECTO: APLICACIÓN DE NOTAS CON PERSISTENCIA
===============================================================================

```javascript
class AppNotas {
    #notas = [];

    constructor() {
        this.#cargar();
    }

    #cargar() {
        try {
            let datos = localStorage.getItem("app-notas");
            this.#notas = datos ? JSON.parse(datos) : [];
        } catch {
            this.#notas = [];
        }
    }

    #guardar() {
        localStorage.setItem("app-notas", JSON.stringify(this.#notas));
    }

    agregar(titulo, contenido) {
        this.#notas.push({
            id: Date.now(),
            titulo: titulo,
            contenido: contenido,
            creada: new Date().toLocaleString()
        });
        this.#guardar();
        console.log("✅ Nota guardada:", titulo);
        return this;
    }

    eliminar(id) {
        this.#notas = this.#notas.filter(n => n.id !== id);
        this.#guardar();
        console.log("🗑️ Nota eliminada");
    }

    listar() {
        console.log("=== NOTAS (" + this.#notas.length + ") ===");
        this.#notas.forEach(n => {
            console.log("  [" + n.id + "] " + n.titulo);
            console.log("     " + n.contenido.substring(0, 50) + "...");
            console.log("     🕐 " + n.creada);
        });
    }

    buscar(texto) {
        return this.#notas.filter(n =>
            n.titulo.toLowerCase().includes(texto.toLowerCase()) ||
            n.contenido.toLowerCase().includes(texto.toLowerCase())
        );
    }

    exportar() {
        let json = JSON.stringify(this.#notas, null, 2);
        console.log("📦 JSON exportado (" + json.length + " chars)");
        return json;
    }

    get total() { return this.#notas.length; }
}

let notas = new AppNotas();
notas.agregar("Compras", "Leche, pan, huevos");
notas.agregar("Estudio", "Repasar JavaScript - Promesas");
notas.listar();

let resultado = notas.buscar("JavaScript");
console.log("🔍 Resultados:", resultado.length);
```
