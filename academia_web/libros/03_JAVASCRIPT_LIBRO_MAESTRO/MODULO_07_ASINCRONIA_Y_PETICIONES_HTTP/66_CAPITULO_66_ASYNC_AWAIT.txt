BIENVENIDA DEL PROFESOR BÚHO
===============================================================================

Querido estudiante…

Las promesas mejoraron los callbacks, pero aún tienes .then().

Async/await es AZÚCAR SINTÁCTICO sobre promesas.
Hace que el código asíncrono SE LEA como código síncrono.

- async: declara que una función devuelve una promesa.
- await: espera a que una promesa se resuelva.

OBJETIVOS DEL CAPÍTULO

✓ Comparar .then() vs async/await.

===============================================================================
1. FUNCIÓN ASYNC
===============================================================================

```javascript
// Una función async SIEMPRE devuelve una Promesa
async function saludar() {
    return "¡Hola!";
}

console.log(saludar());  // Promise { <fulfilled>: "¡Hola!" }

saludar().then(function(msg) {
    console.log(msg);  // "¡Hola!"
});

// Equivalente sin async:
function saludarNormal() {
    return Promise.resolve("¡Hola!");
}
```

===============================================================================
2. AWAIT
===============================================================================

await pausa la ejecución hasta que la promesa se resuelva.
SOLO funciona dentro de funciones async.

```javascript
function esperar(ms) {
    return new Promise(function(resolve) {
        setTimeout(resolve, ms);
    });
}

async function hacerAlgo() {
    console.log("⏳ Inicio");

    await esperar(2000);  // Pausa 2 segundos

    console.log("⏰ Pasaron 2 segundos");
    return "✅ Hecho";
}

hacerAlgo().then(function(r) {
    console.log(r);
});
```

===============================================================================
3. ASYNC/AWAIT ENCADENADO
===============================================================================

```javascript
function paso(numero, ms) {
    return new Promise(function(resolve) {
        setTimeout(function() {
            console.log("✅ Paso " + numero + " (" + ms + "ms)");
            resolve(numero);
        }, ms);
    });
}

// Con .then()
function conThen() {
    paso(1, 1000)
        .then(function() { return paso(2, 1500); })
        .then(function() { return paso(3, 500); });
}

// Con async/await (mucho más legible)
async function conAsync() {
    console.log("🚀 Iniciando secuencia...");
    await paso(1, 1000);
    await paso(2, 1500);
    await paso(3, 500);
    console.log("🏁 Secuencia completada");
}

conAsync();
```

===============================================================================
4. TRY/CATCH CON ASYNC/AWAIT
===============================================================================

```javascript
function buscarUsuario(id) {
    return new Promise(function(resolve, reject) {
        setTimeout(function() {
            if (id === 0) {
                reject("ID inválido");
            } else {
                resolve({ id: id, nombre: "Ana" });
            }
        }, 1000);
    });
}

async function mostrarUsuario(id) {
    try {
        let usuario = await buscarUsuario(id);
        console.log("✅ Usuario:", usuario.nombre);
        return usuario;
    } catch (error) {
        console.log("❌ Error:", error);
        return null;
    } finally {
        console.log("🏁 Operación finalizada");
    }
}

mostrarUsuario(1);  // ✅ Usuario: Ana
mostrarUsuario(0);  // ❌ Error: ID inválido
```

===============================================================================
5. PARALELISMO CON ASYNC/AWAIT
===============================================================================

```javascript
function descargar(url, tiempo) {
    return new Promise(function(resolve) {
        setTimeout(function() {
            resolve("📦 " + url + " completado");
        }, tiempo);
    });
}

// ❌ Secuencial (lento)
async function descargarSecuencial() {
    let a = await descargar("foto.jpg", 2000);
    console.log(a);
    let b = await descargar("video.mp4", 2000);
    console.log(b);
    // Total: 4 segundos
}

// ✅ Paralelo con Promise.all (rápido)
async function descargarParalelo() {
    let urls = [
        descargar("foto.jpg", 2000),
        descargar("video.mp4", 2000),
        descargar("doc.pdf", 1500)
    ];

    let resultados = await Promise.all(urls);
    resultados.forEach(function(r) {
        console.log(r);
    });
    // Total: ~2 segundos
}
```

MINI PROYECTO: PROCESO DE REGISTRO
===============================================================================

```javascript
function validarEmail(email) {
    return new Promise(function(resolve, reject) {
        setTimeout(function() {
            if (email.includes("@")) {
                resolve("✅ Email válido");
            } else {
                reject("❌ Email inválido");
            }
        }, 500);
    });
}

function verificarDisponible(usuario) {
    return new Promise(function(resolve, reject) {
        setTimeout(function() {
            if (usuario !== "admin") {
                resolve("✅ Usuario disponible");
            } else {
                reject("❌ Usuario ya existe");
            }
        }, 1000);
    });
}

function crearCuenta(usuario) {
    return new Promise(function(resolve) {
        setTimeout(function() {
            resolve({
                usuario: usuario,
                creado: true,
                fecha: new Date().toISOString()
            });
        }, 1500);
    });
}

async function registrarse(usuario, email) {
    try {
        console.log("📝 Iniciando registro...");

        let emailResult = await validarEmail(email);
        console.log("  " + emailResult);

        let userResult = await verificarDisponible(usuario);
        console.log("  " + userResult);

        let cuenta = await crearCuenta(usuario);
        console.log("🎉 Cuenta creada para " + cuenta.usuario);
        console.log("   ID:", cuenta.creado ? "✅" : "❌");
        return cuenta;
    } catch (error) {
        console.log("⛔ Registro fallido:", error);
        return null;
    }
}

registrarse("ana_dev", "ana@email.com");
// 📝 Iniciando registro...
// ✅ Email válido
// ✅ Usuario disponible
// 🎉 Cuenta creada para ana_dev
```

===============================================================================
RESUMEN
===============================================================================

✔ async function: siempre devuelve una Promesa.
✔ await promesa: espera a que se resuelva (solo en async).
✔ try/catch: maneja errores de manera natural.
✔ Promise.all + await: paralelismo fácil.
✔ Async/await es azúcar sintáctico sobre promesas.
✔ Más legible que .then() para secuencias complejas.
