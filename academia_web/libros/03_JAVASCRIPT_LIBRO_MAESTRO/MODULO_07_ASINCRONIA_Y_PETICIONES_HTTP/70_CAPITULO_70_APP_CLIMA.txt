```javascript
// Usaremos la API gratuita: https://open-meteo.com/
// (No necesita API key)

// URL de ejemplo:
// https://api.open-meteo.com/v1/forecast?
// latitude=40.4168&longitude=-3.7038&
// current_weather=true&
// daily=temperature_max,temperature_min&
// timezone=auto

class ClimaAPI {
    #baseURL = "https://api.open-meteo.com/v1/forecast";

    async obtenerClimaActual(lat, lon) {
        let url = this.#baseURL + "?" +
            "latitude=" + lat +
            "&longitude=" + lon +
            "&current_weather=true" +
            "&daily=temperature_max,temperature_min,precipitation_sum" +
            "&timezone=auto";

        try {
            let respuesta = await fetch(url);

            if (!respuesta.ok) {
                throw new Error("HTTP " + respuesta.status);
            }

            return await respuesta.json();
        } catch (error) {
            if (error.message.startsWith("HTTP")) {
                throw new Error("Servicio de clima no disponible");
            }
            throw new Error("Error de conexión: verifica tu internet");
        }
    }

    // Coordenadas de ciudades famosas
    static ciudades = {
        "Madrid": { lat: 40.4168, lon: -3.7038 },
        "Barcelona": { lat: 41.3874, lon: 2.1686 },
        "Buenos Aires": { lat: -34.6037, lon: -58.3816 },
        "México DF": { lat: 19.4326, lon: -99.1332 },
        "Bogotá": { lat: 4.7110, lon: -74.0721 },
        "Lima": { lat: -12.0464, lon: -77.0428 },
        "Santiago": { lat: -33.4489, lon: -70.6693 },
        "Londres": { lat: 51.5074, lon: -0.1278 },
        "Nueva York": { lat: 40.7128, lon: -74.0060 },
        "Tokio": { lat: 35.6762, lon: 139.6503 }
    };
}
```

```javascript
class ProcesadorClima {
    procesarRespuesta(datos) {
        let actual = datos.current_weather;
        let diario = datos.daily;

        return {
            // Clima actual
            temperatura: actual.temperature,
            viento: actual.windspeed,
            codigoClima: actual.weathercode,
            // Pronóstico
            maxHoy: diario.temperature_max[0],
            minHoy: diario.temperature_min[0],
            precipitacion: diario.precipitation_sum[0] || 0,
            // Info extra
            unidad: "°C",
            actualizado: new Date().toLocaleTimeString("es-ES")
        };
    }

    obtenerIcono(codigo) {
        // Códigos WMO
        if (codigo === 0) return "☀️";       // Despejado
        if (codigo <= 3) return "⛅";         // Parcialmente nublado
        if (codigo <= 48) return "☁️";        // Nublado
        if (codigo <= 57) return "🌧️";        // Llovizna
        if (codigo <= 67) return "🌧️";        // Lluvia
        if (codigo <= 77) return "❄️";        // Nieve
        if (codigo <= 82) return "🌦️";        // Chubascos
        return "🌈";                           // Otro
    }

    obtenerDescripcion(codigo) {
        let desc = {
            0: "Despejado", 1: "Mayormente despejado",
            2: "Parcialmente nublado", 3: "Nublado",
            45: "Niebla", 51: "Llovizna", 61: "Lluvia",
            71: "Nieve", 95: "Tormenta"
        };
        return desc[codigo] || "Condición variable";
    }
}
```

3. APLICACIÓN PRINCIPAL
===============================================================================

```javascript
class AppClima {
    #api = new ClimaAPI();
    #procesador = new ProcesadorClima();
    #ciudadActual = null;

    constructor() {
        this.#cargarPreferencias();
    }

    #cargarPreferencias() {
        try {
            let ciudad = localStorage.getItem("clima-ciudad");
            if (ciudad && ClimaAPI.ciudades[ciudad]) {
                this.#ciudadActual = ciudad;
            }
        } catch { /* ignorar */ }
    }

    #guardarPreferencias() {
        if (this.#ciudadActual) {
            localStorage.setItem("clima-ciudad", this.#ciudadActual);
        }
    }

    async obtenerPorCiudad(nombre) {
        let coords = ClimaAPI.ciudades[nombre];
        if (!coords) {
            throw new Error("Ciudad no encontrada: " + nombre);
        }

        this.#ciudadActual = nombre;
        this.#guardarPreferencias();
        return this.obtenerPorCoordenadas(coords.lat, coords.lon);
    }

    async obtenerPorCoordenadas(lat, lon) {
        let datos = await this.#api.obtenerClimaActual(lat, lon);
        return this.#procesador.procesarRespuesta(datos);
    }

    async obtenerUbicacionAutomatica() {
        return new Promise(function(resolve, reject) {
            if (!navigator.geolocation) {
                reject("Geolocalización no soportada");
                return;
            }

            navigator.geolocation.getCurrentPosition(
                function(pos) {
                    resolve({
                        lat: pos.coords.latitude,
                        lon: pos.coords.longitude
                    });
                },
                function(error) {
                    let msgs = {
                        1: "Permiso denegado",
                        2: "No disponible",
                        3: "Timeout"
                    };
                    reject(msgs[error.code] || "Error desconocido");
                },
                { timeout: 10000 }
            );
        });
    }

    async iniciarConGeolocalizacion() {
        try {
            let coords = await this.obtenerUbicacionAutomatica();
            return await this.obtenerPorCoordenadas(coords.lat, coords.lon);
        } catch (error) {
            console.log("⚠️ Geolocalización:", error);
            // Fallback a Madrid
            return this.obtenerPorCiudad("Madrid");
        }
    }

    listarCiudades() {
        return Object.keys(ClimaAPI.ciudades);
    }

    get ciudad() { return this.#ciudadActual; }
}
```

```javascript
async function mostrarClima(nombreCiudad) {
    let app = new AppClima();

    try {
        console.log("⏳ Consultando clima...");

            ? await app.obtenerPorCiudad(nombreCiudad)
            : await app.iniciarConGeolocalizacion();

        let icono = app._procesador?.obtenerIcono(clima.codigoClima) || "🌤️";
        let desc = app._procesador?.obtenerDescripcion(clima.codigoClima) || "";
```

```javascript
async function compararClima(...ciudades) {
    if (ciudades.length === 0) {
        ciudades = ["Madrid", "Barcelona", "Londres"];
    }

    let app = new AppClima();

    try {
        console.log("=== COMPARATIVA DE CLIMA ===");
        console.log("Ciudad".padEnd(15) + "Temp".padEnd(8) + "Mín".padEnd(8) +
            "Máx".padEnd(8) + "Viento".padEnd(8) + "Condición");

        let promesas = ciudades.map(function(c) {
            return app.obtenerPorCiudad(c)
                .then(function(clima) {
                    let desc = app._procesador?.obtenerDescripcion(clima.codigoClima) || "";
                    console.log(
                        c.padEnd(15) +
                        (clima.temperatura + "°C").padEnd(8) +
                        (clima.minHoy + "°C").padEnd(8) +
                        (clima.maxHoy + "°C").padEnd(8) +
                        (clima.viento + "km/h").padEnd(8) +
                        desc
                    );
                })
                .catch(function(e) {
                    console.log(c.padEnd(15) + "❌ " + e.message);
                });
        });

        await Promise.all(promesas);

    } catch (error) {
        console.log("❌ Error general:", error.message);
    }
}
```
