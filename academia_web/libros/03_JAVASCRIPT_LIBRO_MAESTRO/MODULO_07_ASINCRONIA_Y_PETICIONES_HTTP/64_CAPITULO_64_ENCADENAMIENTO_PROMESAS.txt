===============================================================================
BIENVENIDA DEL PROFESOR BÚHO
===============================================================================

Querido estudiante…

Cada then() puede devolver un valor o una nueva promesa.
Si devuelves una promesa, el siguiente then espera a que se resuelva.

Esto permite:
- Secuencias de operaciones.
- Pasar datos de un paso al siguiente.
- Evitar el callback hell.

===============================================================================
OBJETIVOS DEL CAPÍTULO
===============================================================================

✓ Encadenar múltiples then.
✓ Devolver valores y promesas en then.
✓ Propagar errores en la cadena.
✓ Transformar datos entre pasos.

===============================================================================
1. ENCADENAMIENTO BÁSICO
===============================================================================

```javascript
function paso(numero, datos) {
    return new Promise(function(resolve) {
        setTimeout(function() {
            let resultado = datos + " → Paso " + numero;
            console.log("✅ Paso " + numero + " completado");
            resolve(resultado);
        }, 1000);
    });
}

// Encadenamiento
paso(1, "Inicio")
    .then(function(r) {
        return paso(2, r);  // Devuelve nueva promesa
    })
    .then(function(r) {
        return paso(3, r);
    })
    .then(function(r) {
        console.log("🏁 Resultado final:", r);
    });
// Inicio → Paso 1 → Paso 2 → Paso 3
```

2. TRANSFORMAR DATOS ENTRE THEN
===============================================================================

```javascript
function obtenerUsuario(id) {
    return new Promise(function(resolve) {
        setTimeout(function() {
            resolve({ id: id, nombre: "Ana García", email: "ana@email.com" });
        }, 1000);
    });
}

function obtenerPedidos(usuario) {
    return new Promise(function(resolve) {
        setTimeout(function() {
            resolve({
                usuario: usuario.nombre,
                pedidos: [
                    { id: 1, total: 250, producto: "Laptop" },
                    { id: 2, total: 35, producto: "Mouse" }
                ]
            });
        }, 1000);
    });
}

function calcularTotal(datos) {
    let total = datos.pedidos.reduce(function(acc, p) {
        return acc + p.total;
    }, 0);
    return {
        cliente: datos.usuario,
        total: total,
        productos: datos.pedidos.map(p => p.producto)
    };
}

// Cadena completa
obtenerUsuario(1)
    .then(function(usuario) {
        console.log("👤 Usuario:", usuario.nombre);
        return obtenerPedidos(usuario);
    })
    .then(function(pedidos) {
        return calcularTotal(pedidos);
    })
    .then(function(resumen) {
        console.log("📊 Resumen:", resumen);
    });
```

3. PROPAGACIÓN DE ERRORES
===============================================================================

```javascript
function validarUsuario(usuario) {
    if (!usuario.nombre) {
        return Promise.reject("❌ Nombre requerido");
    }
    if (usuario.edad < 18) {
        return Promise.reject("❌ Debe ser mayor de edad");
    }
    return Promise.resolve(usuario);
}

function guardarUsuario(usuario) {
    return new Promise(function(resolve) {
        setTimeout(function() {
            resolve({ ...usuario, id: 42, guardado: true });
        }, 1000);
    });
}

// El error salta todos los then hasta el catch más cercano
validarUsuario({ nombre: "Ana", edad: 25 })
    .then(function(u) {
        console.log("✅ Usuario válido");
        return guardarUsuario(u);
    })
    .then(function(u) {
        console.log("💾 Guardado con ID:", u.id);
    })
    .catch(function(error) {
        console.log("❌ Error:", error);  // Se salta aquí si hay error
    });
```

===============================================================================
4. THEN DEVUELVE PROMESA
===============================================================================

```javascript
// Si devuelves un valor que NO es promesa, se envuelve en Promise.resolve
Promise.resolve(5)
    .then(function(n) {
        return n * 2;  // 10 (se envuelve en Promise.resolve(10))
    })
    .then(function(n) {
        return n + 3;  // 13
    })
    .then(function(n) {
        console.log("Resultado:", n);  // 13
    });

// Equivalente a:
Promise.resolve(5)
    .then(n => Promise.resolve(n * 2))
    .then(n => Promise.resolve(n + 3))
    .then(n => console.log(n));
```

===============================================================================
5. RECUPERACIÓN DE ERRORES
===============================================================================

```javascript
function operacionRiesgosa(numero) {
    return new Promise(function(resolve, reject) {
        if (numero < 0) {
            reject("Número negativo no permitido");
        } else {
            resolve(numero * 2);
        }
    });
}

// Puedes recuperarte de un error
operacionRiesgosa(-5)
    .catch(function(error) {
        console.log("⚠️ Error manejado:", error);
        return 0;  // Valor por defecto
    })
    .then(function(resultado) {
        console.log("✅ Resultado:", resultado);  // 0 (se recuperó)
    });

// Error no recuperado
operacionRiesgosa(-5)
    .then(function(r) {
        console.log("Nunca llega aquí");
    })
    .catch(function(e) {
        console.log("❌ Error final:", e);
    });
```

MINI PROYECTO: PROCESADOR DE PEDIDOS
===============================================================================

```javascript
function recibirPedido(pedido) {
    return new Promise(function(resolve) {
        console.log("📦 Recibido:", pedido.producto);
        setTimeout(function() {
            resolve({ ...pedido, recibido: true });
        }, 500);
    });
}

function prepararPedido(pedido) {
    return new Promise(function(resolve) {
        console.log("👨‍🍳 Preparando:", pedido.producto);
        setTimeout(function() {
            resolve({ ...pedido, preparado: true });
        }, 1000);
    });
}

function entregarPedido(pedido) {
    return new Promise(function(resolve) {
        console.log("🚚 Entregando a:", pedido.cliente);
        setTimeout(function() {
            resolve({ ...pedido, entregado: true });
        }, 800);
    });
}

function procesarPedido(cliente, producto) {
    let pedido = { cliente: cliente, producto: producto };

    return recibirPedido(pedido)
        .then(p => prepararPedido(p))
        .then(p => entregarPedido(p))
        .then(p => {
            console.log("🎉 Pedido completado para", p.cliente);
            console.log("   Estado:", p);
            return p;
        });
}

procesarPedido("Ana", "Pizza")
    .catch(e => console.log("❌ Error:", e));
```

RESUMEN
===============================================================================

✔ then() devuelve una nueva Promesa.
✔ Si devuelves un valor, se envuelve en Promise.resolve().
✔ Si devuelves una Promesa, espera a que se resuelva.
✔ Los errores saltan al catch más cercano.
✔ Puedes recuperarte de errores en un catch y continuar.
