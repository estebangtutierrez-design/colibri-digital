1. OBJETOS ANIDADOS
===============================================================================

```javascript
let estudiante = {
    nombre: "Ana García",
    edad: 25,
    direccion: {
        calle: "Av. Principal 123",
        ciudad: "Madrid",
        codigoPostal: "28001",
        pais: "España"
    },
    cursos: [
        { nombre: "JavaScript", nota: 95 },
        { nombre: "HTML", nota: 90 }
    ]
};

// Acceder a propiedades anidadas
console.log(estudiante.direccion.ciudad);     // Madrid
console.log(estudiante.cursos[0].nombre);      // JavaScript
console.log(estudiante.cursos[1].nota);        // 90

// Acceso seguro con ?.
console.log(estudiante.telefono?.codigo);      // undefined (no existe)
```

===============================================================================
2. SPREAD EN OBJETOS
===============================================================================

```javascript
let datosBasicos = { nombre: "Ana", edad: 25 };
let datosContacto = { email: "ana@email.com", telefono: "555-1234" };

// Combinar objetos
let perfil = { ...datosBasicos, ...datosContacto };
console.log(perfil);
// { nombre: "Ana", edad: 25, email: "ana@email.com", telefono: "555-1234" }

// Copia superficial
let copia = { ...estudiante };
copia.direccion.ciudad = "Barcelona";  // ❌ También modifica original
console.log(estudiante.direccion.ciudad);  // Barcelona
```

===============================================================================
3. ¿QUÉ ES JSON?
===============================================================================

JSON (JavaScript Object Notation):
- Formato de texto para intercambiar datos.
- Similar a los objetos de JS, pero con reglas estrictas.
- JSON.stringify: objeto → string JSON.
- JSON.parse: string JSON → objeto.

Diferencias:
- Claves deben llevar comillas dobles.
- No permite funciones ni undefined.
- Strings con comillas dobles.

```javascript
// Objeto normal
let persona = {
    nombre: "Ana",
    edad: 25,
    saludar() { console.log("Hola"); }  // Las funciones se pierden
};
```

===============================================================================
4. JSON.STRINGIFY Y JSON.PARSE
===============================================================================

```javascript
let producto = {
    id: 1,
    nombre: "Laptop",
    precio: 1200,
    disponible: true,
    tags: ["tecnología", "portátil"]
};

// Objeto → String JSON
let jsonString = JSON.stringify(producto);
console.log(jsonString);
// {"id":1,"nombre":"Laptop","precio":1200,"disponible":true,"tags":["tecnología","portátil"]}
console.log(typeof jsonString);  // string

// String JSON → Objeto
let objetoRecuperado = JSON.parse(jsonString);
console.log(objetoRecuperado.nombre);  // Laptop
```

JSON.stringify con formato:

```javascript
console.log(JSON.stringify(producto, null, 2));
// {
//   "id": 1,
//   "nombre": "Laptop",
//   "precio": 1200,
//   ...
// }
```

===============================================================================
5. USO PRÁCTICO: ALMACENAR EN LOCALSTORAGE
===============================================================================

```javascript
// Guardar objeto en localStorage
function guardarPreferencias(prefs) {
    localStorage.setItem("preferencias", JSON.stringify(prefs));
}

// Recuperar objeto de localStorage
function cargarPreferencias() {
    let datos = localStorage.getItem("preferencias");
    return datos ? JSON.parse(datos) : {};
}

// Uso
let prefs = {
    tema: "oscuro",
    idioma: "es",
    notificaciones: true
};

guardarPreferencias(prefs);
let cargadas = cargarPreferencias();
console.log(cargadas.tema);  // oscuro
```

MINI PROYECTO: PERFIL DE USUARIO ANIDADO

```javascript
let sistemaUsuarios = {
    usuarios: [
        {
            id: 1,
            nombre: "Ana García",
            credenciales: { email: "ana@email.com", rol: "admin" },
            direccion: { calle: "Calle Mayor 10", ciudad: "Madrid" },
            pedidos: [
                { id: 101, total: 250, fecha: "2024-01-15" },
                { id: 102, total: 180, fecha: "2024-02-20" }
            ]
        },
        {
            id: 2,
            nombre: "Luis Pérez",
            credenciales: { email: "luis@email.com", rol: "usuario" },
            direccion: { calle: "Av. Sol 5", ciudad: "Barcelona" },
            pedidos: [
                { id: 103, total: 90, fecha: "2024-03-10" }
            ]
        }
    ],
    buscarUsuario(email) {
        return this.usuarios.find(u => u.credenciales.email === email);
    },
    totalGastado(email) {
        let usuario = this.buscarUsuario(email);
        if (!usuario) return 0;
        return usuario.pedidos.reduce((acc, p) => acc + p.total, 0);
    },
    exportarJSON() {
        return JSON.stringify(this.usuarios, null, 2);
    }
};

let ana = sistemaUsuarios.buscarUsuario("ana@email.com");
console.log("Ciudad: " + ana.direccion.ciudad);  // Madrid
console.log("Rol: " + ana.credenciales.rol);      // admin
console.log("Total gastado: $" + sistemaUsuarios.totalGastado("ana@email.com"));  // $430
```
