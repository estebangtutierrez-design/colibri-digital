function crearProducto(id, nombre, precio, categoria, stock) {
    return {
        id: id,
        nombre: nombre,
        precio: precio,
        categoria: categoria,
        stock: stock,
        fechaAgregado: new Date().toISOString(),
        activo: true
    };
}

2. SISTEMA DE INVENTARIO
===============================================================================

```javascript
let inventario = {
    productos: [],

    // CREATE
    agregar(id, nombre, precio, categoria, stock) {
        if (this.productos.find(p => p.id === id)) {
            console.log("❌ ID duplicado");
            return this;
        }
        let producto = crearProducto(id, nombre, precio, categoria, stock);
        this.productos.push(producto);
        console.log("✅ Producto agregado: " + nombre);
        return this;
    },

    // READ
    listar() {
        console.log("=== INVENTARIO ===");
        if (this.productos.length === 0) {
            console.log("📭 Sin productos");
            return this;
        }
        this.productos.forEach(p => {
            console.log(p.id + " | " + p.nombre.padEnd(15) +
                " | $" + p.precio.toString().padStart(5) +
                " | Stock: " + p.stock);
        });
        return this;
    },

    // UPDATE
    actualizarPrecio(id, nuevoPrecio) {
        let producto = this.productos.find(p => p.id === id);
        if (!producto) return console.log("❌ Producto no encontrado"), this;
        producto.precio = nuevoPrecio;
        console.log("✅ Precio actualizado: " + producto.nombre + " → $" + nuevoPrecio);
        return this;
    },

    // DELETE
    eliminar(id) {
        let idx = this.productos.findIndex(p => p.id === id);
        if (idx === -1) return console.log("❌ Producto no encontrado"), this;
        this.productos.splice(idx, 1);
        console.log("🗑️ Producto eliminado");
        return this;
    },

    // Buscar por categoría
    filtrarPorCategoria(categoria) {
        return this.productos.filter(p => p.categoria === categoria);
    },

    // Productos con bajo stock
    stockBajo(limite = 5) {
        return this.productos.filter(p => p.stock <= limite);
    },

    // Valor total del inventario
    valorTotal() {
        return this.productos.reduce((acc, p) => acc + (p.precio * p.stock), 0);
    },

    // Exportar a JSON
    exportarJSON() {
        return JSON.stringify(this.productos, null, 2);
    },

    // Importar desde JSON
    importarJSON(jsonString) {
        try {
            let datos = JSON.parse(jsonString);
            this.productos = datos;
            console.log("✅ Inventario importado: " + datos.length + " productos");
        } catch (e) {
            console.log("❌ Error al importar JSON");
        }
        return this;
    },

    // Reporte en consola
    reporte() {
        console.log("========== REPORTE DE INVENTARIO ==========");
        console.log("Total productos: " + this.productos.length);
        console.log("Valor total: $" + this.valorTotal());

        let categorias = [...new Set(this.productos.map(p => p.categoria))];
        console.log("Categorías: " + categorias.join(", "));

        let bajoStock = this.stockBajo();
        if (bajoStock.length > 0) {
            console.log("⚠️ Productos con stock bajo:");
            bajoStock.forEach(p => {
                console.log("   - " + p.nombre + " (stock: " + p.stock + ")");
            });
        }

        let masCaro = this.productos.reduce((a, b) => a.precio > b.precio ? a : b);
        console.log("💰 Producto más caro: " + masCaro.nombre + " ($" + masCaro.precio + ")");
        console.log("===========================================");
        return this;
    }
};
```

3. PROBANDO EL SISTEMA
===============================================================================

```javascript
inventario
    .agregar(1, "Laptop", 1200, "tecnología", 10)
    .agregar(2, "Mouse", 25, "tecnología", 50)
    .agregar(3, "Camisa", 35, "ropa", 20)
    .agregar(4, "Pantalón", 50, "ropa", 3)
    .agregar(5, "Monitor", 300, "tecnología", 8)
    .listar();

inventario.actualizarPrecio(1, 1100);
inventario.eliminar(3);
inventario.listar();
inventario.reporte();

// Filtrar
let tecnologia = inventario.filtrarPorCategoria("tecnología");
console.log("Tecnología:", tecnologia.map(p => p.nombre).join(", "));

// Bajo stock
let bajo = inventario.stockBajo();
console.log("Bajo stock:", bajo.map(p => p.nombre + " (" + p.stock + ")"));

// Exportar
console.log("=== JSON EXPORTADO ===");
console.log(inventario.exportarJSON());
```

// 1. Producto más vendido (simulado)
// 2. Ordenar por precio (asc/desc)
// 3. Buscar por nombre

// Ejemplo de cómo empezar:
function buscarPorNombre(invent, texto) {
    return invent.productos.filter(p =>
        p.nombre.toLowerCase().includes(texto.toLowerCase())
    );
}

function ordenarPorPrecio(invent, ascendente = true) {
    return [...invent.productos].sort((a, b) =>
        ascendente ? a.precio - b.precio : b.precio - a.precio
    );
}

console.log("Buscar 'lap':", buscarPorNombre(inventario, "lap").map(p => p.nombre));
console.log("Más barato primero:", ordenarPorPrecio(inventario, true).map(p => p.nombre));
