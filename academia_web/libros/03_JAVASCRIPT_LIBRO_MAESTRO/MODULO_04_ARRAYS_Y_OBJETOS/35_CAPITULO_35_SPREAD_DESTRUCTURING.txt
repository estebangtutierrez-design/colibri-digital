BIENVENIDA DEL PROFESOR BÚHO

Querido estudiante…

JavaScript moderno trajo dos formas poderosas
de trabajar con arrays:

- SPREAD (...): expande un array en elementos individuales.
- DESTRUCTURING: extrae valores en variables individuales.

Hacen tu código más corto y legible.

OBJETIVOS DEL CAPÍTULO

✓ Usar destructuring para extraer valores.
✓ Combinar ambos en la práctica.
✓ Parámetros rest con ...

===============================================================================
1. SPREAD OPERATOR (...)
===============================================================================

Copia arrays:

```javascript
let original = [1, 2, 3];
let copia = [...original];

copia.push(4);
console.log(original);  // [1, 2, 3] (sin cambios)
console.log(copia);     // [1, 2, 3, 4]
```

Combina arrays:

```javascript
let frutas = ["manzana", "banana"];
let verduras = ["zanahoria", "brócoli"];

let alimentos = [...frutas, ...verduras];
console.log(alimentos);  // ["manzana", "banana", "zanahoria", "brócoli"]

let mezclado = [...frutas, "naranja", ...verduras];
console.log(mezclado);   // ["manzana", "banana", "naranja", "zanahoria", "brócoli"]
```

Pasar elementos a funciones:

```javascript
let numeros = [5, 10, 15, 20];

let maximo = Math.max(...numeros);
console.log(maximo);  // 20
```

===============================================================================
2. DESTRUCTURING DE ARRAYS
===============================================================================

Extrae valores en variables en una línea:

```javascript
let colores = ["rojo", "azul", "verde"];

let [primero, segundo, tercero] = colores;

console.log(primero);  // rojo
console.log(segundo);  // azul
console.log(tercero);  // verde
```

Saltar elementos:

```javascript
let [a, , c] = [1, 2, 3];
console.log(a);  // 1
console.log(c);  // 3
```

===============================================================================
3. VALORES POR DEFECTO E INTERCAMBIO
===============================================================================

```javascript
// Valores por defecto
let [nombre = "Invitado", edad = 18] = ["Ana"];
console.log(nombre);  // Ana
console.log(edad);    // 18

// Intercambiar variables (swap)
let x = 10, y = 20;
[x, y] = [y, x];
console.log(x, y);  // 20 10
```

===============================================================================
4. REST (...)
===============================================================================

Captura el resto de elementos:

```javascript
let [primero2, ...resto] = [1, 2, 3, 4, 5];
console.log(primero2);  // 1
console.log(resto);     // [2, 3, 4, 5]

// Rest en funciones
function sumarTodos(...numeros) {
    return numeros.reduce((acc, n) => acc + n, 0);
}
console.log(sumarTodos(1, 2, 3, 4, 5));  // 15
```

===============================================================================
MINI PROYECTO: FUSIONADOR DE LISTAS
===============================================================================

```javascript
function fusionarListas(...listas) {
    // Combinar todas las listas con spread
    let combinado = [].concat(...listas);

    // Eliminar duplicados
    return [...new Set(combinado)];
}

let lista1 = [1, 2, 3, 4];
let lista2 = [3, 4, 5, 6];
let lista3 = [5, 6, 7, 8];

let fusion = fusionarListas(lista1, lista2, lista3);
console.log(fusion);  // [1, 2, 3, 4, 5, 6, 7, 8]

// Destructuring con split
let [nombreCompleto, ...restoInfo] = "Ana García Méndez".split(" ");
console.log("Nombre:", nombreCompleto);  // Ana
console.log("Apellidos:", restoInfo);    // ["García", "Méndez"]

// Copia segura con spread
function agregarElemento(arr, elemento) {
    return [...arr, elemento];  // No modifica el original
}

let base = [1, 2, 3];
let extendido = agregarElemento(base, 4);
console.log(base);       // [1, 2, 3]
console.log(extendido);  // [1, 2, 3, 4]
```
