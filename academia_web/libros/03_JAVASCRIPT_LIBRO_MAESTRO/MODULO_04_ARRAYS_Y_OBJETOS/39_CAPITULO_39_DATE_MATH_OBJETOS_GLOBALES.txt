1. DATE: TRABAJAR CON FECHAS
===============================================================================

```javascript
// Fecha actual
let ahora = new Date();
console.log(ahora);  // Wed Jul 01 2026 ...

// Fecha específica
let navidad = new Date(2026, 11, 25);  // Mes: 0-11
console.log(navidad);  // Fri Dec 25 2026

// Métodos de Date
console.log(ahora.getFullYear());   // 2026
console.log(ahora.getMonth());      // 6 (julio = 6)
console.log(ahora.getDate());       // 1 (día del mes)
console.log(ahora.getDay());        // 3 (miércoles = 3)
console.log(ahora.getHours());      // 10
console.log(ahora.getMinutes());    // 30
console.log(ahora.getSeconds());    // 0
```

2. FORMATEAR FECHAS
===============================================================================

```javascript
let fecha = new Date(2026, 6, 15);  // 15 de julio 2026

console.log(fecha.toLocaleDateString("es-ES"));
// 15/7/2026

console.log(fecha.toLocaleDateString("es-ES", {
    weekday: "long",
    year: "numeric",
    month: "long",
    day: "numeric"
}));
// miércoles, 15 de julio de 2026

console.log(fecha.toLocaleTimeString("es-ES"));
// 0:00:00
```

===============================================================================
3. MATH: OPERACIONES MATEMÁTICAS
===============================================================================

```javascript
// Constantes
console.log(Math.PI);   // 3.141592653589793
console.log(Math.E);    // 2.718281828459045

// Redondeo
console.log(Math.round(4.7));   // 5
console.log(Math.round(4.3));   // 4
console.log(Math.ceil(4.1));    // 5 (arriba)
console.log(Math.floor(4.9));   // 4 (abajo)
console.log(Math.trunc(4.9));   // 4 (sin decimales)

// Potencias y raíces
console.log(Math.pow(2, 3));    // 8 (2³)
console.log(Math.sqrt(16));     // 4
console.log(Math.abs(-5));      // 5

// Máximo y mínimo
console.log(Math.max(10, 5, 20));  // 20
console.log(Math.min(10, 5, 20));  // 5
```

===============================================================================
4. NÚMEROS ALEATORIOS
===============================================================================

```javascript
// Aleatorio entre 0 y 1 (sin incluir 1)
console.log(Math.random());  // ej: 0.7234

// Aleatorio entre 1 y 10
let aleatorio = Math.floor(Math.random() * 10) + 1;
console.log(aleatorio);  // ej: 7

// Función reusable
function numeroAleatorio(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
}

console.log(numeroAleatorio(1, 100));  // ej: 42
```

===============================================================================
5. MÉTODOS DE NUMBER Y STRING
===============================================================================

```javascript
// Number
let precio = 19.99;
console.log(precio.toFixed(1));       // "20.0"
console.log(precio.toFixed(0));       // "20"
console.log(Number.isInteger(5));     // true
console.log(Number.isInteger(5.5));   // false

// String (objeto global)
let texto = "  JavaScript es genial  ";

console.log(texto.length);            // 24
console.log(texto.trim());            // "JavaScript es genial"
console.log(texto.trim().toUpperCase());  // "JAVASCRIPT ES GENIAL"
console.log(texto.trim().substring(0, 10));  // "JavaScript"
console.log(texto.trim().includes("genial"));  // true
console.log("100".padStart(5, "0"));  // "00100"
```

MINI PROYECTO: CALENDARIO DE EVENTOS
===============================================================================

```javascript
let eventos = [];

function agregarEvento(nombre, año, mes, dia) {
    let fecha = new Date(año, mes - 1, dia);
    eventos.push({
        nombre: nombre,
        fecha: fecha,
        fechaFormateada: fecha.toLocaleDateString("es-ES", {
            weekday: "long",
            day: "numeric",
            month: "long",
            year: "numeric"
        })
    });
    console.log("📅 Evento agregado: " + nombre);
}

function eventosProximos(diasLimite = 30) {
    let hoy = new Date();
    let limite = new Date();
    limite.setDate(limite.getDate() + diasLimite);

    let proximos = eventos.filter(e => e.fecha >= hoy && e.fecha <= limite);

    if (proximos.length === 0) {
        console.log("📭 No hay eventos próximos");
        return;
    }

    console.log("=== PRÓXIMOS EVENTOS ===");
    proximos.sort((a, b) => a.fecha - b.fecha);
    proximos.forEach(e => {
        let diasFaltan = Math.floor((e.fecha - new Date()) / (1000 * 60 * 60 * 24));
        console.log("📌 " + e.nombre + " - " + e.fechaFormateada + " (en " + diasFaltan + " días)");
    });
}

function eventoAleatorio() {
    let nombres = ["Taller JS", "Conferencia Web", "Meetup Dev", "Hackathon"];
    let año = 2026;
    let mes = numeroAleatorio(7, 12);
    let dia = numeroAleatorio(1, 28);
    let nombre = nombres[Math.floor(Math.random() * nombres.length)];
    agregarEvento(nombre, año, mes, dia);
}

agregarEvento("Examen JavaScript", 2026, 8, 15);
agregarEvento("Entrega Proyecto", 2026, 7, 20);
eventoAleatorio();
eventosProximos(60);
```
