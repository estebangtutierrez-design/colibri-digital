===============================================================================
1. CREAR ARRAYS
===============================================================================

```javascript
// Forma 1: Literal (recomendada)
let frutas = ["manzana", "banana", "naranja"];

// Forma 2: Constructor (evitar)
let numeros = new Array(1, 2, 3);

// Array vacío
let vacio = [];

// Array con diferentes tipos
let mixto = [1, "hola", true, null, { nombre: "Ana" }];

console.log(frutas);  // ["manzana", "banana", "naranja"]
```

===============================================================================
2. ÍNDICES Y ACCESO
===============================================================================

Los índices empiezan en 0.

```javascript
let colores = ["rojo", "azul", "verde", "amarillo"];

console.log(colores[0]);    // rojo
console.log(colores[2]);    // verde
console.log(colores[4]);    // undefined (no existe)

// Último elemento
console.log(colores[colores.length - 1]);  // amarillo
```

===============================================================================
3. MODIFICAR ELEMENTOS
===============================================================================

```javascript
let numeros = [10, 20, 30, 40];

numeros[0] = 100;
numeros[2] = 300;

console.log(numeros);  // [100, 20, 300, 40]

numeros[10] = 999;  // ⚠️ CUIDADO: crea espacios vacíos
console.log(numeros);  // [100, 20, 300, 40, vacío x6, 999]
```

===============================================================================
4. PROPIEDAD LENGTH
===============================================================================

```javascript
let letras = ["a", "b", "c"];

console.log(letras.length);  // 3

// length también se puede modificar
letras.length = 2;
console.log(letras);  // ["a", "b"] (se eliminó "c")

letras.length = 0;    // Vaciar array
console.log(letras);  // []
```

===============================================================================
5. RECORRER ARRAYS CON BUCLES
===============================================================================

```javascript
let estudiantes = ["Ana", "Luis", "Carlos", "Sofía"];

// Bucle for clásico
for (let i = 0; i < estudiantes.length; i++) {
    console.log("Estudiante " + (i + 1) + ": " + estudiantes[i]);
}

// Bucle for...of (más legible)
for (let nombre of estudiantes) {
    console.log("Hola " + nombre);
}
```

===============================================================================
MINI PROYECTO: GESTOR DE LISTA DE COMPRAS
===============================================================================

```javascript
let listaCompras = [];

function agregar(item) {
    listaCompras.push(item);
    console.log("✅ " + item + " agregado");
}

function mostrarLista() {
    if (listaCompras.length === 0) {
        console.log("📭 Lista vacía");
        return;
    }
    console.log("🛒 LISTA DE COMPRAS:");
    for (let i = 0; i < listaCompras.length; i++) {
        console.log((i + 1) + ". " + listaCompras[i]);
    }
}

function buscar(item) {
    for (let i = 0; i < listaCompras.length; i++) {
        if (listaCompras[i] === item) {
            return "🔍 Encontrado en posición " + i;
        }
    }
    return "🔍 " + item + " no está en la lista";
}

agregar("Pan");
agregar("Leche");
agregar("Huevos");
mostrarLista();
console.log(buscar("Leche"));
console.log(buscar("Arroz"));
```
