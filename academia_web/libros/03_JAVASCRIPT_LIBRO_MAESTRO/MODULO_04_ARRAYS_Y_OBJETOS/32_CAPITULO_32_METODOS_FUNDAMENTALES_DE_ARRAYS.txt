===============================================================================
1. AGREGAR Y QUITAR AL FINAL
===============================================================================

```javascript
let numeros = [1, 2, 3];

numeros.push(4);        // Agrega al final
console.log(numeros);   // [1, 2, 3, 4]

let ultimo = numeros.pop();  // Quita del final
console.log(ultimo);    // 4
console.log(numeros);   // [1, 2, 3]
```

===============================================================================
2. AGREGAR Y QUITAR AL INICIO
===============================================================================

```javascript
let letras = ["b", "c"];

letras.unshift("a");    // Agrega al inicio
console.log(letras);    // ["a", "b", "c"]

let primero = letras.shift();  // Quita del inicio
console.log(primero);   // "a"
console.log(letras);    // ["b", "c"]
```

===============================================================================
3. BUSCAR ELEMENTOS
===============================================================================

```javascript
let frutas = ["manzana", "banana", "naranja", "banana"];

// indexOf: primera posición
console.log(frutas.indexOf("banana"));      // 1
console.log(frutas.indexOf("uva"));         // -1 (no existe)

// lastIndexOf: última posición
console.log(frutas.lastIndexOf("banana"));  // 3

// includes: ¿existe?
console.log(frutas.includes("manzana"));     // true
console.log(frutas.includes("pera"));        // false
```

===============================================================================
4. CONVERTIR
===============================================================================

```javascript
// join: array → string
let palabras = ["Hola", "mundo", "JS"];
console.log(palabras.join(" "));   // "Hola mundo JS"
console.log(palabras.join("-"));   // "Hola-mundo-JS"
console.log(palabras.join(""));    // "HolamundoJS"

// split: string → array
let texto = "manzana,banana,naranja";
let frutasSeparadas = texto.split(",");
console.log(frutasSeparadas);  // ["manzana", "banana", "naranja"]
```

===============================================================================
5. ORDENAR E INVERTIR
===============================================================================

```javascript
let numeros = [3, 1, 4, 1, 5, 9];

// reverse: invierte orden
numeros.reverse();
console.log(numeros);  // [9, 5, 1, 4, 1, 3]

// sort: ordena (como strings por defecto)
numeros.sort();
console.log(numeros);  // [1, 1, 3, 4, 5, 9]

// ⚠️ sort() ordena como STRINGS
let mezclados = [10, 2, 1, 20];
mezclados.sort();
console.log(mezclados);  // [1, 10, 2, 20] (incorrecto numéricamente)

// sort numérico correcto
mezclados.sort((a, b) => a - b);
console.log(mezclados);  // [1, 2, 10, 20]
```

MINI PROYECTO: EDITOR DE LISTA DE TAREAS
===============================================================================

```javascript
let tareas = [];

function agregar(descripcion) {
    tareas.push({ desc: descripcion, completada: false });
    console.log("📝 Tarea agregada: " + descripcion);
}

function completar(descripcion) {
    let idx = -1;
    for (let i = 0; i < tareas.length; i++) {
        if (tareas[i].desc === descripcion) {
            idx = i;
            break;
        }
    }
    if (idx !== -1) {
        tareas[idx].completada = true;
        console.log("✅ Tarea completada: " + descripcion);
    } else {
        console.log("❌ Tarea no encontrada");
    }
}

function eliminarUltima() {
    if (tareas.length === 0) return console.log("📭 Sin tareas");
    let eliminada = tareas.pop();
    console.log("🗑️ Eliminada: " + eliminada.desc);
}

function mostrar() {
    console.log("=== TAREAS (" + tareas.length + ") ===");
    for (let t of tareas) {
        let estado = t.completada ? "✅" : "⬜";
        console.log(estado + " " + t.desc);
    }
}

agregar("Estudiar arrays");
agregar("Hacer ejercicio");
agregar("Leer un libro");
completar("Estudiar arrays");
mostrar();
eliminarUltima();
mostrar();
```
