===============================================================================
1. forEach
===============================================================================

Ejecuta una función por cada elemento.
No devuelve un nuevo array.

```javascript
let frutas = ["manzana", "banana", "naranja"];

frutas.forEach(function(fruta, indice) {
    console.log((indice + 1) + ". " + fruta);
});

// Con arrow function
frutas.forEach(f => console.log("🍎 " + f));
```

===============================================================================
2. Parámetros de forEach
===============================================================================

```javascript
let numeros = [10, 20, 30];

numeros.forEach(function(valor, indice, arrayCompleto) {
    console.log(
        "Valor: " + valor +
        ", Índice: " + indice +
        ", Array: " + arrayCompleto
    );
});
// Valor: 10, Índice: 0, Array: [10,20,30]
// Valor: 20, Índice: 1, Array: [10,20,30]
// Valor: 30, Índice: 2, Array: [10,20,30]
```

===============================================================================
3. map
===============================================================================

Crea un NUEVO array transformando cada elemento.
El array original no se modifica.

```javascript
let numeros = [1, 2, 3, 4, 5];

let dobles = numeros.map(function(n) {
    return n * 2;
});

console.log(dobles);    // [2, 4, 6, 8, 10]
console.log(numeros);   // [1, 2, 3, 4, 5] (sin cambios)

// Con arrow function
let cuadrados = numeros.map(n => n * n);
console.log(cuadrados);  // [1, 4, 9, 16, 25]
```

===============================================================================
4. forEach vs map
===============================================================================

```javascript
let precios = [100, 200, 300];

// forEach: ejecuta efecto secundario (mostrar en consola)
precios.forEach(p => {
    console.log("Precio: $" + p);
});

// map: transforma datos y devuelve nuevo array
let preciosConIVA = precios.map(p => p * 1.21);
console.log(preciosConIVA);  // [121, 242, 363]

// ⚠️ NO hacer: map para efectos secundarios
precios.map(p => console.log("❌ No hagas esto"));
```

===============================================================================
5. EJEMPLOS PRÁCTICOS
===============================================================================

```javascript
let usuarios = [
    { nombre: "Ana", edad: 25 },
    { nombre: "Luis", edad: 30 },
    { nombre: "Sofía", edad: 22 }
];

// Extraer solo los nombres
let nombres = usuarios.map(u => u.nombre);
console.log(nombres);  // ["Ana", "Luis", "Sofía"]

// Formatear mensajes
let mensajes = usuarios.map(u => u.nombre + " tiene " + u.edad + " años");
console.log(mensajes);  // ["Ana tiene 25 años", ...]

// forEach para mostrar en tabla
usuarios.forEach(u => {
    console.log("| " + u.nombre.padEnd(6) + " | " + u.edad + " años |");
});
```

===============================================================================
MINI PROYECTO: TRANSFORMADOR DE PRODUCTOS
===============================================================================

```javascript
let productos = [
    { nombre: "Camisa", precio: 25, descuento: 10 },
    { nombre: "Pantalón", precio: 40, descuento: 15 },
    { nombre: "Zapatos", precio: 60, descuento: 20 }
];

function calcularPrecioFinal(precio, descuento) {
    return precio - (precio * descuento / 100);
}

// Transformar con map
let productosFinal = productos.map(p => ({
    nombre: p.nombre,
    precioOriginal: p.precio,
    precioFinal: calcularPrecioFinal(p.precio, p.descuento),
    ahorro: p.precio * p.descuento / 100
}));

console.log("=== PRODUCTOS CON DESCUENTO ===");
productosFinal.forEach(p => {
    console.log(p.nombre + ": $" + p.precioOriginal +
        " → $" + p.precioFinal +
        " (Ahorras $" + p.ahorro + ")");
});

// Obtener solo precios finales
let preciosFinales = productos.map(p => calcularPrecioFinal(p.precio, p.descuento));
let total = preciosFinales.reduce((a, b) => a + b, 0);
console.log("💰 Total: $" + total);
```

RESUMEN
===============================================================================

✔ forEach: ejecuta función por cada elemento. No devuelve nada.
✔ map: crea NUEVO array transformando cada elemento.
✔ forEach → efectos secundarios (console.log, modificar externo).
✔ map → transformar datos.
✔ Ambos reciben: (valor, índice, array).
