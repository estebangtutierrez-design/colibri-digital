// optimizaciones.js

const Optimizacion = {
    // 1. Evitar reflows innecesarios
    buenRenderizado() {
        // ❌ Mal: múltiples cambios individuales
        let elemento = document.querySelector("#mi-elemento");
        elemento.style.width = "100px";
        elemento.style.height = "100px";
        elemento.style.background = "red";

        // ✅ Bien: un solo cambio
        elemento.style.cssText = "width:100px;height:100px;background:red";

        // ✅ Mejor: usar clases CSS
        elemento.className = "caja-roja";
    },

    // 2. Fragmentos para múltiples inserciones
    insertarMuchos(datos, contenedor) {
        // ❌ Mal: appendChild en cada iteración
        datos.forEach(function(d) {
            let div = document.createElement("div");
            div.textContent = d;
            contenedor.appendChild(div);
        });

        // ✅ Bien: fragmento
        let fragmento = document.createDocumentFragment();
        datos.forEach(function(d) {
            let div = document.createElement("div");
            div.textContent = d;
            fragmento.appendChild(div);
        });
        contenedor.appendChild(fragmento);
    },

    // 3. Debounce para eventos frecuentes
    debounce(funcion, espera) {
        let timeout;
        return function(...args) {
            clearTimeout(timeout);
            timeout = setTimeout(() => funcion.apply(this, args), espera);
        };
    },

    // 4. Memoización de resultados
    memoizar(funcion) {
        const cache = new Map();
        return function(...args) {
            let clave = JSON.stringify(args);
            if (cache.has(clave)) {
                return cache.get(clave);
            }
            let resultado = funcion.apply(this, args);
            cache.set(clave, resultado);
            return resultado;
        };
    }
};

// Ejemplo de memoización
const calcularCostoso = Optimizacion.memoizar(function(n) {
    console.log("🔄 Calculando para", n);
    let resultado = 0;
    for (let i = 0; i < n * 100000; i++) {
        resultado += i;
    }
    return resultado;
});

// Virtual scrolling: renderizar solo elementos visibles

class VirtualList {
    constructor(contenedor, items, itemAltura = 50) {
        this.contenedor = contenedor;
        this.items = items;
        this.itemAltura = itemAltura;
        this.visibleItems = Math.ceil(contenedor.clientHeight / itemAltura) + 2;
        this.scrollTop = 0;

        this._configurar();
    }

    _configurar() {
        this.contenedor.style.overflow = "auto";
        this.contenedor.style.position = "relative";
        this.contenedor.style.height = "400px";

        // Elemento espaciador para el scroll total
        this.espaciador = document.createElement("div");
        this.espaciador.style.height = (this.items.length * this.itemAltura) + "px";
        this.contenedor.appendChild(this.espaciador);

        this.contenedor.addEventListener("scroll", () => this._render());
        this._render();
    }

    _render() {
        this.scrollTop = this.contenedor.scrollTop;
        let startIdx = Math.floor(this.scrollTop / this.itemAltura);
        let endIdx = Math.min(startIdx + this.visibleItems, this.items.length);

        // Eliminar items anteriores
        let itemsActuales = this.contenedor.querySelectorAll(".virtual-item");
        itemsActuales.forEach(el => el.remove());

        // Crear solo los items visibles
        for (let i = startIdx; i < endIdx; i++) {
            let div = document.createElement("div");
            div.className = "virtual-item";
            div.style.cssText = `position:absolute;top:${i * this.itemAltura}px;
                                 height:${this.itemAltura}px;width:100%;padding:8px;
                                 border-bottom:1px solid #eee;`;
            div.textContent = `${i + 1}. ${this.items[i]}`;
            this.espaciador.appendChild(div);
        }
    }
}

```yaml
# .github/workflows/deploy.yml - GitHub Actions

name: Deploy a GitHub Pages
on:
  push:
    branches: [main]

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Setup Node
        uses: actions/setup-node@v3
        with:
          node-version: 18

      - name: Install dependencies
        run: npm ci

      - name: Build
        run: npm run build

      - name: Deploy to GitHub Pages
        uses: peaceiris/actions-gh-pages@v3
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: ./dist
```
