```javascript
// =============================================
// GESTOR DE PROYECTOS - APLICACIÓN COMPLETA
// JavaScript Master Book - Capítulo 80
// =============================================

(function() {
    "use strict";

    // ===== INICIALIZACIÓN =====
    async function iniciarApp() {
        console.log("%c📋 Gestor de Proyectos", "font-size:24px;font-weight:bold");
        console.log("%cJavaScript Master Book - Capítulo 80",
                    "font-size:14px;color:#4A90D9");

        // 1. Inicializar store
        store.init();
        console.log("✅ Store inicializado");

        // 2. Conectar base de datos local
        try {
            await dbLocal.conectar();
            console.log("✅ IndexedDB conectado");
        } catch (e) {
            console.log("⚠️ IndexedDB no disponible:", e.message);
        }

        // 3. Iniciar servicio de sincronización
        await syncService.iniciar();
        console.log("✅ Sync service iniciado");

        // 4. Configurar logger
        Logger.setNivel("INFO");
        Logger.info("App iniciada", { version: "1.0.0" });

        // 5. Registrar todas las rutas
        configurarRutas();

        // 6. Iniciar router
        router.iniciar();

        // 7. Configurar debug
        Debug.init();

        // 8. Crear indicador de red
        crearIndicadorRed();

        // 9. Cargar datos iniciales si es necesario
        if (store.getProyectos().length === 0) {
            cargarDatosDemo();
        }

        console.log("🚀 App lista para usar");
    }

    // ===== RUTAS =====
    function configurarRutas() {
        // Página principal
        router.registrar("/", function() {
            let stats = selectores.getDashboardStats();

            return `
                <div class="card" style="text-align:center;padding:40px">
                    <h1>📋 Gestor de Proyectos</h1>
                    <p style="color:var(--text-light);margin:15px 0">
                        Organiza tus proyectos y tareas de forma sencilla
                    </p>
                    ${store.estaAutenticado() ? `
                        <div style="display:grid;grid-template-columns:repeat(4,1fr);
                                    gap:15px;margin:30px 0;text-align:center">
                            <div class="card">
                                <div style="font-size:32px">${stats.totalProyectos}</div>
                                <small>Proyectos</small>
                            </div>
                            <div class="card">
                                <div style="font-size:32px">${stats.totalTareas}</div>
                                <small>Tareas</small>
                            </div>
                            <div class="card">
                                <div style="font-size:32px">${stats.tareasPendientes}</div>
                                <small>Pendientes</small>
                            </div>
                            <div class="card">
                                <div style="font-size:32px">${stats.progresoGeneral}%</div>
                                <small>Progreso</small>
                            </div>
                        </div>
                        <button onclick="router.irA('/proyectos')"
                                class="btn-primary">Ver Proyectos</button>
                    ` : `
                        <p style="margin:20px 0">
                            <a href="#/login" class="btn-primary"
                               style="text-decoration:none;padding:10px 20px">
                                Iniciar Sesión
                            </a>
                            <a href="#/registro" class="btn-success"
                               style="text-decoration:none;padding:10px 20px;margin-left:10px">
                                Registrarse
                            </a>
                        </p>
                    `}
                </div>
            `;
        });

        // Autenticación
        router.registrar("/login", function() {
            return '<div class="card" style="max-width:400px;margin:40px auto">' +
                   '<h2>Iniciar Sesión</h2>' +
                   '<form id="login-form">' +
                   '<div class="form-group"><label>Email</label>' +
                   '<input type="email" id="login-email" required></div>' +
                   '<div class="form-group"><label>Contraseña</label>' +
                   '<input type="password" id="login-password" required></div>' +
                   '<button type="submit" class="btn-primary" style="width:100%">Entrar</button>' +
                   '<p style="margin-top:10px;text-align:center">' +
                   '¿No tienes cuenta? <a href="#/registro">Regístrate</a></p>' +
                   '</form></div>';
        });

        router.registrar("/registro", function() {
            return '<div class="card" style="max-width:400px;margin:40px auto">' +
                   '<h2>Crear Cuenta</h2>' +
                   '<form id="registro-form">' +
                   '<div class="form-group"><label>Nombre</label>' +
                   '<input type="text" id="reg-nombre" required></div>' +
                   '<div class="form-group"><label>Email</label>' +
                   '<input type="email" id="reg-email" required></div>' +
                   '<div class="form-group"><label>Contraseña</label>' +
                   '<input type="password" id="reg-password" minlength="6" required></div>' +
                   '<button type="submit" class="btn-success" style="width:100%">' +
                   'Crear Cuenta</button>' +
                   '<p style="margin-top:10px;text-align:center">' +
                   '¿Ya tienes cuenta? <a href="#/login">Inicia sesión</a></p>' +
                   '</form></div>';
        });

        // Proyectos (requiere auth)
        router.registrar("/proyectos", renderizarProyectos, true);

        // Detalle de proyecto
        router.registrar("/proyecto/:id", function() {
            let id = window.location.hash.split("/").pop();
            renderizarProyecto(id);
        }, true);
    }

    // ===== DATOS DEMO =====
    function cargarDatosDemo() {
        if (!store.estaAutenticado()) return;

        Logger.info("Cargando datos de demostración");

        let p1 = store.crearProyecto("Aprender JavaScript",
            "Completar el JavaScript Master Book");
        let p2 = store.crearProyecto("Proyecto Personal",
            "Construir mi portfolio");

        store.crearTarea(p1.id, "Estudiar módulos 1-4", "Repasar fundamentos", "alta");
        store.crearTarea(p1.id, "Practicar con proyectos", "Hacer los mini proyectos", "media");
        store.crearTarea(p1.id, "Completar el libro", "Los 80 capítulos", "media");
        store.crearTarea(p2.id, "Diseñar layout", "Crear el diseño en Figma", "alta");
        store.crearTarea(p2.id, "Desarrollar componentes", "HTML, CSS, JS", "media");

        Logger.info("Datos demo cargados: 2 proyectos, 5 tareas");
    }

    // ===== INICIAR =====
    document.addEventListener("DOMContentLoaded", iniciarApp);
})();
```
