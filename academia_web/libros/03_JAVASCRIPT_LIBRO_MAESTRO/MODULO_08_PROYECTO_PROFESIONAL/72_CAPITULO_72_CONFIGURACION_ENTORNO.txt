3. ROUTER SIMPLE
===============================================================================

```javascript
// router.js - Sistema de rutas SPA

class Router {
    #rutas = {};
    #actual = null;

    constructor() {
        window.addEventListener("hashchange", () => this.#navegar());
    }

    registrar(ruta, componente) {
        this.#rutas[ruta] = componente;
    }

    #navegar() {
        let hash = window.location.hash.slice(1) || "/";
        let componente = this.#rutas[hash];

        if (!componente) {
            componente = this.#rutas["/404"] || (() => "<h2>404 - No encontrado</h2>");
        }

        let contenido = componente();
        document.querySelector("#main-content").innerHTML = contenido;
        this.#actual = hash;
    }

    irA(ruta) {
        window.location.hash = ruta;
    }

    iniciar() {
        this.#navegar();
    }

    get rutaActual() { return this.#actual; }
}

// Exportar instancia única
const router = new Router();
```

```javascript
// app.js - Punto de entrada

(function() {
    console.log("🚀 Gestor de Proyectos iniciando...");

    // Registrar rutas
    router.registrar("/", function() {
        return "<h2>Bienvenido al Gestor de Proyectos</h2>" +
               "<p>Selecciona una opción del menú.</p>";
    });

    router.registrar("/login", function() {
        return '<div class="card">' +
               '<h2>Iniciar Sesión</h2>' +
               '<form id="login-form">' +
               '<div class="form-group">' +
               '<label>Email</label>' +
               '<input type="email" id="login-email" required>' +
               '</div>' +
               '<div class="form-group">' +
               '<label>Contraseña</label>' +
               '<input type="password" id="login-password" required>' +
               '</div>' +
               '<button type="submit" class="btn-primary">Entrar</button>' +
               '</form></div>';
    });

    router.registrar("/proyectos", function() {
        let proyectos = store.getProyectos();
        if (proyectos.length === 0) {
            return "<h2>Mis Proyectos</h2><p>No hay proyectos aún.</p>" +
                   '<button onclick="router.irA(\'/proyecto/nuevo\')" ' +
                   'class="btn-primary">+ Nuevo Proyecto</button>';
        }

        let html = "<h2>Mis Proyectos</h2>" +
                   '<button onclick="router.irA(\'/proyecto/nuevo\')" ' +
                   'class="btn-primary">+ Nuevo</button><div class="proyectos-grid">';

        proyectos.forEach(function(p) {
            html += '<div class="card proyecto-card" onclick="router.irA(\'/proyecto/' +
                    p.id + '\')">' +
                    '<h3>' + p.nombre + '</h3>' +
                    '<p>' + (p.descripcion || "") + '</p>' +
                    '<small>' + store.getTareas(p.id).length + ' tareas</small>' +
                    '</div>';
        });

        return html + "</div>";
    });

    // Iniciar
    router.iniciar();
    console.log("✅ App lista");
})();
```
