```javascript
// tests.js - Pruebas unitarias simples

const Tests = {
    _pasadas: 0,
    _falladas: 0,

    assert(condicion, mensaje) {
        if (condicion) {
            this._pasadas++;
            console.log("✅ " + mensaje);
        } else {
            this._falladas++;
            console.log("❌ " + mensaje);
        }
    },

    assertEquals(actual, esperado, mensaje) {
        if (actual === esperado) {
            this._pasadas++;
            console.log("✅ " + mensaje);
        } else {
            this._falladas++;
            console.log("❌ " + mensaje);
            console.log("   Esperado: " + esperado);
            console.log("   Actual:   " + actual);
        }
    },

    reporte() {
        let total = this._pasadas + this._falladas;
        console.log("========== RESULTADOS ==========");
        console.log("Total: " + total);
        console.log("✅ Pasadas: " + this._pasadas);
        console.log("❌ Falladas: " + this._falladas);
        console.log("📊 " + (total > 0
            ? Math.round((this._pasadas / total) * 100) : 0) + "% éxito");

        if (this._falladas > 0) {
            console.log("⚠️ Hay pruebas falladas");
        } else {
            console.log("🎉 Todas las pruebas pasaron");
        }
    }
};

// ===== PRUEBAS =====

function ejecutarPruebas() {
    console.log("🧪 Iniciando pruebas...\n");

    // Pruebas de lógica de negocio
    Tests.assertEquals(
        store.registrar("Test", "test@test.com", "123456").exito,
        true,
        "Registro de nuevo usuario"
    );

    Tests.assertEquals(
        store.registrar("Test", "test@test.com", "123456").exito,
        false,
        "Registro con email duplicado"
    );

    Tests.assertEquals(
        store.login("test@test.com", "123456").exito,
        true,
        "Login con credenciales correctas"
    );

    Tests.assertEquals(
        store.login("test@test.com", "wrong").exito,
        false,
        "Login con contraseña incorrecta"
    );

    // Pruebas de proyectos
    let proyecto = store.crearProyecto("Proyecto Test");
    Tests.assert(proyecto.id.startsWith("p"), "Proyecto tiene ID válido");
    Tests.assertEquals(proyecto.nombre, "Proyecto Test", "Nombre del proyecto");
    Tests.assertEquals(store.getProyectos().length, 1, "1 proyecto en la lista");

    // Pruebas de tareas
    let tarea = store.crearTarea(proyecto.id, "Tarea 1");
    Tests.assert(tarea.id.startsWith("t"), "Tarea tiene ID válido");
    Tests.assertEquals(tarea.estado, "pendiente", "Estado inicial pendiente");
    Tests.assertEquals(store.getTareas(proyecto.id).length, 1, "1 tarea en el proyecto");

    store.cambiarEstadoTarea(tarea.id, "progreso");
    let tareaActualizada = store.getTareas(proyecto.id)[0];
    Tests.assertEquals(tareaActualizada.estado, "progreso", "Cambio de estado");

    store.eliminarTarea(tarea.id);
    Tests.assertEquals(store.getTareas(proyecto.id).length, 0, "Tarea eliminada");

    // Pruebas de utilidades
    Tests.assert(validarEmail("ana@email.com"), "Email válido");
    Tests.assert(!validarEmail("invalido"), "Email inválido (sin @)");
    Tests.assert(!validarEmail(""), "Email vacío");

    Tests.assertEquals(
        selectores.getDashboardStats().totalProyectos,
        1,
        "Dashboard: 1 proyecto"
    );

    console.log("\n");
    Tests.reporte();
}

// utilidades para testing
function validarEmail(email) {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}
```

```javascript
// logger.js - Sistema de logging

const Logger = {
    niveles: { DEBUG: 0, INFO: 1, WARN: 2, ERROR: 3 },
    nivelActual: 0,  // DEBUG

    setNivel(nivel) {
        this.nivelActual = this.niveles[nivel] || 0;
    },

    _formatear(nivel, mensaje, datos) {
        let timestamp = new Date().toISOString();
        let base = `[${timestamp}] [${nivel}] ${mensaje}`;
        return datos ? base + " " + JSON.stringify(datos) : base;
    },

    debug(mensaje, datos) {
        if (this.nivelActual <= this.niveles.DEBUG) {
            console.debug(this._formatear("DEBUG", mensaje, datos));
        }
    },

    info(mensaje, datos) {
        if (this.nivelActual <= this.niveles.INFO) {
            console.info(this._formatear("INFO", mensaje, datos));
        }
    },

    warn(mensaje, datos) {
        if (this.nivelActual <= this.niveles.WARN) {
            console.warn(this._formatear("WARN", mensaje, datos));
        }
    },

    error(mensaje, error) {
        if (this.nivelActual <= this.niveles.ERROR) {
            console.error(this._formatear("ERROR", mensaje, {
                message: error?.message,
                stack: error?.stack?.split("\n").slice(0, 3).join("; ")
            }));
        }
    },

    // Logging de rendimiento
    tiempo(inicio, operacion) {
        let duracion = performance.now() - inicio;
        this.info(`${operacion}: ${duracion.toFixed(2)}ms`);
        return duracion;
    }
};
```

```javascript
// debug.js - Utilidades de depuración

const Debug = {
    // Inspeccionar estado
    inspeccionar() {
        console.log("========== DEBUG STATE ==========");
        console.log("Usuario:", store.getUsuarioActual()?.nombre || "No logueado");
        console.log("Proyectos:", store.getProyectos().length);
        console.log("Tareas:", store._tareas.length);
        console.log("Auth:", store.estaAutenticado());
        console.log("Sync pendientes:", syncService.cantidadPendientes);

        // Mostrar todas las tareas
        console.log("\n📋 Tareas:");
        store._tareas.forEach(function(t) {
            console.log(`  [${t.id}] ${t.titulo} - ${t.estado} (${t.prioridad})`);
        });
    },

    // Simular carga lenta
    simularCargaLenta(ms = 2000) {
        return new Promise(function(resolve) {
            Logger.info(`Simulando carga de ${ms}ms...`);
            setTimeout(resolve, ms);
        });
    },

    // Resetear datos
    resetearDatos() {
        if (confirm("¿Resetear todos los datos?")) {
            localStorage.removeItem("gp-usuarios");
            localStorage.removeItem("gp-proyectos");
            localStorage.removeItem("gp-tareas");
            sessionStorage.removeItem("sesion");
            location.reload();
        }
    },

    // Monitorear rendimiento
    monitorearRendimiento() {
        let tiempos = [];

        function medir(nombre, fn) {
            let inicio = performance.now();
            fn();
            let fin = performance.now();
            tiempos.push({ nombre, duracion: fin - inicio });
            return fin - inicio;
        }

        return {
            medir: medir,
            reporte: function() {
                console.log("=== REPORTE DE RENDIMIENTO ===");
                tiempos.forEach(function(t) {
                    let barra = "█".repeat(Math.round(t.duracion / 10));
                    console.log(`${t.nombre.padEnd(30)} ${t.duracion.toFixed(1)}ms ${barra}`);
                });
            }
        };
    },

    // Atajo de teclado: Ctrl+Shift+D para debug
    init() {
        document.addEventListener("keydown", function(e) {
            if (e.ctrlKey && e.shiftKey && e.key === "D") {
                e.preventDefault();
                Debug.inspeccionar();
            }
            if (e.ctrlKey && e.shiftKey && e.key === "R") {
                e.preventDefault();
                Debug.resetearDatos();
            }
        });
        Logger.info("🔧 Debug mode: Ctrl+Shift+D para inspeccionar");
    }
};
```
