```javascript
// components.js - Sistema de componentes

class Componente {
    constructor(props = {}) {
        this.props = props;
        this._elemento = null;
    }

    render() {
        throw new Error("Método render() debe implementarse");
    }

    montar(contenedor) {
        this._elemento = this._crearElemento(this.render());
        if (contenedor) {
            contenedor.appendChild(this._elemento);
        }
        this._afterMount();
        return this._elemento;
    }

    actualizar(nuevasProps) {
        this.props = { ...this.props, ...nuevasProps };
        let nuevoHtml = this.render();
        let nuevoElemento = this._crearElemento(nuevoHtml);
        if (this._elemento && this._elemento.parentNode) {
            this._elemento.parentNode.replaceChild(nuevoElemento, this._elemento);
        }
        this._elemento = nuevoElemento;
        this._afterMount();
    }

    _crearElemento(html) {
        let template = document.createElement("div");
        template.innerHTML = html.trim();
        return template.firstChild;
    }

    _afterMount() {
        // Hook para eventos después de montar
    }

    get elemento() { return this._elemento; }
}
```

```javascript
class TaskCard extends Componente {
    render() {
        let t = this.props.tarea;
        let colores = {
            pendiente: "#f39c12",
            progreso: "#3498db",
            hecho: "#2ecc71"
        };
        let iconos = {
            pendiente: "⬜",
            progreso: "🔄",
            hecho: "✅"
        };

        return `
            <div class="card task-card" data-id="${t.id}"
                 style="border-left:4px solid ${colores[t.estado]};
                        margin-bottom:10px;padding:12px;cursor:grab">
                <div style="display:flex;justify-content:space-between;
                            align-items:start">
                    <div>
                        <strong>${iconos[t.estado]} ${t.titulo}</strong>
                        ${t.prioridad === "alta"
                            ? '<span style="color:red;font-size:12px">🔴 Alta</span>'
                            : t.prioridad === "baja"
                                ? '<span style="color:green;font-size:12px">🟢 Baja</span>'
                                : '<span style="color:orange;font-size:12px">🟡 Media</span>'}
                    </div>
                    <button class="btn-danger" style="padding:2px 8px;font-size:12px"
                            onclick="eliminarTarea('${t.id}')">✗</button>
                </div>
                ${t.descripcion
                    ? `<p style="font-size:13px;color:var(--text-light);margin:5px 0">
                        ${t.descripcion}</p>`
                    : ""}
                <div style="display:flex;gap:5px;margin-top:8px">
                    <button class="btn-sm" style="background:#f39c12;color:white;
                            ${t.estado === 'pendiente' ? 'opacity:0.5' : ''}"
                            onclick="cambiarEstado('${t.id}','pendiente')">⬜</button>
                    <button class="btn-sm" style="background:#3498db;color:white;
                            ${t.estado === 'progreso' ? 'opacity:0.5' : ''}"
                            onclick="cambiarEstado('${t.id}','progreso')">🔄</button>
                    <button class="btn-sm" style="background:#2ecc71;color:white;
                            ${t.estado === 'hecho' ? 'opacity:0.5' : ''}"
                            onclick="cambiarEstado('${t.id}','hecho')">✅</button>
                </div>
            </div>
        `;
    }
}

// Funciones globales para los onclick
function cambiarEstado(tareaId, estado) {
    store.cambiarEstadoTarea(tareaId, estado);
    let proyectoId = window.location.hash.split("/").pop();
    renderizarProyecto(proyectoId);
}

function eliminarTarea(tareaId) {
    if (confirm("¿Eliminar esta tarea?")) {
        store.eliminarTarea(tareaId);
        let proyectoId = window.location.hash.split("/").pop();
        renderizarProyecto(proyectoId);
    }
}
```

```javascript
function renderizarTablero(tareas) {
    let estados = [
        { id: "pendiente", titulo: "📋 Pendientes", icono: "⬜" },
        { id: "progreso", titulo: "🔄 En Progreso", icono: "🔄" },
        { id: "hecho", titulo: "✅ Completadas", icono: "✅" }
    ];

    let colores = {
        pendiente: { bg: "#fff8e1", border: "#f39c12" },
        progreso: { bg: "#e3f2fd", border: "#3498db" },
        hecho: { bg: "#e8f5e9", border: "#2ecc71" }
    };

    let html = `<div style="display:grid;grid-template-columns:repeat(3,1fr);
                            gap:15px;align-items:start">`;

    estados.forEach(function(e) {
        let tareasColumna = tareas.filter(t => t.estado === e.id);
        let color = colores[e.id];

        html += `
            <div style="background:${color.bg};border-radius:8px;padding:10px;
                        min-height:200px">
                <h3 style="border-bottom:3px solid ${color.border};padding-bottom:8px;
                           margin-bottom:10px">
                    ${e.titulo}
                    <span style="font-size:13px;color:var(--text-light)">
                        (${tareasColumna.length})
                    </span>
                </h3>
                <div class="columna-tareas" data-estado="${e.id}"
                     ondrop="onDrop(event)" ondragover="onDragOver(event)">
        `;

        if (tareasColumna.length === 0) {
            html += '<p style="color:var(--text-light);font-size:13px;' +
                    'text-align:center;padding:20px">Sin tareas</p>';
        } else {
            tareasColumna.forEach(function(t) {
                let card = new TaskCard({ tarea: t });
                html += card.render();
            });
        }

        html += '</div></div>';
    });

    return html + "</div>";
}
```

```javascript
// Funciones para drag and drop entre columnas
function onDragOver(e) {
    e.preventDefault();
    e.currentTarget.style.background = "#f0f0f0";
}

function iniciarDrag(e, tareaId) {
    e.dataTransfer.setData("text/plain", tareaId);
    e.currentTarget.style.opacity = "0.5";
}

document.addEventListener("dragend", function(e) {
    document.querySelectorAll(".columna-tareas").forEach(function(col) {
        col.style.background = "";
    });
});

document.addEventListener("drop", function(e) {
    e.preventDefault();
    let tareaId = e.dataTransfer.getData("text/plain");
    let columna = e.target.closest(".columna-tareas");

    if (columna && tareaId) {
        let nuevoEstado = columna.dataset.estado;
        store.cambiarEstadoTarea(tareaId, nuevoEstado);
        let proyectoId = window.location.hash.split("/").pop();
        renderizarProyecto(proyectoId);
    }
});

// Agregar drag a las task cards (se llama después de renderizar)
function habilitarDrag() {
    document.querySelectorAll(".task-card").forEach(function(card) {
        card.draggable = true;
        card.addEventListener("dragstart", function(e) {
            iniciarDrag(e, this.dataset.id);
        });
    });
}
```

```javascript
class Modal extends Componente {
    mostrar(titulo, contenidoHtml) {
        let overlay = document.querySelector("#modal-overlay");
        let container = document.querySelector("#modal-container");

        overlay.classList.remove("hidden");
        container.classList.remove("hidden");
        container.innerHTML = `
            <div style="display:flex;justify-content:space-between;align-items:center;
                        margin-bottom:15px">
                <h3>${titulo}</h3>
                <button onclick="modal.cerrar()"
                        style="background:none;border:none;font-size:24px">&times;</button>
            </div>
            ${contenidoHtml}
        `;

        overlay.addEventListener("click", () => this.cerrar());
    }

    cerrar() {
        document.querySelector("#modal-overlay").classList.add("hidden");
        document.querySelector("#modal-container").classList.add("hidden");
    }

    get estaVisible() {
        return !document.querySelector("#modal-overlay").classList.contains("hidden");
    }
}

const modal = new Modal();

// Función para mostrar formulario de tarea en modal
function mostrarFormTarea(proyectoId) {
    modal.mostrar("Nueva Tarea", `
        <form id="tarea-form">
            <input type="hidden" id="tarea-proyecto" value="${proyectoId}">
            <div class="form-group">
                <label>Título</label>
                <input type="text" id="tarea-titulo" required>
            </div>
            <div class="form-group">
                <label>Descripción</label>
                <textarea id="tarea-descripcion" rows="2"></textarea>
            </div>
            <div class="form-group">
                <label>Prioridad</label>
                <select id="tarea-prioridad">
                    <option value="baja">🟢 Baja</option>
                    <option value="media" selected>🟡 Media</option>
                    <option value="alta">🔴 Alta</option>
                </select>
            </div>
            <button type="submit" class="btn-success">Crear Tarea</button>
        </form>
    `);
}

document.addEventListener("submit", function(e) {
    if (e.target.id === "tarea-form") {
        e.preventDefault();
        let proyectoId = document.querySelector("#tarea-proyecto").value;
        let titulo = document.querySelector("#tarea-titulo").value;
        let desc = document.querySelector("#tarea-descripcion").value;
        let prioridad = document.querySelector("#tarea-prioridad").value;

        store.crearTarea(proyectoId, titulo, desc, prioridad);
        modal.cerrar();
        renderizarProyecto(proyectoId);
    }
});
```
