BIENVENIDA DEL PROFESOR BÚHO

Querido estudiante…

Hay dos formas de cambiar la apariencia con JS:

1. style: estilos en línea (directamente en el elemento).
2. classList: agregar/quitar clases CSS definidas en tu hoja de estilos.

La regla de oro:
- Usa clases CSS para cambios predefinidos.
- Usa style para valores dinámicos (calculados en el momento).

OBJETIVOS DEL CAPÍTULO

✓ Modificar estilos inline con style.
✓ Agregar y quitar clases con classList.
✓ Leer estilos computados con getComputedStyle.
✓ Animar elementos con JS.

===============================================================================
1. STYLE: ESTILOS EN LÍNEA
===============================================================================

```javascript
let caja = document.querySelector(".caja");

// Propiedades CSS en camelCase
caja.style.backgroundColor = "blue";
caja.style.color = "white";
caja.style.fontSize = "20px";
caja.style.padding = "15px";
caja.style.borderRadius = "8px";
caja.style.marginTop = "10px";

// Las propiedades con guión van en camelCase
// background-color → backgroundColor
// font-size → fontSize
// border-radius → borderRadius

// Leer estilo inline
console.log(caja.style.backgroundColor);  // "blue"
```

===============================================================================
2. PROPIEDADES ESPECIALES
===============================================================================

```javascript
let elemento = document.querySelector("div");

// cssText: múltiples estilos a la vez
elemento.style.cssText = "color: red; background: yellow; font-weight: bold;";

// setProperty: útil para nombres con guiones
elemento.style.setProperty("--mi-variable", "42px");

// removeProperty: eliminar un estilo
elemento.style.removeProperty("background");

// float es palabra reservada
elemento.style.cssFloat = "left";  // en vez de "float"
```

===============================================================================
3. getComputedStyle
===============================================================================

Lee los estilos REALES del elemento (incluyendo CSS externo).

```javascript
let parrafo = document.querySelector("p");

// Estilo inline (solo lo que puso JS o el atributo style)
console.log(parrafo.style.color);

// Estilo computado (todo: inline + CSS + herencia)
let estilos = getComputedStyle(parrafo);
console.log(estilos.color);        // rgb(0, 0, 0)
console.log(estilos.fontSize);     // "16px"
console.log(estilos.fontFamily);   // "Times New Roman"
console.log(estilos.display);      // "block"
```

===============================================================================
4. CLASES VS STYLE
===============================================================================

```javascript
// ✅ BUENA PRÁCTICA: clases CSS
// CSS: .error { color: red; border: 1px solid red; }
// CSS: .success { color: green; border: 1px solid green; }

function mostrarEstado(elemento, tipo) {
    elemento.classList.remove("error", "success", "warning");
    elemento.classList.add(tipo);
}

// ❌ MALA PRÁCTICA: estilos inline repetidos
function mostrarEstadoMal(elemento, tipo) {
    if (tipo === "error") {
        elemento.style.color = "red";
        elemento.style.border = "1px solid red";
    } else if (tipo === "success") {
        elemento.style.color = "green";
        elemento.style.border = "1px solid green";
    }
}
```

function animarCaja(elemento) {
    let posicion = 0;

    function mover() {
        posicion += 2;
        elemento.style.transform = "translateX(" + posicion + "px)";
        elemento.style.backgroundColor =
            "rgb(" + posicion + ", 100, 200)";

        if (posicion < 200) {
            requestAnimationFrame(mover);
        }
    }

    requestAnimationFrame(mover);
}

function parpadear(elemento) {

MINI PROYECTO: TEMA OSCURO/CLARO
===============================================================================

```javascript
function toggleTema() {
    let body = document.body;

    if (body.classList.contains("tema-oscuro")) {
        body.classList.remove("tema-oscuro");
        body.classList.add("tema-claro");
        localStorage.setItem("tema", "claro");
        console.log("☀️ Tema claro");
    } else {
        body.classList.remove("tema-claro");
        body.classList.add("tema-oscuro");
        localStorage.setItem("tema", "oscuro");
        console.log("🌙 Tema oscuro");
    }
}

function cargarTema() {
    let tema = localStorage.getItem("tema");
    if (tema === "oscuro") {
        document.body.classList.add("tema-oscuro");
    } else {
        document.body.classList.add("tema-claro");
    }
}

// CSS necesario:
// .tema-oscuro { background: #1a1a2e; color: #eee; }
// .tema-claro { background: #fff; color: #333; }

// Resaltar elemento al pasar el mouse
function resaltar(elemento) {
    elemento.addEventListener("mouseenter", function() {
        this.style.boxShadow = "0 0 10px rgba(0,0,0,0.3)";
        this.style.transform = "scale(1.05)";
    });
    elemento.addEventListener("mouseleave", function() {
        this.style.boxShadow = "none";
        this.style.transform = "scale(1)";
    });
}
```
