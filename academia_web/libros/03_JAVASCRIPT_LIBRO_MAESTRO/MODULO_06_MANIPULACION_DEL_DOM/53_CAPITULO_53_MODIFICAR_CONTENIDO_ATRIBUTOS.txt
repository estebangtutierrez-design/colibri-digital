1. TEXT CONTENT VS INNER HTML
===============================================================================

```javascript
// textContent: solo texto (seguro)
// innerHTML: interpreta HTML (peligroso con datos externos)

// <div id="info">Hola <strong>Mundo</strong></div>

let div = document.getElementById("info");

console.log(div.textContent);  // "Hola Mundo"
console.log(div.innerHTML);    // "Hola <strong>Mundo</strong>"

// Modificar texto
div.textContent = "Nuevo texto";
// Resultado: <div id="info">Nuevo texto</div>

// Modificar HTML
div.innerHTML = "<em>Texto</em> con <strong>HTML</strong>";
// Resultado: <div id="info"><em>Texto</em> con <strong>HTML</strong></div>
```

⚠️ innerHTML con datos de usuario es peligroso (XSS).

2. ATRIBUTOS: GET, SET, REMOVE
===============================================================================

```html
<img id="logo" src="logo.png" alt="Logo" title="Mi Logo">
<a id="link" href="https://ejemplo.com">Ir</a>
```

```javascript
let img = document.getElementById("logo");

// Obtener atributos
console.log(img.getAttribute("src"));   // "logo.png"
console.log(img.getAttribute("alt"));   // "Logo"

// Modificar atributos
img.setAttribute("src", "nuevo-logo.png");
img.setAttribute("alt", "Nuevo Logo");

// Eliminar atributos
img.removeAttribute("title");

// Acceso directo a propiedades comunes
let link = document.getElementById("link");
console.log(link.href);     // URL completa
console.log(link.id);       // "link"
link.href = "https://nuevo.com";
link.target = "_blank";
```

===============================================================================
3. ATRIBUTOS DATA-*
===============================================================================

```html
<div id="usuario"
     data-id="123"
     data-rol="admin"
     data-nombre="Ana García">
  Ana
</div>
```

```javascript
let user = document.getElementById("usuario");

// Acceder con dataset (camelCase)
console.log(user.dataset.id);      // "123"
console.log(user.dataset.rol);     // "admin"
console.log(user.dataset.nombre);  // "Ana García"

// Modificar data-*
user.dataset.rol = "usuario";
user.dataset.ultimoAcceso = "2026-07-01";  // Crea data-ultimo-acceso

// El HTML se actualiza automáticamente
```

===============================================================================
4. CLASES: classList
===============================================================================

```html
<p id="texto" class="destacado">Texto importante</p>
```

```javascript
let p = document.getElementById("texto");

// classList es más seguro que className
p.classList.add("nueva-clase");
p.classList.remove("destacado");
p.classList.toggle("visible");  // Agrega si no tiene, quita si tiene
p.classList.replace("vieja", "nueva");

console.log(p.classList.contains("visible"));  // true

// className: reemplaza TODAS las clases
p.className = "solo-esta";  // ⚠️ Elimina las demás
```

MINI PROYECTO: EDITOR DE PERFIL
===============================================================================

```html
<div id="perfil" data-usuario-id="42">
  <h2 id="nombre">Invitado</h2>
  <img id="avatar" src="default.png" alt="Avatar">
  <p id="descripcion">Sin descripción</p>
  <a id="web" href="#">Sitio web</a>
</div>
```

```javascript
function actualizarPerfil(datos) {
    document.getElementById("nombre").textContent = datos.nombre;
    document.getElementById("avatar").src = datos.avatar;
    document.getElementById("avatar").alt = "Avatar de " + datos.nombre;
    document.getElementById("descripcion").textContent = datos.descripcion;
    document.getElementById("web").href = datos.web;
    document.getElementById("web").textContent = datos.web;

    let perfil = document.getElementById("perfil");
    perfil.dataset.rol = datos.rol || "usuario";

    console.log("✅ Perfil actualizado");
}

actualizarPerfil({
    nombre: "Ana García",
    avatar: "https://ejemplo.com/ana.jpg",
    descripcion: "Desarrolladora JavaScript",
    web: "https://ana.dev",
    rol: "admin"
});
```
