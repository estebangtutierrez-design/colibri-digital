===============================================================================
1. CREAR ELEMENTOS
===============================================================================

```javascript
// Crear un nuevo elemento (no está en el DOM aún)
let nuevoParrafo = document.createElement("p");
nuevoParrafo.textContent = "Este es un párrafo creado con JS";
nuevoParrafo.className = "destacado";
nuevoParrafo.id = "mi-parrafo";

// Crear un nodo de texto
let texto = document.createTextNode("Texto independiente");
```

===============================================================================
2. AGREGAR AL DOM
===============================================================================

```javascript
// appendChild: agrega al final del padre
let lista = document.querySelector("ul");
let nuevoItem = document.createElement("li");
nuevoItem.textContent = "Item nuevo";
lista.appendChild(nuevoItem);

// insertBefore: agrega antes de un elemento de referencia
let primerItem = lista.firstElementChild;
let itemAlInicio = document.createElement("li");
itemAlInicio.textContent = "Primer item";
lista.insertBefore(itemAlInicio, primerItem);

// prepend: agrega al inicio
lista.prepend(document.createElement("li"));

// after / before: agrega adyacente
let referencia = document.querySelector("#referencia");
referencia.after(document.createElement("div"));   // Después
referencia.before(document.createElement("div"));  // Antes
```

===============================================================================
3. INSERTADJACENTHTML
===============================================================================

```javascript
let contenedor = document.querySelector("#contenedor");

// Cuatro posiciones
contenedor.insertAdjacentHTML("beforebegin", "<p>Antes del contenedor</p>");
contenedor.insertAdjacentHTML("afterbegin", "<p>Primero dentro</p>");
contenedor.insertAdjacentHTML("beforeend", "<p>Último dentro</p>");
contenedor.insertAdjacentHTML("afterend", "<p>Después del contenedor</p>");

// Más eficiente que innerHTML +=
```

===============================================================================
4. CLONAR ELEMENTOS
===============================================================================

```javascript
let original = document.querySelector("#plantilla");

// cloneNode(false): clon superficial (sin hijos)
let copiaSuperficial = original.cloneNode(false);

// cloneNode(true): clon profundo (con todos los hijos)
let copiaProfunda = original.cloneNode(true);

// Modificar el clon antes de agregarlo
copiaProfunda.id = "copia-" + Date.now();
copiaProfunda.classList.add("clonado");
document.body.appendChild(copiaProfunda);
```

===============================================================================
5. ELIMINAR ELEMENTOS
===============================================================================

```javascript
let elemento = document.querySelector("#eliminarme");

// remove(): elimina directamente (moderno)
elemento.remove();

// removeChild: elimina hijo del padre
let padre = document.querySelector("#lista");
let hijo = padre.querySelector("li:last-child");
padre.removeChild(hijo);

// Vaciar un contenedor
function vaciar(contenedor) {
    while (contenedor.firstChild) {
        contenedor.removeChild(contenedor.firstChild);
    }
}

// innerHTML = "" también vacía, pero es menos eficiente
```

MINI PROYECTO: GENERADOR DE TARJETAS
===============================================================================

```javascript
function crearTarjeta(titulo, descripcion, color) {
    let tarjeta = document.createElement("div");
    tarjeta.className = "tarjeta";
    tarjeta.style.borderLeft = "4px solid " + color;

    let h3 = document.createElement("h3");
    h3.textContent = titulo;
    tarjeta.appendChild(h3);

    let p = document.createElement("p");
    p.textContent = descripcion;
    tarjeta.appendChild(p);

    let btn = document.createElement("button");
    btn.textContent = "Eliminar";
    btn.addEventListener("click", function() {
        tarjeta.remove();
        console.log("🗑️ Tarjeta eliminada: " + titulo);
    });
    tarjeta.appendChild(btn);

    return tarjeta;
}

function cargarTarjetas(datos) {
    let contenedor = document.querySelector("#tarjetas");
    // Usar fragmento para mejor rendimiento
    let fragmento = document.createDocumentFragment();

    datos.forEach(function(d) {
        let tarjeta = crearTarjeta(d.titulo, d.desc, d.color);
        fragmento.appendChild(tarjeta);
    });

    contenedor.appendChild(fragmento);
    console.log("✅ " + datos.length + " tarjetas cargadas");
}

// Uso:
let datos = [
    { titulo: "JavaScript", desc: "Lenguaje de programación", color: "yellow" },
    { titulo: "HTML", desc: "Estructura web", color: "orange" },
    { titulo: "CSS", desc: "Estilos visuales", color: "blue" }
];
// cargarTarjetas(datos);  // Cuando exista #tarjetas en el DOM
```
