function validarFormulario(formulario) {
    let errores = [];
    let datos = {};

    // Recorrer todos los inputs
    let inputs = formulario.querySelectorAll("input, select, textarea");
    inputs.forEach(function(input) {
        if (input.type === "submit" || input.type === "button") return;

        let valor = input.value.trim();
        datos[input.name] = valor;

        if (input.required && !valor) {
            errores.push("âťŚ " + input.name + " es obligatorio");
        }

        if (input.type === "email" && valor && !validarEmail(valor)) {
            errores.push("âťŚ Email invĂˇlido");
        }

        if (input.type === "number" && valor) {
            let num = Number(valor);
            let min = Number(input.min) || -Infinity;
            let max = Number(input.max) || Infinity;
            if (num < min || num > max) {
                errores.push("âťŚ " + input.name + " debe estar entre " + min + " y " + max);
            }
        }
    });

    return { valido: errores.length === 0, errores: errores, datos: datos };
}

document.querySelector("#miFormulario").addEventListener("submit", function(e) {
    e.preventDefault();
    let resultado = validarFormulario(this);

    if (resultado.valido) {
        console.log("âś… Formulario vĂˇlido");
        console.log("đź“¦ Datos:", resultado.datos);

    } else {
