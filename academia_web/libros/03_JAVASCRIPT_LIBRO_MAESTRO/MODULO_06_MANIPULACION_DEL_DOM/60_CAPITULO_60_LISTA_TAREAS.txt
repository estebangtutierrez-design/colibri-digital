2. CLASE TAREA
===============================================================================

```javascript
class Tarea {
    static #contador = 0;

    constructor(texto, prioridad = "media") {
        this.id = ++Tarea.#contador;
        this.texto = texto;
        this.prioridad = prioridad;
        this.completada = false;
        this.creada = new Date().toISOString();
    }
}
```

3. APLICACIÓN COMPLETA
===============================================================================

```javascript
class AppTareas {
    #tareas = [];
    #filtroActual = "todas";

    constructor() {
        this.$form = document.querySelector("#form-tarea");
        this.$input = document.querySelector("#input-tarea");
        this.$select = document.querySelector("#select-prioridad");
        this.$lista = document.querySelector("#lista-tareas");
        this.$total = document.querySelector("#total");
        this.$pendientes = document.querySelector("#pendientes");
        this.$completadas = document.querySelector("#completadas");

        this.#cargarDeLocalStorage();
        this.#configurarEventos();
        this.#renderizar();
    }

    #configurarEventos() {
        // Agregar tarea
        this.$form.addEventListener("submit", (e) => {
            e.preventDefault();
            this.#agregarTarea(
                this.$input.value.trim(),
                this.$select.value
            );
        });

        // Delegación en la lista
        this.$lista.addEventListener("click", (e) => {
            let item = e.target.closest("li");
            if (!item) return;

            let id = Number(item.dataset.id);

            if (e.target.classList.contains("toggle")) {
                this.#toggleTarea(id);
            } else if (e.target.classList.contains("eliminar")) {
                this.#eliminarTarea(id);
            }
        });

        // Filtros con delegación
        document.querySelector("#filtros").addEventListener("click", (e) => {
            let btn = e.target.closest(".filtro");
            if (!btn) return;

            document.querySelectorAll(".filtro").forEach(b => b.classList.remove("activo"));
            btn.classList.add("activo");
            this.#filtroActual = btn.dataset.filtro;
            this.#renderizar();
        });

        // Atajo de teclado: Escape para limpiar input
        this.$input.addEventListener("keydown", (e) => {
            if (e.key === "Escape") {
                this.$input.value = "";
                this.$input.blur();
            }
        });
    }

    #agregarTarea(texto, prioridad) {
        if (!texto) return;

        let tarea = new Tarea(texto, prioridad);
        this.#tareas.push(tarea);
        this.#guardarEnLocalStorage();
        this.#renderizar(tarea.id);

        this.$input.value = "";
        this.$input.focus();

        this.#mostrarNotificacion("✅ Tarea agregada", "exito");
    }

    #toggleTarea(id) {
        let tarea = this.#tareas.find(t => t.id === id);
        if (tarea) {
            tarea.completada = !tarea.completada;
            this.#guardarEnLocalStorage();
            this.#renderizar();

            if (tarea.completada) {
                this.#mostrarNotificacion("🎯 Tarea completada", "exito");
            }
        }
    }

    #eliminarTarea(id) {
        this.#tareas = this.#tareas.filter(t => t.id !== id);
        this.#guardarEnLocalStorage();
        this.#renderizar();
        this.#mostrarNotificacion("🗑️ Tarea eliminada", "info");
    }

    #getTareasFiltradas() {
        switch (this.#filtroActual) {
            case "pendientes":
                return this.#tareas.filter(t => !t.completada);
            case "completadas":
                return this.#tareas.filter(t => t.completada);
            default:
                return this.#tareas;
        }
    }

    #renderizar(idDestacado) {
        let tareas = this.#getTareasFiltradas();
        this.$lista.innerHTML = "";

        if (tareas.length === 0) {
            let empty = document.createElement("li");
            empty.className = "vacio";
            empty.textContent = "📭 No hay tareas " + this.#filtroActual;
            this.$lista.appendChild(empty);
        } else {
            tareas.forEach(t => {
                let li = document.createElement("li");
                li.dataset.id = t.id;
                li.className = "tarea prioridad-" + t.prioridad +
                    (t.completada ? " completada" : "");

                if (t.id === idDestacado) {
                    li.style.animation = "fadeIn 0.3s ease";
                }

                li.innerHTML =
                    '<span class="toggle">' + (t.completada ? "✅" : "⬜") + '</span>' +
                    '<span class="texto">' + this.#escaparHTML(t.texto) + '</span>' +
                    '<span class="prioridad-badge">' + t.prioridad + '</span>' +
                    '<button class="eliminar" title="Eliminar">✗</button>';

                this.$lista.appendChild(li);
            });
        }

        // Actualizar estadísticas
        this.$total.textContent = this.#tareas.length;
        this.$pendientes.textContent = this.#tareas.filter(t => !t.completada).length;
        this.#completadas.textContent = this.#tareas.filter(t => t.completada).length;
    }

    #escaparHTML(texto) {
        let div = document.createElement("div");
        div.textContent = texto;
        return div.innerHTML;
    }

    #mostrarNotificacion(mensaje, tipo) {
        let toast = document.createElement("div");
        toast.className = "toast toast-" + (tipo || "info");
        toast.textContent = mensaje;
        toast.style.cssText = "position:fixed;bottom:20px;right:20px;" +
            "background:#333;color:#fff;padding:12px 20px;border-radius:5px;" +
            "opacity:0;transition:opacity 0.3s;z-index:9999;";
        document.body.appendChild(toast);

        setTimeout(() => toast.style.opacity = "1", 10);
        setTimeout(() => {
            toast.style.opacity = "0";
            setTimeout(() => toast.remove(), 300);
        }, 2000);
    }

    #guardarEnLocalStorage() {
        localStorage.setItem("tareas", JSON.stringify(this.#tareas));
    }

    #cargarDeLocalStorage() {
        try {
            let datos = localStorage.getItem("tareas");
            if (datos) {
                let tareasGuardadas = JSON.parse(datos);
                this.#tareas = tareasGuardadas.map(t => {
                    let tarea = new Tarea(t.texto, t.prioridad);
                    tarea.id = t.id;
                    tarea.completada = t.completada;
                    tarea.creada = t.creada;
                    if (t.id > Tarea.#contador) {
                        Tarea.#contador = t.id;
                    }
                    return tarea;
                });
            }
        } catch (e) {
            console.log("Error cargando datos:", e);
            this.#tareas = [];
        }
    }
}
```
