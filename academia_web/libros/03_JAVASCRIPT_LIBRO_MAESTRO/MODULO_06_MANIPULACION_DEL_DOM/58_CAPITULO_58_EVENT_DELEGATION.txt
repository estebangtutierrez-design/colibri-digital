"Event delegation: en lugar de asignar un evento a cada elemento,
asignas UN SOLO evento a un padre y dejas que los eventos
'suban' (burbujeen) desde los hijos."
                                  — Profesor Búho AI

===============================================================================
BIENVENIDA DEL PROFESOR BÚHO
===============================================================================

Querido estudiante…

Imagina una lista de 1000 elementos.
Si agregas un evento a cada uno, gastas mucha memoria.

Con EVENT DELEGATION pones un solo evento en el padre.
Cuando haces clic en un hijo, el evento "burbujea" hasta el padre.

Ventajas:
- Menos memoria.
- Funciona para elementos creados después.

===============================================================================
1. BURBUJEO DE EVENTOS
===============================================================================

```html
<div id="abuelo">
  <div id="padre">
    <button id="hijo">Clic aquí</button>
  </div>
</div>
```

```javascript
document.getElementById("abuelo").addEventListener("click", function() {
    console.log("Abuelo");
});
document.getElementById("padre").addEventListener("click", function() {
    console.log("Padre");
});
document.getElementById("hijo").addEventListener("click", function() {
    console.log("Hijo");
});

// Al hacer clic en el botón:
// Hijo → Padre → Abuelo (el evento burbujea hacia arriba)

// Detener burbujeo
hijo.addEventListener("click", function(e) {
    e.stopPropagation();  // No sigue subiendo
});
```

===============================================================================
2. DELEGATION BÁSICO
===============================================================================

```html
<ul id="lista">
  <li>Item 1</li>
  <li>Item 2</li>
  <li>Item 3</li>
</ul>
```

```javascript
// ❌ Sin delegation: evento en cada li
let items = document.querySelectorAll("#lista li");
items.forEach(function(li) {
    li.addEventListener("click", function() {
        console.log("Clic en: " + this.textContent);
    });
});

// ✅ Con delegation: UN SOLO evento en el padre
document.getElementById("lista").addEventListener("click", function(e) {
    // e.target es el elemento que recibió el clic
    if (e.target.tagName === "LI") {
        console.log("🖱️ Clic en: " + e.target.textContent);
        e.target.style.backgroundColor = "lightyellow";
    }
});
```

===============================================================================
3. DELEGATION CON ELEMENTOS DINÁMICOS
===============================================================================

```javascript
// Solo funciona con delegation:
// los elementos creados después NO tienen evento si usas forEach

let lista = document.querySelector("#lista-tareas");

// Delegation: funciona para elementos existentes y futuros
lista.addEventListener("click", function(e) {
    let item = e.target.closest("li");  // Encuentra el li más cercano
    if (!item) return;

    if (e.target.classList.contains("eliminar")) {
        item.remove();
        console.log("🗑️ Tarea eliminada");
    } else if (e.target.classList.contains("completar")) {
        item.classList.toggle("completada");
        console.log("✅ Tarea " + (item.classList.contains("completada") ? "completada" : "pendiente"));
    }
});

// Agregar nueva tarea (funciona porque usamos delegation)
function agregarTarea(texto) {
    let li = document.createElement("li");
    li.innerHTML = '<span class="texto">' + texto + '</span>' +
                   '<button class="completar">✓</button>' +
                   '<button class="eliminar">✗</button>';
    lista.appendChild(li);
}

agregarTarea("Nueva tarea dinámica");  // ✅ El clic funciona
```

4. PATRÓN COMÚN DE DELEGATION
===============================================================================

```javascript
// Patrón: delegar en el contenedor más cercano
// Usar closest() para encontrar el elemento deseado

document.querySelector("#aplicacion").addEventListener("click", function(e) {
    let boton = e.target.closest("button");
    if (!boton) return;

    let accion = boton.dataset.accion;
    let item = boton.closest("[data-id]");
    let id = item ? item.dataset.id : null;

    switch (accion) {
        case "editar":
            console.log("✏️ Editar elemento " + id);
            break;
        case "eliminar":
            console.log("🗑️ Eliminar elemento " + id);
            item?.remove();
            break;
        case "duplicar":
            let clon = item.cloneNode(true);
            item.after(clon);
            console.log("📋 Duplicado " + id);
            break;
        default:
            console.log("❓ Acción desconocida: " + accion);
    }
});
```

===============================================================================
5. EVENTOS QUE NO BURBUJEAN
===============================================================================

Algunos eventos NO burbujean y no se pueden delegar:

```javascript
// ❌ NO burbujean:
// focus, blur, scroll, load, unload, resize

// ✅ Solución para focus/blur:
document.querySelector("#formulario").addEventListener("focusin", function(e) {
    console.log("🔵 Enfocado: " + e.target.name);
});

document.querySelector("#formulario").addEventListener("focusout", function(e) {
    console.log("⚪ Perdió foco: " + e.target.name);
});
```

MINI PROYECTO: TABLA INTERACTIVA CON DELEGATION
===============================================================================

```javascript
function crearTabla() {
    let tabla = document.createElement("table");
    tabla.id = "tabla-datos";
    tabla.style.cssText = "border-collapse: collapse; width: 100%;";

    // Encabezados
    let thead = document.createElement("thead");
    thead.innerHTML = '<tr><th>Nombre</th><th>Email</th><th>Rol</th><th>Acciones</th></tr>';
    tabla.appendChild(thead);

    let tbody = document.createElement("tbody");
    tbody.id = "cuerpo-tabla";
    tabla.appendChild(tbody);

    document.body.appendChild(tabla);

    // Delegation: clic en fila resalta, clic en botones acciona
    tbody.addEventListener("click", function(e) {
        let fila = e.target.closest("tr");
        if (!fila) return;

        if (e.target.classList.contains("btn-editar")) {
            let nombre = fila.cells[0].textContent;
            console.log("✏️ Editando: " + nombre);
        } else if (e.target.classList.contains("btn-eliminar")) {
            fila.remove();
            console.log("🗑️ Fila eliminada");
        } else {
            // Clic en la fila: resaltar
            tbody.querySelectorAll("tr").forEach(function(r) {
                r.style.backgroundColor = "";
            });
            fila.style.backgroundColor = "#f0f0ff";
        }
    });
}

function agregarFila(nombre, email, rol) {
    let tbody = document.getElementById("cuerpo-tabla");
    let tr = document.createElement("tr");
    tr.innerHTML = '<td>' + nombre + '</td>' +
                   '<td>' + email + '</td>' +
                   '<td>' + rol + '</td>' +
                   '<td><button class="btn-editar">✏️</button> ' +
                   '<button class="btn-eliminar">🗑️</button></td>';
    tbody.appendChild(tr);
}

crearTabla();
agregarFila("Ana García", "ana@email.com", "Admin");
agregarFila("Luis Pérez", "luis@email.com", "Usuario");
```

===============================================================================
RESUMEN
===============================================================================

✔ Burbujeo: el evento sube desde el hijo hasta el padre.
✔ stopPropagation(): detiene el burbujeo.
✔ Event delegation: un evento en el padre para todos los hijos.
✔ e.target: el elemento que originó el evento.
✔ closest(): busca el ancestro más cercano que coincida.
✔ Funciona con elementos dinámicos.
✔ focus/blur no burbujean → usar focusin/focusout.
