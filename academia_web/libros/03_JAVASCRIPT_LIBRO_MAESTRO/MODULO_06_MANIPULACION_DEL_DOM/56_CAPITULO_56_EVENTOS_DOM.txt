===============================================================================
BIENVENIDA DEL PROFESOR BÚHO
===============================================================================

Querido estudiante…

Una página web sin eventos es como un libro: solo se lee.

Los eventos hacen la página INTERACTIVA:
- El usuario hace clic → JS responde.
- El usuario escribe → JS valida.
- El usuario mueve el mouse → JS anima.

===============================================================================
1. ADDEVENTLISTENER
===============================================================================

```javascript
let boton = document.querySelector("#miBoton");

// Sintaxis básica
boton.addEventListener("click", function() {
    console.log("✅ Botón clickeado");
});

// Con función nombrada (recomendado para remover después)
function handleClick() {
    console.log("¡Clic!");
}
boton.addEventListener("click", handleClick);
```

===============================================================================
2. EL OBJETO EVENT
===============================================================================

```javascript
document.querySelector("button").addEventListener("click", function(event) {
    console.log(event.type);        // "click"
    console.log(event.target);      // El elemento que disparó el evento
    console.log(event.currentTarget); // El elemento que escucha
    console.log(event.clientX);     // Posición X del mouse
    console.log(event.clientY);     // Posición Y del mouse
    console.log(event.timeStamp);   // Tiempo desde que cargó la página
});
```

===============================================================================
3. EVENTOS DE MOUSE
===============================================================================

```javascript
let elemento = document.querySelector("#caja");

elemento.addEventListener("click", function(e) {
    console.log("🖱️ Click");
});

elemento.addEventListener("dblclick", function(e) {
    console.log("🖱️🖱️ Doble click");
});

elemento.addEventListener("mouseenter", function(e) {
    console.log("➡️ Mouse entró");
    this.style.backgroundColor = "lightblue";
});

elemento.addEventListener("mouseleave", function(e) {
    console.log("⬅️ Mouse salió");
    this.style.backgroundColor = "";
});

elemento.addEventListener("mousemove", function(e) {
    console.log("🎯 Mouse en: " + e.clientX + ", " + e.clientY);
});

elemento.addEventListener("contextmenu", function(e) {
    e.preventDefault();  // Evita menú contextual
    console.log("🖱️ Click derecho");
});
```

===============================================================================
4. EVENTOS DE TECLADO
===============================================================================

```javascript
document.addEventListener("keydown", function(e) {
    console.log("⬇️ Tecla: " + e.key);
    console.log("  Código: " + e.code);
    console.log("  Shift: " + e.shiftKey);
    console.log("  Ctrl: " + e.ctrlKey);

    if (e.key === "Enter") {
        console.log("✅ Enter presionado");
    }
    if (e.key === "Escape") {
        console.log("❌ Escape: cancelar");
    }
});

document.addEventListener("keyup", function(e) {
    console.log("⬆️ Tecla liberada: " + e.key);
});

// Evento input en un campo de texto
let input = document.querySelector("#nombre");
input.addEventListener("input", function(e) {
    console.log("📝 Escribiendo: " + this.value);
});

input.addEventListener("focus", function() {
    this.style.borderColor = "blue";
    console.log("🔵 Enfocado");
});

input.addEventListener("blur", function() {
    this.style.borderColor = "";
    console.log("⚪ Perdió foco");
});
```

===============================================================================
5. REMOVER EVENTOS
===============================================================================

```javascript
function mostrarMensaje() {
    console.log("Este mensaje solo aparece una vez");
}

let btn = document.querySelector("#unaVez");
btn.addEventListener("click", mostrarMensaje);
btn.addEventListener("click", function() {
    btn.removeEventListener("click", mostrarMensaje);
    console.log("🛑 Evento removido (no volverá a ejecutarse)");
});

// Con { once: true } es más simple:
btn2.addEventListener("click", function() {
    console.log("Solo una vez con once:true");
}, { once: true });
```

MINI PROYECTO: SEGUIDOR DE RATÓN
===============================================================================

```javascript
let indicador = document.querySelector("#coordenadas") || (function() {
    let div = document.createElement("div");
    div.id = "coordenadas";
    div.style.cssText = "position:fixed;bottom:10px;right:10px;background:#333;color:#fff;padding:10px;border-radius:5px;font-family:monospace;";
    document.body.appendChild(div);
    return div;
})();

document.addEventListener("mousemove", function(e) {
    indicador.textContent = "X: " + e.clientX + " | Y: " + e.clientY;
});

document.addEventListener("keydown", function(e) {
    if (e.key === "h") {
        indicador.style.display =
            indicador.style.display === "none" ? "block" : "none";
    }
});

console.log("🖱️ Mueve el mouse para ver coordenadas");
console.log("⌨️ Presiona 'h' para ocultar/mostrar");
```

===============================================================================
RESUMEN
===============================================================================

✔ addEventListener("evento", handler): escuchar eventos.
✔ event: objeto con detalles del evento (target, type, clientX).
✔ Eventos de mouse: click, dblclick, mouseenter, mouseleave.
✔ Eventos de teclado: keydown, keyup (key, code, ctrlKey).
✔ focus/blur: eventos de enfoque.
✔ removeEventListener: dejar de escuchar.
✔ { once: true }: ejecutar una sola vez.
