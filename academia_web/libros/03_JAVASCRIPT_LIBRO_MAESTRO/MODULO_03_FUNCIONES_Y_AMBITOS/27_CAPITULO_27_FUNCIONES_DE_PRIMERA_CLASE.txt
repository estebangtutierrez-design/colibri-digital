Querido estudiante…

En JavaScript, las funciones no son especiales.
Son valores, como los números, strings y objetos.

Puedes:
- Asignarlas a variables.
- Pasarlas como argumentos.
- Devolverlas desde otras funciones.
- Guardarlas en arrays y objetos.

Esto se llama "funciones de primera clase".
Es lo que hace a JavaScript tan flexible y poderoso.

===============================================================================
1. ASIGNAR A VARIABLES
===============================================================================

```javascript
let sumar = function(a, b) {
    return a + b;
};

let restar = (a, b) => a - b;

console.log(sumar(5, 3));   // 8
console.log(restar(5, 3));  // 2
```

===============================================================================
2. PASAR COMO ARGUMENTOS
===============================================================================

```javascript
function ejecutarOperacion(a, b, operacion) {
    return operacion(a, b);
}

let suma = ejecutarOperacion(10, 5, function(a, b) { return a + b; });
let multi = ejecutarOperacion(10, 5, (a, b) => a * b);

console.log(suma);   // 15
console.log(multi);  // 50
```

Uso real: métodos de arrays

```javascript
let numeros = [1, 2, 3, 4, 5];

let dobles = numeros.map(n => n * 2);
let pares = numeros.filter(n => n % 2 === 0);
let sumaTotal = numeros.reduce((acc, n) => acc + n, 0);
```

===============================================================================
3. DEVOLVER FUNCIONES
===============================================================================

```javascript
function crearPotencia(exponente) {
    return function(base) {
        return base ** exponente;
    };
}

let alCuadrado = crearPotencia(2);
let alCubo = crearPotencia(3);

console.log(alCuadrado(4));  // 16
console.log(alCubo(3));      // 27
```

===============================================================================
4. GUARDAR EN ESTRUCTURAS
===============================================================================

```javascript
let operaciones = {
    sumar: (a, b) => a + b,
    restar: (a, b) => a - b,
    multiplicar: (a, b) => a * b,
    dividir: (a, b) => b !== 0 ? a / b : "Error"
};

console.log(operaciones.sumar(10, 5));     // 15
console.log(operaciones.dividir(10, 2));   // 5

let metodos = [operaciones.sumar, operaciones.restar];
console.log(metodos[0](8, 2));  // 10
```

MINI PROYECTO: PROCESADOR DE TEXTOS FLEXIBLE
===============================================================================

```javascript
function procesarTexto(texto, ...transformaciones) {
    let resultado = texto;

    for (let i = 0; i < transformaciones.length; i++) {
        resultado = transformaciones[i](resultado);
    }

    return resultado;
}

let aMayusculas = t => t.toUpperCase();
let quitarEspacios = t => t.trim();
let agregarSignos = t => "¡" + t + "!";
let invertir = t => t.split("").reverse().join("");

let texto = "  hola mundo  ";

console.log(procesarTexto(texto, aMayusculas, quitarEspacios));
// HOLA MUNDO

console.log(procesarTexto(texto, quitarEspacios, agregarSignos));
// ¡hola mundo!

console.log(procesarTexto(texto, quitarEspacios, invertir, aMayusculas));
// ODNUM ALOH
```
