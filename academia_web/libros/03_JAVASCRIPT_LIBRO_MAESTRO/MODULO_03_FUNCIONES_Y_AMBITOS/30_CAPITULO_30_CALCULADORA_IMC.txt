1. ¿QUÉ ES EL IMC?
===============================================================================

Fórmula: IMC = peso(kg) / (altura(m) * altura(m))

Clasificación:
- < 18.5: Bajo peso
- 18.5 - 24.9: Normal
- 25 - 29.9: Sobrepeso
- >= 30: Obesidad

===============================================================================
2. FUNCIONES DEL PROGRAMA
===============================================================================

Módulo de validación:

```javascript
function validarEntrada(peso, altura) {
    if (typeof peso !== "number" || typeof altura !== "number") {
        return "Error: ambos valores deben ser números";
    }
    if (peso <= 0 || peso > 500) {
        return "Error: peso debe estar entre 1 y 500 kg";
    }
    if (altura <= 0 || altura > 3) {
        return "Error: altura debe estar entre 0.1 y 3 metros";
    }
    return null;  // Sin errores
}
```

Módulo de cálculo (arrow function):

```javascript
let calcularIMC = (peso, altura) => peso / (altura * altura);
```

Módulo de clasificación:

```javascript
function clasificarIMC(imc) {
    if (imc < 18.5) return "Bajo peso";
    if (imc < 24.9) return "Normal";
    if (imc < 29.9) return "Sobrepeso";
    return "Obesidad";
}
```

3. CLOSURE PARA HISTORIAL
===============================================================================

```javascript
function crearHistorial() {
    let registros = [];

    return {
        agregar: function(peso, altura, imc, clasificacion) {
            registros.push({
                fecha: new Date().toLocaleString(),
                peso: peso,
                altura: altura,
                imc: imc.toFixed(2),
                clasificacion: clasificacion
            });
        },
        mostrar: function() {
            if (registros.length === 0) {
                console.log("📭 No hay registros aún");
                return;
            }
            console.log("=== HISTORIAL ===");
            for (let i = 0; i < registros.length; i++) {
                let r = registros[i];
                console.log((i + 1) + ". " + r.fecha + " → " + r.imc + " (" + r.clasificacion + ")");
            }
        },
        obtener: function() {
            return registros;
        }
    };
}
```

===============================================================================
4. PROCESADOR CON CALLBACK
===============================================================================

```javascript
function procesarCalculo(peso, altura, callbackExito, callbackError) {
    let error = validarEntrada(peso, altura);

    if (error) {
        callbackError(error);
        return;
    }

    let imc = calcularIMC(peso, altura);
    let clasificacion = clasificarIMC(imc);

    callbackExito({
        peso: peso,
        altura: altura,
        imc: imc,
        clasificacion: clasificacion
    });
}
```

```javascript
let historial = crearHistorial();

function mostrarResultado(datos) {
    console.log("==============================");
    console.log("📊 RESULTADO IMC");
    console.log("Peso: " + datos.peso + " kg");
    console.log("Altura: " + datos.altura + " m");

    console.log("Clasificación: " + datos.clasificacion);
    console.log("==============================");

    historial.agregar(
        datos.peso,
        datos.altura,
        datos.imc,
        datos.clasificacion
    );
}

// Función para manejar error
function mostrarError(mensaje) {
    console.log("❌ " + mensaje);
}

// Pruebas
console.log("🔬 TEST 1: Datos normales");
procesarCalculo(70, 1.75, mostrarResultado, mostrarError);

procesarCalculo(50, 1.70, mostrarResultado, mostrarError);
```

6. VERSIÓN CON ARROW FUNCTIONS
===============================================================================

```javascript
let validar = (p, a) => {
    if (typeof p !== "number" || typeof a !== "number") return "Tipo inválido";
    if (p <= 0 || p > 500) return "Peso inválido";
    if (a <= 0 || a > 3) return "Altura inválida";
    return null;
};

let calcular = (p, a) => p / (a * a);
let clasificar = i => i < 18.5 ? "Bajo peso" : i < 24.9 ? "Normal" : i < 29.9 ? "Sobrepeso" : "Obesidad";

function procesarConArrow(peso, altura, onExito, onError) {
    let error = validar(peso, altura);
    if (error) return onError(error);
    let imc = calcular(peso, altura);
    onExito({ peso, altura, imc, clasificacion: clasificar(imc) });
}
```

```javascript
function crearCalculadoraAvanzada() {
    let historial = crearHistorial();

    return {
        calcular: function(peso, altura) {
            procesarCalculo(peso, altura,
                function(r) {
                    console.log("✅ IMC: " + r.imc.toFixed(2) + " - " + r.clasificacion);
                    console.log("Peso recomendado: " +
                        (18.5 * altura * altura).toFixed(1) + " - " +
                        (24.9 * altura * altura).toFixed(1) + " kg");
                },
                function(e) {
                    console.log("❌ " + e);
                }
            );
        },
        mostrarHistorial: function() {
            historial.mostrar();
        }
    };
}

let app = crearCalculadoraAvanzada();
app.calcular(70, 1.75);
// ✅ IMC: 22.86 - Normal
// Peso recomendado: 56.7 - 76.3 kg
```
