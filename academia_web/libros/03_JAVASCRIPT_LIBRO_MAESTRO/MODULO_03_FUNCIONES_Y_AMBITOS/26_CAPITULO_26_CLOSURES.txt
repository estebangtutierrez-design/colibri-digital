BIENVENIDA DEL PROFESOR BÚHO

Querido estudiante…

Este es uno de los conceptos más fascinantes de JavaScript.

Un closure ocurre cuando una función interna recuerda
las variables de la función externa, incluso después
de que la función externa haya terminado.

Es como si una persona recordara su ciudad natal
aunque viva en otro país.

1. ¿QUÉ ES UN CLOSURE?
===============================================================================

```javascript
function saludar(nombre) {
    let mensaje = "Hola " + nombre;  // Variable local

    function mostrar() {  // Función interna
        console.log(mensaje);  // Recuerda 'mensaje'
    }

    return mostrar;
}

let saludo = saludar("Ana");
saludo();  // Hola Ana
```

saludar() terminó, pero mostrar() recuerda la variable mensaje.
Eso es un closure.

===============================================================================
2. CLOSURES PRÁCTICOS
===============================================================================

Contador privado:

```javascript
function crearContador() {
    let cuenta = 0;

    return {
        incrementar: function() {
            cuenta++;
            return cuenta;
        },
        decrementar: function() {
            cuenta--;
            return cuenta;
        },
        obtener: function() {
            return cuenta;
        }
    };
}

let contador = crearContador();
console.log(contador.incrementar());  // 1
console.log(contador.incrementar());  // 2
console.log(contador.decrementar());  // 1
// No hay forma de acceder a 'cuenta' directamente
```

Fábrica de funciones:

```javascript
function crearMultiplicador(multiplicador) {
    return function(numero) {
        return numero * multiplicador;
    };
}

let duplicar = crearMultiplicador(2);
let triplicar = crearMultiplicador(3);

console.log(duplicar(5));   // 10
console.log(triplicar(5));  // 15
```

===============================================================================
3. CLOSURES EN BUCLES
===============================================================================

Problema clásico con var:

```javascript
for (var i = 1; i <= 3; i++) {
    setTimeout(function() {
        console.log(i);  // 4, 4, 4 (no 1, 2, 3)
    }, 100);
}
```

Solución con closure:

```javascript
for (var i = 1; i <= 3; i++) {
    (function(j) {
        setTimeout(function() {
            console.log(j);  // 1, 2, 3
        }, 100);
    })(i);
}
```

Solución moderna con let:

```javascript
for (let i = 1; i <= 3; i++) {
    setTimeout(function() {
        console.log(i);  // 1, 2, 3
    }, 100);
}
```

===============================================================================
LA ANALOGÍA DEL PROFESOR BÚHO: LA MOCHILA
===============================================================================

Un closure es como una mochila invisible.

Cuando creas una función dentro de otra, la función interna
carga una mochila con todas las variables de la externa.
Donde quiera que vaya, lleva su mochila.

```javascript
function viajar(ciudad) {
    let recuerdo = "Recuerdo de " + ciudad;

    return function() {
        console.log("Abriendo mochila...");
        console.log(recuerdo);
    };
}

let viajeParis = viajar("París");
let viajeTokio = viajar("Tokio");

viajeParis();  // Abriendo mochila... Recuerdo de París
viajeTokio();  // Abriendo mochila... Recuerdo de Tokio

Cada closure tiene SU PROPIA mochila.
```

MINI PROYECTO: BANCO PRIVADO
===============================================================================

```javascript
function crearBanco(nombreTitular) {
    let saldo = 0;
    let movimientos = [];

    return {
        depositar: function(monto) {
            if (monto <= 0) return "Monto inválido";
            saldo += monto;
            movimientos.push({ tipo: "depósito", monto: monto });
            return "Depósito de $" + monto + ". Saldo: $" + saldo;
        },
        retirar: function(monto) {
            if (monto > saldo) return "Saldo insuficiente";
            if (monto <= 0) return "Monto inválido";
            saldo -= monto;
            movimientos.push({ tipo: "retiro", monto: monto });
            return "Retiro de $" + monto + ". Saldo: $" + saldo;
        },
        consultarSaldo: function() {
            return "Titular: " + nombreTitular + ", Saldo: $" + saldo;
        },
        obtenerMovimientos: function() {
            return movimientos;
        }
    };
}

let cuenta = crearBanco("Ana");
console.log(cuenta.depositar(1000));  // Depósito de $1000
console.log(cuenta.retirar(300));     // Retiro de $300
console.log(cuenta.consultarSaldo()); // Saldo: $700
// Nadie puede modificar 'saldo' directamente
```
