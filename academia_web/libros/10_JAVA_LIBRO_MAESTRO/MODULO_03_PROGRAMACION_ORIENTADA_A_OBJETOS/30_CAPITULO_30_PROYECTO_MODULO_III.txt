1. DESCRIPCIÓN DEL PROYECTO

Sistema de Gestión Hospitalaria (SGH)

El sistema debe permitir:
- Gestionar personas (pacientes, doctores, enfermeros, administrativos).
- Registrar citas médicas.
- Gestionar historiales clínicos.

// com/hospital/util/ValidadorDatos.java
package com.hospital.util;

public class ValidadorDatos {

    public static boolean validarNombre(String nombre) {
        return nombre != null && nombre.trim().length() >= 3;
    }

    public static boolean validarEmail(String email) {
        return email != null && email.matches("^[\\w.-]+@[\\w.-]+\\.\\w{2,}$");
    }

    public static boolean validarTelefono(String telefono) {
        return telefono != null && telefono.matches("\\+?\\d{7,15}");
    }

    public static boolean validarEdad(int edad) {
        return edad >= 0 && edad <= 150;
    }

    public static boolean validarEspecialidad(String especialidad) {
        String[] validas = {"CARDIOLOGIA", "PEDIATRIA", "NEUROLOGIA",
                           "TRAUMATOLOGIA", "MEDICINA GENERAL", "DERMATOLOGIA",
                           "OFTALMOLOGIA", "PSIQUIATRIA"};
        for (String v : validas) {
            if (v.equalsIgnoreCase(especialidad.trim())) return true;
        }
        return false;
    }
}

// com/hospital/util/FechaUtil.java
package com.hospital.util;

import java.time.LocalDate;
import java.time.LocalTime;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeParseException;

public class FechaUtil {
    private static final DateTimeFormatter FORMATO_FECHA =
        DateTimeFormatter.ofPattern("dd/MM/yyyy");
    private static final DateTimeFormatter FORMATO_HORA =
        DateTimeFormatter.ofPattern("HH:mm");

    public static String formatearFecha(LocalDate fecha) {
        return fecha.format(FORMATO_FECHA);
    }

    public static LocalDate parsearFecha(String fechaStr) {
        try {
            return LocalDate.parse(fechaStr, FORMATO_FECHA);
        } catch (DateTimeParseException e) {
            throw new IllegalArgumentException("Formato de fecha inválido: " +
                                              fechaStr + " (use dd/MM/yyyy)");
        }
    }

    public static String formatearHora(LocalTime hora) {
        return hora.format(FORMATO_HORA);
    }

    public static boolean esFechaFutura(LocalDate fecha) {
        return fecha.isAfter(LocalDate.now()) || fecha.isEqual(LocalDate.now());
    }
}

// com/hospital/modelo/Empleado.java
package com.hospital.modelo;

import com.hospital.interfaz.IAutenticable;

public abstract class Empleado extends Persona implements IAutenticable {
    protected String cargo;
    protected double salario;
    protected String turno;
    protected String usuario;
    protected String passwordHash;
    protected boolean activo;

    public Empleado(String id, String nombreCompleto, int edad,
                    String genero, String telefono, String email,
                    String cargo, double salario, String turno,
                    String usuario, String password) {
        super(id, nombreCompleto, edad, genero, telefono, email);
        this.cargo = cargo;
        this.salario = salario;
        this.turno = turno;
        this.usuario = usuario;
        this.passwordHash = Integer.toHexString(password.hashCode());
        this.activo = true;
    }

    public String getCargo() { return cargo; }
    public double getSalario() { return salario; }
    public String getTurno() { return turno; }
    public boolean isActivo() { return activo; }

    public void setSalario(double salario) { this.salario = salario; }
    public void setTurno(String turno) { this.turno = turno; }

    public void desactivar() { this.activo = false; }
    public void activar() { this.activo = true; }

    public abstract double calcularBonoAnual();

    @Override
    public boolean autenticar(String usuario, String password) {
        if (!validarCredenciales(usuario, password)) return false;
        String hash = Integer.toHexString(password.hashCode());
        return this.usuario.equals(usuario) && this.passwordHash.equals(hash);
    }

    @Override
    public String getNombreUsuario() { return usuario; }

    @Override
    public boolean tieneAcceso(String modulo) {
        return activo;
    }

    public String toFileString() {
        return id + "|" + nombreCompleto + "|" + cargo + "|" + salario;
    }
}

package com.hospital.modelo;

import com.hospital.interfaz.IConsultable;

public class Habitacion implements IConsultable {
    private int numero;
    private String tipo; // INDIVIDUAL, DOBLE, SUITE, UCI
    private double precioPorDia;
    private boolean disponible;
    private Paciente pacienteAsignado;
    private String[] equipamiento;

    public Habitacion(int numero, String tipo, double precioPorDia,
                      String[] equipamiento) {
        this.numero = numero;
        this.tipo = tipo;
        this.precioPorDia = precioPorDia;
        this.disponible = true;
        this.pacienteAsignado = null;
        this.equipamiento = equipamiento;
    }

    public int getNumero() { return numero; }
    public String getTipo() { return tipo; }
    public double getPrecioPorDia() { return precioPorDia; }
    public boolean isDisponible() { return disponible; }
    public Paciente getPacienteAsignado() { return pacienteAsignado; }

    public boolean asignarPaciente(Paciente paciente) {
        if (!disponible) return false;
        this.pacienteAsignado = paciente;
        this.disponible = false;
        return true;
    }

    public void liberar() {
        this.pacienteAsignado = null;
        this.disponible = true;
    }

    @Override
    public String consultarInformacion() {
        String eq = String.join(", ", equipamiento);
        String estado = disponible ? "Disponible" :
                        "Ocupada por: " + pacienteAsignado.getNombreCompleto();
        return "Habitación " + numero + " (" + tipo + ")\n" +
               "Precio/día: $" + String.format("%.2f", precioPorDia) + "\n" +
               "Estado: " + estado + "\n" +
               "Equipamiento: " + eq;
    }

    @Override
    public String getIdentificadorUnico() {
        return "HAB-" + numero;
    }
}

// com/hospital/modelo/CitaMedica.java
package com.hospital.modelo;

import java.time.LocalDate;
import java.time.LocalTime;

public class CitaMedica {
    private String idCita;
    private Paciente paciente;
    private Doctor doctor;
    private LocalDate fecha;
    private LocalTime hora;
    private String consultorio;
    private String motivo;
    private String estado; // PROGRAMADA, CONFIRMADA, COMPLETADA, CANCELADA
    private String notas;

    public CitaMedica(String idCita, Paciente paciente, Doctor doctor,
                      LocalDate fecha, LocalTime hora,
                      String consultorio, String motivo) {
        this.idCita = idCita;
        this.paciente = paciente;
        this.doctor = doctor;
        this.fecha = fecha;
        this.hora = hora;
        this.consultorio = consultorio;
        this.motivo = motivo;
        this.estado = "PROGRAMADA";
        this.notas = "";
    }

    public String getIdCita() { return idCita; }
    public Paciente getPaciente() { return paciente; }
    public Doctor getDoctor() { return doctor; }
    public LocalDate getFecha() { return fecha; }
    public LocalTime getHora() { return hora; }
    public String getConsultorio() { return consultorio; }
    public String getEstado() { return estado; }

    public void confirmar() {
        if (estado.equals("PROGRAMADA")) {
            this.estado = "CONFIRMADA";
        }
    }

    public void completar(String notas) {
        if (estado.equals("CONFIRMADA") || estado.equals("PROGRAMADA")) {
            this.estado = "COMPLETADA";
            this.notas = notas;
            doctor.atenderPaciente();
        }
    }

    public void cancelar(String razon) {
        if (!estado.equals("COMPLETADA")) {
            this.estado = "CANCELADA";
            this.notas = "Cancelada: " + razon;
        }
    }

    public String mostrarDetalle() {
        return "CITA " + idCita + "\n" +
               "Paciente: " + paciente.getNombreCompleto() + "\n" +
               "Doctor: " + doctor.getNombreCompleto() + "\n" +
               "Fecha: " + fecha + " | Hora: " + hora + "\n" +
               "Consultorio: " + consultorio + "\n" +
               "Motivo: " + motivo + "\n" +
               "Estado: " + estado + "\n" +
               (notas.isEmpty() ? "" : "Notas: " + notas);
    }
}

        this.contador = 0;
        agregarEntrada("Historial clínico creado para: " +
                      paciente.getNombreCompleto());
    }

    public void agregarEntrada(String entrada) {
        if (contador < MAX_ENTRADAS) {
            entradas[contador++] = java.time.LocalDateTime.now() +
                                  " - " + entrada;
        }
    }

    public String[] getEntradas() {
        String[] activas = new String[contador];
        System.arraycopy(entradas, 0, activas, 0, contador);
        return activas;
    }

    @Override
    public String generarReporte() {
        StringBuilder sb = new StringBuilder();
        sb.append("HISTORIAL CLÍNICO: ").append(paciente.getNombreCompleto())
          .append(" (ID: ").append(paciente.getId()).append(")\n");
        sb.append(paciente.consultarInformacion()).append("\n\n");
        sb.append("=== ENTRADAS DEL HISTORIAL ===\n");
        for (int i = 0; i < contador; i++) {
            sb.append((i+1) + ". " + entradas[i] + "\n");
        }
        return sb.toString();
    }

    @Override
    public String getTipoReporte() {
        return "HISTORIAL CLÍNICO - " + paciente.getNombreCompleto();
    }
}

    private Habitacion[] habitaciones;
    private int contador;

    public HabitacionServicio(int capacidad) {
        this.habitaciones = new Habitacion[capacidad];
        this.contador = 0;
    }

    public void agregarHabitacion(Habitacion habitacion) {
        if (contador < habitaciones.length) {
            habitaciones[contador++] = habitacion;
        }
    }

    public void asignarHabitacion(int numero, Paciente paciente)
            throws HabitacionNoDisponibleException {
        for (int i = 0; i < contador; i++) {
            if (habitaciones[i].getNumero() == numero) {
                if (habitaciones[i].asignarPaciente(paciente)) {
                    System.out.println("Paciente " + paciente.getNombreCompleto() +
                                     " asignado a habitación " + numero);
                    return;
                } else {
                    throw new HabitacionNoDisponibleException(numero);
                }
            }
        }
        System.out.println("Habitación " + numero + " no encontrada.");
    }

    public void liberarHabitacion(int numero) {
        for (int i = 0; i < contador; i++) {
            if (habitaciones[i].getNumero() == numero) {
                habitaciones[i].liberar();
                System.out.println("Habitación " + numero + " liberada.");
                return;
            }
        }
    }

    public void listarDisponibles() {
        System.out.println("\n=== HABITACIONES DISPONIBLES ===");
        for (int i = 0; i < contador; i++) {
            if (habitaciones[i].isDisponible()) {
                System.out.println("  " + habitaciones[i].consultarInformacion());
                System.out.println();
            }
        }
    }

    public void listarTodas() {
        System.out.println("\n=== TODAS LAS HABITACIONES ===");
        for (int i = 0; i < contador; i++) {
