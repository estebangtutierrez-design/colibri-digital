package sge.modelo;

import java.time.LocalDate;
import java.util.Objects;

public class Empleado implements Comparable<Empleado> {
    private String id;
    private String nombre;
    private String apellido;
    private String email;
    private double salario;
    private LocalDate fechaIngreso;
    private Departamento departamento;

    public Empleado(String id, String nombre, String apellido,
                    String email, double salario, LocalDate fechaIngreso) {
        this.id = id;
        this.nombre = nombre;
        this.apellido = apellido;
        this.email = email;
        this.salario = salario;
        this.fechaIngreso = fechaIngreso;
    }

    // Getters y setters
    public String getId() { return id; }
    public String getNombre() { return nombre; }
    public void setNombre(String nombre) { this.nombre = nombre; }
    public String getApellido() { return apellido; }
    public void setApellido(String apellido) { this.apellido = apellido; }
    public String getEmail() { return email; }
    public void setEmail(String email) { this.email = email; }
    public double getSalario() { return salario; }
    public void setSalario(double salario) { this.salario = salario; }
    public LocalDate getFechaIngreso() { return fechaIngreso; }
    public Departamento getDepartamento() { return departamento; }
    public void setDepartamento(Departamento departamento) {
        this.departamento = departamento;
    }

    public int getAntiguedad() {
        return (int) java.time.temporal.ChronoUnit.YEARS
            .between(fechaIngreso, LocalDate.now());
    }

    @Override
    public int compareTo(Empleado otro) {
        int cmp = this.apellido.compareTo(otro.apellido);
        if (cmp != 0) return cmp;
        cmp = this.nombre.compareTo(otro.nombre);
        if (cmp != 0) return cmp;
        return this.id.compareTo(otro.id);
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof Empleado)) return false;
        Empleado e = (Empleado) o;
        return Objects.equals(id, e.id);
    }

    @Override
    public int hashCode() {
        return Objects.hash(id);
    }

    @Override
    public String toString() {
        return String.format("%s %s (%s) - $%.2f - %s",
            nombre, apellido, id, salario,
            departamento != null ? departamento.getNombre() : "Sin depto");
    }
}

package sge.modelo;

import java.time.LocalDate;
import java.util.Objects;

public class Tarea implements Comparable<Tarea> {
    public enum Prioridad { BAJA, MEDIA, ALTA, CRITICA }
    public enum Estado { PENDIENTE, EN_PROGRESO, COMPLETADA }

    private String id;
    private String titulo;
    private String descripcion;
    private Prioridad prioridad;
    private Estado estado;
    private LocalDate fechaLimite;
    private Empleado asignadoA;
    private Proyecto proyecto;

    public Tarea(String id, String titulo, Prioridad prioridad,
                 LocalDate fechaLimite, Proyecto proyecto) {
        this.id = id;
        this.titulo = titulo;
        this.prioridad = prioridad;
        this.estado = Estado.PENDIENTE;
        this.fechaLimite = fechaLimite;
        this.proyecto = proyecto;
    }

    public String getId() { return id; }
    public String getTitulo() { return titulo; }
    public Prioridad getPrioridad() { return prioridad; }
    public Estado getEstado() { return estado; }
    public void setEstado(Estado estado) { this.estado = estado; }
    public LocalDate getFechaLimite() { return fechaLimite; }
    public Empleado getAsignadoA() { return asignadoA; }
    public void setAsignadoA(Empleado e) { this.asignadoA = e; }
    public Proyecto getProyecto() { return proyecto; }

    public boolean estaVencida() {
        return estado != Estado.COMPLETADA &&
               LocalDate.now().isAfter(fechaLimite);
    }

    @Override
    public int compareTo(Tarea otra) {
        return this.prioridad.compareTo(otra.prioridad);
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof Tarea)) return false;
        return Objects.equals(id, ((Tarea) o).id);
    }

    @Override
    public int hashCode() { return Objects.hash(id); }

    @Override
    public String toString() {
        String vencida = estaVencida() ? " ¡VENCIDA!" : "";
        return String.format("[%s] %s (%s) → %s%s",
            prioridad, titulo, estado,
            asignadoA != null ? asignadoA.getNombre() : "Sin asignar",
            vencida);
    }
}

package sge.repositorio;

import java.util.*;
import java.util.function.Predicate;
import java.util.stream.Collectors;

public class Repositorio<T> {
    protected Map<String, T> almacen = new LinkedHashMap<>();

    public T guardar(String id, T elemento) {
        return almacen.put(id, elemento);
    }

    public T obtener(String id) {
        return almacen.get(id);
    }

    public T eliminar(String id) {
        return almacen.remove(id);
    }

    public boolean existe(String id) {
        return almacen.containsKey(id);
    }

    public List<T> listarTodos() {
        return new ArrayList<>(almacen.values());
    }

    public List<T> buscar(Predicate<? super T> criterio) {
        return almacen.values().stream()
            .filter(criterio)
            .collect(Collectors.toList());
    }

    public long contar() {
        return almacen.size();
    }

    public void clear() {
        almacen.clear();
    }
}

package sge.repositorio;

import sge.modelo.Empleado;
import java.util.*;
import java.util.stream.Collectors;

public class RepositorioEmpleado extends Repositorio<Empleado> {

    public List<Empleado> buscarPorDepartamento(String deptoId) {
        return buscar(e -> e.getDepartamento() != null &&
            e.getDepartamento().getId().equals(deptoId));
    }

    public List<Empleado> buscarPorSalario(double min, double max) {
        return buscar(e -> e.getSalario() >= min && e.getSalario() <= max);
    }

    public Optional<Empleado> buscarPorEmail(String email) {
        return almacen.values().stream()
            .filter(e -> e.getEmail().equalsIgnoreCase(email))
            .findFirst();
    }

    public List<Empleado> ordenar(Comparator<? super Empleado> comparator) {
        List<Empleado> lista = listarTodos();
        lista.sort(comparator);
        return lista;
    }

    public Map<Double, List<Empleado>> agruparPorSalario() {
        return almacen.values().stream()
            .collect(Collectors.groupingBy(
                e -> Math.floor(e.getSalario() / 10000) * 10000));
    }

    public Map<String, Long> contarPorDepartamento() {
        return almacen.values().stream()
            .filter(e -> e.getDepartamento() != null)
            .collect(Collectors.groupingBy(
                e -> e.getDepartamento().getNombre(),
                Collectors.counting()));
    }

    public double calcularSalarioPromedio() {
        return almacen.values().stream()
            .mapToDouble(Empleado::getSalario)
            .average()
            .orElse(0);
    }

    public Empleado empleadoConMayorSalario() {
        return Collections.max(almacen.values(),
            Comparator.comparingDouble(Empleado::getSalario));
    }

    public Empleado empleadoConMenorSalario() {
        return Collections.min(almacen.values(),
            Comparator.comparingDouble(Empleado::getSalario));
    }

    public Map<String, Double> salarioPromedioPorDepartamento() {
        return almacen.values().stream()
            .filter(e -> e.getDepartamento() != null)
            .collect(Collectors.groupingBy(
                e -> e.getDepartamento().getNombre(),
                Collectors.averagingDouble(Empleado::getSalario)));
    }

    public double medianaSalarial() {
        List<Double> salarios = almacen.values().stream()
            .map(Empleado::getSalario)
            .sorted()
            .collect(Collectors.toList());
        int n = salarios.size();
        if (n == 0) return 0;
        if (n % 2 == 1) return salarios.get(n / 2);
        return (salarios.get(n / 2 - 1) + salarios.get(n / 2)) / 2.0;
    }
}

package sge.servicio;

import sge.modelo.*;
import sge.repositorio.*;
import sge.util.*;
import java.util.*;
import java.util.stream.*;

public class ServicioEmpleado {
    private RepositorioEmpleado repo;
    private RepositorioDepartamento repoDepto;

    public ServicioEmpleado(RepositorioEmpleado repo,
                            RepositorioDepartamento repoDepto) {
        this.repo = repo;
        this.repoDepto = repoDepto;
    }

    public Empleado crearEmpleado(String id, String nombre, String apellido,
                                   String email, double salario,
                                   java.time.LocalDate fechaIngreso) {
        Empleado e = new Empleado(id, nombre, apellido, email, salario, fechaIngreso);
        repo.guardar(id, e);
        return e;
    }

    public boolean asignarADepartamento(String empleadoId, String deptoId) {
        Empleado emp = repo.obtener(empleadoId);
        Departamento dept = repoDepto.obtener(deptoId);
        if (emp == null || dept == null) return false;
        emp.setDepartamento(dept);
        return true;
    }

    public List<Empleado> listarEmpleados() {
        return repo.listarTodos();
    }

    public List<Empleado> buscarPorNombre(String texto) {
        return repo.buscar(e ->
            e.getNombre().toLowerCase().contains(texto.toLowerCase()) ||
            e.getApellido().toLowerCase().contains(texto.toLowerCase()));
    }

    public boolean aumentarSalario(String id, double porcentaje) {
        Empleado e = repo.obtener(id);
        if (e == null) return false;
        e.setSalario(e.getSalario() * (1 + porcentaje / 100));
        return true;
    }

    public List<Empleado> empleadosPorDepartamento(String deptoId) {
        return repo.buscarPorDepartamento(deptoId);
    }

    public Map<String, List<Empleado>> empleadosGroupByDepartamento() {
        return repo.listarTodos().stream()
            .filter(e -> e.getDepartamento() != null)
            .collect(Collectors.groupingBy(
                e -> e.getDepartamento().getNombre(),
                TreeMap::new,
                Collectors.toList()));
    }

    public void aplicarAumentoPorDepartamento(
            String deptoId, double porcentaje) {
        repo.buscarPorDepartamento(deptoId)
            .forEach(e -> e.setSalario(
                e.getSalario() * (1 + porcentaje / 100)));
    }

    public boolean eliminarEmpleado(String id) {
        return repo.eliminar(id) != null;
    }
}

package sge.servicio;

import sge.modelo.*;
import sge.repositorio.*;
import java.util.*;
import java.util.stream.*;

public class ServicioReporte {
    private RepositorioEmpleado repoEmp;
    private RepositorioDepartamento repoDepto;

    public ServicioReporte(RepositorioEmpleado repoEmp,
                           RepositorioDepartamento repoDepto) {
        this.repoEmp = repoEmp;
        this.repoDepto = repoDepto;
    }

    // Reporte 1: Distribución salarial por departamento
    public void reporteSalariosPorDepartamento() {
        System.out.println("\n=== REPORTE SALARIAL POR DEPARTAMENTO ===");
        Map<String, Double> promedios =
            repoEmp.salarioPromedioPorDepartamento();
        promedios.forEach((depto, prom) ->
            System.out.printf("%-20s $%.2f%n", depto, prom));

        System.out.println("\nSalario promedio general: $" +
            String.format("%.2f", repoEmp.calcularSalarioPromedio()));
        System.out.println("Mediana salarial: $" +
            String.format("%.2f", repoEmp.medianaSalarial()));
        System.out.println("Mayor salario: " + repoEmp.empleadoConMayorSalario());
        System.out.println("Menor salario: " + repoEmp.empleadoConMenorSalario());
    }

    // Reporte 2: Antigüedad de empleados
    public void reporteAntiguedad() {
        System.out.println("\n=== REPORTE DE ANTIGÜEDAD ===");
        List<Empleado> empleados = repoEmp.listarTodos();
        empleados.sort(Comparator.comparingInt(Empleado::getAntiguedad)
            .reversed());

        System.out.println("Empleados ordenados por antigüedad:");
        empleados.forEach(e ->
            System.out.printf("%-20s %d años%n",
                e.getNombre() + " " + e.getApellido(),
                e.getAntiguedad()));
    }

    // Reporte 3: Estadísticas por departamento
    public void reporteEstadisticasDepartamento() {
        System.out.println("\n=== ESTADÍSTICAS POR DEPARTAMENTO ===");
        Map<String, Long> conteo = repoEmp.contarPorDepartamento();
        conteo.forEach((depto, count) ->
            System.out.printf("%-20s %d empleados%n", depto, count));

        System.out.println("\nDepartamentos sin empleados:");
        List<Departamento> sinEmpleados = repoDepto.listarTodos().stream()
            .filter(d -> !conteo.containsKey(d.getNombre()))
            .collect(Collectors.toList());
        sinEmpleados.forEach(d ->
            System.out.println("  " + d.getNombre()));
    }

    // Reporte 4: Top N salarios
    public void reporteTopSalarios(int n) {
        System.out.println("\n=== TOP " + n + " SALARIOS ===");
        List<Empleado> top = repoEmp.listarTodos().stream()
            .sorted(Comparator.comparingDouble(Empleado::getSalario)
                .reversed())
            .limit(n)
            .collect(Collectors.toList());

        for (int i = 0; i < top.size(); i++) {
            Empleado e = top.get(i);
            System.out.printf("%d. %-20s $%.2f%n",
                i + 1, e.getNombre() + " " + e.getApellido(),
                e.getSalario());
        }
    }

    // Reporte 5: Búsqueda por rango salarial
    public void reporteRangoSalarial(double min, double max) {
        System.out.printf("\n=== EMPLEADOS CON SALARIO ENTRE $%.0f Y $%.0f ===%n",
            min, max);
        List<Empleado> enRango = repoEmp.buscarPorSalario(min, max);
        if (enRango.isEmpty()) {
            System.out.println("No hay empleados en este rango.");
            return;
        }
        enRango.sort(Comparator.comparingDouble(Empleado::getSalario));
        enRango.forEach(e ->
            System.out.printf("%-20s $%.2f%n",
                e.getNombre() + " " + e.getApellido(),
                e.getSalario()));
    }

    // Reporte 6: Reporte completo de la empresa
    public void reporteCompleto() {
        System.out.println("╔══════════════════════════════════════════╗");
        System.out.println("║   REPORTE COMPLETO DE LA EMPRESA       ║");
        System.out.println("╚══════════════════════════════════════════╝");

        System.out.println("\nTotal empleados: " + repoEmp.contar());
        System.out.println("Total departamentos: " + repoDepto.contar());

        reporteSalariosPorDepartamento();
        reporteAntiguedad();
        reporteEstadisticasDepartamento();
        reporteTopSalarios(5);
    }
}

package sge.util;

import sge.modelo.*;
import java.util.Comparator;

public final class Comparadores {

    private Comparadores() { }

    // Empleado comparators
    public static final Comparator<Empleado> EMPLEADO_POR_SALARIO_ASC =
        Comparator.comparingDouble(Empleado::getSalario);

    public static final Comparator<Empleado> EMPLEADO_POR_SALARIO_DESC =
        Comparator.comparingDouble(Empleado::getSalario).reversed();

    public static final Comparator<Empleado> EMPLEADO_POR_ANTIGUEDAD =
        Comparator.comparingInt(Empleado::getAntiguedad).reversed();

    public static final Comparator<Empleado> EMPLEADO_POR_NOMBRE =
        Comparator.comparing(Empleado::getApellido)
            .thenComparing(Empleado::getNombre);

    public static final Comparator<Empleado> EMPLEADO_POR_EMAIL =
        Comparator.comparing(Empleado::getEmail);

    // Tarea comparators
    public static final Comparator<Tarea> TAREA_POR_PRIORIDAD =
        Comparator.comparing(Tarea::getPrioridad);

    public static final Comparator<Tarea> TAREA_POR_FECHA_LIMITE =
        Comparator.comparing(Tarea::getFechaLimite);

    // Proyecto comparators
    public static final Comparator<Proyecto> PROYECTO_POR_ESTADO =
        Comparator.comparing(p -> p.getEstado().ordinal());

    // Métodos de fábrica
    public static <T extends Comparable<? super T>> Comparator<T> ordenReverso() {
        return Comparator.reverseOrder();
    }

    public static <T> Comparator<T> comparadorNuloPrimero(
            Comparator<T> comp) {
        return Comparator.nullsFirst(comp);
    }
}

        Departamento it = new Departamento("D001", "Tecnología", "Piso 3");
        Departamento ventas = new Departamento("D002", "Ventas", "Piso 2");
        Departamento rh = new Departamento("D003", "Recursos Humanos", "Piso 1");
        Departamento finanzas = new Departamento("D004", "Finanzas", "Piso 4");

        repoDepto.guardar("D001", it);
        repoDepto.guardar("D002", ventas);
        repoDepto.guardar("D003", rh);
        repoDepto.guardar("D004", finanzas);

        // Crear empleados
        Empleado emp1 = servicioEmp.crearEmpleado("E001", "Ana", "García",
            "ana@empresa.com", 85000, LocalDate.of(2018, 3, 15));
        Empleado emp2 = servicioEmp.crearEmpleado("E002", "Luis", "Pérez",
            "luis@empresa.com", 72000, LocalDate.of(2019, 7, 1));
        Empleado emp3 = servicioEmp.crearEmpleado("E003", "Carlos", "Ruiz",
            "carlos@empresa.com", 95000, LocalDate.of(2015, 1, 20));
        Empleado emp4 = servicioEmp.crearEmpleado("E004", "María", "López",
            "maria@empresa.com", 65000, LocalDate.of(2020, 11, 5));
        Empleado emp5 = servicioEmp.crearEmpleado("E005", "Pedro", "Martínez",
            "pedro@empresa.com", 78000, LocalDate.of(2017, 6, 10));
        Empleado emp6 = servicioEmp.crearEmpleado("E006", "Sofía", "Ramírez",
            "sofia@empresa.com", 92000, LocalDate.of(2016, 9, 25));

        // Asignar a departamentos
        servicioEmp.asignarADepartamento("E001", "D001"); // Ana → IT
        servicioEmp.asignarADepartamento("E002", "D002"); // Luis → Ventas
        servicioEmp.asignarADepartamento("E003", "D001"); // Carlos → IT
        servicioEmp.asignarADepartamento("E004", "D003"); // María → RH
        servicioEmp.asignarADepartamento("E005", "D002"); // Pedro → Ventas
        servicioEmp.asignarADepartamento("E006", "D004"); // Sofía → Finanzas

        // === DEMOSTRACIÓN DE COLECCIONES Y OPERACIONES ===

        // 1. List: empleados ordenados por salario (usando Comparator)
        System.out.println("\n--- EMPLEADOS ORDENADOS POR SALARIO (DESC) ---");
        List<Empleado> porSalario = repoEmp.ordenar(Comparadores.EMPLEADO_POR_SALARIO_DESC);
        porSalario.forEach(System.out::println);

        // 2. Set: departamentos únicos (sin duplicados)
        System.out.println("\n--- DEPARTAMENTOS (TreeSet ordenado) ---");
        Set<Departamento> deptosSet = new TreeSet<>(repoDepto.listarTodos());
        deptosSet.forEach(System.out::println);

        // 3. Map: empleados agrupados por departamento (TreeMap para orden)
        System.out.println("\n--- EMPLEADOS POR DEPARTAMENTO ---");
        Map<String, List<Empleado>> porDepto = servicioEmp.empleadosGroupByDepartamento();
        porDepto.forEach((depto, empleados) -> {
            System.out.println(depto + ":");
            empleados.forEach(e -> System.out.println("  - " + e.getNombre() + " " + e.getApellido()));
        });

        // 4. Queue: procesamiento de empleados por antigüedad (PriorityQueue)
        System.out.println("\n--- PROCESANDO EMPLEADOS POR ANTIGÜEDAD ---");
        Queue<Empleado> colaAntiguedad = new PriorityQueue<>(
            Comparator.comparingInt(Empleado::getAntiguedad).reversed());
        colaAntiguedad.addAll(repoEmp.listarTodos());
        while (!colaAntiguedad.isEmpty()) {
            Empleado e = colaAntiguedad.poll();
            System.out.println("Procesando: " + e.getNombre() + " (" +
                e.getAntiguedad() + " años)");
        }

        // 5. Deque: historial de operaciones
        System.out.println("\n--- HISTORIAL DE OPERACIONES (Deque) ---");
        Deque<String> historial = new ArrayDeque<>();
        historial.push("Crear empleado E001");
        historial.push("Asignar E001 a IT");
        historial.push("Actualizar salario E001");
        historial.push("Crear departamento D001");
        System.out.println("Últimas 2 operaciones:");
        for (int i = 0; i < 2 && !historial.isEmpty(); i++) {
            System.out.println("  " + historial.pop());
        }

        // === REPORTES ===
        servicioReporte.reporteCompleto();

        // === DEMOSTRACIÓN DE WILDCARDS GENÉRICOS ===
        System.out.println("\n--- WILDCARDS: SUMAR SALARIOS ---");
        double totalSalarios = sumarSalarios(repoEmp.listarTodos());
        System.out.printf("Total salarios: $%.2f%n", totalSalarios);

        // === DEMOSTRACIÓN DE COLLECTIONS UTILITY ===
        System.out.println("\n--- COLLECTIONS UTILITY ---");

        // Barajar lista de empleados
        List<Empleado> empleadosCopy = new ArrayList<>(repoEmp.listarTodos());
        Collections.shuffle(empleadosCopy);
        System.out.println("Empleados barajados:");
        empleadosCopy.forEach(e -> System.out.println("  " + e.getNombre()));

        // Buscar empleado por salario (binarySearch requiere lista ordenada)
        List<Empleado> ordenadosSalario = repoEmp.ordenar(Comparadores.EMPLEADO_POR_SALARIO_ASC);
        Empleado referencia = new Empleado("", "", "", "", 78000, null);
        int pos = Collections.binarySearch(ordenadosSalario, referencia,
            Comparator.comparingDouble(Empleado::getSalario));
        if (pos >= 0) {
            System.out.println("Empleado con salario $78000 encontrado en posición: " + pos);
        }

        // Frecuencia de departamentos
        Map<String, Long> frecuenciaDeptos = repoEmp.contarPorDepartamento();
        String deptoMasEmpleados = Collections.max(frecuenciaDeptos.entrySet(),
            Map.Entry.comparingByValue()).getKey();
        System.out.println("Departamento con más empleados: " + deptoMasEmpleados);
    }

    // Método genérico con wildcard
    public static double sumarSalarios(List<? extends Empleado> empleados) {
        double suma = 0;
        for (Empleado e : empleados) {
            suma += e.getSalario();
        }
        return suma;
    }
}

PREGUNTAS FRECUENTES

P: ¿Por qué usar un Repositorio genérico en lugar de específicos?
R: El repositorio genérico encapsula la lógica común de CRUD
   y búsqueda. Los específicos añaden operaciones de dominio.
   Esto reduce duplicación y mejora mantenibilidad.

P: ¿Cómo decido qué tipo de colección usar?
R: • List: cuando importa el orden y posición
   • Set: cuando necesito unicidad
   • Map: cuando necesito asociaciones clave-valor
   • Queue: para procesamiento programado
   • Deque: para operaciones en ambos extremos

P: ¿Cómo manejar la persistencia de datos?
R: Este proyecto usa almacenamiento en memoria. Para persistencia
   real, añadirías serialización, base de datos o archivos. El
   diseño con repositorios facilita esta transición.

P: ¿Este código es thread-safe?
R: No. Para acceso concurrente, deberías usar colecciones
   sincronizadas (Collections.synchronizedX()) o las del
   paquete java.util.concurrent.
