package com.biblioteca.modelo;

public enum EstadoLibro {
    DISPONIBLE("Disponible para préstamo", true),
    PRESTADO("Actualmente prestado", false),
    RESERVADO("Reservado", false),
    EN_REPARACION("En reparación", false),
    PERDIDO("Perdido", false),
    BAJA("Dado de baja", false);

    private final String descripcion;
    private final boolean disponible;

    EstadoLibro(String descripcion, boolean disponible) {
        this.descripcion = descripcion;
        this.disponible = disponible;
    }

    public String getDescripcion() { return descripcion; }
    public boolean isDisponible() { return disponible; }

    public boolean puedeTransicionarA(EstadoLibro destino) {
        return switch (this) {
            case DISPONIBLE -> destino == PRESTADO || destino == RESERVADO
                    || destino == EN_REPARACION || destino == BAJA
                    || destino == PERDIDO;
            case PRESTADO -> destino == DISPONIBLE || destino == PERDIDO
                    || destino == EN_REPARACION;
            case RESERVADO -> destino == PRESTADO || destino == DISPONIBLE;
            case EN_REPARACION -> destino == DISPONIBLE || destino == BAJA;
            case PERDIDO -> destino == BAJA;
            case BAJA -> false;
        };
    }
}

package com.biblioteca.modelo;

import java.time.LocalDate;
import java.util.Objects;

public class Libro {
    private String isbn;
    private String titulo;
    private String autor;
    private int anioPublicacion;
    private String editorial;
    private int numPaginas;
    private String genero;
    private EstadoLibro estado;
    private LocalDate fechaAdquisicion;

    public Libro(String isbn, String titulo, String autor,
            int anio, String editorial, int paginas, String genero) {
        this.isbn = isbn;
        this.titulo = titulo;
        this.autor = autor;
        this.anioPublicacion = anio;
        this.editorial = editorial;
        this.numPaginas = paginas;
        this.genero = genero;
        this.estado = EstadoLibro.DISPONIBLE;
        this.fechaAdquisicion = LocalDate.now();
    }

    public String getIsbn() { return isbn; }
    public String getTitulo() { return titulo; }
    public String getAutor() { return autor; }
    public int getAnioPublicacion() { return anioPublicacion; }
    public String getEditorial() { return editorial; }
    public int getNumPaginas() { return numPaginas; }
    public String getGenero() { return genero; }
    public EstadoLibro getEstado() { return estado; }
    public LocalDate getFechaAdquisicion() { return fechaAdquisicion; }

    public void setEstado(EstadoLibro nuevoEstado) {
        assert estado.puedeTransicionarA(nuevoEstado)
                : "Transición inválida: " + estado + " -> " + nuevoEstado;
        this.estado = nuevoEstado;
    }

    @Override
    public String toString() {
        return String.format("%-15s %-30s %-20s [%s]",
                isbn, titulo, autor, estado);
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof Libro libro)) return false;
        return Objects.equals(isbn, libro.isbn);
    }

    @Override
    public int hashCode() { return Objects.hash(isbn); }
}

package com.biblioteca.modelo;

import java.time.LocalDate;
import java.time.Period;
import java.util.*;

public class Usuario {
    private String id;
    private String nombre;
    private String email;
    private LocalDate fechaRegistro;
    private boolean activo;
    private List<Prestamo> prestamos;

    public Usuario(String id, String nombre, String email) {
        this.id = id;
        this.nombre = nombre;
        this.email = email;
        this.fechaRegistro = LocalDate.now();
        this.activo = true;
        this.prestamos = new ArrayList<>();
    }

    public String getId() { return id; }
    public String getNombre() { return nombre; }
    public String getEmail() { return email; }
    public LocalDate getFechaRegistro() { return fechaRegistro; }
    public boolean isActivo() { return activo; }
    public List<Prestamo> getPrestamos() {
        return Collections.unmodifiableList(prestamos);
    }

    public void setActivo(boolean activo) { this.activo = activo; }
    public void agregarPrestamo(Prestamo p) { prestamos.add(p); }
    public void removerPrestamo(Prestamo p) { prestamos.remove(p); }

    public int getNumPrestamosActivos() {
        return (int) prestamos.stream()
                .filter(Prestamo::estaActivo).count();
    }

    public boolean tieneMultasPendientes() {
        return prestamos.stream()
                .filter(Prestamo::estaActivo)
                .anyMatch(p -> Period.between(
                        p.getFechaDevolucionEsperada(),
                        LocalDate.now()).getDays() > 0);
    }

    @Override
    public String toString() {
        return String.format("%-10s %-25s %-25s %s",
                id, nombre, email,
                activo ? "Activo" : "Inactivo");
    }
}

package com.biblioteca.modelo;

import java.time.LocalDate;
import java.time.temporal.ChronoUnit;

public class Prestamo {
    private String id;
    private Libro libro;
    private Usuario usuario;
    private LocalDate fechaPrestamo;
    private LocalDate fechaDevolucionEsperada;
    private LocalDate fechaDevolucionReal;
    private double multa;
    private static final int DIAS_PRESTAMO = 14;
    private static final double MULTA_POR_DIA = 0.50;

    public Prestamo(String id, Libro libro, Usuario usuario) {
        this.id = id;
        this.libro = libro;
        this.usuario = usuario;
        this.fechaPrestamo = LocalDate.now();
        this.fechaDevolucionEsperada = fechaPrestamo.plusDays(DIAS_PRESTAMO);
        this.multa = 0.0;
    }

    public String getId() { return id; }
    public Libro getLibro() { return libro; }
    public Usuario getUsuario() { return usuario; }
    public LocalDate getFechaPrestamo() { return fechaPrestamo; }
    public LocalDate getFechaDevolucionEsperada() { return fechaDevolucionEsperada; }
    public LocalDate getFechaDevolucionReal() { return fechaDevolucionReal; }
    public double getMulta() { return multa; }

    public boolean estaActivo() { return fechaDevolucionReal == null; }

    public boolean estaVencido() {
        return estaActivo()
                && LocalDate.now().isAfter(fechaDevolucionEsperada);
    }

    public double calcularMulta() {
        if (fechaDevolucionReal == null) {
            long dias = ChronoUnit.DAYS.between(
                    fechaDevolucionEsperada, LocalDate.now());
            return Math.max(0, dias * MULTA_POR_DIA);
        }
        if (fechaDevolucionReal.isAfter(fechaDevolucionEsperada)) {
            long dias = ChronoUnit.DAYS.between(
                    fechaDevolucionEsperada, fechaDevolucionReal);
            return dias * MULTA_POR_DIA;
        }
        return 0;
    }

    public void devolver() {
        this.fechaDevolucionReal = LocalDate.now();
        this.multa = calcularMulta();
        this.libro.setEstado(EstadoLibro.DISPONIBLE);
    }

    @Override
    public String toString() {
        String estado = estaActivo()
                ? (estaVencido() ? "VENCIDO" : "ACTIVO")
                : "DEVUELTO";
        return String.format("%-12s %-30s %-20s %s %s",
                id, libro.getTitulo(), usuario.getNombre(),
                estado, estaActivo() ? "" : ("Multa: $" + multa));
    }
}

package com.biblioteca.servicio;

import com.biblioteca.modelo.*;
import com.biblioteca.excepcion.*;
import java.util.*;
import java.util.stream.*;
import java.util.function.*;

public class BibliotecaService {
    private Map<String, Libro> libros = new HashMap<>();
    private List<Prestamo> prestamos = new ArrayList<>();

    public void agregarLibro(Libro libro) {
        Objects.requireNonNull(libro, "El libro no puede ser nulo");
        libros.put(libro.getIsbn(), libro);
    }

    public void agregarLibros(Libro... librosArray) {
        Arrays.stream(librosArray).forEach(this::agregarLibro);
    }

    public Libro buscarPorIsbn(String isbn) throws LibroNoEncontradoException {
        Libro libro = libros.get(isbn);
        if (libro == null) throw new LibroNoEncontradoException(isbn);
        return libro;
    }

    public List<Libro> buscarPorTitulo(String titulo) {
        return libros.values().stream()
                .filter(l -> l.getTitulo().toLowerCase()
                        .contains(titulo.toLowerCase()))
                .collect(Collectors.toList());
    }

    public List<Libro> buscarPorAutor(String autor) {
        return libros.values().stream()
                .filter(l -> l.getAutor().toLowerCase()
                        .contains(autor.toLowerCase()))
                .collect(Collectors.toList());
    }

    public List<Libro> buscarPorGenero(String genero) {
        return libros.values().stream()
                .filter(l -> l.getGenero().equalsIgnoreCase(genero))
                .collect(Collectors.toList());
    }

    public List<Libro> buscar(Predicate<Libro> criterio) {
        return libros.values().stream()
                .filter(criterio)
                .collect(Collectors.toList());
    }

    public List<Libro> listarTodos() { return new ArrayList<>(libros.values()); }

    public List<Libro> listarDisponibles() {
        return libros.values().stream()
                .filter(l -> l.getEstado() == EstadoLibro.DISPONIBLE)
                .collect(Collectors.toList());
    }

    public long contarLibros() { return libros.size(); }

    public long contarDisponibles() {
        return libros.values().stream()
                .filter(l -> l.getEstado() == EstadoLibro.DISPONIBLE)
                .count();
    }

    public Prestamo prestarLibro(String isbn, Usuario usuario)
            throws LibroNoEncontradoException, LibroNoDisponibleException,
                   LimitePrestamosExcedidoException {
        Libro libro = buscarPorIsbn(isbn);
        if (libro.getEstado() != EstadoLibro.DISPONIBLE) {
            throw new LibroNoDisponibleException(isbn, libro.getEstado().name());
        }
        int prestamosActivos = usuario.getNumPrestamosActivos();
        if (prestamosActivos >= 5) {
            throw new LimitePrestamosExcedidoException(
                    usuario.getId(), prestamosActivos, 5);
        }
        String idPrestamo = "PR-" + UUID.randomUUID().toString()
                .substring(0, 8).toUpperCase();
        Prestamo prestamo = new Prestamo(idPrestamo, libro, usuario);
        libro.setEstado(EstadoLibro.PRESTADO);
        usuario.agregarPrestamo(prestamo);
        prestamos.add(prestamo);
        return prestamo;
    }

    public double devolverLibro(String idPrestamo) throws BibliotecaException {
        Prestamo prestamo = prestamos.stream()
                .filter(p -> p.getId().equals(idPrestamo))
                .findFirst()
                .orElseThrow(() -> new BibliotecaException(
                        "PR-NF", "Préstamo no encontrado: " + idPrestamo));
        if (!prestamo.estaActivo()) {
            throw new BibliotecaException("PR-DEV",
                    "El préstamo ya fue devuelto");
        }
        prestamo.devolver();
        return prestamo.getMulta();
    }

    public List<Prestamo> listarPrestamosActivos() {
        return prestamos.stream()
                .filter(Prestamo::estaActivo)
                .collect(Collectors.toList());
    }

    public List<Prestamo> listarPrestamosVencidos() {
        return prestamos.stream()
                .filter(Prestamo::estaVencido)
                .collect(Collectors.toList());
    }

    public Map<String, Long> contarPorGenero() {
        return libros.values().stream()
                .collect(Collectors.groupingBy(
                        Libro::getGenero, Collectors.counting()));
    }

    public Map<String, List<Libro>> agruparPorAutor() {
        return libros.values().stream()
                .collect(Collectors.groupingBy(Libro::getAutor));
    }

    public Map<EstadoLibro, Long> contarPorEstado() {
        return libros.values().stream()
                .collect(Collectors.groupingBy(
                        Libro::getEstado, Collectors.counting()));
    }

    public Optional<Libro> libroMasAntiguo() {
        return libros.values().stream()
                .min(Comparator.comparingInt(Libro::getAnioPublicacion));
    }

    public Optional<Libro> libroMasReciente() {
        return libros.values().stream()
                .max(Comparator.comparingInt(Libro::getAnioPublicacion));
    }

    public List<String> topLibrosMasPrestados() {
        return prestamos.stream()
                .collect(Collectors.groupingBy(
                        p -> p.getLibro().getTitulo(), Collectors.counting()))
                .entrySet().stream()
                .sorted(Map.Entry.<String, Long>comparingByValue().reversed())
                .limit(5)
                .map(e -> e.getKey() + " (" + e.getValue() + " veces)")
                .collect(Collectors.toList());
    }

    public double calcularMultasPendientes() {
        return prestamos.stream()
                .filter(Prestamo::estaActivo)
                .mapToDouble(Prestamo::calcularMulta)
                .sum();
    }

    public IntSummaryStatistics estadisticasAnioPublicacion() {
        return libros.values().stream()
                .mapToInt(Libro::getAnioPublicacion)
                .summaryStatistics();
    }

    public List<Libro> ordenarPorTitulo() {
        List<Libro> resultado = new ArrayList<>(libros.values());
        resultado.sort(new Comparator<Libro>() {
            @Override
            public int compare(Libro a, Libro b) {
                return a.getTitulo().compareToIgnoreCase(b.getTitulo());
            }
        });
        return resultado;
    }

    public List<Libro> ordenarPorAnioDesc() {
        List<Libro> resultado = new ArrayList<>(libros.values());
        resultado.sort((a, b) -> Integer.compare(
                b.getAnioPublicacion(), a.getAnioPublicacion()));
        return resultado;
    }
}

package com.biblioteca.servicio;

public class UsuarioService {
    private Map<String, Usuario> usuarios = new HashMap<>();

    public void registrarUsuario(Usuario usuario) {
        Objects.requireNonNull(usuario, "Usuario no puede ser nulo");
        usuarios.put(usuario.getId(), usuario);
    }

    public Usuario buscarPorId(String id) throws UsuarioNoEncontradoException {
        Usuario usuario = usuarios.get(id);
        if (usuario == null) throw new UsuarioNoEncontradoException(id);
        return usuario;
    }

        return usuarios.values().stream()
                .filter(u -> u.getNombre().toLowerCase()
                        .contains(nombre.toLowerCase()))
                .collect(Collectors.toList());
    }

    public List<Usuario> listarTodos() { return new ArrayList<>(usuarios.values()); }
    public long contarUsuarios() { return usuarios.size(); }

    public long contarActivos() {
        return usuarios.values().stream()
                .filter(Usuario::isActivo).count();
    }

    public void desactivarUsuario(String id) throws UsuarioNoEncontradoException {
        Usuario u = buscarPorId(id);
        u.setActivo(false);
    }

    public List<Usuario> usuariosConMultas() {
        return usuarios.values().stream()
                .filter(Usuario::tieneMultasPendientes)
                .collect(Collectors.toList());
    }

    public List<String> topLectores() {
        return usuarios.values().stream()
                .sorted((a, b) -> Integer.compare(
                        b.getPrestamos().size(), a.getPrestamos().size()))
                .limit(5)
                .map(u -> u.getNombre() + " ("
                        + u.getPrestamos().size() + " préstamos)")
                .collect(Collectors.toList());
    }
}

package com.biblioteca.servicio;

import com.biblioteca.modelo.*;
import java.util.*;
import java.util.stream.*;

public class ReporteService {
    private BibliotecaService bibliotecaService;
    private UsuarioService usuarioService;

    public ReporteService(BibliotecaService bs, UsuarioService us) {
        this.bibliotecaService = bs;
        this.usuarioService = us;
    }

    public void generarReporteCompleto() {
        System.out.println("\n" + "=".repeat(70));
        System.out.println("  REPORTE COMPLETO DE BIBLIOTECA");
        System.out.println("=".repeat(70));

        System.out.println("\n--- RESUMEN GENERAL ---");
        System.out.println("Total libros: " + bibliotecaService.contarLibros());
        System.out.println("Disponibles: " + bibliotecaService.contarDisponibles());
        System.out.println("Usuarios: " + usuarioService.contarUsuarios());
        System.out.println("Activos: " + usuarioService.contarActivos());
        System.out.println("Préstamos activos: "
                + bibliotecaService.listarPrestamosActivos().size());
        System.out.println("Vencidos: "
                + bibliotecaService.listarPrestamosVencidos().size());
        System.out.printf("Multas pendientes: $%.2f%n",
                bibliotecaService.calcularMultasPendientes());

        System.out.println("\n--- POR GÉNERO ---");
        bibliotecaService.contarPorGenero().forEach((g, c) ->
                System.out.printf("  %-20s: %d%n", g, c));

package com.biblioteca.util;

import java.util.function.Predicate;

public class Validador {

    public static void validarIsbn(String isbn) {
        if (isbn == null || !isbn.matches("\\d{13}|\\d{10}")) {
            throw new IllegalArgumentException(
                    "ISBN inválido: debe ser 10 o 13 dígitos: " + isbn);
        }
    }

    public static void validarEmail(String email) {
        if (email == null || !email.matches("^[\\w.-]+@[\\w.-]+\\.\\w{2,}$")) {
            throw new IllegalArgumentException("Email inválido: " + email);
        }
    }

    public static void validarTexto(String valor, String campo) {
        if (valor == null || valor.trim().isEmpty()) {
            throw new IllegalArgumentException(campo + " no puede estar vacío");
        }
    }

    public static Predicate<String> minimoCaracteres(int min) {
        return s -> s != null && s.length() >= min;
    }

    public static Predicate<String> maximoCaracteres(int max) {
        return s -> s != null && s.length() <= max;
    }

    public static Predicate<String> soloDigitos() {
        return s -> s != null && s.matches("\\d+");
    }
}

public class FechaUtils {
    private static final DateTimeFormatter FORMATO =
            DateTimeFormatter.ofPattern("dd/MM/yyyy");

    public static LocalDate parsear(String fechaStr)
            throws FechaInvalidaException {
        try {
            return LocalDate.parse(fechaStr, FORMATO);
        } catch (DateTimeParseException e) {
            throw new FechaInvalidaException(fechaStr,
                    "Formato esperado: dd/MM/yyyy");
        }
    }

    public static String formatear(LocalDate fecha) {
        return fecha != null ? fecha.format(FORMATO) : "N/A";
    }
}

     javac -d out src/com/biblioteca/**/*.java
     java -ea -cp out com.biblioteca.Main
     (El flag -ea activa las aserciones)

  3. ¿Qué UUID usa prestamo?
     java.util.UUID.randomUUID() genera identificadores únicos.

  4. ¿Puedo usar Maven o Gradle?
     Sí, pero no es necesario para este proyecto. Se puede compilar
     directamente con javac.

  5. ¿Los datos persisten entre ejecuciones?
     No en la versión base. Para persistencia, implementa el
     ejercicio de extensión 5 sobre serialización.

  6. ¿Qué pasa si ingreso datos inválidos?
     Las validaciones (Validador) y el manejo de excepciones
     capturan los errores y muestran mensajes sin cerrar el programa.

  7. ¿Cómo verifico que las aserciones funcionan?
     Ejecuta con java -ea. Intenta una transición de estado inválida
     (ej: de BAJA a DISPONIBLE) y verás el AssertionError.

  8. ¿Puedo agregar más funcionalidades?
     ¡Por supuesto! El proyecto es extensible. Revisa los ejercicios
     de extensión para ideas.

  DESAFÍO DEL ARQUITECTO (PROYECTO FINAL)

  Extiende el proyecto de biblioteca para convertirlo en un sistema
  multi-biblioteca con las siguientes características avanzadas:

  1. Red de Bibliotecas: Soporte para múltiples sucursales (clase
     Sucursal) con su propio inventario. Usa herencia múltiple
     de interfaces (Gestionable, Reportable, Transferible).

  2. Transferencia entre Sucursales: Implementa transferencia de
     libros entre sucursales usando una interfaz Transferible con
     métodos default y resolución de conflictos.

  3. Motor de Recomendación: Usa Streams y lambdas para recomendar
     libros a usuarios basado en: género más leído por el usuario,
     autor más leído, libros populares en su sucursal. Implementa
     una interfaz funcional Recomendador.

  4. Sistema de Multas con Strategy: Usa un enum EstrategiaMulta
     con métodos abstractos: MULTA_FIJA ($5/día), MULTA_PROGRESIVA
     ($1/día + $1 por cada día adicional), MULTA_MAXIMA (tope de
     $50). Usa el patrón Strategy.

  5. Reportes Avanzados: Genera reportes en formato JSON simulado
     usando Streams y Collectors.joining(). Incluye: ingresos por
     multas por sucursal, top 10 libros más prestados (con conteo),
     usuarios morosos con más de 3 multas.

  6. Concurrencia: Implementa un pool de hilos (ExecutorService)
     para procesar tareas en segundo plano: envío de notificaciones
     de libros vencidos, generación de reportes, backup automático
     cada 30 segundos.

  7. Plugin System: Define una interfaz Plugin con métodos init(),
     execute(), shutdown(). Usa clases anónimas para crear plugins
     como: EmailPlugin, SMSPlugin, BackupPlugin. Carga los plugins
     desde una lista y ejecútalos en orden.

  8. Pruebas Unitarias: Escribe pruebas JUnit para:
     - Validación de transiciones de EstadoLibro
     - Cálculo de multas (días exactos)
     - Flujo completo: crear libro -> prestar -> devolver -> verificar
     - Excepciones: libro no encontrado, límite excedido
     - Búsquedas con streams (título, autor, género)

  Formato de entrega:
  - Código fuente completo en un repositorio Git
  - README.md con instrucciones de compilación y ejecución
  - Diagrama de clases en formato Mermaid o imagen
  - Tests unitarios con cobertura > 70%

  Criterios de evaluación:
  - Uso correcto de todos los temas del módulo (30%)
  - Funcionamiento completo del sistema (30%)
  - Calidad del código (20%)
  - Tests y documentación (20%)
