────────────────────────────────────────────────────────────────────────────────────
  BIENVENIDA
────────────────────────────────────────────────────────────────────────────────────

Bienvenido al Capítulo 47 del Ruby Master Book. Hasta ahora hemos trabajado
con métodos que existen realmente en clases y módulos. Pero Ruby tiene un
secreto fascinante: cuando llamas a un método que no existe, Ruby no se rinde
inmediatamente. En lugar de lanzar un error, llama a un método especial llamado
`method_missing`.

Esto abre un mundo de posibilidades: puedes crear objetos que respondan a
cualquier mensaje, delegar llamadas dinámicamente a otros objetos, construir
DSLs elegantes, implementar proxies, y crear estructuras de datos flexibles
como `OpenStruct`.

En este capítulo aprenderás a usar `method_missing` de forma responsable,
a implementar `respond_to_missing?` para mantener la introspección, y a
evitar las trampas que este poder conlleva.

────────────────────────────────────────────────────────────────────────────────────
  DESARROLLO DEL CAPÍTULO — METHOD_MISSING
────────────────────────────────────────────────────────────────────────────────────

Cuando Ruby recibe un mensaje (llamada a método), busca en:
  1. Métodos singleton del objeto
  2. Métodos de instancia de la clase (y módulos prependidos)
  3. Métodos de instancia de la superclase (y módulos)
  4. Así sucesivamente hasta `BasicObject`
  5. Llama a `method_missing` si no encuentra nada

El `method_missing` por defecto (en `Kernel`) lanza `NoMethodError`.

Al sobrescribir `method_missing`, puedes interceptar cualquier llamada a
un método inexistente y manejarla como desees.

Siempre debes sobrescribir también `respond_to_missing?` para que la
introspección (`respond_to?`, `method`, etc.) funcione correctamente.

```ruby
# archivo: 47_01_method_missing_basico.rb
# Sobrescritura básica de method_missing

class Capturador
  def method_missing(name, *args, &block)
    puts "Método llamado: #{name}"
    puts "Argumentos: #{args.inspect}"
    puts "Bloque: #{block ? 'sí' : 'no'}"
    "Has llamado a #{name} con #{args}"
  end

  def respond_to_missing?(name, include_private = false)
    true
  end
end

obj = Capturador.new
puts obj.hola
puts obj.suma(1, 2, 3)
puts obj.saludar("Mundo") { "bloque" }
puts obj.any_method_you_want

puts "\nrespond_to? pruebas:"
puts "  respond_to?(:hola): #{obj.respond_to?(:hola)}"
puts "  respond_to?(:metodo_inexistente): #{obj.respond_to?(:metodo_inexistente)}"
```

```ruby
# archivo: 47_02_ghost_methods.rb
# Atributos dinámicos con method_missing

class ObjetoFlexible
  def initialize
    @datos = {}
  end

  def method_missing(name, *args, &block)
    nombre = name.to_s

    if nombre.end_with?('=')
      clave = nombre[0..-2]
      @datos[clave] = args.first
    elsif @datos.key?(nombre)
      @datos[nombre]
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    nombre = name.to_s
    nombre.end_with?('=') || @datos.key?(nombre[0..-2]) || @datos.key?(nombre) || super
  end

  def atributos
    @datos.keys
  end

  def to_s
    @datos.map { |k, v| "#{k}: #{v.inspect}" }.join(", ")
  end
end

obj = ObjetoFlexible.new
obj.nombre = "Alice"
obj.edad = 30
obj.email = "alice@example.com"
obj.activo = true

puts "Nombre: #{obj.nombre}"
puts "Edad: #{obj.edad}"
puts "Email: #{obj.email}"
puts "Activo: #{obj.activo}"
puts "Atributos: #{obj.atributos.join(', ')}"
puts "Objeto: #{obj}"

puts "\nrespond_to?:"
puts "  nombre: #{obj.respond_to?(:nombre)}"
puts "  nombre=: #{obj.respond_to?(:nombre=)}"
puts "  inexistente: #{obj.respond_to?(:inexistente)}"

begin
  obj.no_existe
rescue => e
  puts "\nError al llamar a no_existe: #{e.class}: #{e.message}"
end
```

```ruby
# archivo: 47_03_delegacion_dinamica.rb
# Delegar llamadas a otro objeto

class Delegador
  def initialize(target)
    @target = target
  end

  def method_missing(name, *args, &block)
    if @target.respond_to?(name)
      @target.send(name, *args, &block)
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    @target.respond_to?(name) || super
  end

  def delegado
    @target
  end
end

class ServicioReal
  def saludar(nombre)
    "Hola, #{nombre}!"
  end

  def sumar(a, b)
    a + b
  end

  def longitud(texto)
    texto.length
  end
end

real = ServicioReal.new
delegado = Delegador.new(real)

puts delegado.saludar("Mundo")
puts delegado.sumar(5, 3)
puts delegado.longitud("Ruby")

puts "\nrespond_to?:"
puts "  saludar: #{delegado.respond_to?(:saludar)}"
puts "  volar: #{delegado.respond_to?(:volar)}"

begin
  delegado.volar
rescue => e
  puts "Error: #{e.class}: #{e.message}"
end
```

```ruby
# archivo: 47_04_proxy_registro.rb
# Proxy que registra todas las llamadas

class ProxyConRegistro
  def initialize(target, nombre = "Proxy")
    @target = target
    @nombre = nombre
    @llamadas = []
  end

  def method_missing(name, *args, &block)
    if @target.respond_to?(name)
      timestamp = Time.now
      resultado = @target.send(name, *args, &block)
      @llamadas << {
        metodo: name,
        argumentos: args,
        resultado: resultado,
        tiempo: timestamp,
        duracion: Time.now - timestamp
      }
      resultado
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    @target.respond_to?(name) || super
  end

  def llamadas
    @llamadas
  end

  def reporte_llamadas
    @llamadas.map.with_index do |ll, i|
      "#{i + 1}. #{ll[:metodo]}(#{ll[:argumentos].map(&:inspect).join(', ')}) => #{ll[:resultado].inspect}"
    end.join("\n")
  end

  def limpiar_registro
    @llamadas.clear
  end
end

class Calculadora
  def sumar(a, b)
    a + b
  end

  def multiplicar(a, b)
    a * b
  end

  def saludar(nombre)
    "¡Hola #{nombre}!"
  end

  def lento
    sleep(0.05)
    "hecho"
  end
end

calc = Calculadora.new
proxy = ProxyConRegistro.new(calc, "CalcProxy")

puts proxy.sumar(3, 4)
puts proxy.multiplicar(5, 6)
puts proxy.saludar("Ruby")
puts proxy.lento

puts "\nReporte de llamadas:"
puts proxy.reporte_llamadas

puts "\nTotal de llamadas: #{proxy.llamadas.length}"
```

```ruby
# archivo: 47_05_openstruct_casero.rb
# Implementación de OpenStruct usando method_missing

class MiOpenStruct
  def initialize(atributos = {})
    @datos = {}
    atributos.each { |k, v| self[k] = v }
  end

  def [](clave)
    @datos[clave.to_s]
  end

  def []=(clave, valor)
    @datos[clave.to_s] = valor
  end

  def method_missing(name, *args, &block)
    nombre = name.to_s

    if nombre.end_with?('=')
      clave = nombre[0..-2]
      @datos[clave] = args.first
    elsif @datos.key?(nombre)
      @datos[nombre]
    elsif nombre.end_with?('?')
      @datos.key?(nombre[0..-2]) ? !!@datos[nombre[0..-2]] : false
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    nombre = name.to_s
    if nombre.end_with?('=')
      true
    elsif nombre.end_with?('?')
      @datos.key?(nombre[0..-2]) || super
    else
      @datos.key?(nombre) || super
    end
  end

  def to_h
    @datos.dup
  end

  def each(&block)
    @datos.each(&block)
  end

  def delete_field(name)
    @datos.delete(name.to_s)
  end

  def attributes
    @datos.keys
  end

  def inspect
    "#<MiOpenStruct #{@datos.map { |k, v| "#{k}=#{v.inspect}" }.join(', ')}>"
  end

  def ==(otro)
    return false unless otro.is_a?(MiOpenStruct)
    @datos == otro.to_h
  end
end

persona = MiOpenStruct.new(nombre: "Alice", edad: 30, activo: true)
puts persona.nombre
puts persona.edad
persona.email = "alice@example.com"
puts persona.email
puts "¿activo?: #{persona.activo?}"
puts "¿admin?: #{persona.admin?}"
puts "Hash: #{persona.to_h}"
puts persona.inspect

persona2 = MiOpenStruct.new(nombre: "Bob", edad: 25)
puts "\nIterar atributos:"
persona2.each { |k, v| puts "  #{k} => #{v}" }
```

```ruby
# archivo: 47_06_dsl_method_missing.rb
# Construcción de DSL usando method_missing

class DSLRouter
  def initialize
    @rutas = []
  end

  def method_missing(name, *args, &block)
    if [:get, :post, :put, :delete, :patch].include?(name)
      ruta = {
        metodo: name.upcase,
        path: args.first,
        bloque: block
      }
      @rutas << ruta
      ruta
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    [:get, :post, :put, :delete, :patch].include?(name) || super
  end

  def rutas
    @rutas
  end

  def mostrar_rutas
    @rutas.each do |r|
      puts "  #{r[:metodo]} #{r[:path]}"
    end
  end

  def ejecutar(metodo, path)
    ruta = @rutas.find { |r| r[:metodo] == metodo.upcase && r[:path] == path }
    if ruta && ruta[:bloque]
      ruta[:bloque].call
    else
      "404 No encontrado: #{metodo} #{path}"
    end
  end
end

router = DSLRouter.new

router.get "/" do
  "Bienvenido a la página principal"
end

router.get "/usuarios" do
  "Lista de usuarios: Alice, Bob, Carlos"
end

router.post "/usuarios" do
  "Usuario creado exitosamente"
end

router.get "/productos" do
  "Productos: Laptop, Mouse, Teclado"
end

router.put "/usuarios/1" do
  "Usuario #1 actualizado"
end

router.delete "/usuarios/1" do
  "Usuario #1 eliminado"
end

puts "Rutas registradas:"
router.mostrar_rutas

puts "\nEjecutando rutas:"
puts router.ejecutar("GET", "/")
puts router.ejecutar("POST", "/usuarios")
puts router.ejecutar("GET", "/productos")
puts router.ejecutar("DELETE", "/usuarios/1")
puts router.ejecutar("GET", "/no-existe")
```

```ruby
# archivo: 47_07_builder_dsl.rb
# DSL para construir objetos complejos

class QueryBuilder
  def initialize(tabla)
    @tabla = tabla
    @seleccion = "*"
    @condiciones = []
    @orden = nil
    @limite = nil
    @offset = nil
  end

  def select(*campos)
    @seleccion = campos.join(", ")
    self
  end

  def where(**condiciones)
    condiciones.each do |campo, valor|
      @condiciones << "#{campo} = #{valor.is_a?(String) ? "'#{valor}'" : valor}"
    end
    self
  end

  def order(campo, direccion = :asc)
    @orden = "#{campo} #{direccion.upcase}"
    self
  end

  def limit(n)
    @limite = n
    self
  end

  def offset(n)
    @offset = n
    self
  end

  def method_missing(name, *args, &block)
    if name.to_s.start_with?('where_')
      campo = name.to_s.sub('where_', '')
      valor = args.first
      @condiciones << "#{campo} = #{valor.is_a?(String) ? "'#{valor}'" : valor}"
      self
    elsif name.to_s.start_with?('order_by_')
      campo = name.to_s.sub('order_by_', '')
      @orden = "#{campo} #{args.first || 'ASC'}"
      self
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    name.to_s.start_with?('where_') || name.to_s.start_with?('order_by_') || super
  end

  def to_sql
    sql = "SELECT #{@seleccion} FROM #{@tabla}"
    sql += " WHERE #{@condiciones.join(' AND ')}" unless @condiciones.empty?
    sql += " ORDER BY #{@orden}" if @orden
    sql += " LIMIT #{@limite}" if @limite
    sql += " OFFSET #{@offset}" if @offset
    sql
  end
end

query = QueryBuilder.new("usuarios")
  .select(:id, :nombre, :email)
  .where_edad(25)
  .where_activo(true)
  .where_ciudad("Madrid")
  .order_by_nombre("DESC")
  .limit(10)
  .offset(20)

puts query.to_sql

puts "\nOtra consulta:"
query2 = QueryBuilder.new("productos")
  .where_categoria("electronica")
  .where_precio(100)
  .where_stock(0)
  .order(:precio)

puts query2.to_sql
```

```ruby
# archivo: 47_08_proxy_seguridad.rb
# Proxy de seguridad que controla acceso

class ProxySeguridad
  METODOS_PERMITIDOS = {
    admin: [:todos],
    editor: [:leer, :escribir, :buscar],
    viewer: [:leer, :buscar]
  }

  def initialize(target, rol, usuario)
    @target = target
    @rol = rol
    @usuario = usuario
  end

  def method_missing(name, *args, &block)
    if @target.respond_to?(name)
      if METODOS_PERMITIDOS[@rol]&.include?(:todos) ||
         METODOS_PERMITIDOS[@rol]&.include?(name)
        @target.send(name, *args, &block)
      else
        raise SecurityError, "#{@usuario} (#{@rol}) no tiene permiso para #{name}"
      end
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    @target.respond_to?(name) || super
  end
end

class Documento
  attr_reader :titulo, :contenido

  def initialize(titulo, contenido)
    @titulo = titulo
    @contenido = contenido
  end

  def leer
    "Contenido: #{@contenido[0..50]}..."
  end

  def escribir(nuevo_contenido)
    @contenido = nuevo_contenido
    "Documento actualizado"
  end

  def buscar(patron)
    @contenido.include?(patron) ? "Encontrado" : "No encontrado"
  end

  def eliminar
    @contenido = nil
    "Documento eliminado"
  end

  def compartir(usuario)
    "Compartido con #{usuario}"
  end
end

doc = Documento.new("Secreto", "Este es el contenido confidencial del documento.")

admin = ProxySeguridad.new(doc, :admin, "root")
editor = ProxySeguridad.new(doc, :editor, "alice")
viewer = ProxySeguridad.new(doc, :viewer, "bob")

puts "Admin:"
puts "  #{admin.leer}"
puts "  #{admin.escribir('nuevo contenido')}"
puts "  #{admin.eliminar}"

puts "\nEditor:"
puts "  #{editor.leer}"
puts "  #{editor.buscar('confidencial')}"

begin
  editor.eliminar
rescue => e
  puts "  Error editor.eliminar: #{e.message}"
end

puts "\nViewer:"
puts "  #{viewer.leer}"
puts "  #{viewer.buscar('Ruby')}"

begin
  viewer.escribir("hack")
rescue => e
  puts "  Error viewer.escribir: #{e.message}"
```

```ruby
class ProxyCache
  def initialize(target, ttl = 60)
    @target = target
    @ttl = ttl
    @cache = {}
  end

  def method_missing(name, *args, &block)
    if @target.respond_to?(name)
      clave = "#{name}:#{args.hash}"
      ahora = Time.now

      if @cache.key?(clave)
        entrada = @cache[clave]
        if ahora - entrada[:tiempo] < @ttl
          return entrada[:valor]
        else
          @cache.delete(clave)
        end
      end

      valor = @target.send(name, *args, &block)
      @cache[clave] = { valor: valor, tiempo: ahora }
      valor
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    @target.respond_to?(name) || super
  end

  def clear_cache
    @cache.clear
  end

  def cache_size
    @cache.length
  end

  def cache_keys
    @cache.keys
  end
end

class ServicioCostoso

    puts "  Consultando #{id}..."
    sleep(0.1)
    "Resultado para #{id}"
  end

  def calcular(x, y)
    puts "  Calculando #{x} + #{y}..."
    sleep(0.05)
    x + y
  end
end

servicio = ServicioCostoso.new
cache = ProxyCache.new(servicio, 5)
```

```ruby
class FormateadorDinamico
  def initialize(valor)
    @valor = valor
  end

  def method_missing(name, *args, &block)
    nombre = name.to_s

    if nombre.start_with?('como_')
      tipo = nombre.sub('como_', '')
      formatear_como(tipo, *args)
    elsif nombre.start_with?('a_')
      tipo = nombre.sub('a_', '')
      convertir_a(tipo)
    elsif nombre.end_with?('?')
      preguntar(nombre[0..-2])
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    nombre = name.to_s
    nombre.start_with?('como_') || nombre.start_with?('a_') || nombre.end_with?('?') || super
  end

  private

  def formatear_como(tipo, *args)
    case tipo
    when 'moneda' then "$#{format('%.2f', @valor)}"
    when 'porcentaje' then "#{format('%.1f', @valor * 100)}%"
    when 'cientifico' then format('%.2e', @valor)
    when 'entero' then @valor.to_i.to_s
    when 'decimal' then format('%.2f', @valor)
    when 'hexadecimal' then @valor.to_s(16)
    when 'binario' then @valor.to_s(2)
    when 'romano' then convertir_a_romano(@valor)
    when 'humano' then numero_humano(@valor)
    when 'telefono' then formatear_telefono(@valor.to_s)
    else "Formato '#{tipo}' no soportado"
    end
  end

  def convertir_a(tipo)
    case tipo
    when 'string' then @valor.to_s
    when 'entero' then @valor.to_i
    when 'flotante' then @valor.to_f
    when 'arreglo' then Array(@valor)
    when 'simbolo' then @valor.to_s.to_sym
    else "Conversión a '#{tipo}' no soportada"
    end
  end

  def preguntar(atributo)
    case atributo
    when 'numero' then @valor.is_a?(Numeric)
    when 'entero' then @valor.is_a?(Integer)
    when 'string' then @valor.is_a?(String)
    when 'positivo' then @valor.respond_to?(:>) && @valor > 0
    when 'negativo' then @valor.respond_to?(:<) && @valor < 0
    when 'par' then @valor.respond_to?(:even?) && @valor.even?
    when 'impar' then @valor.respond_to?(:odd?) && @valor.odd?
    else false
    end
  end

  def convertir_a_romano(num)
    return "0" if num == 0
    valores = [
      [1000, "M"], [900, "CM"], [500, "D"], [400, "CD"],
      [100, "C"], [90, "XC"], [50, "L"], [40, "XL"],
      [10, "X"], [9, "IX"], [5, "V"], [4, "IV"], [1, "I"]
    ]
    resultado = ""
    num = num.to_i
    valores.each do |val, letra|
      while num >= val
        resultado << letra
        num -= val
      end
    end
    resultado
  end

  def numero_humano(num)
    num = num.to_i
    case num
    when 0 then "cero"
    when 1 then "uno"
    when 2 then "dos"
    when 3 then "tres"
    when 4 then "cuatro"
    when 5 then "cinco"
    when 6 then "seis"
    when 7 then "siete"
    when 8 then "ocho"
    when 9 then "nueve"
    when 10 then "diez"
    else num.to_s
    end
  end

  def formatear_telefono(tel)
    tel = tel.gsub(/[^0-9]/, '')
    if tel.length == 10
      "(#{tel[0..2]}) #{tel[3..5]}-#{tel[6..9]}"
    elsif tel.length == 7
      "#{tel[0..2]}-#{tel[3..6]}"
    else
      tel
    end
  end
```

```ruby
      recurso = nombre.sub('get_', '')

    elsif nombre.start_with?('put_')
      recurso = nombre.sub('put_', '')
      hacer_put(recurso, args.first, args.second || {})
    elsif nombre.start_with?('delete_')
      recurso = nombre.sub('delete_', '')
      hacer_delete(recurso, args.first)
    else
      super
    end

      nombre.start_with?('put_') || nombre.start_with?('delete_') || super

    { status: 200, data: { recurso: recurso, id: id } }

    puts "PUT #{uri} #{datos.to_json}"
    { status: 200, data: { id: id, **datos } }
  end

  def hacer_delete(recurso, id)
    uri = URI("#{@base_url}/#{recurso}/#{id}")
    puts "DELETE #{uri}"
    { status: 204 }
  end
```

```ruby
class Validador
  attr_reader :errores

  def initialize(objeto)
    @objeto = objeto
    @errores = []
  end

  def method_missing(name, *args, &block)
    nombre = name.to_s

    if nombre.start_with?('validar_que_')
      condicion = nombre.sub('validar_que_', '')
      unless @objeto.send(condicion, *args)
        @errores << "Validación falló: #{condicion}"
      end
    elsif nombre.start_with?('validar_')
      campo = nombre.sub('validar_', '')
      valor = @objeto.send(campo) rescue nil
      opciones = args.first || {}

      if opciones[:presencia] && (valor.nil? || valor.to_s.empty?)
        @errores << "#{campo} no puede estar vacío"
      end

      if opciones[:longitud]
        min = opciones[:longitud][:min] || 0
        max = opciones[:longitud][:max] || Float::INFINITY
        if valor && (valor.length < min || valor.length > max)
          @errores << "#{campo} debe tener entre #{min} y #{max} caracteres"
        end
      end

      if opciones[:formato] && valor
        unless valor.match?(opciones[:formato])
          @errores << "#{campo} no tiene el formato correcto"
        end
      end

      if opciones[:en] && valor
        unless opciones[:en].include?(valor)
          @errores << "#{campo} debe ser uno de: #{opciones[:en].join(', ')}"
        end
      end

      if opciones[:numerico] && !valor.is_a?(Numeric)
        @errores << "#{campo} debe ser numérico"
      end
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    nombre = name.to_s
    nombre.start_with?('validar_') || super
  end

  def valido?
    @errores.empty?
```

```ruby
# archivo: 47_15_consultas_dinamicas.rb
# Consultas dinámicas tipo ActiveRecord

module Consultable
  def self.included(base)
    base.extend(ClassMethods)
  end

  module ClassMethods
    def registros
      @registros || []
    end

    def method_missing(name, *args, &block)
      nombre = name.to_s

      if nombre.start_with?('buscar_por_')
        campos = nombre.sub('buscar_por_', '').split('_y_')
        valor = args.first
        if campos.length == 1
          registros.find { |r| r.send(campos.first) == valor }
        else
          registros.select { |r| campos.all? { |c| r.send(c) == valor } }
        end
      elsif nombre.start_with?('donde_')
        condiciones_str = nombre.sub('donde_', '')
        condiciones = parsear_condiciones(condiciones_str)
        registros.select do |r|
          condiciones.all? { |campo, op, val| evaluar_condicion(r, campo, op, val) }
        end
      elsif nombre.start_with?('ordenar_por_')
        campo = nombre.sub('ordenar_por_', '')
        direccion = args.first || :asc
        registros.sort do |a, b|
          compare = a.send(campo) <=> b.send(campo)
          direccion == :asc ? compare : -compare
        end
      elsif nombre.start_with?('contar_')
        campo = nombre.sub('contar_', '')
        registros.group_by { |r| r.send(campo) }.transform_values(&:length)
      else
        super
      end
    end

    def respond_to_missing?(name, include_private = false)
      nombre = name.to_s
      nombre.start_with?('buscar_por_') || nombre.start_with?('donde_') ||
        nombre.start_with?('ordenar_por_') || nombre.start_with?('contar_') || super
    end

    private

    def parsear_condiciones(str)
      condiciones = []
      str.split('_y_').each do |parte|
        if parte.include?('_mayor_que_')
          campos = parte.split('_mayor_que_')
          condiciones << [campos[0], :>, campos[1].to_i]
        elsif parte.include?('_menor_que_')
          campos = parte.split('_menor_que_')
          condiciones << [campos[0], :<, campos[1].to_i]
        elsif parte.include?('_diferente_de_')
          campos = parte.split('_diferente_de_')
          condiciones << [campos[0], :!=, campos[1]]
        else
          condiciones << [parte, :==, nil]
        end
      end
      condiciones
    end

    def evaluar_condicion(record, campo, op, val)
      valor_record = record.send(campo)
      case op
      when :== then valor_record == val
      when :> then valor_record > val
      when :< then valor_record < val
      when :!= then valor_record != val
      else true
      end
    end
  end
end

  attr_reader :id, :nombre, :departamento, :salario, :edad

  def initialize(id:, nombre:, departamento:, salario:, edad:)
    @id = id
    @nombre = nombre
    @departamento = departamento
    @salario = salario
    @edad = edad
    (@@registros ||= []) << self
  end

  def to_s
    "#{@nombre} (#{@departamento}, $#{@salario}, #{@edad}a)"
  end
end
```

```ruby
# Tracing de llamadas con method_missing

class TrazaLlamadas
  def initialize(obj, nombre = nil)
    @target = obj
    @nombre = nombre || obj.class.name
    @trazas = []
  end

  def method_missing(name, *args, &block)
    if @target.respond_to?(name)
      timestamp = Time.now
      linea = caller_locations(1, 1)&.first
      traza = {
        metodo: name,
        args: args,
        bloque: !block.nil?,
        tiempo: timestamp,
        origen: "#{linea&.path}:#{linea&.lineno}",
        objeto: @nombre
      }
      @trazas << traza
      resultado = @target.send(name, *args, &block)
      traza[:resultado] = resultado
      traza[:duracion] = Time.now - timestamp
      resultado
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    @target.respond_to?(name) || super
  end

  def trazas
    @trazas
  end

  def informe
    puts "=== INFORME DE TRAZAS ==="
    puts "Objeto: #{@nombre}"
    puts "Total llamadas: #{@trazas.length}"
    puts ""

    @trazas.group_by { |t| t[:metodo] }.each do |metodo, llamadas|
      puts "  #{metodo}: #{llamadas.length} vez/veces"
      puts "    Tiempo total: #{format('%.4f', llamadas.sum { |l| l[:duracion] || 0 })}s"
    end
  end

  def limpiar
    @trazas.clear
  end
end

class Ordenador
  def ordenar(lista)
    lista.sort
  end

  def invertir(lista)
    lista.reverse
  end

  def mezclar(lista1, lista2)
    (lista1 + lista2).sort
  end

  def buscar(lista, elemento)
    lista.include?(elemento)
  end
end

ord = Ordenador.new

puts traza.invertir([1, 2, 3])
puts traza.mezclar([1, 3, 5], [2, 4, 6])
puts traza.buscar([1, 2, 3], 2)
```

```ruby
      promedios = {}
```

```ruby
# archivo: 47_19_persistencia_dinamica.rb
# Objetos con persistencia automática

module PersistenciaDinamica
  def self.included(base)
    base.extend(ClassMethods)
  end

  module ClassMethods
    def almacen
      @almacen ||= {}
    end

    def method_missing(name, *args, &block)
      nombre = name.to_s

      if nombre.start_with?('buscar_por_')
        campo = nombre.sub('buscar_por_', '')
        valor = args.first
        almacen.values.find { |obj| obj.send(campo) == valor }
      elsif nombre.start_with?('donde_')
        campo = nombre.sub('donde_', '')
        valor = args.first
        almacen.values.select { |obj| obj.send(campo) == valor }
      elsif name == :todos
        almacen.values
      elsif name == :count
        almacen.length
      elsif name == :primero
        almacen.values.first
      elsif name == :ultimo
        almacen.values.last
      else
        super
      end
    end

    def respond_to_missing?(name, include_private = false)
      nombre = name.to_s
      nombre.start_with?('buscar_por_') || nombre.start_with?('donde_') ||
        [:todos, :count, :primero, :ultimo].include?(name) || super
    end
  end

  def method_missing(name, *args, &block)
    nombre = name.to_s

    if nombre.end_with?('=')
      super
    elsif @datos&.key?(nombre)
      @datos[nombre]
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    @datos&.key?(name.to_s) || super
  end

  def initialize(atributos = {})
    @datos = {}
    atributos.each do |k, v|
      @datos[k.to_s] = v
    end
    self.class.almacen[self.object_id] = self
  end

  def atributos
    @datos.dup
  end

  def [](clave)
    @datos[clave.to_s]
  end

  def []=(clave, valor)
    @datos[clave.to_s] = valor
  end

  def to_s
    "#{self.class.name}(#{@datos.map { |k, v| "#{k}=#{v}" }.join(', ')})"
  end
end

class UsuarioDinamico
  include PersistenciaDinamica
end
```

```ruby
# Mini ORM con method_missing

class MiniORM
  def self.inherited(subclass)
    subclass.instance_variable_set(:@tabla, subclass.name.downcase + "s")
    subclass.instance_variable_set(:@registros, [])
    subclass.instance_variable_set(:@columnas, [])
  end

  class << self
    attr_accessor :tabla, :columnas

    def columna(nombre, tipo = nil)
      @columnas ||= []
      @columnas << { nombre: nombre, tipo: tipo }
      attr_accessor nombre
    end

    def todos
      @registros || []
    end

    def count
      todos.length
    end

    def primero
      todos.first
    end

    def ultimo
      todos.last
    end

    def crear(atributos)
      nuevo = new(atributos)
      @registros << nuevo
      nuevo
    end

    def method_missing(name, *args, &block)
      nombre = name.to_s

      if nombre.start_with?('donde_')
        campo = nombre.sub('donde_', '')
        valor = args.first
        todos.select { |r| r.send(campo) == valor }
      elsif nombre.start_with?('buscar_por_')
        campo = nombre.sub('buscar_por_', '')
        valor = args.first
        todos.find { |r| r.send(campo) == valor }
      elsif nombre.start_with?('ordenar_por_')
        campo = nombre.sub('ordenar_por_', '')
        direccion = args.first || :asc
        todos.sort { |a, b| direccion == :asc ? a.send(campo) <=> b.send(campo) : b.send(campo) <=> a.send(campo) }
      else
        super
      end
    end

    def respond_to_missing?(name, include_private = false)
      nombre = name.to_s
      nombre.start_with?('donde_') || nombre.start_with?('buscar_por_') ||
        nombre.start_with?('ordenar_por_') || super
    end
  end

  def initialize(atributos = {})
    atributos.each do |k, v|
      send("#{k}=", v) if respond_to?("#{k}=")
    end
  end

  def method_missing(name, *args, &block)
    nombre = name.to_s
    if nombre.end_with?('=')
      instance_variable_set("@#{nombre[0..-2]}", args.first)
    elsif instance_variable_defined?("@#{nombre}")
      instance_variable_get("@#{nombre}")
    else
      super
    end
  end

  def respond_to_missing?(name, include_private = false)
    nombre = name.to_s
    nombre.end_with?('=') || instance_variable_defined?("@#{nombre}") || super
  end

  def to_s
    cols = self.class.columnas.map { |c| "#{c[:nombre]}=#{send(c[:nombre])}" }
    "#{self.class.name}(#{cols.join(', ')})"
  end

  def salvar
    self.class.todos << self unless self.class.todos.include?(self)
    true
  end
end

class ProductoORM < MiniORM
  columna :nombre, String
  columna :precio, Numeric
  columna :categoria, String
  columna :stock, Integer
  columna :activo, [TrueClass, FalseClass]
end

ProductoORM.crear(nombre: "Laptop", precio: 1200, categoria: "Electrónica", stock: 10, activo: true)
ProductoORM.crear(nombre: "Mouse", precio: 25, categoria: "Periféricos", stock: 50, activo: true)

puts "Total: #{ProductoORM.count}"
puts "Primero: #{ProductoORM.primero}"
puts "Donde categoría 'Electrónica':"
ProductoORM.donde_categoria('Electrónica').each { |p| puts "  #{p}" }
puts "Buscar por nombre 'Laptop': #{ProductoORM.buscar_por_nombre('Laptop')}"
puts "Ordenar por precio descendente:"
ProductoORM.ordenar_por_precio(:desc).each { |p| puts "  #{p}" }
```

  RESUMEN DEL CAPÍTULO
────────────────────────────────────────────────────────────────────────────────────

`method_missing` es una herramienta poderosa para metaprogramación:

  1. Se llama cuando un método no se encuentra en la cadena de ancestros.
  2. `respond_to_missing?` debe sobrescribirse junto con `method_missing`.
  3. Permite crear ghost methods (métodos fantasma) dinámicamente.
  4. Es ideal para delegación dinámica a otros objetos.
  5. Permite construir DSLs elegantes y flexibles.
  6. Es la base de `OpenStruct` y objetos dinámicos.
  7. Útil para proxies (caché, seguridad, registro).
  8. El Null Object Pattern se beneficia de `method_missing`.
  9. Peligros: puede ocultar errores y afectar rendimiento.
  10. Siempre llama a `super` si no manejas el método.

────────────────────────────────────────────────────────────────────────────────────
  FAQ — PREGUNTAS FRECUENTES
────────────────────────────────────────────────────────────────────────────────────

P: ¿Siempre debo sobrescribir `respond_to_missing?`?
R: Sí. Si no lo haces, `respond_to?` devolverá false para tus métodos
   fantasma, rompiendo la introspección.

P: ¿Cuál es la diferencia entre `method_missing` y `const_missing`?
R: `method_missing` es para métodos, `const_missing` es para constantes.
   Ambos son hooks de resolución.

P: ¿`method_missing` afecta el rendimiento?
R: Sí, más que los métodos normales porque Ruby recorre toda la cadena
   de ancestros antes de llamarlo. Para código crítico, define métodos
   reales con `define_method`.

P: ¿Puedo encadenar `method_missing`?
R: Sí, puedes llamar a `super` para pasar el mensaje a la cadena de
   ancestros, permitiendo que varios niveles manejen el missing.

P: ¿Cómo evito conflictos entre métodos reales y fantasma?
R: Verifica siempre si el método existe realmente antes de procesarlo
   como fantasma. Usa `respond_to?` o `method_defined?`.

P: ¿Por qué no usar siempre `method_missing`?
R: Debuggeo más difícil, rendimiento inferior, y posibilidad de errores
   silenciosos. Úsalo con moderación y documenta bien.

  DESAFÍO DEL ARQUITECTO

Construye un "Framework de Serialización Dinámica" usando `method_missing`:

  1. Define un módulo `Serializable` que, cuando se incluye, permita
     serializar cualquier objeto a múltiples formatos mediante métodos
     dinámicos como `to_json`, `to_xml`, `to_yaml`, `to_csv`.

  3. Los métodos de serialización deben ser ghost methods generados por
     `method_missing` según el formato solicitado.

  4. Soporta serialización anidada: si un atributo es serializable,
     serialízalo recursivamente.

  5. Implementa `respond_to_missing?` correctamente para que la
     introspección funcione.

  6. Añade opciones de configuración por formato:
     - JSON: pretty_print, only, except, methods
     - XML: root_name, namespace
     - CSV: headers, separator
     - YAML: options

Este desafío integra method_missing, respond_to_missing?, proxies, formato

────────────────────────────────────────────────────────────────────────────────────
  CIERRE DEL CAPÍTULO
────────────────────────────────────────────────────────────────────────────────────

`method_missing` es una de las herramientas más fascinantes de Ruby. Te permite
crear objetos que responden dinámicamente a cualquier mensaje, construir DSLs
elegantes, implementar proxies y delegación, y crear sistemas flexibles que
se adaptan en tiempo de ejecución.

Pero con gran poder viene gran responsabilidad. Úsalo con moderación, siempre
acompañado de `respond_to_missing?`, y documenta claramente qué métodos son
reales y cuáles son fantasma.

El verdadero arte está en saber cuándo usar `method_missing` y cuándo es mejor
definir métodos explícitamente. Ambas herramientas tienen su lugar.
