Vamos a construir una clase `AnalizadorTexto` con los siguientes módulos:

  1. Estadísticas básicas: palabras, caracteres, líneas, oraciones.
  2. Frecuencia de palabras: las palabras más y menos frecuentes.
  3. Distribución de longitudes: cuántas palabras de cada longitud.
  4. Análisis de caracteres: vocales, consonantes, dígitos, especiales.
  5. Detección de patrones: emails, URLs, números de teléfono.
  6. Generación de resumen ejecutivo.
  7. Exportación de resultados.

El proyecto se desarrolla en varias fases, cada una construyendo
sobre la anterior.

Ejemplo 1 — Clase base y constructor

```ruby
class AnalizadorTexto
  attr_reader :texto, :palabras, :lineas, :oraciones

  def initialize(texto)
    @texto = texto
    @palabras = extraer_palabras
    @lineas = extraer_lineas
    @oraciones = extraer_oraciones
  end

  private

  def extraer_palabras
    @texto.downcase.scan(/\b[a-záéíóúüñ]+\b/)
  end

  def extraer_lineas
    @texto.each_line.map(&:chomp)
  end

  def extraer_oraciones
    @texto.scan(/[^.!?]+[.!?]/).map(&:strip)
  end
end

# Prueba
analizador = AnalizadorTexto.new("Ruby es genial. Ruby es poderoso!")
puts analizador.palabras.inspect    # => ["ruby", "es", "genial", "ruby", "es", "poderoso"]
puts analizador.lineas.length       # => 1
puts analizador.oraciones.inspect   # => ["Ruby es genial.", "Ruby es poderoso!"]
```

```ruby
class AnalizadorTexto
  def frecuencia_palabras
    @palabras.each_with_object(Hash.new(0)) { |p, h| h[p] += 1 }
             .sort_by { |_, v| -v }
             .to_h
  end

  def palabras_top(n = 10)
    frecuencia_palabras.first(n).to_h
  end

  def palabras_bottom(n = 10)
    frecuencia_palabras.sort_by { |_, v| v }.first(n).to_h
  end

  def palabras_por_frecuencia(min = 2)
    frecuencia_palabras.select { |_, count| count >= min }
  end
end

texto = "ruby python ruby java ruby python javascript java ruby python"
analizador = AnalizadorTexto.new(texto)

puts analizador.palabras_top(3).inspect
```

Ejemplo 4 — Distribución de longitudes

```ruby
class AnalizadorTexto
  def distribucion_longitudes
    @palabras.group_by(&:length)
             .transform_values(&:count)
             .sort_by { |long, _| long }
             .to_h
  end

  def palabras_por_longitud(longitud)
    @palabras.select { |p| p.length == longitud }.uniq.sort
  end

  def longitud_mas_comun
    distribucion_longitudes.max_by { |_, count| count }
  end
end

texto = "Ruby es un lenguaje de programación muy poderoso y elegante"
analizador = AnalizadorTexto.new(texto)
puts analizador.distribucion_longitudes.inspect
# => {2=>2, 3=>1, 4=>1, 5=>2, 7=>1, 9=>1, 11=>1}
puts analizador.longitud_mas_comun.inspect
# => [2, 2] o [5, 2] (depende del desempate)
```

```ruby
class AnalizadorTexto
  def analisis_caracteres
    chars = @texto.downcase.chars

    {
      vocales: chars.count { |c| "aeiouáéíóúü".include?(c) },
      consonantes: chars.count { |c| c.match?(/[a-zñ]/) && !"aeiouáéíóúü".include?(c) },
      digitos: chars.count { |c| c.match?(/\d/) },
      espacios: chars.count { |c| c.match?(/\s/) },
      puntuacion: chars.count { |c| c.match?(/[[:punct:]]/) },
      otros: chars.count { |c| c.match?(/[^\w\s]/) && !c.match?(/[[:punct:]]/) }
    }
  end

  def frecuencia_caracteres
    @texto.downcase
          .gsub(/\s+/, "")
          .chars
          .each_with_object(Hash.new(0)) { |c, h| h[c] += 1 }
          .sort_by { |_, v| -v }
          .to_h
  end
end

analizador = AnalizadorTexto.new("Ruby 2024!!!")
puts analizador.analisis_caracteres.inspect
# => {:vocales=>1, :consonantes=>3, :digitos=>4, :espacios=>1, :puntuacion=>3, :otros=>0}
```

```ruby
class AnalizadorTexto
  def emails
    @texto.scan(/[\w.]+@[\w.]+\.\w+/)
  end

  def urls
    @texto.scan(%r{https?://[^\s]+})
  end

  def telefonos
    @texto.scan(/\d{3}[-.\s]?\d{4}/)
  end

  def hashtags
    @texto.scan(/#\w+/)
  end

  def menciones
    @texto.scan(/@\w+/)
  end

  def patrones_detectados
    {
      emails: emails,
      urls: urls,
      telefonos: telefonos,
      hashtags: hashtags,
      menciones: menciones
    }.reject { |_, v| v.empty? }
  end
end

texto = "Contacto: ana@mail.com Web: https://ruby.org Tel: 123-4567 #ruby @usuario"
analizador = AnalizadorTexto.new(texto)
puts analizador.patrones_detectados.inspect
# => {:emails=>["ana@mail.com"], :urls=>["https://ruby.org"],
#     :telefonos=>["123-4567"], :hashtags=>["#ruby"], :menciones=>["@usuario"]}
```

```ruby
puts "\nPalabras de 10 letras o más:"
puts analizador.palabras_por_longitud(10).join(", ")
```

```ruby
module ExtensionesAnalizador
  # Análisis de sentimiento básico
  def sentimiento
    positivas = %w(bueno excelente genial increíble fantástico hermoso
                   maravilloso perfecto útil interesante)
    negativas = %w(malo terrible horrible pésimo feo aburrido difícil
                   error fallo problema mal)

    positivas_encontradas = @palabras & positivas
    negativas_encontradas = @palabras & negativas

    if positivas_encontradas.length > negativas_encontradas.length
      :positivo
    elsif negativas_encontradas.length > positivas_encontradas.length
      :negativo
    else
      :neutral
    end
  end

  # Palabras clave (excluyendo stop words)
  def palabras_clave
    stop_words = %w(el la los las de del en un una y e o a al con por
                    para es se su como más pero le lo que al sus este esta)
    @palabras.reject { |p| stop_words.include?(p) }
             .each_with_object(Hash.new(0)) { |p, h| h[p] += 1 }
             .sort_by { |_, v| -v }
             .to_h
  end

  # N-gramas (secuencias de n palabras consecutivas)
  def ngramas(n = 2)
    @palabras.each_cons(n).map { |ng| ng.join(" ") }
             .each_with_object(Hash.new(0)) { |ng, h| h[ng] += 1 }
             .sort_by { |_, v| -v }
             .to_h
  end

  # Densidad de vocabulario (palabras únicas / total palabras)
  def densidad_vocabulario
    return 0 if @palabras.empty?
    (@palabras.uniq.length.to_f / @palabras.length).round(4)
  end

  # Legibilidad (índice Flesch simplificado)
  def legibilidad
    num_palabras = @palabras.length
    num_oraciones = @oraciones.length
    num_silabas = @palabras.sum { |p| contar_silabas(p) }

    return 0 if num_oraciones == 0 || num_palabras == 0

    puntuacion = 206.835 - 1.015 * (num_palabras / num_oraciones) -
                 84.6 * (num_silabas / num_palabras)

    case puntuacion
    when 90..100 then :muy_facil
    when 80..89  then :facil
    when 70..79  then :bastante_facil
    when 60..69  then :normal
    when 50..59  then :bastante_dificil
    when 30..49  then :dificil
    else :muy_dificil
    end
  end

  private

  def contar_silabas(palabra)
    palabra.downcase.scan(/[aeiouáéíóúü]+/).length
  end
end

# Incluir las extensiones en la clase
AnalizadorTexto.include(ExtensionesAnalizador)
```

                     FAQ — PREGUNTAS FRECUENTES
═══════════════════════════════════════════════════════════════════════════════════

P: ¿Cómo ejecuto el analizador con un archivo?
R: ruby analizador.rb archivo.txt o cat archivo.txt | ruby analizador.rb.

P: ¿Por qué usamos downcase en las palabras?
R: Para normalizar y contar "Ruby" y "ruby" como la misma palabra.

P: ¿Cómo puedo ignorar palabras comunes (stop words)?
R: Define una lista de stop words y usa reject en el array de palabras.

P: ¿El analizador maneja texto en otros idiomas?
R: Sí, ajustando el patrón de scan de palabras. Para inglés, funciona
igual. Para otros alfabetos, ajusta las clases de caracteres.

P: ¿Cómo procesaría un archivo de 1GB?
R: Usa lazy: File.foreach("archivo.txt").lazy y procesa línea por línea
sin cargar todo en memoria.

P: ¿Por qué algunas palabras no se detectan correctamente?
R: El patrón \b[a-záéíóúüñ]+\b puede no capturar palabras con guiones
o apóstrofes. Ajusta el regex según necesites.

P: ¿Cómo cuento palabras compuestas como "radio-reloj"?
R: Modifica el regex: /[\w-]+/ para incluir guiones.

P: ¿El resumen ejecutivo es configurable?
R: Puedes modificar to_s o crear métodos específicos para diferentes
formatos de reporte.
