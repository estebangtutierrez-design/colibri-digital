# app/models/producto.rb
class Producto < ApplicationRecord
  # == Asociaciones ==
  belongs_to :usuario
  has_many :items_pedido, dependent: :destroy
  has_many :pedidos, through: :items_pedido

  # == Validaciones ==
  validates :nombre, presence: true, length: { maximum: 100 }
  validates :precio, presence: true,
                     numericality: { greater_than: 0 }
  validates :stock, numericality: { greater_than_or_equal_to: 0 }

  # == Scopes ==
  scope :activos, -> { where(activo: true) }
  scope :por_categoria, ->(cat) { where(categoria: cat) }
  scope :baratos, -> { where(precio: ..100) }
  scope :caros, -> { where(precio: 1000..) }
  scope :disponibles, -> { activos.where('stock > 0') }

  # == Callbacks ==
  before_save :normalizar_nombre
  after_create :notificar_nuevo_producto

  # == Métodos de instancia ==
  def precio_con_impuesto(tasa = 0.16)
    precio * (1 + tasa)
  end

  def disponible?
    activo? && stock.positive?
  end

  def descuento_aplicado?
    descuento&.positive?
  end

  def precio_final
    descuento_aplicado? ? precio * (1 - descuento / 100.0) : precio
  end

  private

  def normalizar_nombre
    self.nombre = nombre.strip.titleize if nombre_changed?
  end

  def notificar_nuevo_producto
    ProductoMailer.nuevo_producto(self).deliver_later
  end
end

# app/models/usuario.rb
class Usuario < ApplicationRecord
  has_secure_password
  has_many :productos, dependent: :destroy

  validates :email, presence: true,
                    uniqueness: { case_sensitive: false },
                    format: { with: URI::MailTo::EMAIL_REGEXP }
  validates :nombre, presence: true

  scope :admins, -> { where(admin: true) }

  def admin?
    admin
  end

  def to_s
    nombre
  end
end

# app/models/pedido.rb
class Pedido < ApplicationRecord
  belongs_to :usuario
  has_many :items_pedido, dependent: :destroy
  has_many :productos, through: :items_pedido

  validates :estado, inclusion: { in: %w[pendiente pagado enviado entregado cancelado] }

  scope :recientes, -> { order(created_at: :desc) }
  scope :pendientes, -> { where(estado: 'pendiente') }

  def total
    items_pedido.sum(&:subtotal)
  end

  def completar!
    update!(estado: 'entregado', completado_en: Time.current)
  end
end

# app/models/item_pedido.rb
class ItemPedido < ApplicationRecord
  belongs_to :pedido
  belongs_to :producto

  validates :cantidad, numericality: { greater_than: 0 }

  def subtotal
    producto.precio * cantidad
  end
end

# app/controllers/productos_controller.rb
class ProductosController < ApplicationController
  before_action :set_producto, only: %i[show edit update destroy]
  before_action :authenticate_user!, except: %i[index show]

  # GET /productos
  # GET /productos?categoria=electronica
  def index
    @productos = Producto.disponibles

    if params[:categoria].present?
      @productos = @productos.por_categoria(params[:categoria])
    end

    if params[:busqueda].present?
      @productos = @productos.where(
        'nombre LIKE ?', "%#{params[:busqueda]}%"
      )
    end

    @productos = @productos.page(params[:page]).per(12)
  end

  # GET /productos/1
  def show
    @producto = Producto.find(params[:id])
    @productos_relacionados = Producto.por_categoria(@producto.categoria)
                                      .where.not(id: @producto.id)
                                      .limit(4)
  end

  # GET /productos/new
  def new
    @producto = Producto.new
  end

  # GET /productos/1/edit
  def edit
  end

  # POST /productos
  def create
    @producto = current_user.productos.build(producto_params)

    if @producto.save
      redirect_to @producto, notice: 'Producto creado exitosamente.'
    else
      render :new, status: :unprocessable_entity
    end
  end

  # PATCH/PUT /productos/1
  def update
    if @producto.update(producto_params)
      redirect_to @producto, notice: 'Producto actualizado.'
    else
      render :edit, status: :unprocessable_entity
    end
  end

  # DELETE /productos/1
  def destroy
    @producto.destroy
    redirect_to productos_url, notice: 'Producto eliminado.'
  end

  private

  def set_producto
    @producto = Producto.find(params[:id])
  rescue ActiveRecord::RecordNotFound
    redirect_to productos_url, alert: 'Producto no encontrado.'
  end

  def producto_params
    params.require(:producto).permit(
      :nombre, :descripcion, :precio, :stock,
      :categoria, :descuento, :imagen
    )
  end
end

# app/controllers/application_controller.rb
class ApplicationController < ActionController::Base
  before_action :configurar_idioma
  before_action :cargar_notificaciones, if: :user_signed_in?

  private

  def configurar_idioma
    I18n.locale = params[:locale] || session[:locale] || I18n.default_locale
  end

  def cargar_notificaciones
    @notificaciones = current_user.notificaciones.recientes.limit(5)
  end
end

Rails.application.routes.draw do
  # == Rutas RESTful ==
  resources :productos do
    # Rutas anidadas
    resources :comentarios, only: %i[create destroy]

    # Rutas de colección
    collection do
      get :buscar
      get :baratos
    end

    # Rutas de miembro
    member do
      post :comprar
      patch :activar
    end
  end

  resources :usuarios, only: %i[show edit update]
  resources :pedidos, only: %i[index show create] do
    member do
      patch :cancelar
    end
  end

  # == Rutas con nombres personalizados ==
  get '/carrito', to: 'carritos#show', as: :carrito
  post '/carrito/agregar', to: 'carritos#agregar', as: :agregar_al_carrito
  delete '/carrito/eliminar/:id', to: 'carritos#eliminar', as: :eliminar_del_carrito

  # == Autenticación ==
  get  '/login',  to: 'sesiones#new', as: :login
  post '/login',  to: 'sesiones#create'
  delete '/logout', to: 'sesiones#destroy', as: :logout

  get  '/registro', to: 'registros#new', as: :registro
  post '/registro', to: 'registros#create'

  # == Páginas estáticas ==
  get  '/acerca',    to: 'paginas#acerca'
  get  '/contacto',  to: 'paginas#contacto'
  post '/contacto',  to: 'paginas#enviar_contacto'

  # == Root ==
  root 'productos#index'

  # == API (namespaces) ==
  namespace :api do
    namespace :v1 do
      resources :productos, only: %i[index show]
      resources :usuarios, only: %i[show]
    end
  end

  # == Rails health check ==
  get 'up' => 'rails/health#show', as: :rails_health_check
end

# app/helpers/productos_helper.rb
module ProductosHelper
  def producto_stock_label(producto)
    if producto.stock.zero?
      content_tag(:span, 'Agotado', class: 'badge badge-danger')
    elsif producto.stock < 10
      content_tag(:span, "Últimas #{producto.stock} unidades", class: 'badge badge-warning')
    else
      content_tag(:span, 'En existencia', class: 'badge badge-success')
    end
  end

  def producto_precio_formateado(producto)
    if producto.descuento&.positive?
      content_tag(:span) do
        concat content_tag(:del, number_to_currency(producto.precio))
        concat ' '
        concat content_tag(:strong, number_to_currency(producto.precio_final), class: 'text-danger')
        concat content_tag(:span, " (-#{producto.descuento}%)", class: 'descuento-label')
      end
    else
      number_to_currency(producto.precio)
    end
  end
end

class ProductoMailer < ApplicationMailer
  def nuevo_producto(producto)
    @producto = producto
    @usuario  = producto.usuario

    mail(
      to: @usuario.email,
      subject: "Producto creado: #{@producto.nombre}"
    )
  end

  def producto_agotado(producto)
    @producto = producto
    @admin = Usuario.admins.first

    mail(
      to: @admin.email,
      subject: "Producto agotado: #{@producto.nombre}"
    )
  end
end

class ActualizarPreciosJob < ApplicationJob
  queue_as :default

  def perform(categoria, porcentaje_incremento)
    productos = Producto.por_categoria(categoria)
    productos.find_each do |producto|
      producto.update!(precio: producto.precio * (1 + porcentaje_incremento / 100.0))
    end

    AdminMailer.precios_actualizados(categoria, productos.count).deliver_now
  end
end

# def index
#   @productos = Producto.disponibles.ordenados
# end

# ============================================================
# 2. Usar Service Objects para lógica compleja
# ============================================================
# Mal en controlador:
# def create
#   # 30 líneas de lógica de negocio
# end

# Bien:

# end

# ============================================================
# 3. Rutas RESTful siempre que sea posible
# ============================================================
# resources :productos en lugar de rutas personalizadas

# ============================================================
# 4. Migraciones: una tabla por migración
# ============================================================
# No mezcles cambios no relacionados

# ============================================================
# 5. Usar partials y helpers para vistas limpias
# ============================================================

# ============================================================
# 6. NO usar callbacks para lógica de negocio
# ============================================================
# Mal: after_create :enviar_email
# Bien: llamar al servicio explícitamente en el controlador

# ============================================================
# 7. Siempre usar I18n para textos
# ============================================================
# t('.title') en lugar de texto duro

RESUMEN DEL CAPÍTULO
─────────────────────
- Rails sigue el patrón MVC con Convención sobre Configuración.
- `rails new` genera toda la estructura del proyecto.
- Las migraciones definen el esquema de base de datos en Ruby.
- Active Record proporciona ORM, validaciones y asociaciones.
- Los controladores manejan peticiones HTTP y coordinan la lógica.
- Las rutas mapean URLs a acciones de controladores.
- Las vistas (ERB) presentan datos al usuario.
- Los helpers contienen lógica de presentación reutilizable.
- Mailers envían correos; Jobs ejecutan tareas asíncronas.
- Rails prioriza la felicidad del desarrollador y la productividad.

PREGUNTAS FRECUENTES (FAQ)
───────────────────────────
P: ¿Rails es todavía relevante en 2026?
R: Absolutamente. Rails sigue siendo líder en desarrollo web rápido,
   con una comunidad activa, gran ecosistema y uso en empresas como
   GitHub, Shopify y Basecamp.

P: ¿Rails API o Rails completo?
R: Si solo construyes backend, usa `--api`. Si necesitas vistas,
   usa Rails completo. Puedes mezclar ambos enfoques.

P: ¿Cómo elijo entre migraciones o schema.rb?
R: Las migraciones son el historial. schema.rb es el estado actual.
   Siempre usa migraciones para modificar la base de datos.

P: ¿Debo usar scaffolds?
R: Para prototipos, sí. Para producción, prefiere generar modelos
   y controladores por separado para tener más control.
