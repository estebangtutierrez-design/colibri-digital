ESTRUCTURA DE UN PROYECTO RUBY PROFESIONAL
────────────────────────────────────────────

Un proyecto Ruby profesional sigue una convención de directorios que
facilita la navegación, el testing y el despliegue. Veamos la
estructura canónica:

```
mi_app/
├── Gemfile
├── Gemfile.lock
├── README.md
├── Rakefile
├── config/
│   ├── application.rb
│   ├── database.yml
│   └── routes.rb
├── app/
│   ├── controllers/
│   ├── models/
│   ├── views/
│   ├── services/
│   ├── concerns/
│   ├── serializers/
│   └── validators/
├── lib/
│   ├── mi_app.rb
│   └── tasks/
├── db/
│   └── migrations/
├── spec/
│   ├── spec_helper.rb
│   ├── models/
│   ├── controllers/
│   └── services/
├── public/
└── bin/
    └── setup
```

Cada carpeta tiene una responsabilidad única:
- app/: Núcleo de la aplicación
- config/: Configuración global
- lib/: Código reutilizable no específico de la app
- spec/: Pruebas (RSpec)
- db/: Migraciones y esquemas
- public/: Archivos estáticos
- bin/: Scripts ejecutables

CREANDO UN PROYECTO DESDE CERO
───────────────────────────────

Vamos a construir la estructura base para una tienda en línea. Este
proyecto nos acompañará como referencia durante todo el capítulo.

Paso 1: Crear la estructura de directorios

```ruby
#!/usr/bin/env ruby
# setup_project.rb — Genera la estructura base del proyecto

require 'fileutils'

module ProjectGenerator
  STRUCTURE = %w[
    app/controllers
    app/models
    app/views
    app/services
    app/concerns
    app/serializers
    app/validators
    config
    db/migrations
    lib/tasks
    spec/models
    spec/controllers
    spec/services
    spec/support
    public
    bin
  ].freeze

  def self.call(project_name)
    raise 'Nombre inválido' unless project_name.match?(/^[a-z_]+$/)

    STRUCTURE.each do |dir|
      FileUtils.mkdir_p("#{project_name}/#{dir}")
    end

    create_gemfile(project_name)
    create_rakefile(project_name)
    create_readme(project_name)

    puts "✓ Proyecto #{project_name} creado exitosamente"
  end

  def self.create_gemfile(project_name)
    File.write("#{project_name}/Gemfile", <<~RUBY)
      source 'https://rubygems.org'

      gem 'sinatra'
      gem 'activerecord'
      gem 'sqlite3'
      gem 'puma'
      gem 'rake'

      group :test do
        gem 'rspec'
        gem 'factory_bot'
        gem 'database_cleaner'
      end

      group :development do
        gem 'rubocop'
        gem 'pry'
      end
    RUBY
  end

  def self.create_rakefile(project_name)
    File.write("#{project_name}/Rakefile", <<~RUBY)
      require 'rspec/core/rake_task'
      RSpec::Core::RakeTask.new(:spec)
      task default: :spec
    RUBY
  end

  def self.create_readme(project_name)
    File.write("#{project_name}/README.md", "# #{project_name}\n\nProyecto profesional Ruby\n")
  end
end

ProjectGenerator.call('mi_tienda')
```

EL PATRÓN MVC EN RUBY
──────────────────────

MVC (Model-View-Controller) separa la aplicación en tres capas:
- Model: Datos y lógica de negocio (ActiveRecord, Sequel)
- View: Presentación (ERB, Haml, Slim)
- Controller: Coordina la interacción usuario-sistema

Implementación manual de MVC sin frameworks:

```ruby
class Producto
  attr_accessor :id, :nombre, :precio, :stock

  def initialize(attributes = {})
    @id     = attributes[:id]
    @nombre = attributes[:nombre]
    @precio = attributes[:precio]
    @stock  = attributes[:stock] || 0
```

```ruby
class ProductosController
  def initialize(repositorio)
    @repositorio = repositorio
  end

  def index
    productos = @repositorio.todos
    render(:index, productos: productos)
  end

  def show(id)
    producto = @repositorio.buscar(id)
    if producto
      render(:show, producto: producto)
    else
      render(:not_found, mensaje: "Producto #{id} no encontrado")
    end
  end

  def create(params)
    producto = Producto.new(params)
    if producto.nombre && producto.precio
      @repositorio.guardar(producto)
      render(:show, producto: producto, notice: 'Creado exitosamente')
    else
      render(:new, producto: producto, error: 'Datos inválidos')
    end
  end

  private

  def render(template, locals = {})
    view = ViewRenderer.new(template, locals)
    view.render
  end
end
```

```ruby
  def initialize(template, locals = {})
    @template = template
    @locals   = locals
  end

  def render
    plantilla = File.read("app/views/#{@template}.html.erb")
    @locals.each do |key, value|
      instance_variable_set("@#{key}", value)
    end
    ERB.new(plantilla).result(binding)
  rescue Errno::ENOENT
    "<h1>404 — Vista no encontrada</h1>"
  end
end
```

```ruby
# config/routes.rb
class Router
  def initialize
    @routes = []
  end

  def get(path, to:)
    @routes << { method: 'GET', path: path, controller: to[:controller],
                 action: to[:action] }
  end

  def post(path, to:)
    @routes << { method: 'POST', path: path, controller: to[:controller],
                 action: to[:action] }
  end

  def resolve(method, path)
    @routes.find { |r| r[:method] == method && r[:path] == path }
  end
end

# config/application.rb
require_relative 'routes'
require_relative '../app/controllers/productos_controller'
require_relative '../app/models/producto'
require_relative '../app/views/view_renderer'

class Application
  def initialize
    @router = Router.new
    configurar_rutas
    @repositorio = RepositorioProductos.new
  end

  def call(env)
    metodo  = env['REQUEST_METHOD']
    ruta    = env['PATH_INFO']
    ruta_match = @router.resolve(metodo, ruta)

    if ruta_match
      controller = Object.const_get(ruta_match[:controller]).new(@repositorio)
      controller.send(ruta_match[:action])
    else
      [404, { 'Content-Type' => 'text/html' }, ['<h1>404 Not Found</h1>']]
    end
  end

  private

  def configurar_rutas
    @router.get '/productos', to: { controller: 'ProductosController', action: :index }
    @router.get '/productos/:id', to: { controller: 'ProductosController', action: :show }
    @router.post '/productos', to: { controller: 'ProductosController', action: :create }
  end
end
```

SERVICE OBJECTS
────────────────

Los Service Objects encapsulan una operación de negocio completa.
Son objetos con una única responsabilidad: ejecutar un proceso.

```ruby
# app/services/registrar_usuario.rb
class RegistrarUsuario
  RESULT = Struct.new(:success?, :usuario, :errors, keyword_init: true)

  def initialize(repositorio_usuarios, notificador)
    @repositorio = repositorio_usuarios
    @notificador = notificador
  end

  def call(params)
    errors = validar(params)
    return RESULT.new(success?: false, errors: errors) if errors.any?

    usuario = Usuario.new(
      nombre: params[:nombre],
      email:  params[:email],
      admin:  false
    )

    @repositorio.guardar(usuario)
    @notificador.enviar_bienvenida(usuario)

    RESULT.new(success?: true, usuario: usuario, errors: [])
  rescue StandardError => e
    RESULT.new(success?: false, errors: ["Error interno: #{e.message}"])
  end

  private

  def validar(params)
    errores = []
    errores << 'Nombre requerido' if params[:nombre].to_s.strip.empty?
    errores << 'Email requerido'  if params[:email].to_s.strip.empty?
    errores << 'Email inválido'   unless params[:email]&.include?('@')
    errores << 'Email duplicado'  if @repositorio.existe_email?(params[:email])
    errores
  end
end
```

```ruby
# app/services/procesar_pago.rb
class ProcesarPago
  RESULT = Struct.new(:success?, :transaccion, :error, keyword_init: true)

  def initialize(pasarela_pago, repositorio_pedidos)
    @pasarela  = pasarela_pago
    @pedidos   = repositorio_pedidos
  end

  def call(pedido_id, datos_tarjeta)
    pedido = @pedidos.buscar(pedido_id)
    return RESULT.new(success?: false, error: 'Pedido no encontrado') unless pedido

    unless pedido.items.any?
      return RESULT.new(success?: false, error: 'Pedido vacío')
    end

    resultado = @pasarela.cargar(
      monto:        pedido.total,
      tarjeta:      datos_tarjeta[:numero],
      vencimiento:  datos_tarjeta[:vencimiento],
      cvv:          datos_tarjeta[:cvv]
    )

    if resultado.exitosa?
      pedido.marcar_como_pagado!(resultado.transaccion_id)
      RESULT.new(success?: true, transaccion: resultado.transaccion_id)
    else
      RESULT.new(success?: false, error: resultado.mensaje_error)
    end
  end
end
```

```ruby
# app/services/calcular_descuento.rb
class CalcularDescuento
  REGLAS = [
    ->(carrito) { carrito.total > 1000 && carrito.items.size > 5 ? 0.15 : 0 },
    ->(carrito) { carrito.usuario&.admin? ? 0.10 : 0 },
    ->(carrito) { carrito.codigo_promocional == 'RUBY10' ? 0.10 : 0 }
  ].freeze

  def call(carrito)
    descuento_maximo = REGLAS.map { |regla| regla.call(carrito) }.max
    total = carrito.total * (1 - descuento_maximo)
    { descuento_aplicado: descuento_maximo, total_final: total.round(2) }
  end
end
```

CONCERNS (MÓDULOS COMPARTIDOS)
───────────────────────────────

Los Concerns son módulos que agrupan comportamiento reusable.
Rails los usa extensivamente, pero podemos implementarlos en
cualquier proyecto Ruby.

```ruby
# app/concerns/auditable.rb
module Auditable
  def self.included(base)
    base.extend(ClassMethods)
  end

  module ClassMethods
    def auditar_atributos(*attrs)
      @atributos_auditados = attrs
    end

    def atributos_auditados
      @atributos_auditados || []
    end
  end

  def cambios_auditados
    cambios = {}
    self.class.atributos_auditados.each do |attr|
      valor_anterior = send("__previous_#{attr}") if respond_to?("__previous_#{attr}")
      valor_actual   = send(attr)
      if valor_anterior != valor_actual
        cambios[attr] = { de: valor_anterior, a: valor_actual }
      end
    end
    cambios
  end

  def registrar_auditoria(usuario)
    cambios = cambios_auditados
    return if cambios.empty?

    Auditoria.create!(
      auditable:  self,
      usuario:    usuario,
      cambios:    cambios,
      ocurrido_en: Time.now
    )
  end
end
```

```ruby
# app/concerns/togglable.rb
module Togglable
  def activar!
    @activo = true
    on_activacion if respond_to?(:on_activacion)
  end

  def desactivar!
    @activo = false
    on_desactivacion if respond_to?(:on_desactivacion)
  end

  def activo?
    @activo
  end

  def toggle!
    activo? ? desactivar! : activar!
  end
end
```

```ruby
# app/concerns/serializable.rb
module Serializable
  def to_hash
    instance_variables.each_with_object({}) do |var, hash|
      key = var.to_s.delete('@').to_sym
      hash[key] = instance_variable_get(var)
    end
  end

  def to_json(*args)
    to_hash.to_json(*args)
  end

  def to_xml
    hash = to_hash
    builder = Nokogiri::XML::Builder.new(encoding: 'UTF-8') do |xml|
      xml.send(self.class.name.downcase) do
        hash.each do |key, value|
          xml.send(key, value)
        end
      end
    end
    builder.to_xml
  end
end
```

```ruby
# app/models/repositorio_productos.rb
class RepositorioProductos
  def initialize(almacen = nil)
    @almacen = almacen || []
    @next_id = 1
  end

  def todos
    @almacen
  end

  def buscar(id)
    @almacen.find { |p| p.id == id }
  end

  def buscar_por_nombre(nombre)
    @almacen.select { |p| p.nombre&.downcase&.include?(nombre.downcase) }
  end

  def disponibles
    @almacen.select(&:disponible?)
  end

  def guardar(producto)
    if producto.id.nil?
      producto.id = @next_id
      @next_id += 1
      @almacen << producto
    else
      indice = @almacen.index { |p| p.id == producto.id }
      @almacen[indice] = producto if indice
    end
    producto
  end

  def eliminar(id)
    @almacen.reject! { |p| p.id == id }
  end

  def contar
    @almacen.size
  end

  def existe_email?(email)
    false # implementación para usuarios
  end
end
```

```ruby
# app/services/importador_productos.rb
class ImportadorProductos
  def initialize(archivo, repositorio)
    @archivo     = archivo
    @repositorio = repositorio
  end

  def call
    datos = parsear_archivo
    resultados = { importados: 0, fallidos: [], errores: [] }

    datos.each_with_index do |fila, index|
      begin
        producto = Producto.new(
          nombre: fila[:nombre],
          precio: fila[:precio].to_f,
          stock:  fila[:stock].to_i
        )
        @repositorio.guardar(producto)
        resultados[:importados] += 1
      rescue StandardError => e
        resultados[:fallidos] << index + 1
        resultados[:errores] << e.message
      end
    end

    resultados
  end

  private

  def parsear_archivo
    extension = File.extname(@archivo)
    case extension
    when '.csv'  then parsear_csv
    when '.json' then parsear_json
    else raise "Formato #{extension} no soportado"
    end
  end

  def parsear_csv
    require 'csv'
    CSV.read(@archivo, headers: true).map(&:to_h).map { |h| h.transform_keys(&:to_sym) }
  rescue => e
    raise "Error al leer CSV: #{e.message}"
  end

  def parsear_json
    require 'json'
    JSON.parse(File.read(@archivo), symbolize_names: true)
  rescue => e
    raise "Error al leer JSON: #{e.message}"
  end
end
```

CADENA DE RESPONSABILIDAD (CHAIN OF RESPONSIBILITY)
────────────────────────────────────────────────────

```ruby
# app/services/middleware/autenticacion.rb
module Middleware
  class Base
    def initialize(next_middleware = nil)
      @next = next_middleware
    end

    def call(request)
      raise NotImplementedError
    end

    protected

    def pasar_siguiente(request)
      @next&.call(request)
    end
  end

  class Autenticacion < Base
    def call(request)
      if request[:token] == 'secreto'
        request[:usuario] = { id: 1, nombre: 'Admin' }
        pasar_siguiente(request)
      else
        { error: 'No autenticado', status: 401 }
      end
    end
  end

  class Autorizacion < Base
    def call(request)
      if request[:usuario] && request[:accion] == 'admin'
        { error: 'No autorizado', status: 403 }
      else
        pasar_siguiente(request)
      end
    end
  end

  class Logger < Base
    def call(request)
      puts "[LOG] #{Time.now} — #{request[:accion]} por #{request[:usuario]&.dig(:nombre)}"
      pasar_siguiente(request)
    end
  end

  class Ejecutor < Base
    def call(request)
      { status: 200, mensaje: "Ejecutando #{request[:accion]}" }
    end
  end
end

# Uso:
```

OBSERVER (PUBLISH-SUBSCRIBE)
─────────────────────────────

```ruby
# app/services/evento_bus.rb
class EventoBus
  def initialize
    @suscriptores = Hash.new { |h, k| h[k] = [] }
  end

  def suscribir(evento, &bloque)
    @suscriptores[evento] << bloque
  end

  def emitir(evento, *args)
    (@suscriptores[evento] || []).each { |s| s.call(*args) }
  end
end

# app/services/notificador_pedido.rb
class NotificadorPedido
  def initialize(bus:)
    @bus = bus
    configurar
  end

  def configurar
    @bus.suscribir(:pedido_creado) do |pedido|
      puts "Enviando confirmación por email para pedido #{pedido.id}"
    end

    @bus.suscribir(:pedido_enviado) do |pedido|
      puts "Notificando tracking #{pedido.tracking} al cliente"
    end

    @bus.suscribir(:pedido_entregado) do |pedido|
      puts "Solicitando review del producto al cliente"
    end
  end
end
```

  aplicaciones Ruby complejas.

PREGUNTAS FRECUENTES (FAQ)
───────────────────────────
P: ¿Debo siempre usar Service Objects?
R: No. Úsalos cuando la lógica cruce múltiples modelos o requiera
   coordinación compleja. Para operaciones simples, el modelo basta.

P: ¿Cuándo usar Concerns vs herencia?
R: Concerns para compartir comportamiento no jerárquico. Herencia
   para especialización (e.g., `Admin < Usuario`).

P: ¿Cómo evito que los Service Objects crezcan demasiado?
R: Aplica SRP estrictamente. Si un servicio hace más de una cosa,
   divídelo en servicios más pequeños y orquesta con un coordinador.

P: ¿Es necesario usar namespaces en proyectos pequeños?
R: En proyectos pequeños no es crítico, pero es buena práctica.
   Nunca sabes cuándo el proyecto crecerá.
