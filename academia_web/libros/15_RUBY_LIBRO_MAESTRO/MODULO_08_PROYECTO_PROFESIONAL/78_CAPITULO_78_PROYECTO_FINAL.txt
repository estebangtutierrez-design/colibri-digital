# lib/task_manager/task.rb
module TaskManager
  class Task
    ATTRIBUTES = %i[id title description status priority created_at
                    completed_at].freeze

    VALID_STATUSES = %w[pending in_progress completed cancelled].freeze
    VALID_PRIORITIES = %w[low medium high critical].freeze

    attr_accessor :id, :title, :description, :status, :priority,
                  :created_at, :completed_at

    def initialize(attributes = {})
      @id          = attributes[:id] || SecureRandom.uuid
      @title       = attributes[:title].to_s.strip
      @description = attributes[:description].to_s.strip
      @status      = attributes[:status] || 'pending'
      @priority    = attributes[:priority] || 'medium'
      @created_at  = attributes[:created_at] || Time.now.iso8601
      @completed_at = attributes[:completed_at]
    end

    def valid?
      return false if title.empty?
      return false unless VALID_STATUSES.include?(status)
      return false unless VALID_PRIORITIES.include?(priority)
      true
    end

    def validate!
      raise InvalidTask, 'El título no puede estar vacío' if title.empty?
      unless VALID_STATUSES.include?(status)
        raise InvalidTask, "Estado inválido: #{status}"
      end
      unless VALID_PRIORITIES.include?(priority)
        raise InvalidTask, "Prioridad inválida: #{priority}"
      end
    end

    def complete!
      @status = 'completed'
      @completed_at = Time.now.iso8601
    end

    def completed?
      @status == 'completed'
    end

    def pending?
      @status == 'pending'
    end

    def in_progress?
      @status == 'in_progress'
    end

    def to_h
      ATTRIBUTES.each_with_object({}) do |attr, hash|
        hash[attr] = send(attr)
      end
    end

    def to_s
      "[#{status.upcase}] #{title} (#{priority})"
    end

    def <=>(other)
      priority_order = %w[critical high medium low]
      cmp = priority_order.index(priority).to_i <=>
            priority_order.index(other.priority).to_i
      cmp == 0 ? created_at <=> other.created_at : cmp
    end
  end
end

# lib/task_manager/repository.rb
module TaskManager
  class Repository
    DEFAULT_PATH = File.expand_path('~/.task_manager/tasks.json')

    def initialize(path = DEFAULT_PATH)
      @path = path
      @tasks = []
      ensure_directory!
      load_tasks
    end

    def all
      @tasks.dup
    end

    def find(id)
      @tasks.find { |t| t.id == id } || raise(TaskNotFound, id)
    end

    def where(status: nil, priority: nil)
      result = @tasks.dup
      result = result.select { |t| t.status == status } if status
      result = result.select { |t| t.priority == priority } if priority
      result
    end

    def add(task)
      task.validate!
      @tasks << task
      save_tasks
      task
    end

    def update(id, attributes)
      task = find(id)
      attributes.each do |key, value|
        task.send("#{key}=", value) if Task::ATTRIBUTES.include?(key)
      end
      task.validate!
      save_tasks
      task
    end

    def delete(id)
      task = find(id)
      @tasks.delete(task)
      save_tasks
      task
    end

    def complete(id)
      task = find(id)
      task.complete!
      save_tasks
      task
    end

    def count
      @tasks.size
    end

    def search(query)
      @tasks.select do |t|
        t.title.downcase.include?(query.downcase) ||
        t.description.downcase.include?(query.downcase)
      end
    end

    def pending_count
      @tasks.count(&:pending?)
    end

    def completed_count
      @tasks.count(&:completed?)
    end

    def clear!
      @tasks.clear
      save_tasks
    end

    def import(tasks)
      tasks.each { |t| add(t) }
    end

    private

    def ensure_directory!
      dir = File.dirname(@path)
      FileUtils.mkdir_p(dir) unless Dir.exist?(dir)
    end

    def load_tasks
      return unless File.exist?(@path)

      data = JSON.parse(File.read(@path))
      @tasks = data.map { |attrs| Task.new(transform_keys(attrs)) }
    rescue JSON::ParserError => e
      raise RepositoryError, "Error al cargar tareas: #{e.message}"
    end

    def save_tasks
      ensure_directory!
      data = @tasks.map(&:to_h)
      File.write(@path, JSON.pretty_generate(data))
    rescue IOError => e
      raise RepositoryError, "Error al guardar tareas: #{e.message}"
    end

    def transform_keys(hash)
      hash.transform_keys(&:to_sym)
    end
  end
end

# lib/task_manager/formatter.rb
module TaskManager
  class Formatter
    def initialize(repository)
      @repository = repository
    end

    def list_tasks(tasks = nil)
      tasks ||= @repository.all
      return 'No hay tareas registradas.' if tasks.empty?

      tasks.sort.map.with_index(1) do |task, index|
        format_task(task, index)
      end.join("\n")
    end

    def show_task(id)
      task = @repository.find(id)
      format_task_detail(task)
    end

    def stats
      total     = @repository.count
      pending   = @repository.pending_count
      completed = @repository.completed_count
      progress  = total.positive? ? (completed.to_f / total * 100).round(1) : 0

      <<~STATS
        ┌──────────────────────────┐
        │      ESTADÍSTICAS         │
        ├──────────────────────────┤
        │ Total:    #{total.to_s.rjust(10)} │
        │ Pendientes: #{pending.to_s.rjust(8)} │
        │ Completadas: #{completed.to_s.rjust(6)} │
        │ Progreso:  #{progress.to_s.rjust(8)}% │
        └──────────────────────────┘
      STATS
    end

    private

    def format_task(task, index)
      status_icon = case task.status
                    when 'completed'   then '✓'
                    when 'in_progress' then '▶'
                    when 'pending'     then '○'
                    when 'cancelled'   then '✗'
                    end

      priority_color = case task.priority
                       when 'critical' then '!!!'
                       when 'high'     then '!!'
                       when 'medium'   then '!'
                       else ''
                       end

      "#{index.to_s.rjust(3)}. #{status_icon} #{task.title}" \
        " #{priority_color} [#{task.status}]"
    end

    def format_task_detail(task)
      <<~DETAIL
        ┌──────────────────────────────────────┐
        │ #{task.title.center(36)} │
        ├──────────────────────────────────────┤
        │ ID:       #{task.id.ljust(29)} │
        │ Estado:   #{task.status.ljust(10)} Prioridad: #{task.priority.ljust(14)} │
        │ Creado:   #{task.created_at.ljust(28)} │
        │ #{task.completed_at ? "Completado: #{task.completed_at.ljust(26)}" : 'Completado: —'.ljust(37)}│
        ├──────────────────────────────────────┤
        │ #{task.description.center(36) unless task.description.empty?}│
        └──────────────────────────────────────┘
      DETAIL
    end
  end
end

# lib/task_manager/cli.rb
module TaskManager
  class CLI < Thor
    def initialize(*args)
      super
      @repository = Repository.new
      @formatter  = Formatter.new(@repository)
    end

    desc 'list', 'Lista todas las tareas'
    option :status, type: :string, desc: 'Filtrar por estado'
    option :priority, type: :string, desc: 'Filtrar por prioridad'
    def list
      tasks = if options[:status] || options[:priority]
                @repository.where(
                  status: options[:status],
                  priority: options[:priority]
                )
              else
                @repository.all
              end
      puts @formatter.list_tasks(tasks)
    end

    desc 'show TASK_ID', 'Muestra detalle de una tarea'
    def show(task_id)
      puts @formatter.show_task(task_id)
    rescue TaskNotFound => e
      say "Error: #{e.message}", :red
    end

    desc 'add TITLE', 'Crea una nueva tarea'
    option :description, type: :string, desc: 'Descripción de la tarea'
    option :priority, type: :string, default: 'medium',
           enum: %w[low medium high critical]
    def add(title)
      task = Task.new(
        title: title,
        description: options[:description],
        priority: options[:priority]
      )
      @repository.add(task)
      say "✓ Tarea creada: #{task.title}", :green
    rescue InvalidTask => e
      say "Error: #{e.message}", :red
    end

    desc 'complete TASK_ID', 'Marca una tarea como completada'
    def complete(task_id)
      task = @repository.complete(task_id)
      say "✓ Tarea completada: #{task.title}", :green
    rescue TaskNotFound => e
      say "Error: #{e.message}", :red
    end

    desc 'update TASK_ID', 'Actualiza una tarea'
    option :title, type: :string
    option :description, type: :string
    option :status, type: :string,
           enum: %w[pending in_progress completed cancelled]
    option :priority, type: :string,
           enum: %w[low medium high critical]
    def update(task_id)
      attrs = options.symbolize_keys
      task = @repository.update(task_id, attrs)
      say "✓ Tarea actualizada: #{task.title}", :green
    rescue TaskNotFound, InvalidTask => e
      say "Error: #{e.message}", :red
    end

    desc 'delete TASK_ID', 'Elimina una tarea'
    def delete(task_id)
      task = @repository.delete(task_id)
      say "✗ Tarea eliminada: #{task.title}", :yellow
    rescue TaskNotFound => e
      say "Error: #{e.message}", :red
    end

    desc 'search QUERY', 'Busca tareas por texto'
    def search(query)
      tasks = @repository.search(query)
      puts @formatter.list_tasks(tasks)
    end

    desc 'stats', 'Muestra estadísticas de tareas'
    def stats
      puts @formatter.stats
    end

    desc 'clear', 'Elimina todas las tareas'
    def clear
      @repository.clear!
      say '✓ Todas las tareas han sido eliminadas', :yellow
    end

    desc 'export FILE', 'Exporta tareas a un archivo JSON'
    def export(file)
      tasks = @repository.all.map(&:to_h)
      File.write(file, JSON.pretty_generate(tasks))
      say "✓ Tareas exportadas a #{file}", :green
    end

    desc 'import FILE', 'Importa tareas desde un archivo JSON'
    def import(file)
      data = JSON.parse(File.read(file))
      tasks = data.map { |attrs| Task.new(attrs.transform_keys(&:to_sym)) }
      @repository.import(tasks)
      say "✓ #{tasks.size} tareas importadas desde #{file}", :green
    rescue JSON::ParserError => e
      say "Error al leer archivo: #{e.message}", :red
    end

    desc 'version', 'Muestra la versión'
    def version
      say "task_manager v#{TaskManager::VERSION}"
    end

    default_task :list
  end
end

# lib/task_manager/task.rb (con documentación YARD)
module TaskManager
  # Representa una tarea individual en el sistema.
  #
  # @example Crear una tarea
  #   TaskManager::Task.new(title: 'Comprar comida', priority: 'high')
  #
  # @attr [String] id Identificador único (UUID)
  # @attr [String] title Título de la tarea
  # @attr [String] description Descripción detallada
  # @attr [String] status Estado: pending, in_progress, completed, cancelled
  # @attr [String] priority Prioridad: low, medium, high, critical
  # @attr [String] created_at Fecha de creación ISO8601
  # @attr [String] completed_at Fecha de completado ISO8601
  class Task
    # @return [Array<String>] Estados válidos para una tarea
    VALID_STATUSES = %w[pending in_progress completed cancelled].freeze

    # @return [Array<String>] Prioridades válidas

# Instalación

$ task_manager add "Estudiar Ruby" --description "Leer capítulo 8"

# Listar tareas
$ task_manager list
$ task_manager list --status pending
$ task_manager list --priority critical

# Ver detalle
$ task_manager show <id>

$ task_manager complete <id>

$ task_manager update <id> --title "Nuevo título" --priority critical

$ task_manager search "leche"
