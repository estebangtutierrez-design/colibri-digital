```ruby
# spec/models/producto_spec.rb
require 'spec_helper'
require_relative '../../app/models/producto'

RSpec.describe Producto do
  describe '#initialize' do
    it 'crea un producto con atributos por defecto' do
      producto = Producto.new

      expect(producto.nombre).to be_nil
      expect(producto.precio).to be_nil
      expect(producto.stock).to eq(0)
    end

    it 'asigna atributos correctamente' do
      producto = Producto.new(nombre: 'Laptop', precio: 1500, stock: 10)

      expect(producto.nombre).to eq('Laptop')
      expect(producto.precio).to eq(1500)
      expect(producto.stock).to eq(10)
    end
  end

  describe '#disponible?' do
    it 'es disponible si tiene stock positivo' do
      producto = Producto.new(stock: 5)
      expect(producto).to be_disponible
    end

    it 'no es disponible si stock es cero' do
      producto = Producto.new(stock: 0)
      expect(producto).not_to be_disponible
    end

    it 'no es disponible si stock es negativo' do
      producto = Producto.new(stock: -1)
      expect(producto).not_to be_disponible
    end
  end

  describe '#precio_con_impuesto' do
    it 'calcula precio con impuesto por defecto' do
      producto = Producto.new(precio: 100)
      expect(producto.precio_con_impuesto).to eq(116.0)
    end

    it 'usa la tasa de impuesto proporcionada' do
      producto = Producto.new(precio: 100)
      expect(producto.precio_con_impuesto(0.08)).to eq(108.0)
    end
  end

  describe '#to_s' do
    it 'formatea el producto como string' do
      producto = Producto.new(nombre: 'Mouse', precio: 25.50, stock: 3)
      expect(producto.to_s).to match(/Mouse/)
      expect(producto.to_s).to match(/\$25\.50/)
    end
  end
end
```

```ruby
# spec/models/producto_spec.rb (versión optimizada)
RSpec.describe Producto do
  # let crea una variable perezosa (solo se evalúa cuando se usa)
  let(:atributos) { { nombre: 'Teclado', precio: 80, stock: 15 } }
  subject(:producto) { Producto.new(atributos) }

  describe '#initialize' do
    it 'asigna nombre' do
      expect(producto.nombre).to eq('Teclado')
    end

    it 'asigna precio' do
      expect(producto.precio).to eq(80)
    end

    context 'sin atributos' do
      subject(:producto_sin_atributos) { Producto.new }

      it 'tiene valores por defecto' do
        expect(producto_sin_atributos.stock).to eq(0)
      end
    end
  end

  describe '#disponible?' do
    it { is_expected.to be_disponible }

    context 'cuando no hay stock' do
      let(:atributos) { { stock: 0 } }
      it { is_expected.not_to be_disponible }
    end
  end

  describe '#precio_con_impuesto' do
    it 'calcula correctamente' do
      expect(producto.precio_con_impuesto).to eq(92.8) # 80 * 1.16
    end
  end
end
```

LET! — EVALUACIÓN INMEDIATA
─────────────────────────────

```ruby
RSpec.describe 'let vs let!' do
  $contador = 0

  let(:perezoso) { $contador += 1 }
  let!(:inmediato) { $contador += 1 }

  it 'let! se evalúa antes del test (inmediato)' do
    expect($contador).to eq(1) # inmediato ya se ejecutó
  end

  it 'let se evalúa solo cuando se referencia' do
    expect($contador).to eq(1) # inmediato se ejecutó otra vez
    perezoso # ahora se evalúa let
    expect($contador).to eq(2)
  end
end
```

HOOKS: BEFORE, AFTER, AROUND
─────────────────────────────

```ruby
RSpec.describe 'Hooks en RSpec' do
  before(:suite) { puts 'Antes de TODA la suite' }
  after(:suite)  { puts 'Después de TODA la suite' }

  before(:context) { puts 'Antes de este grupo describe' }
  after(:context)  { puts 'Después de este grupo describe' }

  before(:each) { @valor = 42 }
  after(:each)  { @valor = nil }

  around(:each) do |test|
    puts "  Antes de #{test.description}"
    test.run
    puts "  Después de #{test.description}"
  end

  before do
    # Equivalente a before(:each), el más común
    @usuario = Usuario.new(nombre: 'Test')
  end

  it 'usa hooks' do
    expect(@valor).to eq(42)
  end
end
```

```ruby
RSpec.describe 'Matchers de RSpec' do
  # ============================================================
  # MATCHERS DE IGUALDAD
  # ============================================================
  it 'compara valores' do
    expect(5).to eq(5)
    expect('hola').to eq('hola')
    expect([1, 2]).to eq([1, 2])
  end

  it 'compara identidad de objeto' do
    a = 'hola'
    b = a
    expect(a).to be(b)
    expect('hola').not_to be('hola') # son strings diferentes
  end

  it 'compara con eql (valor y tipo)' do
    expect(5).to eql(5)
    expect(5).not_to eql(5.0) # Integer vs Float
  end

  it 'compara con equal (alias de be)' do
    expect(5).to equal(5) # cuidado con integers pequeños
  end

  # ============================================================
  # MATCHERS BOOLEANOS
  # ============================================================
  it 'evalúa booleanos' do
    expect(true).to be_truthy
    expect(false).to be_falsy
    expect(nil).to be_falsy
    expect(0).to be_truthy # ¡En Ruby 0 es truthy!
    expect([]).to be_truthy
    expect(nil).to be_nil
  end

  # ============================================================
  # MATCHERS DE COMPARACIÓN
  # ============================================================
  it 'compara números' do
    expect(5).to be > 3
    expect(5).to be >= 5
    expect(5).to be < 10
    expect(5).to be <= 5
    expect(5).to be_between(1, 10).inclusive
    expect(5).to be_between(1, 10).exclusive
  end

  # ============================================================
  # MATCHERS DE TIPO
  # ============================================================
  it 'verifica tipos' do
    expect('hola').to be_a(String)
    expect(5).to be_an(Integer)
    expect([]).to be_an(Array)
    expect([1, 2]).to be_kind_of(Enumerable)
    expect(5).to respond_to(:to_s)
  end

  # ============================================================
  # MATCHERS DE STRING
  # ============================================================
  it 'trabaja con strings' do
    expect('Hola Mundo').to include('Mundo')
    expect('Hola Mundo').to start_with('Hola')
    expect('Hola Mundo').to end_with('Mundo')
    expect('hola@email.com').to match(/\A[\w.]+@\w+\.\w+\z/)
  end

  # ============================================================
  # MATCHERS DE ARREGLOS
  # ============================================================
  it 'trabaja con arreglos' do
    expect([1, 2, 3]).to include(2)
    expect([1, 2, 3]).to include(1, 3)
    expect([1, 2, 3]).to match_array([3, 1, 2]) # sin orden
    expect([1, 2, 3]).to contain_exactly(3, 1, 2) # sin orden
    expect([1, 2, 3]).to start_with(1)
    expect([1, 2, 3]).to end_with(3)
  end

  # ============================================================
  # MATCHERS DE HASH
  # ============================================================
  it 'trabaja con hashes' do
    expect({ a: 1, b: 2 }).to include(:a)
    expect({ a: 1, b: 2 }).to include(a: 1)
    expect({ a: 1, b: 2 }).to have_key(:a)
    expect({ a: 1, b: 2 }).to have_value(2)
  end

  # ============================================================
  # MATCHERS DE ERRORES
  # ============================================================
  it 'captura errores' do
    expect { 1 / 0 }.to raise_error(ZeroDivisionError)
    expect { raise 'Error' }.to raise_error('Error')
    expect { raise ArgumentError }.to raise_error(ArgumentError)
    expect { 'no block' }.not_to raise_error
  end

  # ============================================================
  # MATCHERS DE CAMBIOS
  # ============================================================
  it 'detecta cambios' do
    arreglo = []
    expect { arreglo << 1 }.to change { arreglo.size }.by(1)
    expect { arreglo << 2 }.to change { arreglo.size }.from(1).to(2)
    expect { arreglo.clear }.to change { arreglo }.to([])
  end

  # ============================================================
  # MATCHERS DE SATISFACCIÓN
  # ============================================================
  it 'usa predicados y bloques' do
    expect(5).to satisfy { |n| n.positive? && n.even? == false }
    expect('ruby').to satisfy(:include?, 'ru')
  end

  # ============================================================
  # COMPOSICIÓN DE MATCHERS
  # ============================================================
  it 'compone matchers' do
    expect('RSpec es genial').to(
      start_with('RSpec').and(
        include('genial').or(
          include('increíble')
        )
      )
    )
  end
end
```

```ruby
class CalculadoraImpuestos
  TASA_POR_DEFECTO = 0.16

  def initialize(tasa_por_defecto: TASA_POR_DEFECTO)
    @tasa_por_defecto = tasa_por_defecto
  end

  def calcular(monto:, tasa: nil)
    raise ArgumentError, 'Monto debe ser positivo' if monto <= 0

    tasa ||= @tasa_por_defecto
    raise ArgumentError, 'Tasa debe ser positiva' if tasa < 0

    impuesto = (monto * tasa).round(2)
    { monto: monto, tasa: tasa, impuesto: impuesto,
      total: (monto + impuesto).round(2), tasa_usada: tasa }
  end
end

# PASO 3: REFACTOR — Mejorar el diseño

# app/services/calculadora_impuestos.rb (refactorizado)
class CalculadoraImpuestos
  RESULTADO = Struct.new(:monto, :tasa, :impuesto, :total,
                          keyword_init: true)

  TASA_POR_DEFECTO = 0.16

  def initialize(tasa_por_defecto: TASA_POR_DEFECTO)
    @tasa_por_defecto = tasa_por_defecto
  end

  def calcular(monto:, tasa: nil)
    validar_monto!(monto)
    tasa ||= @tasa_por_defecto
    validar_tasa!(tasa)

    impuesto = (monto * tasa).round(2)

    RESULTADO.new(
      monto: monto,
      tasa: tasa,
      impuesto: impuesto,
      total: (monto + impuesto).round(2)
    )
  end

  private

  def validar_monto!(monto)
    raise ArgumentError, 'Monto debe ser positivo' unless monto.is_a?(Numeric)
    raise ArgumentError, 'Monto debe ser positivo' if monto <= 0
  end

  def validar_tasa!(tasa)
    raise ArgumentError, 'Tasa debe ser positiva' if tasa < 0
  end
end
```

```ruby
    it 'registra los atributos a auditar' do
      expect(clase_con_auditoria.atributos_auditados).to eq([:nombre, :precio])
    end
  end

  describe '#cambios_auditados' do
    it 'detecta cambios en atributos auditados' do
      objeto.nombre = 'Nuevo Nombre'
      cambios = objeto.cambios_auditados
      expect(cambios).to have_key(:nombre)
      expect(cambios[:nombre][:de]).to eq('Test')
      expect(cambios[:nombre][:a]).to eq('Nuevo Nombre')
    end

    it 'ignora atributos no auditados' do
      objeto.stock = 50
```

RESUMEN DEL CAPÍTULO
─────────────────────
- RSpec es el framework de testing más usado en Ruby.
- describe/context/it organizan las especificaciones.
- let crea variables perezosas; let! las evalúa de inmediato.
- subject define el objeto bajo prueba.
- before/after/around son hooks del ciclo de vida.
- Los matchers permiten afirmaciones expresivas (eq, be, include).
- Los doubles reemplazan objetos reales; mocks verifican interacciones.
- Factory Bot genera datos de prueba de forma declarativa.
- TDD sigue el ciclo RED → GREEN → REFACTOR.
- SimpleCov mide la cobertura de código.
- Los shared examples reutilizan pruebas entre modelos.

PREGUNTAS FRECUENTES (FAQ)
───────────────────────────
P: ¿Debo escribir tests antes o después del código?
R: TDD recomienda antes. Pero es mejor escribir tests después que
   no escribirlos nunca. El hábito ideal es TDD.

P: ¿Qué porcentaje de cobertura es adecuado?
R: 80-90% es un buen objetivo. El 100% puede ser contraproducente
   (tests triviales que solo inflan la métrica).

P: ¿Cuándo usar mocks vs objetos reales?
R: Mocks para servicios externos (API, email, base de datos).
   Objetos reales para lógica de negocio pura.

P: ¿let o variables de instancia (@)?
R: let. Es perezoso, memoizado y limpio entre tests.

P: ¿Cómo organizo mis archivos de test?
R: Espeja la estructura de app/ en spec/. Cada clase tiene su
   archivo _spec.rb correspondiente.
