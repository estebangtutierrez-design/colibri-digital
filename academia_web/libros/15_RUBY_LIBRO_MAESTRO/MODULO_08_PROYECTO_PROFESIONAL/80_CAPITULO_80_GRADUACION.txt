#!/usr/bin/env ruby
# graduation.rb — Script de graduación simbólica

module RubyMasterBook
  module Graduation
    class Student
      attr_reader :name, :completed_modules

      def initialize(name)
        @name = name
        @completed_modules = []
      end

      def complete(module_name)
        @completed_modules << module_name
        puts "✓ Módulo #{module_name} completado"
      end

      def graduated?
        @completed_modules.size >= 8
      end

      def celebrate!
        puts '╔══════════════════════════════════════╗'
        puts '║                                      ║'
        puts '║   🎉 FELICIDADES, MAESTRO RUBY! 🎉   ║'
        puts '║                                      ║'
        puts '╚══════════════════════════════════════╝'
        puts
        puts "Nombre: #{@name}"
        puts "Módulos completados: #{@completed_modules.size}/8"
        puts "Estado: #{graduated? ? 'GRADUADO' : 'EN PROGRESO'}"
        puts
        puts 'Recuerda siempre:'
        puts '  "El código es poesía. Ruby es la pluma."'
        puts '   — Profesor Búho AI'
      end
    end

    CERTIFICATE = <<~CERT
      ╔══════════════════════════════════════════════════╗
      ║                                                  ║
      ║         RUBY MASTER BOOK — CERTIFICADO           ║
      ║                                                  ║
      ║   Por la presente se certifica que               ║
      ║                                                  ║
      ║   #{'[ESTUDIANTE]'.center(42)}   ║
      ║                                                  ║
      ║   ha completado el programa completo              ║
      ║   de 80 capítulos y 8 módulos.                   ║
      ║                                                  ║
      ║   Fecha: #{Time.now.strftime('%d/%m/%Y').center(32)}   ║
      ║                                                  ║
      ║   ─────────────────────────────                   ║
      ║   Profesor Búho AI                                ║
      ║   Ruby Master Book                                ║
      ║                                                  ║
      ╚══════════════════════════════════════════════════╝
    CERT
  end
end
