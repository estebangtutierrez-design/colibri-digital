  module Utilidades
    COLORES = {
      rojo:     "\e[31m",
      verde:    "\e[32m",
      amarillo: "\e[33m",
      azul:     "\e[34m",
      magenta:  "\e[35m",
      cian:     "\e[36m",
      blanco:   "\e[37m",
      reset:    "\e[0m"
    }.freeze

    def self.colorear(texto, color = :reset)
      "#{COLORES.fetch(color, COLORES[:reset])}#{texto}#{COLORES[:reset]}"
    end

    def self.titulo(texto)
      ancho = 60
      linea = "=" * ancho
      "#{linea}\n#{texto.center(ancho)}\n#{linea}"
    end

    def self.subtitulo(texto)
      "--- #{texto} ---"
    end

    def self.limpiar_pantalla
      system("clear") || system("cls")
    end

    def self.pausa
      print "\nPresiona Enter para continuar..."
      gets
    end

    def self.entero_valido?(entrada)
      entrada =~ /\A-?\d+\z/
    end

    def self.positivo?(entrada)
      entero_valido?(entrada) && entrada.to_i > 0
    end

    def self.seleccion_valida?(entrada, min, max)
      entero_valido?(entrada) && entrada.to_i.between?(min, max)
    end

    def self.formatear_tiempo(segundos)
      minutos = (segundos / 60).to_i
      segs = (segundos % 60).to_i
      "#{minutos}m #{segs}s"
    end
  end

  class Jugador
    attr_reader :nombre, :puntaje_total, :partidas_jugadas, :partidas_ganadas

    def initialize(nombre: "Invitado")
      @nombre = nombre.strip.empty? ? "Invitado" : nombre.strip.capitalize
      @puntaje_total = 0
      @partidas_jugadas = 0
      @partidas_ganadas = 0
      @historial = []
    end

    def registrar_partida(resultado, **detalles)
      @partidas_jugadas += 1
      @partidas_ganadas += 1 if resultado == :ganada
      @puntaje_total += detalles[:puntos] || 0
      @historial << {
        resultado: resultado,
        puntos: detalles[:puntos] || 0,
        dificultad: detalles[:dificultad] || "N/A",
        modo: detalles[:modo] || "N/A",
        intentos: detalles[:intentos] || 0,
        tiempo: detalles[:tiempo] || 0,
        fecha: Time.now
      }
    end

    def promedio_intentos
      return 0.0 if @historial.empty?
      @historial.map { |h| h[:intentos] }.sum.to_f / @historial.length
    end

    def racha_actual
      return 0 if @historial.empty?
      racha = 0
      @historial.reverse_each do |h|
        break unless h[:resultado] == :ganada
        racha += 1
      end
      racha
    end

    def mejor_juego
      @historial.select { |h| h[:resultado] == :ganada }
                .min_by { |h| h[:intentos] }
    end

    def mostrar_estadisticas
      puts Utilidades.subtitulo("Estadísticas de #{@nombre}")
      puts "  Partidas jugadas: #{@partidas_jugadas}"
      puts "  Partidas ganadas: #{@partidas_ganadas}"
      puts "  Porcentaje victorias: #{porcentaje_victorias}%"
      puts "  Puntaje total: #{@puntaje_total}"
      puts "  Promedio intentos: #{'%.1f' % promedio_intentos}"
      puts "  Racha actual: #{racha_actual}"
      puts "  Mejor juego: #{detalle_mejor_juego}"
    end

    def porcentaje_victorias
      return 0 if @partidas_jugadas.zero?
      (@partidas_ganadas.to_f / @partidas_jugadas * 100).round(1)
    end

    def detalle_mejor_juego
      juego = mejor_juego
      return "Ninguno aún" unless juego
      "#{juego[:intentos]} intentos (nivel #{juego[:dificultad]})"
    end

    def to_s
      "#{@nombre} (#{@puntaje_total} pts, #{@partidas_ganadas}/#{@partidas_jugadas} ganadas)"
    end
  end

  class GeneradorNumeros
    def initialize(rango: 1..100, &personalizador)
      @rango = rango
      @personalizador = personalizador
    end

    def generar
      numero = rand(@rango)
      numero = @personalizador.call(numero) if @personalizador
      numero
    end

    def generar_unicos(cantidad, &filtro)
      numeros = []
      while numeros.length < cantidad
        num = generar
        next if numeros.include?(num)
        next if filtro && !filtro.call(num)
        numeros << num
      end
      numeros
    end

    def self.secuencia(tamaño, rango: 1..100)
      generador = new(rango: rango)
      generador.generar_unicos(tamaño)
    end
  end

  class ModoJuego
    attr_reader :nombre, :descripcion

    def initialize(nombre:, descripcion:, dificultad:, jugador:)
      @nombre = nombre
      @descripcion = descripcion
      @dificultad = dificultad
      @jugador = jugador
      @config = Configuracion::DIFICULTADES[dificultad.to_sym]
      @generador = GeneradorNumeros.new(rango: @config[:rango])
    end

    def jugar(&bloque_finalizacion)
      raise NotImplementedError, "Cada modo debe implementar #jugar"
    end

    protected

    def calcular_puntos(intentos)
      base = @config[:puntos_base]
      max_intentos = @config[:intentos]
      bonus = ((max_intentos - intentos).to_f / max_intentos * base).round
      [base + bonus, base].max
    end

    def dar_pista(numero_secreto, intento)
      case
      when (numero_secreto - intento).abs <= 3
        "¡Estás muy cerca!"
      when (numero_secreto - intento).abs <= 10
        "Estás cerca"
      when intento < numero_secreto
        "El número es mayor"
      else
        "El número es menor"
      end
    end

    def mostrar_progreso(intentos_usados, max_intentos)
      return if max_intentos == Float::INFINITY
      restantes = max_intentos - intentos_usados
      print "  Intentos restantes: #{restantes}"
      restantes.times { print " █" }
      puts
    end
  end

  class ModoClasico < ModoJuego
    def initialize(dificultad:, jugador:)
      config = Configuracion::DIFICULTADES[dificultad.to_sym]
      super(
        nombre: "Clásico",
        descripcion: "Adivina el número en la menor cantidad de intentos",
        dificultad: dificultad,
        jugador: jugador
      )
    end

    def jugar(&bloque_finalizacion)
      numero_secreto = @generador.generar
      intentos_usados = 0
      max_intentos = @config[:intentos]
      inicio = Time.now
      adivinado = false

      puts "\n--- MODO CLÁSICO (#{@config[:nombre]}) ---"
      puts "Adivina el número entre #{@config[:rango].min} y #{@config[:rango].max}"
      puts "Tienes #{max_intentos} intentos."
      puts

      loop do
        break if intentos_usados >= max_intentos

        intentos_usados += 1
        mostrar_progreso(intentos_usados - 1, max_intentos)
        print "Intento ##{intentos_usados}: "

        entrada = gets.chomp
        next if entrada.empty?

        unless Utilidades.entero_valido?(entrada)
          puts Utilidades.colorear("  Entrada inválida. Ingresa un número entero.", :rojo)
          redo
        end

        intento = entrada.to_i

        if intento == numero_secreto
          tiempo = Time.now - inicio
          puntos = calcular_puntos(intentos_usados)
          adivinado = true

          puts Utilidades.colorear("\n  ¡FELICIDADES! Adivinaste el número!", :verde)
          puts "  Número secreto: #{numero_secreto}"
          puts "  Intentos: #{intentos_usados}"
          puts "  Tiempo: #{Utilidades.formatear_tiempo(tiempo)}"
          puts "  Puntos ganados: #{puntos}"

          @jugador.registrar_partida(
            :ganada, puntos: puntos, dificultad: @config[:nombre],
            modo: @nombre, intentos: intentos_usados, tiempo: tiempo
          )

          yield(:ganada, intentos_usados, puntos, tiempo) if bloque_finalizacion
          break
        else
          puts "  #{dar_pista(numero_secreto, intento)}"
        end
      end

      unless adivinado
        puts Utilidades.colorear("\n  Se acabaron los intentos. El número era: #{numero_secreto}", :rojo)

        @jugador.registrar_partida(
          :perdida, puntos: 0, dificultad: @config[:nombre],
          modo: @nombre, intentos: intentos_usados, tiempo: Time.now - inicio
        )

        yield(:perdida, intentos_usados, 0, Time.now - inicio) if bloque_finalizacion
      end
    end
  end

  class ModoContrarreloj < ModoJuego
    def initialize(dificultad:, jugador:, tiempo_limite: 30)
      super(
        nombre: "Contrarreloj",
        descripcion: "Adivina antes de que el tiempo se agote",
        dificultad: dificultad,
        jugador: jugador
      )
      @tiempo_limite = tiempo_limite
    end

    def jugar(&bloque_finalizacion)
      numero_secreto = @generador.generar
      intentos_usados = 0
      inicio = Time.now
      adivinado = false

      puts "\n--- MODO CONTRARRELOL (#{@config[:nombre]}) ---"
      puts "Adivina el número entre #{@config[:rango].min} y #{@config[:rango].max}"
      puts "Tienes #{@tiempo_limite} segundos!"
      puts

      loop do
        tiempo_transcurrido = Time.now - inicio
        break if tiempo_transcurrido >= @tiempo_limite

        tiempo_restante = @tiempo_limite - tiempo_transcurrido
        print "  [Tiempo: #{tiempo_restante.round(1)}s] "
        print "Intento ##{intentos_usados + 1}: "

        entrada = gets.chomp
        next if entrada.empty?

        unless Utilidades.entero_valido?(entrada)
          puts Utilidades.colorear("  Inválido. Ingresa un número.", :rojo)
          redo
        end

        intentos_usados += 1
        intento = entrada.to_i

        if intento == numero_secreto
          tiempo_total = Time.now - inicio
          puntos = calcular_puntos(intentos_usados) + 50 # Bonus por velocidad
          adivinado = true

          puts Utilidades.colorear("\n  ¡GANASTE CONTRARRELOL!", :verde)
          puts "  Tiempo restante: #{(@tiempo_limite - tiempo_total).round(1)}s"
          puts "  Puntos: #{puntos}"

          @jugador.registrar_partida(
            :ganada, puntos: puntos, dificultad: @config[:nombre],
            modo: @nombre, intentos: intentos_usados, tiempo: tiempo_total
          )

          yield(:ganada, intentos_usados, puntos, tiempo_total) if bloque_finalizacion
          break
        else
          puts "  #{dar_pista(numero_secreto, intento)}"
        end
      end

      unless adivinado
        puts Utilidades.colorear("\n  ¡TIEMPO! El número era: #{numero_secreto}", :rojo)
        @jugador.registrar_partida(
          :perdida, puntos: 0, dificultad: @config[:nombre],
          modo: @nombre, intentos: intentos_usados, tiempo: @tiempo_limite
        )
        yield(:perdida, intentos_usados, 0, @tiempo_limite) if bloque_finalizacion
      end
    end
  end

  class SistemaJuego
    def initialize
      @jugador = nil
      @modo_actual = nil
    end

    def iniciar
      loop do
        Utilidades.limpiar_pantalla
        puts Utilidades.colorear(Utilidades.titulo("ADIVINA EL NÚMERO — MASTER EDITION"), :cian)
        puts Utilidades.colorear("  Proyecto Final — Módulo II: Control de Flujo y Métodos", :amarillo)
        puts "  Ruby Master Book\n\n"

        mostrar_menu_principal
        opcion = gets.chomp

        case opcion
        when "1" then iniciar_sesion
        when "2" then seleccionar_modo
        when "3" then ver_estadisticas
        when "4" then ver_historial
        when "5" then mostrar_ayuda
        when "6"
          puts Utilidades.colorear("\n¡Gracias por jugar!", :verde)
          break
        else
          puts Utilidades.colorear("Opción inválida. Presiona Enter para continuar...", :rojo)
          gets
        end
      end
    end

    private

    def mostrar_menu_principal
      puts "  1. Iniciar sesión / Cambiar jugador"
      puts "  2. Seleccionar modo de juego"
      puts "  3. Ver estadísticas"
      puts "  4. Ver historial de partidas"
      puts "  5. Ayuda"
      puts "  6. Salir"
      print "\n  Elige una opción (1-6): "
    end

    def iniciar_sesion
      Utilidades.limpiar_pantalla
      puts Utilidades.subtitulo("INICIO DE SESIÓN")
      print "  Ingresa tu nombre: "
      nombre = gets.chomp
      @jugador = Jugador.new(nombre: nombre)
      puts Utilidades.colorear("  Bienvenido, #{@jugador.nombre}!", :verde)
      Utilidades.pausa
    end

    def seleccionar_modo
      unless @jugador
        puts Utilidades.colorear("  Debes iniciar sesión primero.", :amarillo)
        Utilidades.pausa
        return
      end

      loop do
        Utilidades.limpiar_pantalla
        puts Utilidades.subtitulo("SELECCIONAR MODO DE JUEGO")

        modos = [
          ["1", "Clásico", "Adivina el número"],
          ["2", "Contrarreloj", "Adivina contra el tiempo"],
          ["3", "Volver al menú principal"]
        ]

        modos.each { |num, nombre, desc| puts "  #{num}. #{nombre} — #{desc}" }
        print "\n  Elige: "
        opcion = gets.chomp

        case opcion
        when "1"
          jugar_modo_clasico
        when "2"
          jugar_modo_contrarreloj
        when "3"
          break
        else
          puts Utilidades.colorear("Opción inválida", :rojo)
          Utilidades.pausa
        end
      end
    end

    def seleccionar_dificultad
      Utilidades.limpiar_pantalla
      puts Utilidades.subtitulo("SELECCIONAR DIFICULTAD")

      diffs = Configuracion::DIFICULTADES
      diffs.each_with_index do |(key, config), i|
        puts "  #{i + 1}. #{config[:nombre]} — #{config[:descripcion]}"
      end
      puts "  #{diffs.length + 1}. Cancelar"

      print "\n  Elige dificultad (1-#{diffs.length + 1}): "
      opcion = gets.chomp.to_i

      return nil if opcion == diffs.length + 1

      dificultad = difficas.keys[opcion - 1]
      unless dificultad && opcion.between?(1, diffs.length)
        puts Utilidades.colorear("Opción inválida", :rojo)
        Utilidades.pausa
        return seleccionar_dificultad
      end

      dificultad
    end

    def jugar_modo_clasico
      dificultad = seleccionar_dificultad
      return unless dificultad

      modo = ModoClasico.new(dificultad: dificultad, jugador: @jugador)

      loop do
        modo.jugar do |resultado, intentos, puntos, tiempo|
          # Bloque de finalización
          puts "\n  (Nota: partida registrada para #{@jugador.nombre})"
        end

        print "\n  ¿Jugar otra partida? (s/n): "
        break unless gets.chomp.downcase.start_with?("s")
      end
    end

    def jugar_modo_contrarreloj
      dificultad = seleccionar_dificultad
      return unless dificultad

      print "  Tiempo límite en segundos (default 30): "
      tiempo = gets.chomp
      tiempo_limite = tiempo.empty? ? 30 : [tiempo.to_i, 10].max

      modo = ModoContrarreloj.new(
        dificultad: dificultad,
        jugador: @jugador,
        tiempo_limite: tiempo_limite
      )

      loop do
        modo.jugar
        print "\n  ¿Otra partida contrarreloj? (s/n): "
        break unless gets.chomp.downcase.start_with?("s")
      end
    end

    def ver_estadisticas
      Utilidades.limpiar_pantalla
      unless @jugador
        puts Utilidades.colorear("  No hay jugador registrado.", :amarillo)
        Utilidades.pausa
        return
      end
      @jugador.mostrar_estadisticas
      Utilidades.pausa
    end

    def ver_historial
      Utilidades.limpiar_pantalla
      unless @jugador
        puts Utilidades.colorear("  No hay jugador registrado.", :amarillo)
        Utilidades.pausa
        return
      end

      puts Utilidades.subtitulo("Historial de #{@jugador.nombre}")
      if @jugador.instance_variable_get(:@historial).empty?
        puts "  No hay partidas registradas aún."
      else
        @jugador.instance_variable_get(:@historial).each_with_index do |h, i|
          resultado_color = h[:resultado] == :ganada ? :verde : :rojo
          puts "  #{i + 1}. #{Utilidades.colorear(h[:resultado].to_s, resultado_color)} " \
               "| #{h[:dificultad]} | #{h[:modo]} | " \
               "#{h[:intentos]} intentos | #{h[:puntos]} pts | " \
               "#{h[:tiempo].round(1)}s"
        end
      end
      Utilidades.pausa
    end

    def mostrar_ayuda
      Utilidades.limpiar_pantalla
      puts Utilidades.titulo("AYUDA DEL JUEGO")
      puts <<~AYUDA
        BIENVENIDO al juego de Adivinanzas Master Edition!

        CÓMO JUGAR:
        1. Inicia sesión con tu nombre (opcional, puedes jugar como Invitado).
        2. Selecciona un modo de juego:
           - Clásico: Adivina el número en la menor cantidad de intentos.
           - Contrarreloj: Adivina antes de que el tiempo se agote.
        3. Elige una dificultad: Fácil, Medio, Difícil o Experto.
        4. Ingresa tus intentos hasta adivinar el número secreto.

        PUNTUACIÓN:
        - Ganas puntos base por dificultad + bonus por intentos restantes.
        - El modo contrarreloj da bonus adicional por velocidad.

        CONSEJOS:
        - Usa las pistas (mayor/menor) para ajustar tu búsqueda.
        - Comienza con un rango amplio y ve ajustando.
        - Practica en modo Fácil antes de subir la dificultad.

        TECNOLOGÍA:
        Este juego fue construido con Ruby, utilizando:
        - if/elsif/else, unless, case/when
        - while, until, loop
        - each, times, upto
        - Métodos con def/end, argumentos, keyword args
        - Bloques con yield y &block
        - Clases, módulos y composición
      AYUDA
      Utilidades.pausa
    end
  end

    juego = SistemaJuego.new

  =======================

  Ahora que has visto el proyecto completo, aquí tienes 10 ejercicios

  --- EJERCICIO 1 ---
  Añade un nuevo modo de juego: "Modo Racha".  En este modo, el
  jugador debe adivinar una secuencia de números (ej: 5 números)

  ModoRacha que herede de ModoJuego.

  --- EJERCICIO 2 ---
  Añade un sistema de logros (achievements).  Por ejemplo:
  - "Primera Victoria" (ganar 1 partida)
  - "Maestro" (ganar 10 partidas)
  - "Velocista" (ganar en contrarreloj con más de 15s restantes)
  - "Perfecto" (ganar en 1 intento)
  Implementa un módulo Logros y verifícalos después de cada partida.

  --- EJERCICIO 3 ---
  Implementa persistencia de datos.  Guarda el jugador, estadísticas
  e historial en un archivo JSON.  Al iniciar el juego, carga los
  datos si existen.  Usa la biblioteca json de Ruby.

  --- EJERCICIO 4 ---
  Añade un sistema de "pistas" que el jugador pueda comprar con puntos.
  Por ejemplo: "El número está en el rango X-Y" cuesta 10 puntos.
  Implementa un método comprar_pista en el modo de juego.

  --- EJERCICIO 5 ---
  Implementa un "Modo Multijugador" local donde dos jugadores se
  turnen para adivinar el mismo número.  El que lo adivine en menos
  intentos gana.

  --- EJERCICIO 6 ---
  Añade efectos visuales con colores.  Actualmente ya hay algunos con
  Utilidades.colorear.  Mejora la interfaz: tablas, barras de progreso,
  animaciones simples (como puntos suspensivos mientras "piensa").

  --- EJERCICIO 7 ---
  Implementa el "Modo Práctica".  Sin límite de intentos, sin presión
  de tiempo.  Ideal para principiantes.  Muestra estadísticas de
  aprendizaje (mejora con el tiempo).

  --- EJERCICIO 8 ---
  Añade un sistema de niveles de jugador.  Cada cierto puntaje, el
  jugador sube de nivel (ej: 0-100 Nivel 1, 100-300 Nivel 2, etc.).
  Muestra el nivel en el menú principal.

  --- EJERCICIO 9 ---
  Implementa un "ranking" (tabla de posiciones) que ordene a los

  el ranking se mantiene entre sesiones.

  --- EJERCICIO 10 ---
  Refactoriza el código para que los modos de juego se carguen
  dinámicamente.  Usa un registro de modos donde cada modo se
  auto-registra al cargarse.  Esto permitiría añadir nuevos modos
  sin modificar SistemaJuego.

  PREGUNTAS FRECUENTES (FAQ)
  ==========================

  P: ¿Cómo ejecuto el proyecto?
  R: Guarda el código en un archivo juego.rb y ejecuta: ruby juego.rb

  P: ¿Necesito instalar gemas adicionales?
  R: No. El proyecto usa solo la biblioteca estándar de Ruby.

  P: ¿Puedo modificar el proyecto libremente?
  R: Sí, es tu proyecto de aprendizaje.  Modifícalo, extiéndelo,
     rómpelo y arréglalo.

  P: ¿Por qué el código usa módulos y clases?
  R: Para organizar el código, separar responsabilidades y hacerlo
     mantenible.  Es una buena práctica desde el principio.

  P: ¿Qué hace __FILE__ == $0?
  R: Permite que el archivo sea usado como script (ruby juego.rb) o
     como biblioteca (require). $0 es el nombre del archivo ejecutado.

  P: ¿Por qué algunos métodos terminan en ? y otros en !?
  R: ? para métodos que devuelven boolean; ! para métodos peligrosos
     o modificadores.  Es una convención de Ruby.

  P: ¿Los colores funcionan en Windows?
  R: Sí, desde Windows 10 version 1511 con la consola moderna.

  P: ¿Cómo puedo añadir más dificultades?
  R: Agrega una entrada al hash Configuracion::DIFICULTADES con el
     nombre, rango, intentos y puntos base.

  P: ¿Cómo puedo cambiar el idioma?
  R: El código está en español.  Para cambiarlo a otro idioma, busca
     y reemplaza los strings visibles para el usuario.

  P: ¿Qué es el bloque de finalización en los modos?
  R: Es un bloque opcional (&bloque_finalizacion) que se ejecuta
     cuando termina una partida, permitiendo personalizar el
     comportamiento post-juego.
