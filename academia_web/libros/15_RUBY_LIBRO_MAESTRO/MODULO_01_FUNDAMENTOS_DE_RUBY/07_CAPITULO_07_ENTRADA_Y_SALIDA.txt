1. MÉTODOS DE SALIDA (OUTPUT)
===============================================================================

--- 1.1 PUTS ---

puts es el método de salida más común. Imprime un objeto y agrega salto de
línea.

```ruby
# puts - salida con salto de línea
puts "Hola Mundo"
puts 42
puts [1, 2, 3]
puts nil
puts true

# puts con múltiples argumentos
puts "Uno", "Dos", "Tres"
# Uno
# Dos
# Tres

# puts con arrays
puts [1, 2, 3]
# 1
# 2
# 3

# puts con objetos personalizados
class Persona
  attr_reader :nombre
  def initialize(nombre)
    @nombre = nombre
  end
  def to_s
    "Persona: #{@nombre}"
  end
end

ana = Persona.new("Ana")
puts ana  # Persona: Ana (usa to_s)
```

--- 1.2 PRINT ---

print imprime sin salto de línea al final.

```ruby
# print - salida sin salto de línea
print "Hola "
print "Mundo"
print "!"
# Salida: Hola Mundo!

# print con múltiples argumentos
print 1, 2, 3, "\n"

# Barra de progreso simple
10.times do |i|
  print "."
  sleep(0.1)
end
puts " Hecho!"
```

--- 1.3 P ---

p imprime la representación de inspección del objeto.

```ruby
# p - representación interna
p "Hola Mundo"        # "Hola Mundo"
p 42                  # 42
p [1, 2, 3]           # [1, 2, 3]
p nil                 # nil
p :ruby               # :ruby

# p vs puts
puts "Hola"           # Hola
p "Hola"              # "Hola"

# p para debugging
nombre = "Ana"
edad = 30
p [nombre, edad]      # ["Ana", 30]
```

--- 1.4 PP (PRETTY PRINT) ---

pp formatea objetos complejos legiblemente.

```ruby
require "pp"

datos = {
  usuarios: [
    { nombre: "Ana", edad: 30, hobbies: ["leer", "nadar"] },
    { nombre: "Luis", edad: 25, hobbies: ["musica", "correr"] }
  ]
}

pp datos
```

--- 1.5 SALIDA CON COLOR ---

```ruby
# Códigos ANSI para color
def colorize(texto, color)
  colores = {
    rojo: "\e[31m", verde: "\e[32m", amarillo: "\e[33m",
    azul: "\e[34m", magenta: "\e[35m", cian: "\e[36m", reset: "\e[0m"
  }
  "#{colores[color]}#{texto}#{colores[:reset]}"
end

puts colorize("Rojo", :rojo)
puts colorize("Verde", :verde)
puts colorize("Azul", :azul)
```

2. MÉTODOS DE ENTRADA (INPUT)
===============================================================================

--- 2.1 GETS BÁSICO ---

gets lee una línea de la entrada estándar como string.

```ruby
puts "Como te llamas?"
nombre = gets
puts "Hola, #{nombre}"  # nombre incluye \n
```

--- 2.2 GETS.CHOMP ---

chomp elimina el salto de línea del final.

```ruby
print "Como te llamas? "
nombre = gets.chomp
puts "Hola, #{nombre}!"

# Múltiples entradas
print "Nombre: "
nombre = gets.chomp
print "Edad: "
edad = gets.chomp
puts "Te llamas #{nombre} y tienes #{edad} años."
```

--- 2.3 CONVIRTIENDO ENTRADA ---

gets siempre devuelve string. Debes convertir al tipo deseado.

```ruby
# Convertir a entero
print "Ingresa un numero: "
numero = gets.chomp.to_i
puts "El doble es: #{numero * 2}"

# Convertir a float
print "Precio: "
precio = gets.chomp.to_f
puts "Con IVA: #{precio * 1.16}"

# Validación básica
print "Numero: "
entrada = gets.chomp
if entrada.match?(/^-?\d+(\.\d+)?$/)
  puts "Valido! #{entrada.to_f}"
else
  puts "No es un numero"
end
```

--- 2.4 MÚLTIPLES ENTRADAS ---

```ruby
# Array desde una línea
print "Numeros separados por espacio: "
numeros = gets.chomp.split.map(&:to_i)
puts "Suma: #{numeros.sum}"

# Leyendo hasta que ingrese vacío
datos = []
puts "Nombres (vacio para terminar):"
loop do
  entrada = gets.chomp
  break if entrada.empty?
  datos << entrada
end
puts "Ingresados: #{datos.join(", ")}"
```

--- 2.5 GETC ---

getc lee un solo carácter.

```ruby
puts "Presiona una tecla..."
tecla = STDIN.getc
puts "Tecla: #{tecla.inspect}"
```

--- 2.6 READLINES ---

```ruby
puts "Escribe líneas (Ctrl+D para terminar):"
lineas = STDIN.readlines(chomp: true)
puts "Líneas: #{lineas.length}"
```

3. INTERPOLACIÓN DE STRINGS
===============================================================================

--- 3.1 INTERPOLACIÓN BÁSICA ---

```ruby
nombre = "Ana"
puts "Hola, #{nombre}!"          # Hola, Ana!
puts "2 + 2 = #{2 + 2}"          # 2 + 2 = 4
puts "Hoy es #{Time.now}"        # Hoy es ...
puts "Al revés: #{nombre.reverse}"

# Atajo para variables de instancia
@nombre = "Ana"
puts "Hola, #@nombre"  # Hola, Ana
```

--- 3.2 INTERPOLACIÓN COMPLEJA ---

```ruby
# Con bloques
puts "Resultado: #{(1..5).map { |n| n * 2 }.join(", ")}"

# Con arrays
colores = %w[rojo verde azul]
puts "Colores: #{colores.join(" - ")}"

# Con hashes
persona = { nombre: "Ana", edad: 30 }
puts "#{persona[:nombre]} tiene #{persona[:edad]} años"
```

--- 3.3 INTERPOLACIÓN CON % ---

```ruby
puts "Nombre: %s, Edad: %d" % ["Ana", 30]
puts "Precio: $%.2f" % 99.99
puts "%{nombre} tiene %{edad} años" % { nombre: "Ana", edad: 30 }
```

--- 3.4 INTERPOLACIÓN VS CONCATENACIÓN ---

```ruby
nombre = "Ana"
edad = 30

# Interpolación (recomendada)
puts "Hola, me llamo #{nombre} y tengo #{edad} años."

# Concatenación (menos legible)
puts "Hola, me llamo " + nombre + " y tengo " + edad.to_s + " años."
```

8. PROYECTO: SISTEMA DE DIARIO PERSONAL
===============================================================================

```ruby
# diario.rb - Sistema de diario personal
require "json"

class DiarioPersonal
  ARCHIVO = "diario.txt"

  def initialize
    @entradas = cargar_entradas
  end

  def nueva_entrada
    puts "--- NUEVA ENTRADA ---"
    print "Título: "
    titulo = gets.chomp
    puts "Contenido (línea vacía para terminar):"
    contenido = []
    loop do
      linea = gets.chomp
      break if linea.empty?
      contenido << linea
    end
    entrada = {
      fecha: Time.now.strftime("%Y-%m-%d %H:%M"),
      titulo: titulo,
      contenido: contenido.join("\n")
    }
    @entradas << entrada
    guardar_entradas
    puts "Entrada guardada!"
  end

  def ver_entradas
    if @entradas.empty?
      puts "No hay entradas."
      return
    end
    @entradas.each_with_index do |e, i|
      puts "#{i + 1}. [#{e[:fecha]}] #{e[:titulo]}"
      puts "   #{e[:contenido][0..50]}..."
      puts ""
    end
  end

  def buscar(termino)
    resultados = @entradas.select do |e|
      e[:titulo].downcase.include?(termino.downcase) ||
      e[:contenido].downcase.include?(termino.downcase)
    end
    if resultados.empty?
      puts "Sin resultados para: #{termino}"
    else
      resultados.each { |e| puts "  - #{e[:titulo]} (#{e[:fecha]})" }
    end
  end

  private

  def cargar_entradas
    if File.exist?(ARCHIVO)
      File.readlines(ARCHIVO, chomp: true).map { |l| JSON.parse(l, symbolize_names: true) }
    else
      []
    end
  rescue
    []
  end

  def guardar_entradas
    File.open(ARCHIVO, "w") do |f|
      @entradas.each { |e| f.puts(e.to_json) }
    end
  end
end

# Menú interactivo
diario = DiarioPersonal.new
loop do
  puts "\n=== DIARIO PERSONAL ==="
  puts "1. Nueva entrada"
  puts "2. Ver entradas"
  puts "3. Buscar"
  puts "4. Salir"
  print "Opción: "
  case gets.chomp
  when "1" then diario.nueva_entrada
  when "2" then diario.ver_entradas
  when "3"
    print "Buscar: "
    diario.buscar(gets.chomp)
  when "4" then break
  else puts "Opción inválida"
  end
end
puts "Hasta luego!"
```

  bar = "#" * lleno + "." * (ancho - lleno)

10. PROCESADOR DE ARCHIVOS CSV
===============================================================================

```ruby
require "csv"

class ProcesadorCSV
  def initialize(archivo)
    @archivo = archivo
    @datos = []
  end

  def cargar
    CSV.foreach(@archivo, headers: true) do |fila|
      @datos << fila.to_h
    end
    puts "Cargados #{@datos.length} registros"
  end

  def filtrar(columna, valor)
    @datos.select { |f| f[columna] == valor }
  end

  def exportar(archivo_salida)
    return if @datos.empty?
    CSV.open(archivo_salida, "w") do |csv|
      csv << @datos.first.keys
      @datos.each { |fila| csv << fila.values }
    end
  end

  def resumen(columna)
    valores = @datos.map { |f| f[columna] }
    frecuencias = valores.tally
    puts "Resumen de #{columna}:"
    frecuencias.sort_by { |_, v| -v }.each do |valor, count|
      puts "  #{valor}: #{count}"
    end
  end
end
```

PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Cuál es la diferencia entre puts y print?**
puts agrega un salto de línea al final; print no.

❓ **¿Por qué gets incluye \n al final?**
gets lee exactamente lo que el usuario escribe, incluyendo Enter (\n).
Usa chomp para eliminarlo.

❓ **¿Qué hace chomp exactamente?**
Elimina el separador de registro (\n, \r\n) del final del string.

❓ **¿Cómo convierto entrada a número?**
Con to_i (entero) o to_f (float). Para validación usa match? con regex.

❓ **¿Qué es % en strings?**
El operador % formatea strings al estilo printf. "Hola %s" % ["Mundo"]

❓ **¿Cómo leo un archivo completo?**
File.read para todo, File.readlines para array de líneas, File.foreach
para iterar sin cargar todo en memoria.

❓ **¿Cómo escribo sin sobrescribir?**
File.write("archivo", "texto", mode: "a") o "a" en File.open.

❓ **¿Qué significan los modos de apertura?**
"r" lectura, "w" escritura (sobrescribe), "a" agregar, "b" binario.

🏗️ **Arquitecto de Software** — Diseña un "Sistema de Exportación Multi-formato"

Diseña una clase `Exportador` que:

1. Lea datos desde CSV, JSON o YAML
2. Exporte a CSV, JSON, YAML o TXT formateado
3. Soporte filtros y transformaciones
4. Muestre progreso durante la exportación
5. Genere reportes con resumen estadístico

```ruby
require "csv"
require "json"
require "yaml"

class Exportador
  def initialize
    @datos = []
  end

  def cargar(archivo, formato = nil)
    formato ||= File.extname(archivo).delete(".")
    @datos = case formato
             when "csv" then CSV.foreach(archivo, headers: true).map(&:to_h)
             when "json" then JSON.parse(File.read(archivo))
             when "yaml" then YAML.safe_load(File.read(archivo))
             else raise "Formato no soportado: #{formato}"
             end
    puts "Cargados #{@datos.length} registros"
  end

  def filtrar(&bloque)
    @datos = @datos.select(&bloque)
    self
  end

  def transformar(columna, &bloque)
    @datos.each { |f| f[columna] = bloque.call(f[columna]) if f.key?(columna) }
    self
  end

  def exportar(archivo, formato)
    case formato
    when "csv" then exportar_csv(archivo)
    when "json" then File.write(archivo, JSON.pretty_generate(@datos))
    when "yaml" then File.write(archivo, @datos.to_yaml)
    when "txt" then exportar_txt(archivo)
    end
    puts "Exportados #{@datos.length} registros"
  end

  def resumen
    return {} if @datos.empty?
    @datos.first.keys.each_with_object({}) do |col, h|
      valores = @datos.map { |f| f[col] }
      h[col] = { unicos: valores.uniq.length, tipo: valores.first.class.name }
    end
  end

  private

  def exportar_csv(archivo)
    return if @datos.empty?
    CSV.open(archivo, "w") do |csv|
      csv << @datos.first.keys
      @datos.each { |f| csv << f.values }
    end
  end

  def exportar_txt(archivo)
    File.open(archivo, "w") do |f|
      f.puts @datos.first.keys.join("\t")
      @datos.each { |row| f.puts row.values.join("\t") }
    end
  end
end
```
