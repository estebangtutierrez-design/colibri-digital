1. CREACIÓN DE STRINGS
===============================================================================

--- 1.1 LITERALES DE STRINGS ---

Ruby ofrece múltiples formas de crear strings. Cada una tiene propósitos
específicos.

```ruby
# Comillas simples (no interpola, menos escapes)
simple = 'Hola Mundo'
sin_interpolacion = 'Hola #{nombre}'  # Literal: Hola #{nombre}
escapes_limitados = 'Salto de línea \\n literal'  # Solo \\ y \'

# Comillas dobles (interpola, más escapes)
doble = "Hola Mundo"
con_interpolacion = "Hola #{nombre}"  # Interpola
con_escape = "Salto de\nlínea"  # Interpreta \n

# %q y %Q - sintaxis alternativa
# %q es como comillas simples (no interpola)
q_string = %q(Hola Mundo sin interpolación #{nombre})
puts q_string  # Hola Mundo sin interpolación #{nombre}

# %Q es como comillas dobles (interpola)
Q_string = %Q(Hola #{nombre} esto interpola)
puts Q_string  # Hola Ana esto interpola

# % (con cualquier delimitador) - equivalente a %Q
otro = %(Hola #{nombre})
puts otro  # Hola Ana

# Delimitadores pueden ser cualquier par
puts %{Hola}
puts %[Hola]
puts %|Hola|
puts %!Hola!
puts %(Hola)
```

--- 1.2 STRINGS CON %w, %W, %i ---

```ruby
# %w - array de strings (sin interpolación)
palabras = %w[hola mundo ruby es genial]
puts palabras.inspect  # ["hola", "mundo", "ruby", "es", "genial"]

# %W - array de strings (con interpolación)
nombre = "Ana"
saludos = %W[hola #{nombre} adiós]
puts saludos.inspect  # ["hola", "Ana", "adiós"]

# %i - array de símbolos (sin interpolación)
simbolos = %i[rojo verde azul]
puts simbolos.inspect  # [:rojo, :verde, :azul]

# %I - array de símbolos (con interpolación)
color = "rojo"
colores = %I[#{color} verde]
puts colores.inspect  # [:rojo, :verde]
```

--- 1.3 HEREDOC ---

El heredoc permite crear strings multilínea de forma limpia.

```ruby
# Heredoc básico
texto = <<TEXTO
Esta es la línea 1
Esta es la línea 2
Esta es la línea 3
TEXTO

puts texto

# Heredoc con sangría (<<~)
# Elimina la indentación automáticamente
texto_sangria = <<~TEXTO
  Esta línea está indentada
    Esta también pero con más espacio
  La indentación se normaliza
TEXTO

puts texto_sangria

# Heredoc sin interpolación (<<~'')
nombre = "Ana"
texto_sin_interp = <<~'TEXTO'
  Hola #{nombre}
  Esto se muestra literal
TEXTO

puts texto_sin_interp  # Hola #{nombre}

# Heredoc con interpolación (<<~"")
nombre = "Ana"
texto_con_interp = <<~"TEXTO"
  Hola #{nombre}
  Esto interpola correctamente
TEXTO

puts texto_con_interp  # Hola Ana

# Heredoc con método (squiggly heredoc)
def render_mensaje
  <<~MSG.strip
    Este es un mensaje
    que se forma con heredoc
    y se puede usar en métodos
  MSG
end

puts render_mensaje

# Heredoc anidados (cada uno con su delimitador)
texto_anidado = <<~PRINCIPAL
  Texto principal
  #{<<~SECUNDARIO.strip}
    Texto secundario
    con interpolación
  SECUNDARIO}
PRINCIPAL

puts texto_anidado
```

2. INTERPOLACIÓN DE STRINGS
===============================================================================

--- 2.1 INTERPOLACIÓN BÁSICA ---

```ruby
nombre = "Ana"
edad = 30

# Variables
puts "Nombre: #{nombre}"
puts "Edad: #{edad}"

# Expresiones aritméticas
puts "Suma: #{10 + 20}"
puts "Promedio: #{(85 + 90 + 95) / 3.0}"

# Llamadas a métodos
puts "Mayúsculas: #{nombre.upcase}"
puts "Longitud: #{nombre.length}"

# Variables de instancia y globales
@titulo = "Ruby Master"
$version = "3.3"
puts "Curso: #@titulo v#$version"
```

--- 2.2 INTERPOLACIÓN AVANZADA ---

```ruby
# Bloques
resultado = "#{(1..10).select(&:even?).map { |n| n ** 2 }.join(", ")}"
puts resultado  # 4, 16, 36, 64, 100

# Condicionales dentro de interpolación
edad = 20
mensaje = "#{edad >= 18 ? "Mayor" : "Menor"} de edad"
puts mensaje  # Mayor de edad

# Objetos personalizados
class Persona
  attr_reader :nombre, :edad
  def initialize(nombre, edad)
    @nombre = nombre
    @edad = edad
  end
  def to_s
    "#{@nombre} (#{@edad})"
  end
end

p = Persona.new("Ana", 30)
puts "Persona: #{p}"  # Persona: Ana (30)
```

--- 2.3 INTERPOLACIÓN SEGURA ---

```ruby
# Sin método to_s seguro
class SinToS
  # No define to_s
end

puts "Objeto: #{SinToS.new}"  # Objeto: #<SinToS:0x00007f...>

# Con nil
valor = nil
puts "Valor: #{valor}"  # Valor: (vacio)
puts "Valor: #{valor.inspect}"  # Valor: nil

# Interpolación con safe navigation
usuario = nil
puts "Usuario: #{usuario&.nombre || "Invitado"}"
```

3. MÉTODOS ESENCIALES DE STRINGS
===============================================================================

--- 3.1 LONGITUD Y BÚSQUEDA ---

```ruby
texto = "Ruby es un lenguaje hermoso"

# Longitud
puts texto.length      # 28
puts texto.size        # 28 (alias de length)
puts texto.bytesize    # 28 (bytes, cuenta caracteres multi-byte)
puts texto.empty?      # false

# Búsqueda
puts texto.include?("Ruby")    # true
puts texto.start_with?("Ruby")  # true
puts texto.end_with?("hermoso") # true
puts texto.index("lenguaje")    # 11 (posición de inicio)
puts texto.rindex("e")          # 18 (última ocurrencia)
puts texto.count("aeiou")       # 10 (cuenta vocales)
```

--- 3.2 TRANSFORMACIÓN DE CASE ---

```ruby
texto = "Ruby es Genial"

puts texto.upcase        # RUBY ES GENIAL
puts texto.downcase      # ruby es genial
puts texto.capitalize    # Ruby es genial (primera mayúscula)
puts texto.swapcase      # rUBY ES gENIAL
puts texto.upcase!       # Modifica el original (bang!)
puts texto               # RUBY ES GENIAL

# Comparación case-insensitive
puts "Ruby".casecmp("ruby")   # 0 (iguales)
puts "Ruby".casecmp?("ruby")  # true
```

--- 3.3 MODIFICACIÓN DE CONTENIDO ---

```ruby
texto = "  Hola Mundo Ruby  "

# Eliminar espacios
puts texto.strip         # "Hola Mundo Ruby"
puts texto.lstrip        # "Hola Mundo Ruby  "
puts texto.rstrip        # "  Hola Mundo Ruby"

# Rellenar
puts "Hola".center(15, "=")  # =====Hola======
puts "Hola".ljust(15, "*")   # Hola***********
puts "Hola".rjust(15, "*")   # ***********Hola

# Acortar
puts texto.truncate(10) rescue nil  # Necesita require active_support
# Método nativo:
puts texto[0..9] + "..." if texto.length > 10

# Invertir
puts "Ruby".reverse  # ybuR
puts "Anita lava la tina".reverse  # anit al aval atinA
```

--- 3.4 DIVISIÓN Y UNIÓN ---

```ruby
texto = "manzana,pera,uva,mandarina"

# split - dividir en array
frutas = texto.split(",")
puts frutas.inspect  # ["manzana", "pera", "uva", "mandarina"]

# Con límite
puts texto.split(",", 2).inspect   # ["manzana", "pera,uva,mandarina"]
puts texto.split(",", 3).inspect   # ["manzana", "pera", "uva,mandarina"]

# split con regex
datos = "Ana:30:Ingeniera"
puts datos.split(/:/).inspect  # ["Ana", "30", "Ingeniera"]

# split sin argumento (divide por whitespace)
palabras = "Hola   Mundo  Ruby".split
puts palabras.inspect  # ["Hola", "Mundo", "Ruby"]

# join - unir array en string
colores = ["rojo", "verde", "azul"]
puts colores.join(" - ")      # rojo - verde - azul
puts colores.join(", ")       # rojo, verde, azul
puts colores.join             # rojoverdeazul
```

--- 3.5 CHOMB, STRIP, CHOP ---

```ruby
linea = "Hola Mundo\n"

# chomp - elimina el separador de línea del final
puts linea.chomp.inspect  # "Hola Mundo"

# chomp con argumento
puts "Hola Mundo".chomp("Mundo")  # "Hola "

# chop - elimina el último carácter
puts "Hola Mundo".chop   # "Hola Mun"
puts "Hola\n".chop       # "Hola"

# strip - elimina whitespace al inicio y final
puts "  Hola  ".strip.inspect  # "Hola"

# strip elimina: espacios, tabs, nuevas líneas
puts "\t\n  Hola  \n\t".strip  # "Hola"
```

--- 3.6 SUSTITUCIÓN ---

```ruby
texto = "Hola Mundo, Hola Ruby"

# sub - reemplaza la primera ocurrencia
puts texto.sub("Hola", "Adiós")  # "Adiós Mundo, Hola Ruby"

# gsub - reemplaza todas las ocurrencias
puts texto.gsub("Hola", "Adiós") # "Adiós Mundo, Adiós Ruby"

# gsub con regex
puts texto.gsub(/[aeiou]/, "*")  # "H*l* M*nd*, H*l* R*by"

# gsub con bloque
resultado = texto.gsub(/Hola/) { |match| match.upcase.reverse }
puts resultado  # "ALOH Mundo, ALOH Ruby"

# Versiones destructivas (bang!)
texto.gsub!("Hola", "Hey")
puts texto  # "Hey Mundo, Hey Ruby"
```

--- 3.7 CONVERSIÓN ---

```ruby
puts 42.to_s           # "42"
puts 3.14.to_s         # "3.14"
puts true.to_s         # "true"
puts nil.to_s          # ""
puts [1, 2, 3].to_s    # "[1, 2, 3]"
puts :simbolo.to_s     # "simbolo"

# A entero/float (desde string)
puts "42".to_i         # 42
puts "3.14".to_f       # 3.14
puts "42abc".to_i      # 42 (tolera basura)
puts "abc".to_i        # 0

# A símbolo
puts "nombre".to_sym   # :nombre
puts "mi metodo".to_sym  # :"mi metodo"

# Convertir a otros formatos
puts "255".to_i(16)    # 597 (hexadecimal)
puts "11111111".to_i(2) # 255 (binario)
puts 255.to_s(16)      # "ff"
puts 255.to_s(2)       # "11111111"
```

4. COMPARACIÓN DE STRINGS
===============================================================================

--- 4.1 OPERADORES DE COMPARACIÓN ---

```ruby
# Igualdad (case-sensitive)
puts "Ruby" == "Ruby"    # true
puts "Ruby" == "ruby"    # false

# Desigualdad
puts "Ruby" != "Python"  # true

# Orden alfabético
puts "a" < "b"   # true (a viene antes)
puts "A" < "a"   # true (mayúsculas antes que minúsculas en ASCII)
puts "abc" < "abd"  # true

# Spaceship
puts "a" <=> "b"   # -1
puts "b" <=> "a"   # 1
puts "a" <=> "a"   # 0
```

--- 4.2 COMPARACIÓN CASE-INSENSITIVE ---

```ruby
# casecmp - comparación sin importar mayúsculas
puts "Ruby".casecmp("ruby")   # 0
puts "Ruby".casecmp("RUBY")   # 0
puts "Ruby".casecmp("Python") # -1 (o 1)

# casecmp? - versión booleana (Ruby 2.4+)
puts "Ruby".casecmp?("ruby")   # true
puts "Ruby".casecmp?("RUBY")   # true
puts "Ruby".casecmp?("Python") # false
```

--- 4.3 COMPARACIÓN DE LONGITUD ---

```ruby
puts "Ruby".length <=> "Python".length  # -1
puts "Ruby".size == "Java".size          # false (4 == 4?) true

# Strings vacíos
puts "" == ""         # true
puts "".empty?        # true
puts " ".empty?       # false
puts " ".strip.empty? # true
```

5. EXPRESIONES REGULARES CON STRINGS
===============================================================================

--- 5.1 MATCHING BÁSICO ---

```ruby
# =~ para coincidencia
puts "Hola Ruby" =~ /Ruby/    # 5 (posición)
puts /Ruby/ =~ "Hola Ruby"    # 5 (al revés también funciona)
puts "Hola Ruby" =~ /Python/  # nil

# match? (Ruby 2.4+) - más rápido, solo booleano
puts "Hola Ruby".match?(/Ruby/)    # true
puts "Hola Ruby".match?(/Python/)  # false

# match - devuelve MatchData
match_data = "Hola Ruby".match(/(Hola) (Ruby)/)
puts match_data[0]   # "Hola Ruby" (match completo)
puts match_data[1]   # "Hola" (primer grupo)
puts match_data[2]   # "Ruby" (segundo grupo)
puts match_data.pre_match   # "" (antes del match)
puts match_data.post_match  # "" (después del match)
```

--- 5.2 SCAN ---

```ruby
# scan - encuentra todas las coincidencias
texto = "Ruby es Ruby y más Ruby"

# Sin grupos - array de matches
puts texto.scan(/Ruby/).inspect  # ["Ruby", "Ruby", "Ruby"]

# Con grupos - array de arrays
puts texto.scan(/(R)(u)/).inspect  # [["R", "u"], ["R", "u"], ["R", "u"]]

# Con bloque
texto.scan(/Ruby/) { |match| puts "Encontrado: #{match}" }

# Extraer números
datos = "Edad: 30, Peso: 75.5, Altura: 1.75"
puts datos.scan(/\d+\.?\d*/).inspect  # ["30", "75", "5", "1", "75"]
```

--- 5.3 SUB Y GSUB CON REGEX ---

```ruby
texto = "Hola 123 Mundo 456 Ruby 789"

# sub - primera ocurrencia
puts texto.sub(/\d+/, "XXX")  # "Hola XXX Mundo 456 Ruby 789"

# gsub - todas las ocurrencias
puts texto.gsub(/\d+/, "XXX") # "Hola XXX Mundo XXX Ruby XXX"

# gsub con referencia a grupos
puts texto.gsub(/(\d+)/, '[\1]')  # "Hola [123] Mundo [456] Ruby [789]"

# gsub con bloque
resultado = texto.gsub(/\d+/) { |n| n.to_i * 2 }
puts resultado  # "Hola 246 Mundo 912 Ruby 1578"

# Eliminar caracteres no deseados
puts texto.gsub(/\d+/, "")  # "Hola  Mundo  Ruby "
puts texto.gsub(/\s+/, " ")  # "Hola 123 Mundo 456 Ruby 789"
```

--- 5.4 PATRONES COMUNES ---

```ruby
# Validación de email
email = "usuario@dominio.com"
puts email.match?(/\A[\w.+-]+@[\w-]+\.[\w.]+\z/)  # true

# Validación de teléfono
telefono = "+34 612 345 678"
puts telefono.match?(/\A\+?\d{1,3}[\s-]?\d{3}[\s-]?\d{3}[\s-]?\d{3}\z/)  # true

# Extraer URLs
texto = "Visita https://ruby-lang.org y http://example.com"
urls = texto.scan(/https?:\/\/[\w.-]+/)
puts urls.inspect  # ["https://ruby-lang.org", "http://example.com"]

# Validación de fecha
fecha = "2024-01-15"
puts fecha.match?(/\A\d{4}-\d{2}-\d{2}\z/)  # true
```

--- 6.1 UTF-8 EN RUBY ---

```ruby
# Ruby usa UTF-8 por defecto desde 1.9
texto = "ñandú está aquí"
puts texto.length     # 15 (caracteres reales)
puts texto.bytesize   # 18 (bytes, caracteres multi-byte)

# encoding
puts texto.encoding   # UTF-8

# Cambiar encoding
utf8 = "Hola".encode("UTF-8")
puts utf8.encoding    # UTF-8

# Forzar encoding (peligroso)
bytes = "Hola".b
puts bytes.encoding   # ASCII-8BIT (binario)

# Caracteres especiales
puts "¿Cómo estás?"
puts "¡Bienvenido!"
puts "äöüñçéèêë"
puts "русский язык"
puts "中文"
puts "日本語"
puts "한국어"
```

--- 6.2 MANIPULACIÓN DE ENCODING ---

```ruby
# Detectar encoding
texto = "Hola".encode("ISO-8859-1")
puts texto.encoding.name      # ISO-8859-1
puts texto.valid_encoding?    # true

# Forzar interpretación (no convertir)
original = "Hola".encode("ISO-8859-1")
forzado = original.force_encoding("UTF-8")
puts forzado.encoding  # UTF-8 (pero bytes siguen siendo ISO-8859-1)

# Normalizar Unicode
compuesto = "é"  # e + acento agudo
puts compuesto.bytes.inspect

# Eliminar caracteres no ASCII
texto = "Hola ñandú 123"
ascii = texto.gsub(/[^\x00-\x7F]/, "")
puts ascii  # "Hola nd 123"
```

--- 7.1 STRING#NEXT Y STRING#SUCC ---

```ruby
# next/succ - siguiente secuencia
puts "a".next     # "b"
puts "z".next     # "aa"
puts "A".next     # "B"
puts "Z".next     # "AA"
puts "1".next     # "2"
puts "9".next     # "10"
puts "a9".next    # "b0"

# succ es alias de next
puts "ruby".succ  # "rubz"

# Útil para generación de IDs
id = "A001"
5.times { puts id = id.next }
# A002, A003, A004, A005, A006
```

--- 7.2 STRING#DELETE ---

```ruby
texto = "Hola Mundo Ruby"

# delete - elimina caracteres específicos
puts texto.delete("aeiou")  # "Hl Mnd Rby"

# delete con conjuntos
puts texto.delete("a-z")    # "H M R" (solo mayúsculas)
puts texto.delete("^a-z")   # "olaundo" (solo minúsculas)
puts texto.delete(" ", "^R") # "HolaMundoRuby" (espacios excepto antes de R)
```

--- 7.3 STRING#COUNT ---

```ruby
texto = "Hola Mundo Ruby"

# count - cuenta ocurrencias
puts texto.count("a")        # 1
puts texto.count("aeiou")    # 5
puts texto.count("a-z")      # 10 (minúsculas)
puts texto.count("A-Z")      # 3 (mayúsculas)
puts texto.count("^a-zA-Z ") # 0 (solo espacio y letras)
```

--- 7.4 STRING#TR ---

tr reemplaza caracteres uno a uno (como tr en shell).

```ruby
# tr - traducir caracteres
puts "Hola".tr("aeiou", "AEIOU")  # "HOlA"

# tr con rangos
puts "abc123".tr("a-z", "A-Z")  # "ABC123"
puts "ruby".tr("ry", "RY")      # "RubY"

# tr_s - traducir y squeezear
puts "Hola".tr_s("aeiou", "-")  # "H-l-" (reemplaza y elimina duplicados)

# Eliminar con tr
puts "Hola 123".tr("0-9", "")   # "Hola "
```

--- 7.5 STRING#SQUEEZE ---

```ruby
# squeeze - elimina caracteres repetidos consecutivos
puts "Hoolaa Muundoo Ruby".squeeze  # "Hola Mundo Ruby"

# squeeze con argumento
puts "aaabbbccc".squeeze("a-c")  # "abc"
puts "aaabbbccc".squeeze("a")    # "abbbccc"
```

--- 7.6 STRING#UNPACK ---

```ruby
# unpack - desempaqueta string según formato
puts "Ruby".unpack("C4").inspect  # [82, 117, 98, 121] (bytes)
puts "Ruby".unpack("H*").inspect  # ["52756279"] (hex)
puts "Ruby".unpack("B*").inspect  # binario

# Template directivas comunes
# C - entero de 8 bits sin signo
# H - hex string (high nybble first)
# A - string con espacios eliminados
# Z - string terminado en null
```

8. STRINGS INMUTABLES (FROZEN STRINGS)

--- 8.1 FROZEN STRINGS ---

Ruby permite congelar strings para hacerlos inmutables. Esto mejora el
rendimiento y la seguridad.

```ruby
# frozen_string_literal: true
# Si pones esto al inicio del archivo, todos los strings literales
# se congelan automáticamente

# Congelar un string
texto = "Ruby".freeze
puts texto.frozen?  # true

# Intentar modificar un string congelado da error
# texto.upcase!  # RuntimeError: can't modify frozen String

# Pero los métodos no destructivos funcionan
puts texto.upcase  # RUBY (crea nuevo string)

# frozen_string_literal pragma (archivo completo)
# # frozen_string_literal: true
```

--- 8.2 DEDUPLICACIÓN CON -@ ---

```ruby
# En Ruby 3.2+, el operador -@ devuelve un string congelado deduplicado
str1 = -"Hola"
str2 = -"Hola"
puts str1.object_id == str2.object_id  # true (mismo objeto)

# Sin deduplicación
str3 = "Hola"
str4 = "Hola"
puts str3.object_id == str4.object_id  # false
```

9. PROYECTO: PROCESADOR DE TEXTO
===============================================================================

```ruby
# procesador_texto.rb - Procesador de texto completo
class ProcesadorTexto
  def initialize(texto)
    @texto = texto
  end

  def estadisticas
    {
      caracteres: @texto.length,
      palabras: @texto.split.length,
      lineas: @texto.lines.count,
      oraciones: @texto.split(/[.!?]+/).length,
      bytes: @texto.bytesize,
      espacios: @texto.count(" "),
      vocales: @texto.downcase.count("aeiouáéíóúü"),
      consonantes: @texto.downcase.count("a-z") - @texto.downcase.count("aeiouáéíóúü")
    }
  end

  def palabras_mas_comunes(n = 10)
    frecuencias = @texto.downcase
      .gsub(/[^a-záéíóúüñ\s]/, "")
      .split
      .tally
      .sort_by { |_, v| -v }
      .first(n)
    frecuencias
  end

  def reemplazar(patron, reemplazo)
    @texto.gsub(patron, reemplazo)
  end

  def invertir_palabras
    @texto.split.map(&:reverse).join(" ")
  end

  def a_slug
    @texto.downcase
      .gsub(/[^a-z0-9\s-]/, "")
      .gsub(/\s+/, "-")
      .gsub(/-+/, "-")
      .gsub(/\A-|-\z/, "")
  end

  def censurar(palabras_prohibidas)
    resultado = @texto
    palabras_prohibidas.each do |palabra|
      resultado = resultado.gsub(/#{Regexp.escape(palabra)}/i, "*" * palabra.length)
    end
    resultado
  end

  def mostrar_estadisticas
    stats = estadisticas
    puts "=== ESTADÍSTICAS DEL TEXTO ==="
    stats.each { |k, v| puts "  #{k.capitalize}: #{v}" }
  end
end

texto = "Ruby es un lenguaje de programación hermoso. Ruby fue creado por
Yukihiro Matsumoto en 1995. Ruby es conocido por su sintaxis elegante

puts "\nCensurado: #{procesador.censurar(["Ruby", "Matsumoto"])}"
```

```ruby
class GeneradorPlantilla
  def initialize(template)
    @template = template
  end

  def renderizar(variables)
    resultado = @template.dup

    variables.each do |key, value|
      resultado = resultado.gsub("{{#{key}}}", value.to_s)
    end

    resultado
  end

  def renderizar_con_bloques(variables)
    resultado = @template.dup

    variables.each do |key, value|
      resultado = resultado.gsub("{{#{key}}}") { |_| value.to_s }
    end

    # Procesar condicionales
    resultado = resultado.gsub(/{% if (\w+) %}(.*?){% endif %}/m) do
      variable = $1
      contenido = $2
      variables[variable.to_sym] || variables[variable] ? contenido : ""
    end

    # Procesar bucles
    resultado = resultado.gsub(/{% for (\w+) in (\w+) %}(.*?){% endfor %}/m) do
      item_var = $1
      list_var = $2
      template = $3
      items = variables[list_var.to_sym] || variables[list_var] || []
      items.map { |item| template.gsub("{{#{item_var}}}", item.to_s) }.join
    end

    resultado
  end
end

# Ejemplo de uso
template = <<~TEMPLATE

  <p>Tienes {{edad}} años.</p>

  <ul>
  {% for hobbie in hobbies %}
    <li>{{hobbie}}</li>
  {% endfor %}
  </ul>
TEMPLATE

plantilla = GeneradorPlantilla.new(template)

  nombre: "Ana",
```

PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Cuál es la diferencia entre comillas simples y dobles?**
Las comillas simples no interpretan escapes (\n) ni interpolar (#{}). Las
dobles sí. Usa simples cuando no necesites estas características.

❓ **¿Qué es un heredoc?**
Un heredoc (<<~TEXTO) permite crear strings multilínea de forma legible.
<<~ elimina la indentación automáticamente.

❓ **¿Cómo reemplazo texto en un string?**
sub para primera ocurrencia, gsub para todas. Ambos aceptan strings o regex.

❓ **¿Cuál es la diferencia entre chomp y chop?**
chomp elimina el separador de línea (\n, \r\n) del final. chop elimina el
último carácter siempre.

❓ **¿Cómo divido un string en partes?**
Con split: "a,b,c".split(",") => ["a", "b", "c"]

❓ **¿Cómo uno un array en string?**
Con join: ["a", "b"].join(", ") => "a, b"

❓ **¿Qué significa frozen_string_literal?**
Es un pragma que congela todos los strings literales del archivo, mejorando
rendimiento y seguridad.

❓ **¿Cómo trabajo con caracteres acentuados?**
Ruby usa UTF-8 por defecto. Caracteres como ñ, á, é funcionan naturalmente.
La mayoría de métodos (length, upcase) los manejan correctamente.

❓ **¿Qué métodos modifican el string original?**
Los que terminan en ! (bang): upcase!, downcase!, gsub!, etc.
Los métodos sin ! crean un nuevo string.

🏗️ **Arquitecto de Software** — Diseña un "Motor de Búsqueda de Texto"

Diseña un sistema de búsqueda de texto completo que:

1. Indexe documentos (strings) por palabras clave
2. Soporte búsqueda simple, por frase exacta y por negación
3. Devuelva resultados con contexto (alrededor de la palabra)
4. Resalte las palabras encontradas en el contexto
5. Ordene resultados por relevancia (frecuencia de coincidencias)

```ruby
class Documento
  attr_reader :id, :titulo, :contenido

  def initialize(id, titulo, contenido)
    @id = id
    @titulo = titulo
    @contenido = contenido
  end

  def palabras
    @contenido.downcase.gsub(/[^a-záéíóúüñ\s]/, "").split
  end

  def resaltar(terminos, contexto_chars = 60)
    resultado = @contenido
    terminos.each do |t|
      resultado = resultado.gsub(/#{Regexp.escape(t)}/i) { |m| "**#{m}**" }
    end
    resultado
  end

  def contexto(palabra, chars = 40)
    idx = @contenido.downcase.index(palabra.downcase)
    return nil unless idx
    inicio = [idx - chars, 0].max
    fin = [idx + palabra.length + chars, @contenido.length].min
    contexto = @contenido[inicio..fin]
    contexto = "...#{contexto}" if inicio > 0
    contexto = "#{contexto}..." if fin < @contenido.length
    contexto
  end
end

class MotorBusqueda
  def initialize
    @documentos = []
    @indice = {}  # palabra => { doc_id => frecuencia }
  end

  def agregar(documento)
    @documentos << documento
    idx = @documentos.length - 1
    documento.palabras.each do |palabra|
      @indice[palabra] ||= {}
      @indice[palabra][idx] ||= 0
      @indice[palabra][idx] += 1
    end
  end

  def buscar(termino)
    termino = termino.downcase
    resultados = @indice[termino] || {}
    docs_encontrados = resultados.map do |doc_idx, frecuencia|
      doc = @documentos[doc_idx]
      {
        documento: doc,
        frecuencia: frecuencia,
        contexto: doc.contexto(termino),
        relevancia: frecuencia.to_f / doc.palabras.length
      }
    end
    docs_encontrados.sort_by { |r| -r[:relevancia] }
  end

  def buscar_frase(frase)
    terminos = frase.downcase.split
    resultados = buscar(terminos.first)
    resultados.select do |r|
      texto = r[:documento].contenido.downcase
      terminos.all? { |t| texto.include?(t) } && texto.include?(frase.downcase)
    end
  end

  def buscar_con_negacion(termino, negado)
    resultados = buscar(termino)
    resultados.reject { |r| r[:documento].contenido.downcase.include?(negado.downcase) }
  end
end
```
