```ruby
# modulo_operaciones.rb - Operaciones matemáticas de la calculadora
module ModuloOperaciones
  # Constantes
  PRECISION_POR_DEFECTO = 2

  # Operaciones básicas
  def sumar(a, b)
    a + b
  end

  def restar(a, b)
    a - b
  end

  def multiplicar(a, b)
    a * b
  end

  def dividir(a, b)
    if b == 0
      raise ZeroDivisionError, "No se puede dividir por cero"
    end
    a / b.to_f
  end

  # Operaciones avanzadas
  def potencia(base, exponente)
    base ** exponente
  end

  def modulo(a, b)
    if b == 0
      raise ZeroDivisionError, "No se puede calcular módulo con divisor cero"
    end
    a % b
  end

  def raiz_cuadrada(numero)
    if numero < 0
      raise ArgumentError, "No se puede calcular raíz cuadrada de un número negativo"
    end
    Math.sqrt(numero)
  end

  def porcentaje(valor, total)
    if total == 0
      raise ZeroDivisionError, "El total no puede ser cero"
    end
    (valor.to_f / total) * 100
  end

  def factorial(numero)
    if numero < 0
      raise ArgumentError, "No se puede calcular factorial de un número negativo"
    end
    if numero == 0 || numero == 1
      1
    else
      (1..numero).reduce(:*)
    end
  end

  # Utilidad
  def tabla_multiplicar(numero, limite = 10)
    (1..limite).map { |i| { factor: i, resultado: numero * i } }
  end

  def redondear(numero, decimales = PRECISION_POR_DEFECTO)
    numero.round(decimales)
  end
end
```

```ruby
# modulo_historial.rb - Historial de operaciones
require "json"

class Historial
  attr_reader :operaciones

  def initialize
    @operaciones = []
    @archivo = "historial_calculadora.json"
  end

  def agregar(operacion)
    entrada = {
      fecha: Time.now.strftime("%Y-%m-%d %H:%M:%S"),
      operacion: operacion,
      resultado: operacion[:resultado]
    }
    @operaciones << entrada
  end

  def mostrar(limite = 10)
    if @operaciones.empty?
      puts "  No hay operaciones en el historial."
      return
    end

    puts "  Últimas #{[limite, @operaciones.length].min} operaciones:"
    puts "  " + "-" * 50
    @operaciones.last(limite).each_with_index do |op, i|
      puts "  #{i + 1}. [#{op[:fecha]}] #{op[:operacion][:texto]} = #{op[:resultado]}"
    end
  end

  def limpiar
    @operaciones.clear
    puts "  Historial limpiado."
  end

  def exportar_json
    JSON.pretty_generate(@operaciones)
  end

  def guardar_archivo
    File.write(@archivo, exportar_json)
    puts "  Historial guardado en #{@archivo}"
  end

  def cargar_archivo
    if File.exist?(@archivo)
      datos = JSON.parse(File.read(@archivo))
      @operaciones = datos.map(&:symbolize_keys)
      puts "  Historial cargado: #{@operaciones.length} operaciones."
    end
  rescue JSON::ParserError
    puts "  Error al cargar el historial. Archivo corrupto."
    @operaciones = []
  end
end
```

```ruby
# modulo_validacion.rb - Validación de entrada del usuario
module ModuloValidacion
  def validar_numero(entrada)
    # Permitir números enteros y decimales
    if entrada.match?(/^-?\d+(\.\d+)?$/)
      entrada.include?(".") ? entrada.to_f : entrada.to_i
    else
      nil
    end
  end

  def validar_opcion_menu(entrada, max_opcion)
    numero = validar_numero(entrada)
    return nil unless numero.is_a?(Integer)
    return nil unless numero >= 0 && numero <= max_opcion
    numero
  end

  def solicitar_numero(mensaje)
    loop do
      print mensaje
      entrada = gets.chomp
      numero = validar_numero(entrada)
      return numero if numero
      puts "  Error: Ingresa un número válido (ej: 42, 3.14, -5)"
    end
  end

  def solicitar_entero_positivo(mensaje)
    loop do
      print mensaje
      entrada = gets.chomp
      numero = validar_numero(entrada)
      if numero.is_a?(Integer) && numero >= 0
        return numero
      end
      puts "  Error: Ingresa un número entero positivo (ej: 5)"
    end
  end

  def solicitar_opcion_menu(mensaje, max_opcion)
    loop do
      print mensaje
      entrada = gets.chomp
      opcion = validar_opcion_menu(entrada, max_opcion)
      return opcion if opcion
      puts "  Error: Ingresa un número entre 0 y #{max_opcion}"
    end
  end

  def solicitar_si_no(mensaje)
    loop do
      print mensaje
      respuesta = gets.chomp.downcase
      return true if %w[s si sí y yes].include?(respuesta)
      return false if %w[n no].include?(respuesta)
      puts "  Error: Responde 's' para sí o 'n' para no"
    end
  end
end
```

```ruby
#!/usr/bin/env ruby
# encoding: UTF-8
# frozen_string_literal: true

# calculadora.rb - Calculadora Interactiva Completa
# Ruby Master Book - Módulo I - Proyecto Final
# Autor: Profesor Búho AI
# Versión: 1.0.0

require "json"
require_relative "modulo_operaciones"
require_relative "modulo_menu"
require_relative "modulo_validacion"
require_relative "modulo_historial"

class Calculadora
  include ModuloOperaciones
  include ModuloMenu
  include ModuloValidacion

  def initialize
    @historial = Historial.new
    @historial.cargar_archivo
  end

  def ejecutar
    loop do
      mostrar_banner
      mostrar_menu

      opcion = solicitar_opcion_menu("  Selecciona una opción: ", 12)

      case opcion
      when 0 then break
      when 1 then operacion_sumar
      when 2 then operacion_restar
      when 3 then operacion_multiplicar
      when 4 then operacion_dividir
      when 5 then operacion_modulo
      when 6 then operacion_potencia
      when 7 then operacion_raiz
      when 8 then operacion_porcentaje
      when 9 then operacion_factorial
      when 10 then operacion_tabla
      when 11 then @historial.mostrar
      when 12 then @historial.limpiar
      end

      @historial.guardar_archivo unless opcion == 0
      pausa unless opcion == 0
    end

    mostrar_despedida
    @historial.guardar_archivo
  end

  private

  def registrar_operacion(texto, resultado)
    @historial.agregar({ texto: texto, resultado: resultado })
  end

  def operacion_sumar
    a = solicitar_numero("  Primer número: ")
    b = solicitar_numero("  Segundo número: ")
    resultado = sumar(a, b)
    texto = "#{formatear_numero(a)} + #{formatear_numero(b)}"
    mostrar_resultado("SUMA", resultado, texto)
    registrar_operacion(texto, resultado)
  end

  def operacion_restar
    a = solicitar_numero("  Primer número: ")
    b = solicitar_numero("  Segundo número: ")
    resultado = restar(a, b)
    texto = "#{formatear_numero(a)} - #{formatear_numero(b)}"
    mostrar_resultado("RESTA", resultado, texto)
    registrar_operacion(texto, resultado)
  end

  def operacion_multiplicar
    a = solicitar_numero("  Primer número: ")
    b = solicitar_numero("  Segundo número: ")
    resultado = multiplicar(a, b)
    texto = "#{formatear_numero(a)} * #{formatear_numero(b)}"
    mostrar_resultado("MULTIPLICACIÓN", resultado, texto)
    registrar_operacion(texto, resultado)
  end

  def operacion_dividir
    a = solicitar_numero("  Dividendo: ")
    b = solicitar_numero("  Divisor: ")
    begin
      resultado = dividir(a, b)
      texto = "#{formatear_numero(a)} / #{formatear_numero(b)}"
      mostrar_resultado("DIVISIÓN", resultado, texto)
      registrar_operacion(texto, resultado)
    rescue ZeroDivisionError => e
      mensaje_error(e.message)
    end
  end

  def operacion_modulo
    a = solicitar_numero("  Número: ")
    b = solicitar_numero("  Divisor: ")
    begin
      resultado = modulo(a, b)
      texto = "#{formatear_numero(a)} % #{formatear_numero(b)}"
      mostrar_resultado("MÓDULO", resultado, texto)
      registrar_operacion(texto, resultado)
    rescue ZeroDivisionError => e
      mensaje_error(e.message)
    end
  end

  def operacion_potencia
    base = solicitar_numero("  Base: ")
    exponente = solicitar_numero("  Exponente: ")
    resultado = potencia(base, exponente)
    texto = "#{formatear_numero(base)} ^ #{formatear_numero(exponente)}"
    mostrar_resultado("POTENCIA", resultado, texto)
    registrar_operacion(texto, resultado)
  end

  def operacion_raiz
    a = solicitar_numero("  Número: ")
    begin
      resultado = raiz_cuadrada(a)
      texto = "√(#{formatear_numero(a)})"
      mostrar_resultado("RAÍZ CUADRADA", resultado, texto)
      registrar_operacion(texto, resultado)
    rescue ArgumentError => e
      mensaje_error(e.message)
    end
  end

  def operacion_porcentaje
    valor = solicitar_numero("  Valor: ")
    total = solicitar_numero("  Total: ")
    begin
      resultado = porcentaje(valor, total)
      texto = "#{formatear_numero(valor)} es qué % de #{formatear_numero(total)}"
      mostrar_resultado("PORCENTAJE", "#{resultado}%", texto)
      registrar_operacion(texto, "#{resultado}%")
    rescue ZeroDivisionError => e
      mensaje_error(e.message)
    end
  end

  def operacion_factorial
    a = solicitar_entero_positivo("  Número entero positivo: ")
    begin
      resultado = factorial(a)
      texto = "#{a}!"
      mostrar_resultado("FACTORIAL", resultado, texto)
      registrar_operacion(texto, resultado)
    rescue ArgumentError => e
      mensaje_error(e.message)
    end
  end

  def operacion_tabla
    a = solicitar_entero_positivo("  Número: ")
    limite = solicitar_entero_positivo("  Hasta (ej: 10): ")
    limite = 10 if limite == 0

    puts ""
    puts "  ════════════════════════════════"
    puts "  TABLA DEL #{a}"
    puts "  ════════════════════════════════"

    tabla = tabla_multiplicar(a, limite)
    tabla.each do |item|
      puts "  #{a} x #{item[:factor]} = #{item[:resultado]}"
    end

    puts "  ════════════════════════════════"
    registrar_operacion("Tabla del #{a} (x#{limite})", "completada")
  end

  def formatear_numero(numero)
    if numero == numero.to_i
      numero.to_i.to_s
    else
      numero.to_s
    end
  end
end

# Ejecutar la calculadora si se llama directamente
if __FILE__ == $PROGRAM_NAME
  calculadora = Calculadora.new
  calculadora.ejecutar
end
```

```ruby
# test_calculadora.rb - Pruebas unitarias para la calculadora
require "minitest/autorun"
require_relative "modulo_operaciones"
require_relative "modulo_validacion"

class TestModuloOperaciones < Minitest::Test
  include ModuloOperaciones

  def setup
    @calc = self
  end

  def test_sumar
    assert_equal 5, sumar(2, 3)
    assert_equal 0, sumar(-1, 1)
    assert_equal(-5, sumar(-2, -3))
    assert_in_delta 5.5, sumar(2.5, 3.0)
  end

  def test_restar
    assert_equal 3, restar(5, 2)
    assert_equal(-3, restar(2, 5))
    assert_equal 0, restar(5, 5)
  end

  def test_multiplicar
    assert_equal 15, multiplicar(3, 5)
    assert_equal 0, multiplicar(0, 5)
    assert_equal(-15, multiplicar(-3, 5))
  end

  def test_dividir
    assert_equal 2.5, dividir(5, 2)
    assert_equal 1.0, dividir(5, 5)
    assert_raises(ZeroDivisionError) { dividir(5, 0) }
  end

  def test_potencia
    assert_equal 8, potencia(2, 3)
    assert_equal 1, potencia(5, 0)
    assert_equal 0.5, potencia(2, -1)
  end

  def test_modulo
    assert_equal 1, modulo(10, 3)
    assert_equal 0, modulo(10, 5)
    assert_raises(ZeroDivisionError) { modulo(10, 0) }
  end

  def test_raiz_cuadrada
    assert_equal 3.0, raiz_cuadrada(9)
    assert_equal 0.0, raiz_cuadrada(0)
    assert_raises(ArgumentError) { raiz_cuadrada(-1) }
  end

  def test_porcentaje
    assert_equal 50.0, porcentaje(50, 100)
    assert_equal 0.0, porcentaje(0, 100)
    assert_raises(ZeroDivisionError) { porcentaje(50, 0) }
  end

  def test_factorial
    assert_equal 1, factorial(0)
    assert_equal 1, factorial(1)
    assert_equal 120, factorial(5)
    assert_raises(ArgumentError) { factorial(-1) }
  end

  def test_tabla_multiplicar
    tabla = tabla_multiplicar(5, 3)
    assert_equal 3, tabla.length
    assert_equal({ factor: 1, resultado: 5 }, tabla[0])
    assert_equal({ factor: 3, resultado: 15 }, tabla[2])
  end
end
```

PREGUNTAS FRECUENTES
===============================================================================

❓ **¿Cómo ejecuto la calculadora?**
Guarda todos los archivos en el mismo directorio y ejecuta:
`ruby calculadora.rb`

❓ **¿Qué hago si tengo un error de sintaxis?**
Revisa que todos los `end` estén correctamente cerrados. Cada `def`, `class`,
`module`, `do`, `if`, `case` necesita su `end`.

❓ **¿Cómo agrego una nueva operación?**
1. Agrega el método en ModuloOperaciones
2. Agrega una entrada en el menú
3. Agrega un case en el método ejecutar
4. Crea el método privado para la operación

❓ **¿Cómo ejecuto las pruebas?**
`ruby test_calculadora.rb`

❓ **¿El historial se borra al cerrar?**
No, el historial se guarda en historial_calculadora.json y se carga al iniciar.

❓ **¿Puedo usar la calculadora desde la línea de comandos?**
No directamente, pero puedes implementar un modo batch (Ejercicio 2).

❓ **¿Qué significan los require_relative?**
Carga archivos relativos al archivo actual. Por ejemplo, modulo_operaciones.rb
debe estar en el mismo directorio.

❓ **¿Por qué usar módulos en lugar de clases separadas?**
Los módulos (include) permiten compartir comportamiento sin herencia. Es más
flexible y sigue el principio de composición sobre herencia.

❓ **¿Cómo mantengo el código organizado?**
Sigue el diseño modular: cada módulo tiene una responsabilidad única. Usa
nombres descriptivos y sigue las convenciones de Ruby.

🏗️ **Arquitecto de Software** — Diseña la "Calculadora del Futuro"

Diseña un sistema de calculadora extensible que pueda cargar plugins de
operaciones desde archivos externos. El sistema debe:

1. Cargar dinámicamente operaciones desde archivos Ruby en un directorio
2. Cada plugin de operación debe ser un archivo Ruby independiente
3. Los plugins deben auto-registrarse en la calculadora
4. El menú debe adaptarse dinámicamente a los plugins cargados
5. Debe permitir habilitar/deshabilitar plugins

```ruby
# plugin_sistema.rb - Sistema de plugins para la calculadora
module PluginSistema
  class PluginManager
    attr_reader :plugins

    def initialize(directorio_plugins = "plugins")
      @directorio = directorio_plugins
      @plugins = []
    end

    def cargar_plugins
      return unless Dir.exist?(@directorio)
      Dir.glob("#{@directorio}/*.rb").each do |archivo|
        begin
          load archivo
          nombre_plugin = File.basename(archivo, ".rb")
          clase_plugin = Object.const_get(nombre_plugin.camelize)
          @plugins << clase_plugin.new if clase_plugin.method_defined?(:ejecutar)
          puts "  Plugin cargado: #{nombre_plugin}"
        rescue => e
          puts "  Error cargando #{archivo}: #{e.message}"
        end
      end
    end

    def plugin?(nombre)
      @plugins.any? { |p| p.nombre == nombre }
    end

    def ejecutar_plugin(nombre, *args)
      plugin = @plugins.find { |p| p.nombre == nombre }
      plugin&.ejecutar(*args)
    end
  end

  class PluginBase
    attr_reader :nombre, :descripcion

    def initialize(nombre, descripcion)
      @nombre = nombre
      @descripcion = descripcion
    end

    def ejecutar(*args)
      raise NotImplementedError, "Los plugins deben implementar #ejecutar"
    end

    def menu_texto
      "#{@nombre}: #{@descripcion}"
    end
  end
end
```
