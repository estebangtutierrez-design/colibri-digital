EJEMPLO DE CÓDIGO
----------------------------------------

import { test, expect, Page } from '@playwright/test'

interface LoginPage {
  fillEmail(email: string): Promise<void>
  fillPassword(pwd: string): Promise<void>
  submit(): Promise<void>
}

class LoginPageImpl implements LoginPage {
  constructor(private page: Page) {}
  async fillEmail(email: string) {
    await this.page.fill('[name="email"]', email)
  }
  async fillPassword(pwd: string) {
    await this.page.fill('[name="password"]', pwd)
  }
  async submit() {
    await this.page.click('button[type="submit"]')
  }
}

test('login flow', async ({ page }) => {
  const login = new LoginPageImpl(page)
  await login.fillEmail('user@test.com')
  await login.fillPassword('password')
  await login.submit()
  await expect(page).toHaveURL('/dashboard')
})
