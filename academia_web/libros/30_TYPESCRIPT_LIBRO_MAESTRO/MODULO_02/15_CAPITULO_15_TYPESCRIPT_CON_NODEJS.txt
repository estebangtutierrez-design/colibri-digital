2. TIPOS PARA NODE

import type y export type para type-only imports.



3. PATRONES NODE

Error handling tipado: Result<T, E> pattern. Async/await con tipos.

Middleware tipado con Express. Streams tipados. Worker threads con

tipos. EventEmitter tipado con genéricos.

EJEMPLO DE CÓDIGO
----------------------------------------

import express, { Request, Response, NextFunction } from 'express'

interface AppConfig {
  port: number
  database: {
    host: string
    port: number
  }
}

type AsyncHandler = (req: Request, res: Response, next: NextFunction) => Promise<void>

function asyncHandler(fn: AsyncHandler) {
  return (req: Request, res: Response, next: NextFunction) => {
    Promise.resolve(fn(req, res, next)).catch(next)
  }
}

const app = express()
app.get('/users', asyncHandler(async (req, res) => {
  const users = await getUsers()
  res.json(users)
}))
