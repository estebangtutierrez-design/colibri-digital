CONTENIDO DEL CAPÍTULO
===============================================================================

1. INFER EN CONDITIONAL TYPES

T extends (...args: infer A) => infer R ? [A, R] : never. Extraer

tipos de funciones, promesas, arrays. Deep inference patterns.



2. VARIADIC TUPLE TYPES

[...T, ...U] concatenación. [string, ...T, number] patrones.

Tuple rest elements. Labeled tuple elements. Partial parameter spec.



3. TEMPLATE LITERALS WITH INFER

`${infer Prefix}/${infer Suffix}` parsing en tiempo de tipo.

Parsear rutas, URLs, patrones. Router type-safe paths.

Deep object path typing con infer recursivo.

EJEMPLO DE CÓDIGO
----------------------------------------

// Infer return type
type ReturnOf<T> = T extends (...args: any[]) => infer R ? R : never
type AsyncReturn<T> = T extends Promise<infer R> ? R : never

// Variadic tuples
function concat<T extends any[], U extends any[]>(
  a: [...T],
  b: [...U]
): [...T, ...U] {
  return [...a, ...b]
}

const result = concat([1, 'a'], [true, 2n])
// type: [number, string, boolean, bigint]

// Path parsing
type ParsePath<T extends string> =
  T extends `${infer Prefix}/${infer Rest}`
    ? [Prefix, ...ParsePath<Rest>]
    : [T]

type PathParts = ParsePath<'users/123/posts'>
// ['users', '123', 'posts']
